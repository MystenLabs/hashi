4//! `key-provisioner provision` (per-KP recovery against a fresh withdraw-mode guardian).
5//!
6//! Run by a key provisioner when a new guardian instance is brought up to
7//! replace one that went down. The KP decrypts through their yubikey-backed gpg
8//! setup; plaintext never touches disk, but the raw share scalar is held in this
9//! process' memory long enough to verify and re-encrypt it. The flow:
10//!
11//! 1. Heartbeat audit (check A) selects the single live session.
12//! 2. The new guardian's signed `GuardianInfo` is fetched from its S3 `init/`
13//!    log and verified against the enclave attestation (check B). Bucket,
14//!    limiter config, `mpc_master_g`, and `enclave_btc_pubkey == None` are all
15//!    confirmed.
16//! 3. The authoritative `ceremony/` log is scraped for the secret-sharing
17//!    instance the new guardian was booted with; it must match.
18//! 4. Initial limiter state is recovered from the prior enclave's max-seq
19//!    Success withdrawal log (or genesis) and confirmed equal (check C).
20//! 5. The committee comes from the latest signed `committee-update/` log (or
21//!    on-chain Hashi state); the `state_hash` is recomputed and confirmed (check D).
22//! 6. This KP's encrypted share is read from `shares/{seq}-{session}.json`
23//!    (attestation-anchored), every share's recipients are verified against
24//!    the roster, and this KP's share is located by fingerprint.
25//! 7. The share is decrypted via the yubikey (`gpg --decrypt` over a pipe;
26//!    plaintext stays in memory) and verified against its commitment.
27//! 8. The decrypted share is HPKE-encrypted to the new guardian's
28//!    `encryption_pubkey` (from its `GuardianInfo`), with `state_hash` as AAD,
29//!    producing a `GuardianEncryptedShare` ready for `provisioner_init`.
30//! 9. The share is submitted to the configured relay endpoint, which runs the
31//!    same pre-checks before forwarding it. Submission itself awaits the
32//!    relay's `single_provisioner_init` RPC (TODO).
33

59pub async fn run(cfg: Config) -> anyhow::Result<()> {
60    cfg.kp_roster.validate()?;
61    let guardian_s3 = cfg.guardian_s3.resolve().await?;
62
63    info!(
64        phase = "setup",
65        bucket = guardian_s3.bucket_name(),
66        region = guardian_s3.region(),
67        num_shares = cfg.kp_roster.num_shares,
68        threshold = cfg.kp_roster.threshold,
69        relay_endpoint = %cfg.relay_endpoint,
70        "running provision flow",
71    );
72
73    // One reader for the whole run: it owns the S3 client and the trusted-key
74    // cache, so each session's attestation is verified once whichever check
75    // reads that session first.
76    info!(
77        phase = "s3 connect",
78        bucket = guardian_s3.bucket_name(),
79        region = guardian_s3.region(),
80        current_git_revision = %cfg.kp_roster.pcr_allowlist.current_build().git_revision(),
81        current_pcr0 = hex::encode(cfg.kp_roster.pcr_allowlist.current_build().pcr0()),
82        prev_build_count = cfg.kp_roster.pcr_allowlist.prev_builds().len(),
83        "connecting to guardian log bucket",
84    );
85    let allowlist = cfg.kp_roster.pcr_allowlist();
86    let mut reader = GuardianReader::new(&guardian_s3, allowlist.clone())
87        .await
88        .context("connect to guardian log bucket")?;
89    info!(phase = "s3 connect", "connected to guardian log bucket");
90
91    info!(
92        phase = "sui connect",
93        sui_rpc = %cfg.hashi.sui_rpc,
94        package_id = %cfg.hashi.hashi_ids.package_id,
95        hashi_object_id = %cfg.hashi.hashi_ids.hashi_object_id,
96        "connecting to Sui RPC for Hashi on-chain state",
97    );
98    let onchain_state = cfg.hashi.onchain_state().await?;
99    info!(phase = "sui connect", "connected to Sui RPC");
100
101    let master_g = onchain_state.onchain_verifying_key_g()?;
102    info!(phase = "setup", master_g = ?master_g, "fetched on-chain MPC master G");
103
104    info!(
105        phase = "roster load",
106        cert_count = cfg.kp_roster.kp_pgp_cert_paths.len(),
107        "loading + validating full KP cert roster",
108    );
109    let certs = load_certs(&cfg.kp_roster.kp_pgp_cert_paths)?;
110    info!(
111        phase = "roster load",
112        cert_count = certs.len(),
113        "KP cert roster loaded"
114    );
115
116    let kp_pgp_cert_path = cfg.require_kp_pgp_cert_path("key-provisioner provision")?;
117    let kp_cert = PgpPublicCert::new(
118        std::fs::read_to_string(kp_pgp_cert_path)
119            .with_context(|| format!("read KP cert at {}", kp_pgp_cert_path.display()))?,
120    )
121    .with_context(|| format!("invalid PGP cert at {}", kp_pgp_cert_path.display()))?;
122    let want_fp = kp_cert.fingerprint();
123    info!(
124        phase = "setup",
125        fingerprint = %want_fp,
126        kp_cert_path = %kp_pgp_cert_path.display(),
127        "loaded this KP's cert",
128    );
129    ensure_cert_in_roster(&kp_cert, &certs)?;
130
131    // 1. Check A — no past enclave's heartbeats remain & gather the latest
132    //    enclave's session id.
133    info!(
134        phase = "heartbeat audit",
135        "auditing heartbeats to select the single live session",
136    );
137    let session_id = heartbeat_checks::heartbeat_audit(&mut reader).await?;
138    info!(
139        phase = "heartbeat audit",
140        session_id = %session_id,
141        "selected live session",
142    );
143
144    // 2. Check B — fetch + verify the new guardian's signed `GuardianInfo`.
145    info!(
146        phase = "guardian info",
147        session_id = %session_id,
148        "fetching + verifying new guardian's signed GuardianInfo from S3",
149    );
150    let verified_session = reader
151        .get_session_info(&session_id, BuildPolicy::Current)
152        .await?;
153    let guardian_info = verified_session.info;
154    let (
155        enclave_ss_instance,
156        enclave_bucket_info,
157        enclave_enc_pubkey_bytes,
158        enclave_state_hash,
159        enclave_git_revision,
160        enclave_btc_pubkey,
161        enclave_limiter_state,
162        enclave_limiter_config,
163        enclave_current_committee_epoch,
164        enclave_mpc_master_g,
165    ) = guardian_info
166        .clone()
167        .into_parts()
168        .context("Guardian info has missing fields")?;
169    info!(
170        phase = "guardian info",
171        session_id = %session_id,
172        bucket = %enclave_bucket_info.bucket,
173        region = %enclave_bucket_info.region,
174        enc_pubkey = hex::encode(&enclave_enc_pubkey_bytes),
175        state_hash = hex::encode(enclave_state_hash),
176        btc_pubkey_set = enclave_btc_pubkey.is_some(),
177        committee_epoch = enclave_current_committee_epoch,
178        limiter_refill_rate = enclave_limiter_config.refill_rate,
179        limiter_max_capacity = enclave_limiter_config.max_bucket_capacity,
180        verified_git_revision = %enclave_git_revision,
181        "guardian info verified against current build; cross-checking against config",
182    );
183    anyhow::ensure!(
184        enclave_git_revision == allowlist.current_build().git_revision(),
185        "Guardian git revision mismatch: expected {}, got {}",
186        allowlist.current_build().git_revision(),
187        enclave_git_revision
188    );
189    anyhow::ensure!(
190        guardian_s3.bucket_info == enclave_bucket_info,
191        "Guardian bucket info mismatch: expected {:?}, got {:?}",
192        guardian_s3.bucket_info,
193        enclave_bucket_info
194    );
195    anyhow::ensure!(
196        cfg.limiter_config == enclave_limiter_config,
197        "Guardian limiter config mismatch: expected {:?}, got {:?}",
198        cfg.limiter_config,
199        enclave_limiter_config
200    );
201    anyhow::ensure!(
202        enclave_btc_pubkey.is_none(),
203        "Guardian has a BTC pubkey => provisioner init over"
204    );
205    anyhow::ensure!(
206        master_g == enclave_mpc_master_g,
207        "MPC master g mismatch: expected {:?}, got {:?}",
208        master_g,
209        enclave_mpc_master_g
210    );
211    info!(
212        phase = "guardian info",
213        session_id = %session_id,
214        "guardian info checks passed (bucket, limiter config, mpc_master_g, btc_pubkey=None)",
215    );
216
217    // 3. Confirm the new guardian was booted with the same secret-sharing
218    //    instance the authoritative `ceremony/` log records.
219    info!(
220        phase = "ceremony instance",
221        "scraping authoritative ceremony/ log for the secret-sharing instance",
222    );
223    let (ceremony_session, scraped_instance, roster) = reader
224        .read_latest_ceremony(BuildPolicy::AnyAllowlisted)
225        .await?
226        .context("no ceremony log found in S3; key setup has not run")?;
227    let sharing_seq = scraped_instance.sharing_seq();
228    info!(
229        phase = "ceremony instance",
230        ceremony_session = %ceremony_session,
231        sharing_seq,
232        n = scraped_instance.num_shares(),
233        t = scraped_instance.threshold(),
234        roster_len = roster.len(),
235        "scraped latest ceremony entry",
236    );
237    anyhow::ensure!(
238        scraped_instance == enclave_ss_instance,
239        "Enclave secret sharing instance mismatch: expected {:?}, got {:?}",
240        scraped_instance,
241        enclave_ss_instance
242    );
243    info!(
244        phase = "ceremony instance",
245        ceremony_session = %ceremony_session,
246        sharing_seq,
247        "ceremony instance matches enclave",
248    );
249
250    // 4. Check C — source the initial limiter state. If a prior enclave left
251    //    Success withdrawal logs we recover from them; otherwise (first
252    //    deployment, or a rotation where the prior enclave processed no
253    //    withdrawals) we initialize from genesis.
254    info!(
255        phase = "limiter recovery",
256        "recovering initial limiter state from prior enclave's withdraw logs",
257    );
258    let limiter_state = match limiter_recovery::recover_limiter_state(&mut reader).await? {
259        Some(mut recovered) => {
260            // Cap to the current config's bucket capacity in case max capacity
261            // was lowered across the rotation. (Raising is fine — refill will
262            // fill it.)
263            recovered.num_tokens_available = recovered
264                .num_tokens_available
265                .min(cfg.limiter_config.max_bucket_capacity);
266            info!(
267                phase = "limiter recovery",
268                source = "recovered",
269                next_seq = recovered.next_seq,
270                num_tokens_available = recovered.num_tokens_available,
271                last_updated_at = recovered.last_updated_at,
272                "recovered limiter state from prior enclave's max-seq Success log",
273            );
274            recovered
275        }
276        None => {
277            info!(
278                phase = "limiter recovery",
279                source = "genesis",
280                "no prior Success withdrawal logs found; initializing limiter from genesis",
281            );
282            LimiterState::genesis(&cfg.limiter_config)
283        }
284    };
285    anyhow::ensure!(
286        limiter_state == enclave_limiter_state,
287        "limiter state mismatch: expected {:?}, got {:?}",
288        limiter_state,
289        enclave_limiter_state
290    );
291    info!(
292        phase = "limiter recovery",
293        next_seq = limiter_state.next_seq,
294        num_tokens_available = limiter_state.num_tokens_available,
295        "recovered limiter state matches enclave",
296    );
297
298    // 5. Check D — recompute the init state the operator booted the enclave
299    //    with; its digest is the `state_hash` we bind as the share's AAD. The
300    //    committee comes from the latest signed `committee-update/` log or,
301    //    before any update exists, from authoritative on-chain Hashi state.
302    info!(
303        phase = "state hash",
304        "recomputing state_hash from committee + limiter + master_g",
305    );
306    let committee = match reader
307        .read_latest_committee(BuildPolicy::AnyAllowlisted)
308        .await?
309    {
310        Some(scraped) => {
311            info!(
312                phase = "state hash",
313                epoch = scraped.epoch,
314                source = "committee-update log",
315                "scraped latest committee-update log",
316            );
317            scraped.try_into()?
318        }
319        None => {
320            let committee = onchain_state
321                .current_committee()
322                .context("no current committee on chain (DKG not yet complete?)")?;
323            info!(
324                phase = "state hash",
325                epoch = committee.epoch(),
326                source = "on-chain Hashi state",
327                "no committee-update log; falling back to on-chain current committee",
328            );
329            committee
330        }
331    };
332    anyhow::ensure!(
333        committee.epoch() == enclave_current_committee_epoch,
334        "committee epoch mismatch: expected {}, got {}",
335        committee.epoch(),
336        enclave_current_committee_epoch
337    );
338    let committee_epoch = committee.epoch();
339    let expected_state =
340        WithdrawModeState::new(committee, cfg.limiter_config, limiter_state, master_g)?;
341    let state_hash = expected_state.digest();
342    anyhow::ensure!(
343        state_hash == enclave_state_hash,
344        "state_hash mismatch: expected {}, got {}",
345        hex::encode(state_hash),
346        hex::encode(enclave_state_hash)
347    );
348    info!(
349        phase = "state hash",
350        committee_epoch = committee_epoch,
351        state_hash = hex::encode(state_hash),
352        "recomputed state_hash matches enclave",
353    );
354
355    // 6. Check E — read + verify this KP's encrypted share. The ceremony state
356    //    is constructed directly from the S3 log reads above so we don't pay
357    //    for a second ceremony/ + shares/ walk.
358    info!(
359        phase = "share read",
360        ceremony_session = %ceremony_session,
361        sharing_seq,
362        "reading + verifying this KP's encrypted share from shares/",
363    );
364    let encrypted_shares = reader
365        .read_shares(&ceremony_session, sharing_seq, BuildPolicy::AnyAllowlisted)
366        .await?;
367    let state = VerifiedCeremonyState::from_scraped(
368        ceremony_session.clone(),
369        scraped_instance.clone(),
370        encrypted_shares,
371        &roster,
372        cfg.kp_roster.num_shares,
373        cfg.kp_roster.threshold,
374    )?;
375    state.verify_encrypted_share_recipients(&certs)?;
376    info!(
377        phase = "share read",
378        share_count = state.encrypted_shares.len(),
379        roster_matches = true,
380        all_recipients_verified = true,
381        "shares/ log verified: every share is addressed only to its labeled KP cert",
382    );
383
384    // Find this KP's share by exact fingerprint match. Given the roster checks
385    // above, this should always succeed; keep the error for defensive clarity.
386    let kp_encrypted_share = state
387        .encrypted_shares
388        .iter()
389        .find(|s| s.recipient_fingerprint == want_fp)
390        .ok_or_else(|| {
391            anyhow::anyhow!(
392                "no share in the shares/ log is labeled for this KP's fingerprint \
393                 {want_fp} (labeled fingerprints: {:?})",
394                state
395                    .encrypted_shares
396                    .iter()
397                    .map(|s| s.recipient_fingerprint.clone())
398                    .collect::<Vec<_>>()
399            )
400        })?;
401    info!(
402        phase = "share read",
403        share_id = kp_encrypted_share.id.get(),
404        fingerprint = %kp_encrypted_share.recipient_fingerprint,
405        "located this KP's encrypted share",
406    );
407
408    // 7. Decrypt via yubikey (gpg streams plaintext over a pipe — never hits
409    //    disk) and verify the decrypted share matches its commitment.
410    info!(
411        phase = "share decrypt",
412        share_id = kp_encrypted_share.id.get(),
413        "decrypting share via yubikey (ciphertext piped via stdin; plaintext in memory)",
414    );
415    let decrypted = decrypt_share(kp_encrypted_share)?;
416    state
417        .secret_sharing_instance
418        .commitments()
419        .verify_share(&decrypted)
420        .context("decrypted share does not match its commitment")?;
421    let expected_commitment = state
422        .secret_sharing_instance
423        .commitments()
424        .iter()
425        .find(|c| c.id == decrypted.id)
426        .ok_or_else(|| {
427            anyhow::anyhow!(
428                "commitment for share id {} missing despite verify_share success",
429                decrypted.id
430            )
431        })?;
432    info!(
433        phase = "share decrypt",
434        share_id = decrypted.id.get(),
435        commitment = hex::encode(&expected_commitment.digest),
436        "decrypted share matches its commitment",
437    );
438
439    // 8. HPKE-encrypt the decrypted share to the new guardian's pubkey, binding
440    //    the verified `state_hash` as AAD.
441    info!(
442        phase = "share build",
443        share_id = decrypted.id.get(),
444        enc_pubkey = hex::encode(&enclave_enc_pubkey_bytes),
445        state_hash = hex::encode(state_hash),
446        "HPKE-encrypting share to new guardian's pubkey (state_hash AAD)",
447    );
448    let guardian_pub_key =
449        EncPubKey::from_bytes(&enclave_enc_pubkey_bytes).map_err(anyhow::Error::msg)?;
450    let encrypted_share = ProvisionerInitRequest::build_from_share(
451        &decrypted,
452        &guardian_pub_key,
453        state_hash,
454        &mut thread_rng(),
455    );
456    info!(
457        phase = "share build",
458        share_id = encrypted_share.id.get(),
459        "built GuardianEncryptedShare ready for submission",
460    );
461
462    // 9. Submit. The relay collects T-of-N shares before forwarding them to the
463    //    guardian in one `ProvisionerInit` call.
464    info!(
465        phase = "summary",
466        session_id = %session_id,
467        ceremony_session = %ceremony_session,
468        share_id = decrypted.id.get(),
469        fingerprint = %want_fp,
470        sharing_seq,
471        state_hash = hex::encode(state_hash),
472        enc_pubkey = hex::encode(&enclave_enc_pubkey_bytes),
473        relay_endpoint = %cfg.relay_endpoint,
474        "share built; submitting to relay",
475    );
476    submit_provisioner_init_to_relay(
477        &cfg.relay_endpoint,
478        &session_id,
479        guardian_info,
480        encrypted_share,
481        allowlist.current_build(),
482    )
483    .await?;
484    Ok(())
485}

487/// Submit this KP's share to the relay endpoint. The relay fronts the
488/// guardian's `GetGuardianInfo`, so we verify the relay's reported session +
489/// `GuardianInfo` against the values we already pinned from S3 before any
490/// further step.
491///
492/// TODO: once the relay exposes `single_provisioner_init`, send `encrypted_share`
493/// to it here. The relay collects T-of-N shares before calling the guardian's
494/// batch `provisioner_init`.
495async fn submit_provisioner_init_to_relay(
496    endpoint: &str,
497    expected_session_id: &str,
498    expected_guardian_info: GuardianInfo,
499    encrypted_share: GuardianEncryptedShare,
500    current_build: &BuildPcrs,
501) -> anyhow::Result<()> {
502    info!(
503        phase = "relay submit",
504        endpoint = %endpoint,
505        "connecting to relay endpoint",
506    );
507    let mut client =
508        pb::guardian_service_client::GuardianServiceClient::connect(endpoint.to_string())
509            .await
510            .with_context(|| format!("failed to connect to relay endpoint {endpoint}"))?;
511    info!(phase = "relay submit", endpoint = %endpoint, "connected to relay");
512
513    info!(
514        phase = "relay submit",
515        endpoint = %endpoint,
516        expected_session_id = %expected_session_id,
517        "running relay-side prechecks (GetGuardianInfo + session pin + GuardianInfo match)",
518    );
519    prechecks(
520        &mut client,
521        expected_session_id,
522        expected_guardian_info,
523        current_build,
524    )
525    .await
526    .with_context(|| "relay endpoint pre-check failed")?;
527    info!(
528        phase = "relay submit",
529        endpoint = %endpoint,
530        share_id = encrypted_share.id.get(),
531        "relay prechecks passed; share submission awaits single_provisioner_init",
532    );
533    todo!("submit encrypted_share via single_provisioner_init once the relay exposes it");
534}
535
536async fn prechecks(
537    client: &mut pb::guardian_service_client::GuardianServiceClient<tonic::transport::Channel>,
538    expected_session_id: &str,
539    expected_guardian_info: GuardianInfo,
540    current_build: &BuildPcrs,
541) -> anyhow::Result<()> {
542    let resp_pb = client
543        .get_guardian_info(pb::GetGuardianInfoRequest {})
544        .await
545        .with_context(|| "GetGuardianInfo RPC failed")?
546        .into_inner();
547
548    let resp = GetGuardianInfoResponse::try_from(resp_pb)?;
549
550    let verified = resp.verify(current_build)?;
551    let actual_session_id = verified.session_id;
552    info!(
553        phase = "relay submit",
554        actual_session_id = %actual_session_id,
555        expected_session_id = %expected_session_id,
556        "relay returned GuardianInfo; verifying attestation + signature + session match",
557    );
558
559    anyhow::ensure!(
560        actual_session_id == expected_session_id,
561        "relay endpoint session mismatch: expected {}, got {}",
562        expected_session_id,
563        actual_session_id
564    );
565    anyhow::ensure!(
566        verified.info == expected_guardian_info,
567        "relay endpoint GuardianInfo mismatch: expected {:?}, got {:?}",
568        expected_guardian_info,
569        verified.info
570    );
571    info!(
572        phase = "relay submit",
573        session_id = %actual_session_id,
574        "relay GuardianInfo matches expected (attestation, signature, session, fields)",
575    );
576
577    Ok(())
578}
