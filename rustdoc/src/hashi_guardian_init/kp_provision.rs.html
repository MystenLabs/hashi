4//! `key-provisioner provision` (per-KP recovery against a fresh withdraw-mode guardian).
5//!
6//! Run by a key provisioner when a new guardian instance is brought up to
7//! replace one that went down. The KP decrypts through their yubikey-backed gpg
8//! setup; plaintext never touches disk, but the raw share scalar is held in this
9//! process' memory long enough to verify and re-encrypt it. The flow:
10//!
11//! 1. The relay/standby endpoint's signed `GuardianInfo` is fetched and
12//!    verified against the enclave attestation, pinning the standby session.
13//! 2. The same session's S3 `init/` log is fetched and required to match the
14//!    endpoint `GuardianInfo`. Bucket, limiter config, `mpc_master_g`, and
15//!    `enclave_btc_pubkey == None` are all confirmed.
16//! 3. The authoritative `ceremony/` log is scraped for the secret-sharing
17//!    instance the new guardian was booted with; it must match.
18//! 4. The stable `InitConfig` is recomputed from limiter config, master G, PCR
19//!    allowlist, and network; its `config_hash` is confirmed.
20//! 5. The optional genesis state hash is independently derived from S3 and
21//!    current on-chain state and confirmed against the enclave's pin.
22//! 6. This KP's PGP-encrypted share is read from the latest
23//!    `kp-shares/{seq}/` state (attestation-anchored), every encrypted copy's
24//!    recipient is verified against the roster, and this KP's selected
25//!    ciphertext is located by fingerprint.
26//! 7. The selected ciphertext is decrypted via its yubikey (`gpg --decrypt`
27//!    over a pipe; plaintext stays in memory) and verified against the
28//!    commitment.
29//! 8. The decrypted share is HPKE-encrypted to the new guardian's
30//!    `encryption_pubkey` (from its `GuardianInfo`) while constructing a PI
31//!    request bound to the pinned session and verified config/genesis hashes.
32//! 9. The request is signed and submitted to the configured relay endpoint via
33//!    `SingleProvisionerInit`. The relay accumulates T-of-N signed submissions
34//!    and calls the guardian's batch `provisioner_init` once it has enough; the
35//!    enclave re-verifies every signature and binding.
36

61pub async fn run(cfg: Config, do_genesis: bool) -> anyhow::Result<()> {
62    cfg.kp_roster.validate()?;
63    let guardian_s3 = cfg.guardian_s3.resolve().await?;
64
65    info!(
66        phase = "setup",
67        bucket = guardian_s3.bucket_name(),
68        region = guardian_s3.region(),
69        num_shares = cfg.kp_roster.num_shares,
70        threshold = cfg.kp_roster.threshold,
71        relay_endpoint = %cfg.relay_endpoint,
72        do_genesis,
73        "running provision flow",
74    );
75
76    // One reader for the whole run: it owns the S3 client and the trusted-key
77    // cache, so each session's attestation is verified once whichever check
78    // reads that session first.
79    info!(
80        phase = "s3 connect",
81        bucket = guardian_s3.bucket_name(),
82        region = guardian_s3.region(),
83        current_git_revision = %cfg.kp_roster.pcr_allowlist.current_build().git_revision(),
84        current_pcr0 = hex::encode(cfg.kp_roster.pcr_allowlist.current_build().pcr0()),
85        prev_build_count = cfg.kp_roster.pcr_allowlist.prev_builds().len(),
86        "connecting to guardian log bucket",
87    );
88    let allowlist = cfg.kp_roster.pcr_allowlist();
89    let mut reader = GuardianReader::new(&guardian_s3, allowlist.clone())
90        .await
91        .context("connect to guardian log bucket")?;
92    info!(phase = "s3 connect", "connected to guardian log bucket");
93
94    info!(
95        phase = "sui connect",
96        sui_rpc = %cfg.hashi.sui_rpc,
97        package_id = %cfg.hashi.hashi_ids.package_id,
98        hashi_object_id = %cfg.hashi.hashi_ids.hashi_object_id,
99        "connecting to Sui RPC for Hashi on-chain state",
100    );
101    let onchain_state = cfg.hashi.onchain_state().await?;
102    info!(phase = "sui connect", "connected to Sui RPC");
103
104    let master_g = onchain_state.onchain_verifying_key_g()?;
105    info!(phase = "setup", master_g = ?master_g, "fetched on-chain MPC master G");
106
107    info!(
108        phase = "roster load",
109        share_count = cfg.kp_roster.kp_pgp_cert_paths.len(),
110        certificate_count = cfg.kp_roster.cert_count(),
111        "loading + validating full KP certificate roster",
112    );
113    let certs_roster = cfg.kp_roster.load_certs_roster()?;
114    info!(
115        phase = "roster load",
116        share_count = certs_roster.num_kps(),
117        certificate_count = cfg.kp_roster.cert_count(),
118        "KP certificate roster loaded"
119    );
120
121    let kp_pgp_cert_path = cfg.require_kp_pgp_cert_path("key-provisioner provision")?;
122    let kp_cert = load_kp_cert(kp_pgp_cert_path)?;
123    let kp_fingerprint = kp_cert.fingerprint();
124    anyhow::ensure!(
125        certs_roster
126            .certs_for_fingerprint(&kp_fingerprint)
127            .is_some(),
128        "this KP's cert (fingerprint {kp_fingerprint}) is not among the configured \
129         kp_roster.kp_pgp_cert_paths"
130    );
131    info!(
132        phase = "setup",
133        fingerprint = %kp_fingerprint,
134        "loaded this KP's selected cert",
135    );
136
137    // 1. Ask the relay/standby endpoint which session KPs are provisioning.
138    // Active guardian heartbeats may still exist, so identity comes from the
139    // endpoint KPs will submit to rather than from S3 heartbeat discovery.
140    info!(
141        phase = "guardian endpoint",
142        endpoint = %cfg.relay_endpoint,
143        "fetching + verifying relay endpoint GuardianInfo",
144    );
145    let endpoint_verified =
146        verified_endpoint_guardian_info(&cfg.relay_endpoint, allowlist.current_build()).await?;
147    let session_id = endpoint_verified.session_id;
148    let guardian_info = endpoint_verified.info;
149    info!(
150        phase = "guardian endpoint",
151        session_id = %session_id,
152        "relay endpoint GuardianInfo verified; pinned standby session",
153    );
154
155    // 2. Fetch + verify the same session's signed `GuardianInfo` from S3.
156    info!(
157        phase = "guardian info",
158        session_id = %session_id,
159        "fetching + verifying pinned standby session's signed GuardianInfo from S3",
160    );
161    let verified_session = reader
162        .get_session_info(&session_id, BuildPolicy::Current)
163        .await?;
164    let GuardianInfo {
165        lifecycle,
166        secret_sharing_instance,
167        bucket_info,
168        encryption_pubkey: enclave_enc_pubkey_bytes,
169        config_hash,
170        genesis_state_hash,
171        untrusted_git_revision: enclave_git_revision,
172        enclave_btc_pubkey,
173        limiter_state: enclave_limiter_state,
174        limiter_config,
175        current_committee_epoch: enclave_current_committee_epoch,
176        mpc_master_g,
177    } = &guardian_info;
178    anyhow::ensure!(
179        *lifecycle == WithdrawStage::OperatorInitialized.into(),
180        "Guardian lifecycle is {lifecycle:?}; expected withdraw/operator_initialized"
181    );
182    let enclave_ss_instance = secret_sharing_instance
183        .as_ref()
184        .context("Guardian info missing secret_sharing_instance")?;
185    let enclave_bucket_info = bucket_info
186        .as_ref()
187        .context("Guardian info missing bucket_info")?;
188    let enclave_config_hash = config_hash
189        .as_ref()
190        .copied()
191        .context("Guardian info missing config_hash")?;
192    let enclave_genesis_state_hash = *genesis_state_hash;
193    let enclave_limiter_config = limiter_config
194        .as_ref()
195        .copied()
196        .context("Guardian info missing limiter_config")?;
197    let enclave_mpc_master_g = mpc_master_g
198        .as_ref()
199        .context("Guardian info missing mpc_master_g")?;
200    info!(
201        phase = "guardian info",
202        session_id = %session_id,
203        bucket = %enclave_bucket_info.bucket,
204        region = %enclave_bucket_info.region,
205        enc_pubkey = hex::encode(enclave_enc_pubkey_bytes),
206        config_hash = hex::encode(enclave_config_hash),
207        btc_pubkey_set = enclave_btc_pubkey.is_some(),
208        limiter_refill_rate = enclave_limiter_config.refill_rate,
209        limiter_max_capacity = enclave_limiter_config.max_bucket_capacity,
210        verified_git_revision = %enclave_git_revision,
211        "guardian info verified against current build; cross-checking against config",
212    );
213    anyhow::ensure!(
214        enclave_git_revision == allowlist.current_build().git_revision(),
215        "Guardian git revision mismatch: expected {}, got {}",
216        allowlist.current_build().git_revision(),
217        enclave_git_revision
218    );
219    anyhow::ensure!(
220        &guardian_s3.bucket_info == enclave_bucket_info,
221        "Guardian bucket info mismatch: expected {:?}, got {:?}",
222        guardian_s3.bucket_info,
223        enclave_bucket_info
224    );
225    anyhow::ensure!(
226        cfg.limiter_config == enclave_limiter_config,
227        "Guardian limiter config mismatch: expected {:?}, got {:?}",
228        cfg.limiter_config,
229        enclave_limiter_config
230    );
231    anyhow::ensure!(
232        enclave_btc_pubkey.is_none(),
233        "Guardian has a BTC pubkey => provisioner init over"
234    );
235    anyhow::ensure!(
236        enclave_limiter_state.is_none(),
237        "Guardian has limiter_state => operator activation already ran"
238    );
239    anyhow::ensure!(
240        enclave_current_committee_epoch.is_none(),
241        "Guardian has current_committee_epoch => operator activation already ran"
242    );
243    let oi_info = verified_session.info;
244    ensure_oi_info_matches_post_init(&oi_info, &guardian_info)
245        .with_context(|| format!("S3 GuardianInfo mismatch for session {session_id}"))?;
246    anyhow::ensure!(
247        &master_g == enclave_mpc_master_g,
248        "MPC master g mismatch: expected {:?}, got {:?}",
249        master_g,
250        enclave_mpc_master_g
251    );
252    info!(
253        phase = "guardian info",
254        session_id = %session_id,
255        "guardian info checks passed (bucket, limiter config, mpc_master_g, standby not activated)",
256    );
257
258    // 3. Read the ceremony + KP share state and confirm the new guardian was
259    //    booted with the same secret-sharing instance.
260    info!(
261        phase = "ceremony instance",
262        "scraping authoritative ceremony/ and kp-shares/ logs",
263    );
264    let state = reader
265        .read_latest_ceremony_state(BuildPolicy::AnyAllowlisted)
266        .await?
267        .context("no ceremony log found in S3; key setup has not run")?;
268    let sharing_seq = state.secret_sharing_instance.sharing_seq();
269    info!(
270        phase = "ceremony instance",
271        sharing_seq,
272        n = state.secret_sharing_instance.num_shares(),
273        t = state.secret_sharing_instance.threshold(),
274        "scraped latest ceremony entry",
275    );
276    anyhow::ensure!(
277        state.secret_sharing_instance == *enclave_ss_instance,
278        "Enclave secret sharing instance mismatch: expected {:?}, got {:?}",
279        state.secret_sharing_instance,
280        enclave_ss_instance
281    );
282    info!(
283        phase = "ceremony instance",
284        sharing_seq, "ceremony instance matches enclave",
285    );
286
287    // 4. Recompute the stable config the operator armed the enclave with; its
288    //    digest is the `config_hash` bound into the signed PI submission.
289    info!(
290        phase = "config hash",
291        "recomputing config_hash from limiter config + master G + PCR allowlist + network",
292    );
293    let expected_config = InitConfig::new(
294        cfg.limiter_config,
295        master_g,
296        allowlist.clone(),
297        cfg.bitcoin_network,
298    )?;
299    let config_hash = expected_config.digest();
300    anyhow::ensure!(
301        config_hash == enclave_config_hash,
302        "config_hash mismatch: expected {}, got {}",
303        hex::encode(config_hash),
304        hex::encode(enclave_config_hash)
305    );
306    info!(
307        phase = "config hash",
308        config_hash = hex::encode(config_hash),
309        "recomputed config_hash matches enclave",
310    );
311
312    // Require the explicit intent marker to match S3 state. For genesis, bind
313    // the independently read on-chain committee into this KP's signed PI submission.
314    let latest_committee = reader
315        .read_latest_committee(BuildPolicy::AnyAllowlisted)
316        .await?;
317    let expected_genesis_state_hash = match (do_genesis, latest_committee) {
318        (false, Some(_)) => None,
319        (true, None) => {
320            let committee = onchain_state
321                .current_committee()
322                .context("no current committee on chain (DKG not yet complete?)")?;
323            Some(GenesisState::new(committee).digest())
324        }
325        (true, Some(committee)) => anyhow::bail!(
326            "--do-genesis was supplied, but a serving committee already exists at epoch {}",
327            committee.epoch
328        ),
329        (false, None) => anyhow::bail!(
330            "no serving committee exists; re-run with --do-genesis to explicitly authorize genesis"
331        ),
332    };
333    anyhow::ensure!(
334        expected_genesis_state_hash == enclave_genesis_state_hash,
335        "genesis_state_hash mismatch: expected {:?}, got {:?}",
336        expected_genesis_state_hash.map(hex::encode),
337        enclave_genesis_state_hash.map(hex::encode)
338    );
339    info!(
340        phase = "genesis state hash",
341        genesis_state_hash = ?expected_genesis_state_hash.map(hex::encode),
342        "independently derived genesis state matches enclave",
343    );
344
345    // 5. Verify this KP's encrypted share from the ceremony + KP-share state
346    //    read above.
347    info!(
348        phase = "share read",
349        sharing_seq, "verifying this KP's encrypted share from kp-shares/",
350    );
351    state.validate_sharing_params(cfg.kp_roster.num_shares, cfg.kp_roster.threshold)?;
352    state.encrypted_shares.verify_recipients(&certs_roster)?;
353    info!(
354        phase = "share read",
355        cert_seq = state.cert_seq,
356        share_count = state.encrypted_shares.share_count(),
357        ciphertext_count = state.encrypted_shares.ciphertext_count(),
358        all_recipients_verified = true,
359        "kp-shares log verified: every PGP-encrypted share matches the expected KP cert sets",
360    );
361
362    // 6. Decrypt and commitment-check the ciphertext selected by this KP.
363    let decrypted = decrypt_kp_share_copies(&state, std::slice::from_ref(&kp_cert))?;
364    let expected_commitment = state
365        .secret_sharing_instance
366        .commitments()
367        .iter()
368        .find(|c| c.id == decrypted.id)
369        .ok_or_else(|| {
370            anyhow::anyhow!(
371                "commitment for share id {} missing despite verify_share success",
372                decrypted.id
373            )
374        })?;
375    info!(
376        phase = "share decrypt",
377        share_id = decrypted.id.get(),
378        commitment = hex::encode(&expected_commitment.digest),
379        "decrypted share matches its commitment",
380    );
381
382    // 7. HPKE-encrypt the decrypted share to the new guardian's pubkey. The
383    //    signed relay request below binds it to the verified config hash.
384    info!(
385        phase = "share build",
386        share_id = decrypted.id.get(),
387        enc_pubkey = hex::encode(enclave_enc_pubkey_bytes),
388        config_hash = hex::encode(config_hash),
389        "HPKE-encrypting share to new guardian's pubkey",
390    );
391    let guardian_pub_key =
392        EncPubKey::from_bytes(enclave_enc_pubkey_bytes).map_err(anyhow::Error::msg)?;
393    let request = SingleProvisionerInitRequest::build_from_share(
394        session_id.clone(),
395        config_hash,
396        expected_genesis_state_hash,
397        &decrypted,
398        &guardian_pub_key,
399        &mut thread_rng(),
400    );
401    info!(
402        phase = "share build",
403        share_id = request.encrypted_share().id.get(),
404        "built SingleProvisionerInitRequest ready for signing",
405    );
406
407    // 8. Submit. The relay collects T-of-N shares before forwarding them to the
408    //    guardian in one `ProvisionerInit` call.
409    // The cert that decrypted the share also authorizes its relay submission.
410    let signer_cert = &kp_cert;
411    info!(
412        phase = "summary",
413        session_id = %session_id,
414        share_id = decrypted.id.get(),
415        signer_fingerprint = %signer_cert.fingerprint(),
416        sharing_seq,
417        config_hash = hex::encode(config_hash),
418        genesis_state_hash = ?expected_genesis_state_hash.map(hex::encode),
419        enc_pubkey = hex::encode(enclave_enc_pubkey_bytes),
420        relay_endpoint = %cfg.relay_endpoint,
421        "share built; submitting to relay",
422    );
423    submit_provisioner_init_to_relay(
424        &cfg.relay_endpoint,
425        guardian_info,
426        request,
427        signer_cert,
428        allowlist.current_build(),
429    )
430    .await?;
431    Ok(())
432}

434async fn verified_endpoint_guardian_info(
435    endpoint: &str,
436    current_build: &BuildPcrs,
437) -> anyhow::Result<VerifiedGuardianInfo> {
438    let mut client =
439        pb::guardian_service_client::GuardianServiceClient::connect(endpoint.to_string())
440            .await
441            .with_context(|| format!("failed to connect to relay endpoint {endpoint}"))?;
442    verified_live_guardian_info(&mut client, current_build)
443        .await
444        .with_context(|| format!("verify relay endpoint GuardianInfo at {endpoint}"))
445}
446
447/// Submit this KP's share to the relay endpoint. The relay fronts the
448/// guardian's current session and rejects `SingleProvisionerInit` if it no
449/// longer matches the session or config this KP signed. The relay collects
450/// T-of-N submissions and calls the guardian's batch `provisioner_init` once it
451/// has enough; the guardian re-verifies every signature.
452async fn submit_provisioner_init_to_relay(
453    endpoint: &str,
454    expected_guardian_info: GuardianInfo,
455    request: SingleProvisionerInitRequest,
456    signer_cert: &PgpPublicCert,
457    current_build: &BuildPcrs,
458) -> anyhow::Result<()> {
459    let expected_session_id = request.expected_session_id();
460    info!(
461        phase = "relay submit",
462        endpoint = %endpoint,
463        "connecting to relay endpoint",
464    );
465    let mut client =
466        pb::guardian_service_client::GuardianServiceClient::connect(endpoint.to_string())
467            .await
468            .with_context(|| format!("failed to connect to relay endpoint {endpoint}"))?;
469    info!(phase = "relay submit", endpoint = %endpoint, "connected to relay");
470
471    info!(
472        phase = "relay submit",
473        endpoint = %endpoint,
474        expected_session_id = %expected_session_id,
475        "running relay-side prechecks (GetGuardianInfo + session pin + GuardianInfo match)",
476    );
477    prechecks(
478        &mut client,
479        expected_session_id,
480        &expected_guardian_info,
481        current_build,
482    )
483    .await
484    .with_context(|| "relay endpoint pre-check failed")?;
485    let share_id = request.encrypted_share().id.get();
486    info!(
487        phase = "relay submit",
488        endpoint = %endpoint,
489        share_id,
490        "relay prechecks passed; submitting share via SingleProvisionerInit",
491    );
492
493    let mut relay_client = pb::guardian_relay_service_client::GuardianRelayServiceClient::connect(
494        endpoint.to_string(),
495    )
496    .await
497    .with_context(|| format!("failed to connect to relay endpoint {endpoint}"))?;
498
499    // Detached-sign the exact (session, config, share) bytes with this KP's
500    // offline key. The relay pre-verifies the request before buffering it and
501    // the enclave authoritatively re-verifies it before using the share.
502    let signed_request = KpSigned::new(request, signer_cert.clone(), None)
503        .map_err(anyhow::Error::msg)
504        .context("sign the relay submission with the KP key")?;
505    let resp = relay_client
506        .single_provisioner_init(pb::SignedSingleProvisionerInitRequest::from(signed_request))
507        .await
508        .with_context(|| "SingleProvisionerInit RPC failed")?
509        .into_inner();
510
511    if resp.completed {
512        info!(
513            phase = "relay submit",
514            share_id, "share accepted; the relay has provisioned the guardian (threshold reached)",
515        );
516    } else {
517        info!(
518            phase = "relay submit",
519            share_id,
520            have = resp.have,
521            need = resp.need,
522            "share accepted; the relay is still collecting shares before it provisions the guardian",
523        );
524    }
525    Ok(())
526}

528async fn prechecks(
529    client: &mut pb::guardian_service_client::GuardianServiceClient<tonic::transport::Channel>,
530    expected_session_id: &str,
531    expected_guardian_info: &GuardianInfo,
532    current_build: &BuildPcrs,
533) -> anyhow::Result<()> {
534    let verified = verified_live_guardian_info(client, current_build).await?;
535    let actual_session_id = verified.session_id;
536    info!(
537        phase = "relay submit",
538        actual_session_id = %actual_session_id,
539        expected_session_id = %expected_session_id,
540        "relay returned GuardianInfo; verifying attestation + signature + session match",
541    );
542
543    anyhow::ensure!(
544        actual_session_id.as_str() == expected_session_id,
545        "relay endpoint session mismatch: expected {}, got {}",
546        expected_session_id,
547        actual_session_id
548    );
549    anyhow::ensure!(
550        &verified.info == expected_guardian_info,
551        "relay endpoint GuardianInfo mismatch: expected {:?}, got {:?}",
552        expected_guardian_info,
553        verified.info
554    );
555    info!(
556        phase = "relay submit",
557        session_id = %actual_session_id,
558        "relay GuardianInfo matches expected (attestation, signature, session, fields)",
559    );
560
561    Ok(())
562}
