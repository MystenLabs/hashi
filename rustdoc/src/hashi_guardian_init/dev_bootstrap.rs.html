4//! Drive a fresh hashi-guardian from heartbeating-only to fully-initialized.
5//! Scrapes the on-chain `HashiCommittee`, generates BTC master + Shamir shares
6//! in memory, then runs OperatorInit -> GetGuardianInfo -> ProvisionerInit
7//! until the guardian reaches the secret-sharing threshold.

83pub async fn run(args: Args, onchain_state: &OnchainState) -> Result<()> {
84    // AWS creds stay env-only — passing them as CLI flags would leak via `ps`.
85    let bucket = required_env("AWS_S3_BUCKET")?;
86    let region = required_env("AWS_REGION")?;
87    let access_key = required_env("AWS_ACCESS_KEY_ID")?;
88    let secret_key = required_env("AWS_SECRET_ACCESS_KEY")?;
89    let network = parse_network(&args.bitcoin_network)?;
90
91    let committee = onchain_state
92        .current_committee()
93        .ok_or_else(|| anyhow!("no current committee on chain (DKG not yet complete?)"))?;
94    let committee_epoch = committee.epoch();
95    tracing::info!(
96        committee_epoch,
97        committee_total_weight = committee.total_weight(),
98        num_members = committee.members().len(),
99        "fetched on-chain committee"
100    );
101
102    let mut rng = thread_rng();
103    let n = args.num_shares;
104    let t = args.threshold;
105    let existing_secret = args
106        .master_secret_hex
107        .as_deref()
108        .map(parse_master_secret)
109        .transpose()?;
110    let material = generate_share_material(n, t, &mut rng, existing_secret)?;
111    tracing::info!(master_pubkey = %hex::encode(material.master_pubkey.serialize()),
112        n, t, "generated share material");
113
114    // Derived key must match the pubkey pinned on-chain at publish; a wrong or
115    // missing --master-secret-hex would otherwise provision a key the chain rejects.
116    let onchain_btc_pubkey = onchain_state.guardian_btc_public_key();
117    if let Some(onchain) = onchain_btc_pubkey {
118        let derived = material.master_pubkey.serialize();
119        anyhow::ensure!(
120            derived.as_slice() == onchain.as_slice(),
121            "derived master pubkey {} does not match on-chain \
122             guardian_btc_public_key {}; check HASHI_MASTER_SECRET_HEX",
123            hex::encode(derived),
124            hex::encode(&onchain),
125        );
126        tracing::info!("master pubkey matches on-chain guardian_btc_public_key");
127    }
128
129    let mut client = GuardianServiceClient::connect(args.guardian_endpoint.clone())
130        .await
131        .with_context(|| format!("connect to guardian at {}", args.guardian_endpoint))?;
132
133    // The operator now supplies the full init state at OperatorInit; its digest
134    // is the state_hash KPs bind as their share-encryption AAD.
135    let limiter_config = LimiterConfig {
136        refill_rate: args.refill_rate_sats_per_sec,
137        max_bucket_capacity: args.max_bucket_capacity_sats,
138    };
139    let limiter_state = LimiterState::genesis(&limiter_config);
140    // `hashi_btc_master_pubkey` is the MPC committee `G` the 2-of-2 scripts derive
141    // from — use the on-chain MPC `G` (as hashi does), not the guardian's own key.
142    let master_g = decode_mpc_master_g(&onchain_state.mpc_public_key())?;
143    let secret_sharing_instance = SecretSharingInstance::new(material.commitments.clone(), n, t, 0)
144        .map_err(|e| anyhow!("build SecretSharingInstance: {e:?}"))?;
145    let config = WithdrawModeConfig::new(
146        committee,
147        limiter_config,
148        limiter_state,
149        master_g,
150        secret_sharing_instance,
151        network,
152    )
153    .map_err(|e| anyhow!("build WithdrawModeConfig: {e:?}"))?;
154    let state_hash = config.state().digest();
155
156    let operator_init_req = pb::OperatorInitRequest {
157        s3_config: Some(pb::S3Config {
158            access_key: Some(access_key),
159            secret_key: Some(secret_key),
160            bucket_name: Some(bucket.clone()),
161            region: Some(region.clone()),
162        }),
163        state: Some(
164            withdraw_mode_config_to_pb(config)
165                .map_err(|e| anyhow!("encode WithdrawModeConfig: {e:?}"))?,
166        ),
167    };
168    tracing::info!("calling OperatorInit");
169    client
170        .operator_init(operator_init_req)
171        .await
172        .context("OperatorInit RPC failed")?;
173
174    let info_pb = client
175        .get_guardian_info(pb::GetGuardianInfoRequest {})
176        .await
177        .context("GetGuardianInfo RPC failed")?
178        .into_inner();
179    let resp = GetGuardianInfoResponse::try_from(info_pb)
180        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
181    let session_id = session_id_from_signing_pubkey(&resp.signing_pub_key);
182
183    // Verify the enclave's own signature on the info — without this the
184    // `encryption_pubkey` below would be unauthenticated and a buggy or
185    // hostile endpoint could trick us into encrypting shares to a key it
186    // controls.
187    //
188    // TODO: also authenticate `signing_pub_key` against the AWS Nitro
189    // attestation once the guardian runs in an enclave. Today
190    // `hashi_types::guardian::verify_enclave_attestation` is a no-op, so a
191    // malicious operator with their own signing key can pass all the
192    // checks below (sign their own GuardianInfo, echo the bucket and
193    // commitments we just submitted, return their own encryption key) and
194    // capture `t` encrypted shares. The current deployment runs the
195    // guardian in a k8s cluster — the attestation gate lands when we move
196    // to Nitro.
197    let info = resp
198        .signed_info
199        .verify(&resp.signing_pub_key)
200        .map_err(|e| anyhow!("verify GuardianInfo signature (session={session_id}): {e:?}"))?;
201
202    // Match against what we just sent in OperatorInit. Catches a stale or
203    // wrong enclave echoing back a different config than the one we set up.
204    let returned_bucket = info.bucket_info.as_ref().ok_or_else(|| {
205        anyhow!("guardian info missing bucket_info; OperatorInit may have silently failed")
206    })?;
207    anyhow::ensure!(
208        returned_bucket.bucket == bucket && returned_bucket.region == region,
209        "bucket mismatch: submitted {region}/{bucket}, guardian echoed {}/{}",
210        returned_bucket.region,
211        returned_bucket.bucket,
212    );
213    let returned_instance = info
214        .secret_sharing_instance
215        .as_ref()
216        .ok_or_else(|| anyhow!("guardian info missing secret_sharing_instance"))?;
217    anyhow::ensure!(
218        *returned_instance.commitments() == material.commitments
219            && returned_instance.num_shares() == n
220            && returned_instance.threshold() == t
221            && returned_instance.sharing_seq() == 0,
222        "secret-sharing instance mismatch: guardian echoed different scheme than was submitted"
223    );
224    let returned_state_hash = info
225        .state_hash
226        .ok_or_else(|| anyhow!("guardian info missing state_hash"))?;
227    anyhow::ensure!(
228        returned_state_hash == state_hash,
229        "state_hash mismatch: guardian echoed a different init state than was submitted"
230    );
231
232    let enc_pubkey = EncPubKey::from_bytes(&info.encryption_pubkey)
233        .map_err(|e| anyhow!("decode guardian encryption pubkey (session={session_id}): {e:?}"))?;
234    tracing::info!(session_id = %session_id, "guardian info verified");
235
236    tracing::info!("submitting ProvisionerInit with {t} shares");
237    let encrypted_shares = material
238        .shares
239        .iter()
240        .take(t)
241        .map(|share| {
242            ProvisionerInitRequest::build_from_share(share, &enc_pubkey, state_hash, &mut rng)
243        })
244        .collect();
245    let pb_req = provisioner_init_request_to_pb(ProvisionerInitRequest::new(encrypted_shares))
246        .map_err(|e| anyhow!("encode ProvisionerInitRequest: {e:?}"))?;
247    client
248        .provisioner_init(pb_req)
249        .await
250        .context("ProvisionerInit RPC failed")?;
251
252    // Pin: confirm we're still talking to the same enclave session we set up.
253    // If the guardian restarted between OperatorInit and now, the new session
254    // would have a different signing key — and the shares we just submitted
255    // would belong to a session that no longer exists.
256    let post_resp_pb = client
257        .get_guardian_info(pb::GetGuardianInfoRequest {})
258        .await
259        .context("post-ProvisionerInit GetGuardianInfo RPC failed")?
260        .into_inner();
261    let post_resp = GetGuardianInfoResponse::try_from(post_resp_pb)
262        .map_err(|e| anyhow!("decode post-ProvisionerInit GetGuardianInfoResponse: {e:?}"))?;
263    let post_session_id = session_id_from_signing_pubkey(&post_resp.signing_pub_key);
264    anyhow::ensure!(
265        post_session_id == session_id,
266        "guardian session changed mid-bootstrap: started {session_id}, now {post_session_id} \
267         (enclave likely restarted; rerun bootstrap)"
268    );
269
270    println!("Guardian fully initialized.");
271    println!("  session_id:               {session_id}");
272    println!(
273        "  master pubkey:            {}",
274        hex::encode(material.master_pubkey.serialize())
275    );
276    println!("  committee_epoch:          {committee_epoch}");
277    println!(
278        "  refill_rate_sats_per_sec: {}",
279        args.refill_rate_sats_per_sec
280    );
281    println!(
282        "  max_bucket_capacity_sats: {}",
283        args.max_bucket_capacity_sats
284    );
285    Ok(())
286}

288struct ShareMaterial {
289    shares: Vec<Share>,
290    commitments: ShareCommitments,
291    master_pubkey: BitcoinPubkey,
292}
293
294/// Fresh BTC master key + Shamir shares + matching commitments, all in memory.
295/// `master_pubkey` is the x-only key the guardian reconstructs from any `t`
296/// shares out of `n`. When `existing_secret` is `Some`, the shares are split
297/// from that secret so the resulting `master_pubkey` matches whatever was
298/// already pinned on-chain at publish time.
299fn generate_share_material<R: CryptoRng + RngCore>(
300    n: usize,
301    t: usize,
302    rng: &mut R,
303    existing_secret: Option<k256::SecretKey>,
304) -> Result<ShareMaterial> {
305    let k256_sk = existing_secret.unwrap_or_else(|| k256::SecretKey::random(&mut *rng));
306
307    let params =
308        SecretSharingParams::new(n, t).map_err(|e| anyhow!("invalid sharing params: {e:?}"))?;
309    let shares = split_secret(&k256_sk, &params, rng);
310    let commitments_vec: Vec<ShareCommitment> = shares.iter().map(commit_share).collect();
311    let commitments =
312        ShareCommitments::new(commitments_vec).expect("commitments built from a valid share set");
313
314    let secp = Secp256k1::new();
315    let btc_sk = BtcSecretKey::from_slice(&k256_sk.to_bytes())
316        .expect("k256 secret key bytes are a valid secp256k1 secret");
317    let keypair = Keypair::from_secret_key(&secp, &btc_sk);
318    let (master_pubkey, _parity) = keypair.x_only_public_key();
319
320    Ok(ShareMaterial {
321        shares,
322        commitments,
323        master_pubkey,
324    })
325}
326
327fn parse_master_secret(hex_str: &str) -> Result<k256::SecretKey> {
328    let bytes = hex::decode(hex_str.trim_start_matches("0x"))
329        .context("invalid hex for --master-secret-hex")?;
330    anyhow::ensure!(
331        bytes.len() == 32,
332        "--master-secret-hex must decode to 32 bytes, got {}",
333        bytes.len(),
334    );
335    k256::SecretKey::from_slice(&bytes)
336        .map_err(|e| anyhow!("--master-secret-hex is not a valid secp256k1 scalar: {e}"))
337}

339fn required_env(name: &str) -> Result<String> {
340    env::var(name).map_err(|_| anyhow!("required env var `{name}` is not set"))
341}

343/// Decode the on-chain MPC committee verifying key `G` (`mpc_public_key` = `bcs(G)`),
344/// matching `Hashi::onchain_verifying_key_g` — the point the guardian must derive from.
345fn decode_mpc_master_g(mpc_public_key: &[u8]) -> Result<HashiMasterG> {
346    anyhow::ensure!(
347        !mpc_public_key.is_empty(),
348        "on-chain mpc_public_key is empty (DKG / end_reconfig may not have completed yet)"
349    );
350    bcs::from_bytes(mpc_public_key).map_err(|e| anyhow!("decode on-chain MPC verifying key G: {e}"))
351}
352
353fn parse_network(s: &str) -> Result<Network> {
354    match s.to_ascii_lowercase().as_str() {
355        "mainnet" | "bitcoin" => Ok(Network::Bitcoin),
356        "testnet" => Ok(Network::Testnet),
357        "regtest" => Ok(Network::Regtest),
358        "signet" => Ok(Network::Signet),
359        _ => Err(anyhow!(
360            "unknown BITCOIN_NETWORK `{s}`; expected mainnet/testnet/regtest/signet"
361        )),
362    }

365#[cfg(test)]
366mod tests {
367    use super::*;
368    use hashi_types::guardian::crypto::combine_shares;
369    use hashi_types::guardian::crypto::k256_sk_to_btc_keypair;
370
371    #[test]
372    fn decode_mpc_master_g_round_trips_and_rejects_empty() {
373        use fastcrypto::groups::GroupElement;
374        // The on-chain `mpc_public_key` is `bcs::to_bytes(&G)`; decoding must yield
375        // the same point hashi derives from.
376        let g = HashiMasterG::generator();
377        let onchain_bytes = bcs::to_bytes(&g).expect("serialize G");
378        let decoded = decode_mpc_master_g(&onchain_bytes).expect("decode valid G");
379        assert_eq!(
380            bcs::to_bytes(&decoded).expect("re-serialize"),
381            onchain_bytes,
382            "decoded master G must round-trip the on-chain bytes"
383        );
384        // An empty key (DKG / end_reconfig not yet complete) is rejected, not
385        // silently turned into a bogus master.
386        assert!(decode_mpc_master_g(&[]).is_err());
387    }
388
389    #[test]
390    fn generated_shares_reconstruct_to_master_pubkey() {
391        const TEST_N: usize = 5;
392        const TEST_T: usize = 3;
393        let mut rng = rand::thread_rng();
394        let material = generate_share_material(TEST_N, TEST_T, &mut rng, None).unwrap();
395
396        assert_eq!(material.shares.len(), TEST_N);
397        let subset = &material.shares[..TEST_T];
398        let reconstructed_sk = combine_shares(subset, TEST_T).expect("threshold shares combine");
399        let reconstructed_kp = k256_sk_to_btc_keypair(&reconstructed_sk);
400        let (reconstructed_xonly, _) = reconstructed_kp.x_only_public_key();
401        assert_eq!(reconstructed_xonly, material.master_pubkey);
402    }
403
404    #[test]
405    fn pre_supplied_secret_round_trips_through_shares() {
406        const TEST_N: usize = 5;
407        const TEST_T: usize = 3;
408        let mut rng = rand::thread_rng();
409        let secret = k256::SecretKey::random(&mut rng);
410        let expected_xonly = {
411            let secp = Secp256k1::new();
412            let sk = BtcSecretKey::from_slice(&secret.to_bytes()).unwrap();
413            Keypair::from_secret_key(&secp, &sk).x_only_public_key().0
414        };
415
416        let material =
417            generate_share_material(TEST_N, TEST_T, &mut rng, Some(secret.clone())).unwrap();
418        assert_eq!(material.master_pubkey, expected_xonly);
419
420        let subset = &material.shares[..TEST_T];
421        let reconstructed = combine_shares(subset, TEST_T).expect("threshold shares combine");
422        let reconstructed_kp = k256_sk_to_btc_keypair(&reconstructed);
423        let (reconstructed_xonly, _) = reconstructed_kp.x_only_public_key();
424        assert_eq!(reconstructed_xonly, expected_xonly);
425    }

427    #[test]

430        let sk = parse_master_secret(&hex).unwrap();
