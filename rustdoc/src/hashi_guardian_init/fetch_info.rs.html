4//! Fetch deployed guardian's public keys.
5//!
6//! Calls `GuardianService/GetGuardianInfo` and prints a hex-encoded key
7//! to stdout. Used by the deploy workflow to feed `hashi publish
8//! --guardian-public-key` / `--guardian-btc-public-key` so the keys get
9//! recorded on chain.
10//!
11//! TODO: Remove this bootstrap helper once deploy no longer records guardian
12//! keys by querying a live guardian. This command verifies the returned
13//! GuardianInfo signature, but it does not verify the guardian's Nitro
14//! attestation or PCRs.
15
16use anyhow::Context;
17use anyhow::Result;
18use anyhow::anyhow;
19use clap::Parser;
20use clap::ValueEnum;
21use hashi_types::guardian::GetGuardianInfoResponse;
22use hashi_types::proto as pb;
23use hashi_types::proto::guardian_service_client::GuardianServiceClient;
24
25#[derive(Parser)]
26pub struct Args {
27    /// gRPC endpoint of the deployed guardian (e.g. `http://localhost:3000`).
28    #[arg(long, env = "GUARDIAN_ENDPOINT")]
29    pub endpoint: String,
30
31    /// Which key to print.
32    #[arg(long, value_enum, default_value_t = Field::SigningPubKey)]
33    pub field: Field,
34}
35
36#[derive(Clone, Copy, Debug, ValueEnum)]
37pub enum Field {
38    /// Ed25519 RPC identity key (32 bytes hex).
39    SigningPubKey,
40    /// X-only enclave BTC pubkey (32 bytes hex). Absent before `provisioner_init`.
41    EnclaveBtcPubkey,
42}
43
44pub async fn run(args: Args) -> Result<()> {
45    let mut client = GuardianServiceClient::connect(args.endpoint.clone())
46        .await
47        .with_context(|| format!("connect to guardian at {}", args.endpoint))?;
48    let resp = client
49        .get_guardian_info(pb::GetGuardianInfoRequest {})
50        .await
51        .context("GetGuardianInfo RPC failed")?
52        .into_inner();
53    let verified = GetGuardianInfoResponse::try_from(resp)
54        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
55    let verified = verified
56        // TODO: Accept PCR config here and use `verify` for attestation/PCR checks.
57        .verify_signed_info_without_attestation()
58        .map_err(|e| anyhow!("verify GuardianInfo signature: {e:?}"))?;
59    match args.field {
60        Field::SigningPubKey => {
61            println!("{}", hex::encode(verified.signing_pub_key.as_bytes()));
62        }
63        Field::EnclaveBtcPubkey => {
64            let btc_pk = verified.info.enclave_btc_pubkey.ok_or_else(|| {
65                anyhow!(
66                    "guardian /info did not return enclave_btc_pubkey; \
67                     provisioner_init may not have completed"
68                )
69            })?;
70            println!("{}", hex::encode(btc_pk.serialize()));
71        }
72    }
73    Ok(())
74}
