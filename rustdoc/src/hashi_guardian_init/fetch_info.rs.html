4//! Fetch deployed guardian's public keys.
5//!
6//! Calls `GuardianService/GetGuardianInfo` and prints a hex-encoded key
7//! to stdout. Used by the deploy workflow to feed `hashi publish
8//! --guardian-public-key` / `--guardian-btc-public-key` so the keys get
9//! recorded on chain.
10
11use anyhow::Context;
12use anyhow::Result;
13use anyhow::anyhow;
14use clap::Parser;
15use clap::ValueEnum;
16use hashi_types::guardian::GetGuardianInfoResponse;
17use hashi_types::proto as pb;
18use hashi_types::proto::guardian_service_client::GuardianServiceClient;
19
20#[derive(Parser)]
21pub struct Args {
22    /// gRPC endpoint of the deployed guardian (e.g. `http://localhost:3000`).
23    #[arg(long, env = "GUARDIAN_ENDPOINT")]
24    pub endpoint: String,
25
26    /// Which key to print.
27    #[arg(long, value_enum, default_value_t = Field::SigningPubKey)]
28    pub field: Field,
29}
30
31#[derive(Clone, Copy, Debug, ValueEnum)]
32pub enum Field {
33    /// Ed25519 RPC identity key (32 bytes hex).
34    SigningPubKey,
35    /// X-only enclave BTC pubkey (32 bytes hex). Absent before `provisioner_init`.
36    EnclaveBtcPubkey,
37}
38
39pub async fn run(args: Args) -> Result<()> {
40    let mut client = GuardianServiceClient::connect(args.endpoint.clone())
41        .await
42        .with_context(|| format!("connect to guardian at {}", args.endpoint))?;
43    let resp = client
44        .get_guardian_info(pb::GetGuardianInfoRequest {})
45        .await
46        .context("GetGuardianInfo RPC failed")?
47        .into_inner();
48    let info = GetGuardianInfoResponse::try_from(resp)
49        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
50    match args.field {
51        Field::SigningPubKey => {
52            println!("{}", hex::encode(info.signing_pub_key.as_bytes()));
53        }
54        Field::EnclaveBtcPubkey => {
55            let btc_pk = info.signed_info.data.enclave_btc_pubkey.ok_or_else(|| {
56                anyhow!(
57                    "guardian /info did not return enclave_btc_pubkey; \
58                     provisioner_init may not have completed"
59                )
60            })?;
61            println!("{}", hex::encode(btc_pk.serialize()));
62        }
63    }
64    Ok(())
65}
