4//! Fetch deployed guardian's public keys.
5//!
6//! Calls `GuardianService/GetGuardianInfo` and prints a hex-encoded key
7//! to stdout. Used by the deploy workflow to feed `hashi publish
8//! --guardian-public-key` / `--guardian-btc-public-key` so the keys get
9//! recorded on chain.
10//!
11//! TODO: Remove this bootstrap helper once deploy no longer records guardian
12//! keys by querying a live guardian. This command does not verify the guardian's
13//! signature, Nitro attestation, or PCRs.
14
15use anyhow::Context;
16use anyhow::Result;
17use anyhow::anyhow;
18use clap::Parser;
19use clap::ValueEnum;
20use hashi_types::guardian::GetGuardianInfoResponse;
21use hashi_types::proto as pb;
22use hashi_types::proto::guardian_service_client::GuardianServiceClient;
23
24#[derive(Parser)]
25pub struct Args {
26    /// gRPC endpoint of the deployed guardian (e.g. `http://localhost:3000`).
27    #[arg(long, env = "GUARDIAN_ENDPOINT")]
28    pub endpoint: String,
29
30    /// Which key to print.
31    #[arg(long, value_enum, default_value_t = Field::SigningPubKey)]
32    pub field: Field,
33}
34
35#[derive(Clone, Copy, Debug, ValueEnum)]
36pub enum Field {
37    /// Ed25519 RPC identity key (32 bytes hex).
38    SigningPubKey,
39    /// X-only enclave BTC pubkey (32 bytes hex). Absent before `provisioner_init`.
40    EnclaveBtcPubkey,
41}
42
43pub async fn run(args: Args) -> Result<()> {
44    let mut client = GuardianServiceClient::connect(args.endpoint.clone())
45        .await
46        .with_context(|| format!("connect to guardian at {}", args.endpoint))?;
47    let resp = client
48        .get_guardian_info(pb::GetGuardianInfoRequest {})
49        .await
50        .context("GetGuardianInfo RPC failed")?
51        .into_inner();
52    let response = GetGuardianInfoResponse::try_from(resp)
53        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
54    // TODO: Accept PCR config here and use `verify_live` for attestation/PCR checks.
55    let (info, signing_pub_key) = response.into_info_unchecked();
56    match args.field {
57        Field::SigningPubKey => {
58            println!("{}", hex::encode(signing_pub_key.as_bytes()));
59        }
60        Field::EnclaveBtcPubkey => {
61            let btc_pk = info.enclave_btc_pubkey.ok_or_else(|| {
62                anyhow!(
63                    "guardian /info did not return enclave_btc_pubkey; \
64                     provisioner_init may not have completed"
65                )
66            })?;
67            println!("{}", hex::encode(btc_pk.serialize()));
68        }
69    }
70    Ok(())
71}
