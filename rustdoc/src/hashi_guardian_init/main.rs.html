4use anyhow::Context;
5use clap::Parser;
6use clap::Subcommand;
7use hashi::config::Config;
8use hashi::onchain::OnchainState;
9use std::path::PathBuf;

11mod config;

19#[derive(Parser)]
20#[command(name = "hashi-guardian-init")]
21#[command(about = "Off-enclave tooling to initialize a guardian")]
22struct Cli {
23    #[command(subcommand)]
24    command: Command,
25}
26
27#[derive(Subcommand)]
28enum Command {
29    /// Run a key provisioner's init checks against guardian S3 logs and emit its share.
30    Provision {
31        /// Path to provisioner-init YAML config file.
32        #[arg(long)]
33        config: PathBuf,
34    },
35    /// Guardian helper tooling and dev-only shortcuts.
36    Tools {
37        #[command(subcommand)]
38        command: ToolsCommand,
39    },
40}
41
42#[derive(Subcommand)]
43enum ToolsCommand {
44    /// Drive the current centralized dev guardian bootstrap shortcut.
45    DevBootstrap {
46        #[command(flatten)]
47        config: ConfigArgs,
48        #[command(flatten)]
49        args: dev_bootstrap::Args,
50    },
51    /// Fetch deployed guardian public keys.
52    FetchInfo {
53        #[command(flatten)]
54        args: fetch_info::Args,
55    },
56    /// Generate a fresh BTC master keypair for the dev bootstrap shortcut.
57    GenerateMasterKey {
58        #[command(flatten)]
59        args: generate_master_key::Args,
60    },
61}
62
63#[derive(Parser)]
64struct ConfigArgs {
65    /// Path to a node config TOML file (provides sui-rpc and hashi-ids).
66    #[arg(long)]
67    config: PathBuf,
68}
69
70impl ConfigArgs {
71    fn load(&self) -> anyhow::Result<Config> {
72        let s = std::fs::read_to_string(&self.config)
73            .with_context(|| format!("failed to read config: {}", self.config.display()))?;
74        toml::from_str(&s).context("failed to parse config TOML")
75    }
76}
77
78#[tokio::main]
79async fn main() -> anyhow::Result<()> {
80    hashi_types::telemetry::TelemetryConfig::new()
81        .with_target(false)
82        .with_env()
83        .init();
84    hashi::init_crypto_provider();
85
86    match Cli::parse().command {
87        Command::Provision { config } => {
88            let cfg = provisioner::ProvisionerConfig::load_yaml(&config)?;
89            provisioner::run(cfg).await?;
90        }
91        Command::Tools { command } => match command {
92            ToolsCommand::DevBootstrap { config, args } => {
93                let cfg = config.load()?;
94                let sui_rpc = cfg
95                    .sui_rpc
96                    .as_deref()
97                    .ok_or_else(|| anyhow::anyhow!("config missing sui-rpc"))?;
98                println!("Connecting to Sui RPC: {sui_rpc}");
99                let (onchain_state, _watcher) =
100                    OnchainState::new(sui_rpc, cfg.hashi_ids(), None, None, None)
101                        .await
102                        .context("failed to connect to Sui RPC")?;
103                dev_bootstrap::run(args, &onchain_state).await?;
104            }
105            ToolsCommand::FetchInfo { args } => fetch_info::run(args).await?,
106            ToolsCommand::GenerateMasterKey { args } => generate_master_key::run(args)?,
107        },
108    }
109    Ok(())
110}
