4//! Production guardian key ceremony commands.
5//!
6//! `operator ceremony` drives a fresh ceremony-mode guardian through genesis BTC key setup:
7//! [`OperatorInit`] (ceremony mode, S3-only) -> [`SetupNewKey`] -> inspect each
8//! returned encrypted share to confirm it is addressed to the expected KP cert
9//! (without decrypting) -> cross-check the guardian's `ceremony/` audit log and
10//! `shares/` recovery log.
11//!
12//! [`OperatorInit`]: hashi_types::guardian::OperatorInitRequest
13//! [`SetupNewKey`]: hashi_types::guardian::SetupNewKeyRequest
14//! [`SetupNewKeyResponse`]: hashi_types::guardian::SetupNewKeyResponse
15

38/// Run the one-time production guardian key ceremony.
39///
40/// See the module docs for the full step-by-step flow. Each step is logged via
41/// `tracing` so the operator can follow exactly what is happening.
42pub async fn run(cfg: Config) -> Result<()> {
43    let guardian_s3 = cfg.guardian_s3.resolve().await?;
44
45    info!(
46        phase = "setup",
47        num_shares = cfg.kp_roster.num_shares,
48        threshold = cfg.kp_roster.threshold,
49        cert_count = cfg.kp_roster.kp_pgp_cert_paths.len(),
50        bucket = guardian_s3.bucket_name(),
51        region = guardian_s3.region(),
52        endpoint = %cfg.guardian_endpoint,
53        sui_rpc = %cfg.hashi.sui_rpc,
54        package_id = %cfg.hashi.hashi_ids.package_id,
55        hashi_object_id = %cfg.hashi.hashi_ids.hashi_object_id,
56        current_pcr0 = hex::encode(cfg.kp_roster.pcr_allowlist.current_build().pcr0()),
57        "running guardian key ceremony",
58    );
59
60    // 1. Validate config-level sharing params up front (also re-validated by
61    //    SetupNewKeyRequest::new).
62    cfg.kp_roster.validate()?;
63
64    // 2. Load + validate each KP's PGP cert.
65    info!(
66        phase = "roster load",
67        cert_count = cfg.kp_roster.kp_pgp_cert_paths.len(),
68        "loading + validating full KP cert roster",
69    );
70    let certs = load_certs(&cfg.kp_roster.kp_pgp_cert_paths)?;
71    info!(
72        phase = "roster load",
73        cert_count = certs.len(),
74        "KP cert roster loaded"
75    );
76    let setup_req = SetupNewKeyRequest::new(
77        certs.clone(),
78        cfg.kp_roster.num_shares,
79        cfg.kp_roster.threshold,
80    )
81    .map_err(|e| anyhow!("build SetupNewKeyRequest: {e:?}"))?;
82
83    // 3. Connect to the ceremony-mode guardian.
84    info!(
85        phase = "connect",
86        endpoint = %cfg.guardian_endpoint,
87        "connecting to ceremony-mode guardian",
88    );
89    let mut client = GuardianServiceClient::connect(cfg.guardian_endpoint.clone())
90        .await
91        .with_context(|| format!("connect to guardian at {}", cfg.guardian_endpoint))?;
92    info!(phase = "connect", endpoint = %cfg.guardian_endpoint, "connected to guardian");
93
94    // 4. operator_init (ceremony mode: S3 config only, no WithdrawModeConfig).
95    info!(
96        phase = "operator_init",
97        bucket = guardian_s3.bucket_name(),
98        region = guardian_s3.region(),
99        "calling OperatorInit (ceremony mode: S3 config only)",
100    );
101    let oi_req =
102        operator_init_request_to_pb(OperatorInitRequest::new_ceremony_mode(guardian_s3.clone()))
103            .map_err(|e| anyhow!("encode OperatorInitRequest: {e:?}"))?;
104    client
105        .operator_init(oi_req)
106        .await
107        .context("OperatorInit RPC failed")?;
108    info!(
109        phase = "operator_init",
110        "operator_init complete; guardian S3 logger installed"
111    );
112
113    // 5. get_guardian_info -> verify attestation/PCRs and signed info -> pin the
114    //    session id. This binds `signing_pub_key` (and thus the session) before
115    //    we trust the SetupNewKey response we'll verify against it below.
116    info!(phase = "guardian info", "calling GetGuardianInfo");
117    let info_pb = client
118        .get_guardian_info(pb::GetGuardianInfoRequest {})
119        .await
120        .context("GetGuardianInfo RPC failed")?
121        .into_inner();
122    let info_resp = GetGuardianInfoResponse::try_from(info_pb)
123        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
124    let allowlist = cfg.kp_roster.pcr_allowlist();
125    let verified = info_resp
126        .verify(allowlist.current_build())
127        .map_err(|e| anyhow!("verify GuardianInfo attestation/signature: {e:?}"))?;
128    let signing_pub_key = verified.signing_pub_key;
129    let session_id = verified.session_id;
130    info!(
131        phase = "guardian info",
132        session_id = %session_id,
133        signing_pubkey = hex::encode(signing_pub_key.as_bytes()),
134        "guardian info attestation and signature verified; session pinned",
135    );
136
137    info!(
138        phase = "attestation pin",
139        session_id = %session_id,
140        "connecting to guardian log bucket + verifying attestation against current build",
141    );
142    let mut reader = GuardianReader::new(&guardian_s3, allowlist)
143        .await
144        .context("connect to guardian log bucket")?;
145    let verified_session = reader
146        .get_session_info(&session_id, BuildPolicy::Current)
147        .await?;
148    let attested_signing_pub_key = verified_session.signing_pubkey;
149    ensure!(
150        attested_signing_pub_key == signing_pub_key,
151        "guardian S3 attestation signing pubkey differs from gRPC signing pubkey"
152    );
153    info!(
154        phase = "attestation pin",
155        session_id = %session_id,
156        "guardian S3 attestation matches gRPC signing key",
157    );
158
159    // 6. setup_new_key.
160    info!(
161        phase = "setup_new_key",
162        n = cfg.kp_roster.num_shares,
163        t = cfg.kp_roster.threshold,
164        "calling SetupNewKey",
165    );
166    let signed_resp_pb = client
167        .setup_new_key(setup_new_key_request_to_pb(setup_req))
168        .await
169        .context("SetupNewKey RPC failed")?
170        .into_inner();
171    let signed_resp = GuardianSigned::<SetupNewKeyResponse>::try_from(signed_resp_pb)
172        .map_err(|e| anyhow!("decode SignedSetupNewKeyResponse: {e:?}"))?;
173    info!(
174        phase = "setup_new_key",
175        n = cfg.kp_roster.num_shares,
176        t = cfg.kp_roster.threshold,
177        encrypted_share_count = signed_resp.data.encrypted_shares.len(),
178        "setup_new_key response received",
179    );
180
181    // 7. Verify the response signature under the pinned session's signing key,
182    //    and sanity-check the response shape.
183    let sharing_seq = 0u64;
184    let live = signed_resp
185        .verify(&signing_pub_key)
186        .map_err(|e| anyhow!("verify SetupNewKeyResponse signature: {e:?}"))
187        .and_then(|response| {
188            VerifiedCeremonyState::from_response(
189                response,
190                session_id.clone(),
191                sharing_seq,
192                cfg.kp_roster.num_shares,
193                cfg.kp_roster.threshold,
194            )
195        })?;
196    info!(
197        phase = "setup_new_key",
198        session_id = %live.session_id,
199        sharing_seq = live.secret_sharing_instance.sharing_seq(),
200        "verified SetupNewKeyResponse signature + shape",
201    );
202
203    // 8. Inspect each encrypted share's PGP recipients WITHOUT decrypting, and
204    //    confirm every share is addressed only to its expected cert.
205    info!(
206        phase = "roster verify",
207        share_count = live.encrypted_shares.len(),
208        "verifying every returned share is addressed only to its labeled KP cert (without decrypting)",
209    );
210    live.verify_encrypted_share_recipients(&certs)?;
211    info!(
212        phase = "roster verify",
213        "all returned shares verified against expected KP certs",
214    );
215
216    // 9. Cross-check the latest guardian ceremony/ and shares/ logs.
217    //    KPs will fetch the same shares/ object during key-provisioner ceremony.
218    info!(
219        phase = "log cross-check",
220        "cross-checking the latest guardian ceremony/ and shares/ logs",
221    );
222    let logged = VerifiedCeremonyState::latest_from_s3(
223        &mut reader,
224        cfg.kp_roster.num_shares,
225        cfg.kp_roster.threshold,
226    )
227    .await?;
228    anyhow::ensure!(
229        logged == live,
230        "ceremony/ and shares/ logs differ from the SetupNewKeyResponse"
231    );
232    info!(
233        phase = "log cross-check",
234        "ceremony/ and shares/ logs match the SetupNewKeyResponse",
235    );
236
237    // 10. Summary.
238    info!(
239        phase = "summary",
240        session_id = %live.session_id,
241        sharing_seq = live.secret_sharing_instance.sharing_seq(),
242        shares_key = %SharesLogMessage::object_key(
243            &live.session_id,
244            live.secret_sharing_instance.sharing_seq()
245        ),
246        n = live.secret_sharing_instance.num_shares(),
247        t = live.secret_sharing_instance.threshold(),
248        "guardian key ceremony complete",
249    );
250    for commitment in live.secret_sharing_instance.commitments().iter() {
251        info!(
252            phase = "summary",
253            share_id = commitment.id.get(),
254            commitment = hex::encode(&commitment.digest),
255            "share commitment",
256        );
257    }
258
259    Ok(())
260}
