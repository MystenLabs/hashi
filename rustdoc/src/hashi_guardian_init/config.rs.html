19#[derive(Deserialize)]
20pub struct Config {
21    pub hashi: HashiOnchainConfig,
22    pub guardian_s3: GuardianInitS3Config,
23    #[serde(deserialize_with = "deserialize_network")]
24    pub bitcoin_network: Network,
25    pub kp_roster: KpRosterConfig,
26    pub limiter_config: LimiterConfig,
27    /// Relay endpoint the KP's encrypted share is submitted to.
28    pub relay_endpoint: String,
29    /// gRPC endpoint of the guardian.
30    pub guardian_endpoint: String,
31    /// Path to this KP's armored OpenPGP public cert. Required by
32    /// key-provisioner commands.
33    pub kp_pgp_cert_path: Option<PathBuf>,
34}
35
36impl Config {
37    pub fn load_yaml(path: &Path) -> anyhow::Result<Self> {
38        let bytes = std::fs::read(path).with_context(|| {
39            format!("failed to read guardian init config at {}", path.display())
40        })?;
41        serde_yaml::from_slice(&bytes)
42            .with_context(|| format!("failed to parse guardian init yaml at {}", path.display()))
43    }
44
45    pub fn require_kp_pgp_cert_path(&self, command: &str) -> anyhow::Result<&Path> {
46        self.kp_pgp_cert_path.as_deref().ok_or_else(|| {
47            anyhow::anyhow!("{command} requires kp_pgp_cert_path in guardian init config")
48        })
49    }
50}

52fn deserialize_network<'de, D>(deserializer: D) -> Result<Network, D::Error>
53where
54    D: serde::Deserializer<'de>,
55{
56    let s = String::deserialize(deserializer)?;
57    parse_network(&s).map_err(serde::de::Error::custom)
58}
59
60fn parse_network(s: &str) -> anyhow::Result<Network> {
61    match s.to_ascii_lowercase().as_str() {
62        "mainnet" | "bitcoin" => Ok(Network::Bitcoin),
63        "testnet" => Ok(Network::Testnet),
64        "regtest" => Ok(Network::Regtest),
65        "signet" => Ok(Network::Signet),
66        _ => {
67            anyhow::bail!("unknown bitcoin_network `{s}`; expected mainnet/testnet/regtest/signet")
68        }
69    }
70}

72#[derive(Clone)]
73pub struct HashiOnchainConfig {
74    /// Sui RPC URL used to fetch Hashi on-chain state.
75    pub sui_rpc: String,
76    pub hashi_ids: HashiIds,
77}
78
79impl<'de> Deserialize<'de> for HashiOnchainConfig {
80    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
81    where
82        D: serde::Deserializer<'de>,
83    {
84        #[derive(Deserialize)]
85        struct HashiOnchainConfigWire {
86            sui_rpc: String,
87            package_id: String,
88            hashi_object_id: String,
89        }
90
91        let wire = HashiOnchainConfigWire::deserialize(deserializer)?;
92        Ok(Self {
93            sui_rpc: wire.sui_rpc,
94            hashi_ids: HashiIds {
95                package_id: wire.package_id.parse().map_err(serde::de::Error::custom)?,
96                hashi_object_id: wire
97                    .hashi_object_id
98                    .parse()
99                    .map_err(serde::de::Error::custom)?,
100            },
101        })
102    }
103}
104
105impl HashiOnchainConfig {
106    pub async fn onchain_state(&self) -> anyhow::Result<OnchainState> {
107        let (state, _watcher) = OnchainState::new(&self.sui_rpc, self.hashi_ids, None, None, None)
108            .await
109            .with_context(|| format!("failed to connect to Sui RPC at {}", self.sui_rpc))?;
110        Ok(state)
111    }
112}

114#[derive(Deserialize)]
115pub struct GuardianInitS3Config {
116    pub bucket: String,
117    pub region: String,
118    pub access_key: Option<String>,
119    pub secret_key: Option<String>,
120}
121
122impl GuardianInitS3Config {
123    pub async fn resolve(&self) -> anyhow::Result<S3Config> {
124        let access_key = self
125            .access_key
126            .as_deref()
127            .filter(|value| !value.trim().is_empty());
128        let secret_key = self
129            .secret_key
130            .as_deref()
131            .filter(|value| !value.trim().is_empty());
132
133        let (access_key, secret_key) = match (access_key, secret_key) {
134            (Some(access_key), Some(secret_key)) => {
135                (access_key.to_string(), secret_key.to_string())
136            }
137            (None, None) => {
138                let provider =
139                    aws_config::default_provider::credentials::DefaultCredentialsChain::builder()
140                        .build()
141                        .await;
142                let creds = provider
143                    .provide_credentials()
144                    .await
145                    .context("failed to resolve AWS credentials from the default provider chain")?;
146                (
147                    creds.access_key_id().to_string(),
148                    creds.secret_access_key().to_string(),
149                )
150            }
151            _ => anyhow::bail!(
152                "guardian_s3 access_key and secret_key must either both be set or both be omitted"
153            ),
154        };
155
156        Ok(S3Config {
157            access_key,
158            secret_key,
159            bucket_info: S3BucketInfo {
160                bucket: self.bucket.clone(),
161                region: self.region.clone(),
162            },
163        })
164    }
