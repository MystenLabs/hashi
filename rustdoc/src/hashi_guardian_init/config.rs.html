4use std::path::Path;
5use std::path::PathBuf;

7use anyhow::Context;
8use hashi::config::HashiIds;

10use hashi_types::guardian::LimiterConfig;
11use serde::Deserialize;
12
13use crate::kp_roster::KpRosterConfig;
14
15#[derive(Clone)]
16pub struct HashiOnchainConfig {
17    /// Sui RPC URL used to fetch Hashi on-chain state.
18    pub sui_rpc: String,
19    pub hashi_ids: HashiIds,
20}
21
22impl<'de> Deserialize<'de> for HashiOnchainConfig {
23    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
24    where
25        D: serde::Deserializer<'de>,
26    {
27        #[derive(Deserialize)]
28        struct HashiOnchainConfigWire {
29            sui_rpc: String,
30            package_id: String,
31            hashi_object_id: String,
32        }
33
34        let wire = HashiOnchainConfigWire::deserialize(deserializer)?;
35        Ok(Self {
36            sui_rpc: wire.sui_rpc,
37            hashi_ids: HashiIds {
38                package_id: wire.package_id.parse().map_err(serde::de::Error::custom)?,
39                hashi_object_id: wire
40                    .hashi_object_id
41                    .parse()
42                    .map_err(serde::de::Error::custom)?,
43            },
44        })
45    }
46}
47
48impl HashiOnchainConfig {
49    pub async fn onchain_state(&self) -> anyhow::Result<OnchainState> {
50        let (state, _watcher) = OnchainState::new(&self.sui_rpc, self.hashi_ids, None, None, None)
51            .await
52            .with_context(|| format!("failed to connect to Sui RPC at {}", self.sui_rpc))?;
53        Ok(state)
54    }
55}
56
57#[derive(Deserialize)]
58pub struct Config {
59    pub hashi: HashiOnchainConfig,
60    pub kp_roster: KpRosterConfig,
61    pub limiter_config: LimiterConfig,
62    /// Relay endpoint the KP's encrypted share is submitted to.
63    pub relay_endpoint: String,
64    /// gRPC endpoint of the guardian.
65    pub guardian_endpoint: String,
66    /// Path to this KP's armored OpenPGP public cert. Required by
67    /// key-provisioner commands.
68    pub kp_pgp_cert_path: Option<PathBuf>,
69}
70
71impl Config {
72    pub fn load_yaml(path: &Path) -> anyhow::Result<Self> {
73        let bytes = std::fs::read(path).with_context(|| {
74            format!("failed to read guardian init config at {}", path.display())
75        })?;
76        serde_yaml::from_slice(&bytes)
77            .with_context(|| format!("failed to parse guardian init yaml at {}", path.display()))
78    }
79
80    pub fn require_kp_pgp_cert_path(&self, command: &str) -> anyhow::Result<&Path> {
81        self.kp_pgp_cert_path.as_deref().ok_or_else(|| {
82            anyhow::anyhow!("{command} requires kp_pgp_cert_path in guardian init config")
83        })
84    }
85}
