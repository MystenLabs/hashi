4//! Shared KP-roster config + ceremony-state verification, used by both the
5//! `key-provisioner ceremony` and `key-provisioner provision` commands.
6//!
7//! Both commands need to:
8//! - load a roster of KP OpenPGP certs
9//! - discover the latest attested ceremony from S3
10//! - validate the ceremony's `secret_sharing_instance` against expected params
11//! - confirm every encrypted share in `shares/` is addressed only to its
12//!   labeled cert (without decrypting)
13//!
14//! `provision` additionally decrypts one share via the yubikey and re-encrypts
15//! it to a new guardian. The decryption helper lives here so both commands
16//! share the same gpg-streaming pattern.
17

47/// Common KP-roster config: the sharing params, the guardian's S3 log bucket,
48/// the full KP cert roster, and the PCR allowlist. Shared by every command that
49/// needs to discover and verify a ceremony against an expected KP set.
50#[derive(Deserialize)]
51pub struct KpRosterConfig {
52    /// Total number of shares. Must equal `kp_pgp_cert_paths.len()`.
53    pub num_shares: usize,
54    /// Reconstruction threshold. Must satisfy `2 <= threshold <= num_shares`.
55    pub threshold: usize,
56    /// S3 config for the guardian's log bucket (object-lock enabled).
57    pub guardian_s3: S3Config,
58    /// Paths to each KP's armored OpenPGP public cert. Order matters for
59    /// `operator ceremony` (the cert at index `i` is assigned share id `i + 1`); for
60    /// the read-only commands (`key-provisioner ceremony`, `key-provisioner provision`), shares are
61    /// matched by fingerprint so order is irrelevant.
62    pub kp_pgp_cert_paths: Vec<PathBuf>,
63    #[serde(flatten)]
64    pub pcr_allowlist: PcrAllowlist,
65}
66
67impl KpRosterConfig {
68    pub fn validate(&self) -> Result<()> {
69        SecretSharingParams::new(self.num_shares, self.threshold)
70            .map_err(|e| anyhow!("invalid sharing params: {e:?}"))?;
71
72        let cert_count = self.kp_pgp_cert_paths.len();
73        anyhow::ensure!(
74            self.num_shares == cert_count,
75            "num_shares ({}) must equal the number of KP certs ({cert_count})",
76            self.num_shares
77        );
78
79        Ok(())
80    }
81
82    /// The PCR allowlist decoded from `current_build` + `prev_builds`.
83    pub fn pcr_allowlist(&self) -> PcrAllowlist {
84        self.pcr_allowlist.clone()
85    }
86}

88/// Validated ceremony state. May come from the live `SetupNewKeyResponse`
89/// (`operator ceremony`) or be reconstructed from the guardian's `ceremony/` +
90/// `shares/` logs (`key-provisioner ceremony`, `key-provisioner provision`).
91#[derive(Debug, PartialEq)]
92pub struct VerifiedCeremonyState {
93    pub session_id: String,
94    pub encrypted_shares: KPEncryptedShares,
95    pub secret_sharing_instance: SecretSharingInstance,
96}
97
98impl VerifiedCeremonyState {
99    pub fn from_response(
100        response: SetupNewKeyResponse,
101        session_id: String,
102        expected_sharing_seq: u64,
103        expected_n: usize,
104        expected_t: usize,
105    ) -> Result<Self> {
106        let state = Self {
107            session_id,
108            encrypted_shares: response.encrypted_shares,
109            secret_sharing_instance: response.secret_sharing_instance,
110        };
111        anyhow::ensure!(
112            state.secret_sharing_instance.sharing_seq() == expected_sharing_seq,
113            "ceremony sharing_seq ({}) differs from expected ({expected_sharing_seq})",
114            state.secret_sharing_instance.sharing_seq()
115        );
116        state.validate_shape(expected_n, expected_t)?;
117        Ok(state)
118    }
119
120    pub async fn latest_from_s3(
121        reader: &mut GuardianReader,
122        expected_n: usize,
123        expected_t: usize,
124    ) -> Result<Self> {
125        // TODO: KP-begins discovery for rotations needs to read the latest
126        // historical ceremony with `BuildPolicy::AnyAllowlisted` to learn N.
127        // Keep this helper current-build-only for target verification, and add
128        // a separate discovery path there.
129        let (session_id, instance, roster) = reader
130            .read_latest_ceremony(BuildPolicy::Current)
131            .await?
132            .ok_or_else(|| anyhow!("no ceremony logs found in guardian S3 bucket"))?;
133        let encrypted_shares = reader
134            .read_shares(&session_id, instance.sharing_seq(), BuildPolicy::Current)
135            .await?;
136        Self::from_scraped(
137            session_id,
138            instance,
139            encrypted_shares,
140            &roster,
141            expected_n,
142            expected_t,
143        )
144    }
145
146    /// Assemble + validate from parts already scraped from S3 (the ceremony
147    /// log and the shares log). Use instead of [`Self::latest_from_s3`] when the
148    /// caller has already read those objects and wants to avoid a second walk.
149    pub fn from_scraped(
150        session_id: String,
151        secret_sharing_instance: SecretSharingInstance,
152        encrypted_shares: KPEncryptedShares,
153        roster: &[KPFingerprint],
154        expected_n: usize,
155        expected_t: usize,
156    ) -> Result<Self> {
157        let state = Self {
158            session_id,
159            encrypted_shares,
160            secret_sharing_instance,
161        };
162        state.validate_shape(expected_n, expected_t)?;
163        state.ensure_roster_matches(roster)?;
164        Ok(state)
165    }
166
167    /// Confirm the state uses the expected sharing shape and carries exactly
168    /// `expected_n` encrypted shares.
169    pub fn validate_shape(&self, expected_n: usize, expected_t: usize) -> Result<()> {
170        anyhow::ensure!(
171            self.secret_sharing_instance.num_shares() == expected_n,
172            "ceremony num_shares ({}) differs from expected ({expected_n})",
173            self.secret_sharing_instance.num_shares()
174        );
175        anyhow::ensure!(
176            self.secret_sharing_instance.threshold() == expected_t,
177            "ceremony threshold ({}) differs from expected ({expected_t})",
178            self.secret_sharing_instance.threshold()
179        );
180        anyhow::ensure!(
181            self.encrypted_shares.len() == expected_n,
182            "expected {expected_n} encrypted shares, got {}",
183            self.encrypted_shares.len()
184        );
185        Ok(())
186    }
187
188    /// Confirm the agreed `ceremony/` roster matches the recipient fingerprints
189    /// on the `shares/` ciphertexts. The roster is ordered by share id, so this
190    /// check preserves the ceremony log's share-id-to-KP binding.
191    pub fn ensure_roster_matches(&self, roster: &[KPFingerprint]) -> Result<()> {
192        let got = self.encrypted_shares.recipient_roster();
193        anyhow::ensure!(
194            roster == got.as_slice(),
195            "ceremony/ roster differs from shares/ recipient fingerprints: expected \
196             {roster:?}, got {got:?}"
197        );
198        Ok(())
199    }
200
201    /// For each encrypted share, confirm (a) its `recipient_fingerprint` label
202    /// names exactly one of the operator-supplied certs, and (b) the ciphertext
203    /// is actually encrypted only to that cert (parsed via PKESK without
204    /// decrypting).
205    ///
206    /// Identity is by fingerprint, not positional index — a share is matched to
207    /// its cert by `recipient_fingerprint`, independent of ordering.
208    pub fn verify_encrypted_share_recipients(&self, certs: &[PgpPublicCert]) -> Result<()> {
209        let by_fingerprint: std::collections::HashMap<KPFingerprint, &PgpPublicCert> =
210            certs.iter().map(|c| (c.fingerprint(), c)).collect();
211        anyhow::ensure!(
212            by_fingerprint.len() == certs.len(),
213            "duplicate fingerprints among the supplied KP certs"
214        );
215
216        let mut expected_fingerprints: Vec<KPFingerprint> =
217            by_fingerprint.keys().cloned().collect();
218        expected_fingerprints.sort_unstable();
219        let mut labeled_fingerprints: Vec<KPFingerprint> = self
220            .encrypted_shares
221            .iter()
222            .map(|s| s.recipient_fingerprint.clone())
223            .collect();
224        labeled_fingerprints.sort_unstable();
225        anyhow::ensure!(
226            labeled_fingerprints == expected_fingerprints,
227            "encrypted share recipient roster differs from expected KP certs: expected \
228             {expected_fingerprints:?}, got {labeled_fingerprints:?}"
229        );
230
231        for share in self.encrypted_shares.iter() {
232            let expected_cert = by_fingerprint
233                .get(&share.recipient_fingerprint)
234                .with_context(|| {
235                    format!(
236                        "share id {} is labeled for fingerprint {}, which is not among the \
237                             operator-supplied KP certs",
238                        share.id.get(),
239                        share.recipient_fingerprint
240                    )
241                })?;
242            let recipients = pgp_message_recipients(&share.armored_ciphertext)
243                .with_context(|| format!("parse PGP recipients for share id {}", share.id.get()))?;
244            anyhow::ensure!(
245                !recipients.is_empty(),
246                "share id {} has no PGP recipients",
247                share.id.get()
248            );
249            for handle in &recipients {
250                anyhow::ensure!(
251                    cert_owns_key_handle(expected_cert, handle),
252                    "share id {} (labeled {}) is encrypted to key {handle}, which is not in \
253                     that cert",
254                    share.id.get(),
255                    share.recipient_fingerprint
256                );
257            }
258            info!(
259                share_id = share.id.get(),
260                fingerprint = %share.recipient_fingerprint,
261                recipient_count = recipients.len(),
262                "verified share is encrypted only to its labeled cert"
263            );
264        }
265        info!(
266            "all {} shares encrypted to their labeled certs",
267            self.encrypted_shares.len()
268        );
269        Ok(())
270    }
271}

273/// Confirm this KP's own cert is one of the operator-supplied roster certs.
274/// Catches a typo'd `kp_pgp_cert_path` before any S3 or yubikey work.
275pub fn ensure_cert_in_roster(kp_cert: &PgpPublicCert, certs: &[PgpPublicCert]) -> Result<()> {
276    let want_fp = kp_cert.fingerprint();
277    anyhow::ensure!(
278        certs.iter().any(|c| c.fingerprint() == want_fp),
279        "this KP's cert (fingerprint {want_fp}) is not among the configured kp_pgp_cert_paths"
280    );
281    Ok(())
282}

284/// Decrypt a KP's encrypted share via the yubikey-backed gpg agent, returning
285/// the share wrapped in a [`DecryptedShare`] that wipes its scalar on drop.
286/// Nothing touches disk: gpg reads the ciphertext from a piped stdin and
287/// streams the plaintext over its stdout pipe into memory.
288///
289/// **Zeroization scope:** the gpg plaintext bytes, the intermediate scalar
290/// byte array, and the returned wrapper's inner [`Scalar`] are zeroed on drop.
291/// `k256::Scalar` is `Copy`, so the compiler may produce transient stack
292/// copies (e.g. inside `verify_share` / `build_from_share`) that this can't
293/// reach — those are wiped only when the process exits. The named locations
294/// this code owns are wiped deterministically.
295pub fn decrypt_share(share: &KPEncryptedShare) -> Result<DecryptedShare> {
296    let plaintext = Zeroizing::new(decrypt_armored_via_gpg(&share.armored_ciphertext, None)?);
297    let scalar = scalar_from_decrypted_plaintext(&plaintext)?;
298    Ok(DecryptedShare(Share {
299        id: share.id,
300        value: scalar,
301    }))
302}
303
304/// Owning wrapper around a decrypted [`Share`] that wipes the scalar value on
305/// drop. Use `&*share` to access the inner [`Share`] for commitment
306/// verification / re-encryption. See [`decrypt_share`] for the zeroization
307/// scope.
308pub struct DecryptedShare(Share);
309
310impl Deref for DecryptedShare {
311    type Target = Share;
312
313    fn deref(&self) -> &Share {
314        &self.0
315    }
316}
317
318impl Drop for DecryptedShare {
319    fn drop(&mut self) {
320        self.0.value.zeroize();
321    }
322}

324/// Parse the decrypted plaintext bytes into a secp256k1 scalar. Extracted from
325/// [`decrypt_share`] so the byte-length and canonical-scalar checks are
326/// unit-testable without invoking gpg.
327fn scalar_from_decrypted_plaintext(plaintext: &[u8]) -> Result<Scalar> {
328    let src: &[u8; 32] = plaintext
329        .try_into()
330        .map_err(|_| anyhow!("decrypted share is {} bytes, expected 32", plaintext.len()))?;
331    let mut scalar_bytes = Zeroizing::new([0u8; 32]);
332    scalar_bytes.copy_from_slice(src);
333    Option::<Scalar>::from(Scalar::from_repr(FieldBytes::from(*scalar_bytes)))
334        .ok_or_else(|| anyhow!("decrypted share is not a valid secp256k1 scalar"))
335}
