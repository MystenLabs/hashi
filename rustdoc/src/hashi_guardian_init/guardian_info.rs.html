15pub async fn verified_live_guardian_info(
16    client: &mut GuardianServiceClient<Channel>,
17    current_build: &BuildPcrs,
18) -> anyhow::Result<VerifiedGuardianInfo> {
19    let info_pb = client
20        .get_guardian_info(pb::GetGuardianInfoRequest {})
21        .await
22        .context("GetGuardianInfo RPC failed")?
23        .into_inner();
24    let info_resp = GetGuardianInfoResponse::try_from(info_pb)
25        .map_err(|e| anyhow!("decode GetGuardianInfoResponse: {e:?}"))?;
26    info_resp
27        .verify_live(current_build)
28        .map_err(|e| anyhow!("verify GuardianInfo attestation/signature: {e:?}"))
29}

31/// The OI log captures the final pre-transition snapshot. Apart from the
32/// lifecycle advancing once, it must match the live post-OI GuardianInfo.
33pub fn ensure_oi_info_matches_post_init(
34    oi_info: &GuardianInfo,
35    live_info: &GuardianInfo,
36) -> anyhow::Result<()> {
37    ensure!(
38        live_info.lifecycle.predecessor() == Some(oi_info.lifecycle),
39        "S3 OI lifecycle {:?} is not the predecessor of live lifecycle {:?}",
40        oi_info.lifecycle,
41        live_info.lifecycle
42    );
43
44    let mut expected_live_info = oi_info.clone();
45    expected_live_info.lifecycle = live_info.lifecycle;
46    ensure!(
47        &expected_live_info == live_info,
48        "S3 OI GuardianInfo differs from live post-OperatorInit GuardianInfo"
49    );
50    Ok(())
51}
