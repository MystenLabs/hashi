4//! `key-provisioner ceremony` verifies and decrypts this KP's ceremony share.
5
6use anyhow::Context;
7use anyhow::Result;
8use anyhow::anyhow;
9use hashi_guardian::s3_reader::GuardianReader;
10use hashi_types::pgp::PgpPublicCert;
11use hashi_types::pgp::load_certs;
12use tracing::info;
13
14use crate::config::Config;
15use crate::kp_roster::VerifiedCeremonyState;
16use crate::kp_roster::decrypt_share;
17use crate::kp_roster::ensure_cert_in_roster;
18
19/// Verify this KP can fetch and decrypt its ceremony share.
20///
21/// Trust is anchored entirely to the guardian's S3 attestation log: the
22/// `GuardianReader` resolves the session's attested signing pubkey through the
23/// reader cache, and both the `ceremony/` audit entry and `shares/` recovery
24/// entry are verified under it. Each step is logged.
25///
26/// Security: the ciphertext is piped into `gpg --decrypt` over stdin and the
27/// plaintext streams back over stdout; neither ciphertext nor plaintext is
28/// written to disk by this flow.
29pub async fn run(cfg: Config) -> Result<()> {
30    cfg.kp_roster.validate()?;
31
32    info!(
33        phase = "setup",
34        bucket = cfg.kp_roster.guardian_s3.bucket_name(),
35        region = cfg.kp_roster.guardian_s3.region(),
36        num_shares = cfg.kp_roster.num_shares,
37        threshold = cfg.kp_roster.threshold,
38        sui_rpc = %cfg.hashi.sui_rpc,
39        package_id = %cfg.hashi.hashi_ids.package_id,
40        hashi_object_id = %cfg.hashi.hashi_ids.hashi_object_id,
41        "verifying ceremony share",
42    );
43
44    info!(
45        phase = "roster load",
46        cert_count = cfg.kp_roster.kp_pgp_cert_paths.len(),
47        "loading + validating full KP cert roster",
48    );
49    let certs = load_certs(&cfg.kp_roster.kp_pgp_cert_paths)?;
50    info!(
51        phase = "roster load",
52        cert_count = certs.len(),
53        "KP cert roster loaded"
54    );
55
56    // Load this KP's cert. Its fingerprint finds our share in `shares/`, and
57    // the cert itself lets us confirm the ciphertext is genuinely encrypted to
58    // us before we touch the yubikey.
59    let kp_pgp_cert_path = cfg.require_kp_pgp_cert_path("key-provisioner ceremony")?;
60    let cert_armored = std::fs::read_to_string(kp_pgp_cert_path)
61        .with_context(|| format!("read KP cert at {}", kp_pgp_cert_path.display()))?;
62    let kp_cert = PgpPublicCert::new(cert_armored)
63        .with_context(|| format!("invalid PGP cert at {}", kp_pgp_cert_path.display()))?;
64    let want_fp = kp_cert.fingerprint();
65    info!(
66        phase = "setup",
67        fingerprint = %want_fp,
68        kp_cert_path = %kp_pgp_cert_path.display(),
69        "loaded this KP's cert",
70    );
71    ensure_cert_in_roster(&kp_cert, &certs)?;
72
73    // 1. Discover and verify the latest ceremony from the immutable log
74    //    (attestation-verified once via the reader's session-key cache).
75    info!(
76        phase = "s3 connect",
77        bucket = cfg.kp_roster.guardian_s3.bucket_name(),
78        region = cfg.kp_roster.guardian_s3.region(),
79        current_pcr0 = hex::encode(cfg.kp_roster.pcr_allowlist.current_build().pcr0()),
80        "connecting to guardian log bucket",
81    );
82    let mut reader = GuardianReader::new(&cfg.kp_roster.guardian_s3, cfg.kp_roster.pcr_allowlist())
83        .await
84        .context("connect to guardian log bucket")?;
85    info!(phase = "s3 connect", "connected to guardian log bucket");
86
87    info!(
88        phase = "ceremony scrape",
89        "scraping latest ceremony/ + shares/ logs (attestation-anchored)",
90    );
91    let state = VerifiedCeremonyState::latest_from_s3(
92        &mut reader,
93        cfg.kp_roster.num_shares,
94        cfg.kp_roster.threshold,
95    )
96    .await?;
97    info!(
98        phase = "ceremony scrape",
99        session_id = %state.session_id,
100        sharing_seq = state.secret_sharing_instance.sharing_seq(),
101        n = state.secret_sharing_instance.num_shares(),
102        t = state.secret_sharing_instance.threshold(),
103        share_count = state.encrypted_shares.len(),
104        "discovered + validated latest ceremony session",
105    );
106
107    // 2. Confirm every share is addressed only to its labeled KP cert.
108    info!(
109        phase = "roster verify",
110        share_count = state.encrypted_shares.len(),
111        "verifying every share is addressed only to its labeled KP cert (without decrypting)",
112    );
113    state.verify_encrypted_share_recipients(&certs)?;
114    info!(
115        phase = "roster verify",
116        "ceremony/ and shares/ logs verified against expected params and KP certs",
117    );
118
119    // 3. Find this KP's share by exact fingerprint match (both sides derive
120    //    from PgpPublicCert::fingerprint over the same key, so they're
121    //    canonical and identical: no normalization needed). The matched share
122    //    carries its own crypto `id`.
123    let share = state
124        .encrypted_shares
125        .iter()
126        .find(|s| s.recipient_fingerprint == want_fp)
127        .ok_or_else(|| {
128            anyhow!(
129                "no share in the shares/ log is labeled for this KP's fingerprint \
130                 {want_fp} (labeled fingerprints: {:?})",
131                state
132                    .encrypted_shares
133                    .iter()
134                    .map(|s| s.recipient_fingerprint.clone())
135                    .collect::<Vec<_>>()
136            )
137        })?;
138    let share_id = share.id;
139    info!(
140        phase = "share find",
141        share_id = share_id.get(),
142        fingerprint = %share.recipient_fingerprint,
143        "located this KP's encrypted share",
144    );
145
146    // 4. Decrypt the share with the yubikey via gpg. The ciphertext is piped
147    //    into gpg over stdin; decrypted bytes stream back into memory.
148    info!(
149        phase = "share decrypt",
150        share_id = share_id.get(),
151        "decrypting share via yubikey (ciphertext piped via stdin; plaintext in memory)",
152    );
153    let reconstructed = decrypt_share(share)?;
154    info!(
155        phase = "share decrypt",
156        share_id = share_id.get(),
157        "decrypted share via yubikey",
158    );
159
160    // 5. Verify the decrypted share's commitment is in the set: proves the
161    //    bytes we decrypted are a valid share of the guardian's BTC key.
162    info!(
163        phase = "commitment verify",
164        share_id = share_id.get(),
165        "verifying decrypted share against its commitment",
166    );
167    state
168        .secret_sharing_instance
169        .commitments()
170        .verify_share(&reconstructed)
171        .map_err(|e| anyhow!("decrypted share does not match its commitment: {e:?}"))?;
172    let expected_commitment = state
173        .secret_sharing_instance
174        .commitments()
175        .iter()
176        .find(|c| c.id == share_id)
177        .ok_or_else(|| {
178            anyhow!(
179                "commitment for share id {} missing despite verify_share success",
180                share_id
181            )
182        })?;
183    info!(
184        phase = "commitment verify",
185        share_id = share_id.get(),
186        commitment = hex::encode(&expected_commitment.digest),
187        "decrypted share matches its commitment",
188    );
189
190    info!(
191        phase = "summary",
192        session_id = %state.session_id,
193        share_id = share_id.get(),
194        sharing_seq = state.secret_sharing_instance.sharing_seq(),
195        fingerprint = %want_fp,
196        commitment = hex::encode(&expected_commitment.digest),
197        "ceremony share verified",
198    );
199    Ok(())
200}
