21enum VerifiedWithdrawal {
22    Success(MonitorWithdrawalEvent),
23    Failure,
24}
25
26impl TryFrom<VerifiedLogRecord> for VerifiedWithdrawal {
27    type Error = anyhow::Error;
28
29    fn try_from(log: VerifiedLogRecord) -> Result<Self, Self::Error> {
30        let LogMessage::Withdrawal(withdrawal_message) = log.message else {
31            anyhow::bail!("non-withdrawal logs found");
32        };
33
34        match *withdrawal_message {
35            WithdrawalLogMessage::Success {
36                txid, request_data, ..
37            } => {
38                debug!(
39                    wid = %request_data.wid,
40                    txid = %txid,
41                    "successful guardian withdrawal log"
42                );
43                Ok(VerifiedWithdrawal::Success(MonitorWithdrawalEvent {
44                    event_type: WithdrawalEventType::E2GuardianApproved,
45                    wid: request_data.wid,
46                    timestamp_secs: unix_millis_to_seconds(log.timestamp_ms),
47                    btc_txid: txid,
48                }))
49            }
50            failure @ WithdrawalLogMessage::Failure { .. } => {
51                info!(?failure, "failed guardian withdrawal log");
52                Ok(VerifiedWithdrawal::Failure)
53            }
54        }
55    }
56}

58// Note: current design does not check if multiple concurrent sessions are running.
59//       one way to impl this: store the first & last observed session timestamp & ensure no overlap between time ranges.
60pub struct GuardianWithdrawalsPoller {
61    /// Owns the S3 client + the trusted-key cache, so a session's attestation is
62    /// verified once for the poller's lifetime.
63    reader: GuardianReader,
64    cursor: S3HourScopedDirectory,
65}
66
67impl GuardianWithdrawalsPoller {
68    // Note: Throws an error if there is a S3 connectivity issue
69    pub async fn new(config: &Config, start: UnixSeconds) -> anyhow::Result<Self> {
70        Ok(Self {
71            reader: GuardianReader::new(&config.guardian, config.pcr_allowlist()).await?,
72            cursor: withdraw_cursor(start),
73        })
74    }
75
76    pub fn cursor_seconds(&self) -> UnixSeconds {
77        self.cursor.to_unix_seconds()
78    }
79
80    /// Polls the Guardian S3 bucket for one hour worth of events.
81    /// A more aggressive fetch, e.g., one day at a time, can also be done if needed.
82    pub async fn poll_one_hour(&mut self) -> anyhow::Result<PollOutcome> {
83        if now_timestamp_secs() < self.cursor.write_completion_time() {
84            return Ok(PollOutcome::CursorUnmoved);
85        }
86
87        let verified_logs = self.reader.read_logs_in_dir(&self.cursor).await?;
88        // Withdrawal polling may replay historical buckets during an upgrade, so
89        // this caller accepts any record whose session build verifies against the
90        // configured allowlist. Add a cursor/cutoff policy here if tailing must
91        // require the current build after the upgrade window.
92        let withdrawal_events = verified_logs
93            .into_iter()
94            .map(VerifiedWithdrawal::try_from)
95            .collect::<anyhow::Result<Vec<_>>>()?
96            .into_iter()
97            .filter_map(|e| match e {
98                VerifiedWithdrawal::Success(event) => Some(MonitorEvent::Withdrawal(event)),
99                VerifiedWithdrawal::Failure => None,
100            })
101            .collect::<Vec<MonitorEvent>>();
102
103        self.cursor = self.cursor.next_dir();
104        Ok(PollOutcome::CursorAdvanced(withdrawal_events))
105    }
106}
