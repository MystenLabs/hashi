4//! Read-only HTTP surface for the proxy: `GET /info` (a curated JSON projection
5//! of the enclave's signed `GetGuardianInfo`) and `GET /health` (proxy
6//! liveness), with permissive CORS. It lets browser/`fetch` clients like the
7//! hashi-ts-sdk read the withdrawal rate-limiter state that the gRPC surface
8//! only exposes to nodes. `u64`s serialize as strings (JSON/JS `2^53`), and
9//! `limiter`/`btcPubkey`/`committeeEpoch` are `null` until the guardian is
10//! provisioned. The S3 bucket and KP shares are deliberately not exposed.
11//!
12//! `/info` is public and polled by every SDK client, so it fronts the enclave
13//! with a single-slot TTL cache and single-flight refresh: a burst of hits
14//! collapses to at most one `GetGuardianInfo` per TTL, and while the enclave is
15//! briefly unreachable the last good view is served (its age is visible via
16//! `signedAtMs`) rather than erroring.
17

80/// Why an `/info` refresh could not produce a fresh view. Kept transport-shaped
81/// so the source stays decoupled from the HTTP response contract.
82enum InfoError {
83    /// The enclave gRPC call failed (unreachable / deadline / transport).
84    Unreachable(String),
85    /// The response could not be converted to the guardian domain type.
86    Invalid(String),
87}
88
89impl InfoError {
90    fn into_response(self) -> Response {
91        match self {
92            InfoError::Unreachable(detail) => unavailable("guardian_unreachable", &detail),
93            InfoError::Invalid(detail) => unavailable("guardian_info_invalid", &detail),
94        }
95    }
96}

98/// Source of a fresh [`GuardianInfoView`]. A trait so the cache can be tested
99/// without a live enclave or a signed response to verify against.
100#[tonic::async_trait]
101trait InfoSource: Send + Sync + 'static {
102    async fn fetch(&self) -> Result<GuardianInfoView, InfoError>;
103}
104
105/// [`InfoSource`] backed by the enclave's gRPC `GetGuardianInfo`, verifying and
106/// projecting the signed response.
107struct GrpcInfoSource {
108    client: GuardianServiceClient<Channel>,
109}
110
111#[tonic::async_trait]
112impl InfoSource for GrpcInfoSource {
113    async fn fetch(&self) -> Result<GuardianInfoView, InfoError> {
114        let raw = self
115            .client
116            .clone()
117            .get_guardian_info(proto::GetGuardianInfoRequest {})
118            .await
119            .map_err(|status| InfoError::Unreachable(status.to_string()))?
120            .into_inner();
121
122        // Read the signing timestamp off the raw proto; the domain conversion drops it.
123        let signed_at_ms = raw.signed_info.as_ref().and_then(|s| s.timestamp_ms);
124        let response = GetGuardianInfoResponse::try_from(raw).map_err(|e| {
125            error!(error = ?e, "GetGuardianInfo could not be decoded for /info");
126            InfoError::Invalid(format!("{e:?}"))
127        })?;
128        let (info, signing_pub_key) = response.into_info_unchecked();
129        Ok(project(&info, &signing_pub_key, signed_at_ms))
130    }
131}

133/// The `/info` cache: the last projected view plus a refresh lock. Reads take
134/// the fast path off `slot`; a miss elects one refresher via `refresh` so a
135/// burst of public hits collapses to a single backend call.
136struct InfoCache {
137    slot: Mutex<Option<(Instant, GuardianInfoView)>>,
138    refresh: tokio::sync::Mutex<()>,
139}
140
141/// Backend source plus its single-slot TTL cache, so a burst of public `/info`
142/// hits collapses to at most one `GetGuardianInfo` per `ttl`.
143#[derive(Clone)]
144pub struct InfoState {
145    source: Arc<dyn InfoSource>,
146    cache: Arc<InfoCache>,
147    ttl: Duration,
148}
149
150impl InfoState {
151    pub fn new(client: GuardianServiceClient<Channel>, ttl: Duration) -> Self {
152        Self::with_source(Arc::new(GrpcInfoSource { client }), ttl)
153    }
154
155    fn with_source(source: Arc<dyn InfoSource>, ttl: Duration) -> Self {
156        Self {
157            source,
158            cache: Arc::new(InfoCache {
159                slot: Mutex::new(None),
160                refresh: tokio::sync::Mutex::new(()),
161            }),
162            ttl,
163        }
164    }
165
166    /// Serve `/info`: a fresh cached view, otherwise one single-flighted
167    /// refresh whose result (or, if the enclave is down, the last good view) is
168    /// shared with every concurrent caller.
169    async fn serve(&self) -> Response {
170        if let Some(view) = self.fresh() {
171            return Json(view).into_response();
172        }
173
174        // Elect a single refresher. Concurrent callers serve the last good view
175        // rather than pile onto the enclave; on a cold cache (nothing to serve)
176        // they instead wait for that one refresh.
177        let refreshed = match self.cache.refresh.try_lock() {
178            Ok(_leader) => self.refresh_locked().await,
179            Err(_) => match self.last_good() {
180                Some(view) => return Json(view).into_response(),
181                None => {
182                    let _leader = self.cache.refresh.lock().await;
183                    self.refresh_locked().await
184                }
185            },
186        };
187
188        match refreshed {
189            Ok(view) => Json(view).into_response(),
190            // Enclave unreachable: the last good view (its age is visible via
191            // `signedAtMs`) beats a hard error for an advisory route.
192            Err(err) => self
193                .last_good()
194                .map(|view| Json(view).into_response())
195                .unwrap_or_else(|| err.into_response()),
196        }
197    }
198
199    /// Refresh from the backend. Call with `refresh` held; re-checks the cache
200    /// first so callers that queued behind the leader reuse its fetch.
201    async fn refresh_locked(&self) -> Result<GuardianInfoView, InfoError> {
202        if let Some(view) = self.fresh() {
203            return Ok(view);
204        }
205        let view = self.source.fetch().await?;
206        self.store(view.clone());
207        Ok(view)
208    }
209
210    /// The cached view if it is still within its TTL.
211    fn fresh(&self) -> Option<GuardianInfoView> {
212        let slot = self.cache.slot.lock().expect("info cache mutex poisoned");
213        slot.as_ref()
214            .filter(|(at, _)| at.elapsed() < self.ttl)
215            .map(|(_, view)| view.clone())
216    }
217
218    /// The last successfully fetched view regardless of age.
219    fn last_good(&self) -> Option<GuardianInfoView> {
220        self.cache
221            .slot
222            .lock()
223            .expect("info cache mutex poisoned")
224            .as_ref()
225            .map(|(_, view)| view.clone())
226    }
227
228    fn store(&self, view: GuardianInfoView) {
229        *self.cache.slot.lock().expect("info cache mutex poisoned") = Some((Instant::now(), view));
230    }
231}

233/// Router for the HTTP status surface, with permissive CORS (public read-only
234/// data, so any origin; no credentials).
235pub fn router(state: InfoState) -> Router {
236    let cors = CorsLayer::new()
237        .allow_origin(Any)
238        .allow_methods([Method::GET, Method::OPTIONS])
239        .allow_headers(Any);
240    Router::new()
241        .route("/info", get(get_info))
242        .route("/health", get(health))
243        .layer(cors)
244        .with_state(state)
245}

247/// Proxy liveness only — never gated on the enclave, so an enclave hiccup can't
248/// deregister the target and take `/info` dark.
249async fn health() -> StatusCode {
250    StatusCode::OK
251}
252
253async fn get_info(State(state): State<InfoState>) -> Response {
254    state.serve().await
255}
256
257fn unavailable(error: &str, detail: &str) -> Response {
258    (
259        StatusCode::SERVICE_UNAVAILABLE,
260        Json(json!({ "error": error, "detail": detail })),
261    )
262        .into_response()
263}
264
265fn project(
266    info: &GuardianInfo,
267    signing_pub_key: &GuardianPubKey,
268    signed_at_ms: Option<u64>,
269) -> GuardianInfoView {
270    GuardianInfoView {
271        limiter: limiter_view(info.limiter_state, info.limiter_config),
272        git_revision: info.untrusted_git_revision.clone(),
273        committee_epoch: info.current_committee_epoch.map(|e| e.to_string()),
274        btc_pubkey: info
275            .enclave_btc_pubkey
276            .as_ref()
277            .map(|pk| hex::encode(pk.serialize())),
278        signing_pub_key: hex::encode(signing_pub_key.as_bytes()),
279        signed_at_ms: signed_at_ms.map(|t| t.to_string()),
280    }
281}

283/// Reported only when both halves are present — a half-initialized limiter is
284/// useless, so the client sees `limiter` as a single nullable object.
285fn limiter_view(state: Option<LimiterState>, config: Option<LimiterConfig>) -> Option<LimiterView> {
286    let (state, config) = (state?, config?);
287    Some(LimiterView {
288        state: LimiterStateView {
289            num_tokens_available_sats: state.num_tokens_available.to_string(),
290            last_updated_at_secs: state.last_updated_at.to_string(),
291            next_seq: state.next_seq.to_string(),
292        },
293        config: LimiterConfigView {
294            refill_rate_sats_per_sec: config.refill_rate.to_string(),
295            max_bucket_capacity_sats: config.max_bucket_capacity.to_string(),
296        },
297    })
298}

400    /// Test source: counts calls, can be flipped `down`, and can delay each
401    /// fetch to widen the window a concurrent burst races in.
402    #[derive(Clone)]
403    struct FakeSource {
404        view: GuardianInfoView,
405        calls: Arc<AtomicUsize>,
406        down: Arc<AtomicBool>,
407        delay: Duration,
408    }
409
410    impl FakeSource {
411        fn new(delay: Duration) -> Self {
412            Self {
413                view: sample_view(),
414                calls: Arc::new(AtomicUsize::new(0)),
415                down: Arc::new(AtomicBool::new(false)),
416                delay,
417            }
418        }
419    }
420
421    #[tonic::async_trait]
422    impl InfoSource for FakeSource {
423        async fn fetch(&self) -> Result<GuardianInfoView, InfoError> {
424            self.calls.fetch_add(1, Ordering::SeqCst);
425            if !self.delay.is_zero() {
426                tokio::time::sleep(self.delay).await;
427            }
428            if self.down.load(Ordering::SeqCst) {
429                Err(InfoError::Unreachable("fake guardian down".to_string()))
430            } else {
431                Ok(self.view.clone())
432            }
433        }
434    }

436    fn state_with(fake: &FakeSource, ttl: Duration) -> InfoState {
437        InfoState::with_source(Arc::new(fake.clone()), ttl)
438    }

440    async fn body_json(resp: Response) -> serde_json::Value {
441        let bytes = axum::body::to_bytes(resp.into_body(), usize::MAX)
442            .await
443            .unwrap();
444        serde_json::from_slice(&bytes).unwrap()
445    }

447    #[tokio::test]
448    async fn concurrent_misses_collapse_to_a_single_fetch() {
449        // Six cold-cache hits arrive while the one leader is still fetching; the
450        // rest must reuse its result, not each call the enclave.
451        let fake = FakeSource::new(Duration::from_millis(200));
452        let state = state_with(&fake, Duration::from_secs(10));
453
454        let responses = tokio::join!(
455            state.serve(),
456            state.serve(),
457            state.serve(),
458            state.serve(),
459            state.serve(),
460            state.serve(),
461        );
462        for resp in [
463            responses.0,
464            responses.1,
465            responses.2,
466            responses.3,
467            responses.4,
468            responses.5,
469        ] {
470            assert_eq!(resp.status(), StatusCode::OK);
471        }
472        assert_eq!(
473            fake.calls.load(Ordering::SeqCst),
474            1,
475            "a burst must collapse to one GetGuardianInfo"
476        );
