4//! Wid-keyed response cache for the guardian's `StandardWithdrawal` RPC: an
5//! in-process LRU in front of the guardian's own S3 withdrawal log
6//! ([`crate::widlog`]) as the durable, read-only tier.
7//!
8//! Keyed by `wid`, not `(wid, seq)`: the guardian debits the limiter and
9//! advances `next_seq` when it signs, before hashi has the signed event
10//! on-chain. If hashi retries the same wid at a bumped seq (its limiter mirror
11//! reconciled forward while the event was still pending), re-consuming would
12//! drain the bucket for a withdrawal that was already signed. Replaying by
13//! `wid` avoids that — safe because a committed wid's inputs/outputs are
14//! immutable (only signatures change), so the response is stable.
15//!
16//! The enclave persists every signed withdrawal before releasing it, so the
17//! proxy has nothing durable of its own to lose. On a log hit it verifies the
18//! record's signatures spend the incoming request, then replays them (see
19//! `replay_from_log`); it does not reason about the limiter `seq`, which is the
20//! guardian's to own — the node's mirror self-heals any divergence
21//! (`hashi/src/guardian_limiter.rs`). When the log can't answer, the proxy
22//! fails closed with `UNAVAILABLE` rather than forwarding blind — it cannot
23//! distinguish "never signed" from "signed but unreadable", and forwarding the
24//! latter re-signs a withdrawal the guardian already durably signed, the
25//! double-debit the cache exists to prevent.

57// One entry per withdrawal would leak forever; 10k far exceeds the rate-limited
58// in-flight working set, so a hot wid always outlives its retry window.
59const CACHE_CAPACITY: usize = 10_000;
60
61/// Fail-closed error surfaced when the durable tier can't answer. Kept clear of
62/// "seq mismatch" and "Rate limit exceeded": the node classifies guardian
63/// errors by those substrings (`crates/hashi/src/leader/guardian.rs`) and this
64/// one must land in its retriable bucket.
65pub const WID_CACHE_UNAVAILABLE_MSG: &str = "wid cache unavailable; retry";
66
67fn unavailable() -> Status {
68    Status::unavailable(WID_CACHE_UNAVAILABLE_MSG)
69}

71struct CacheEntry {
72    /// The seq the guardian consumed this wid at. Kept for observability only —
73    /// the cache is keyed by `wid`, so lookups ignore the requester's seq.
74    consumed_seq: u64,
75    response: proto::SignedStandardWithdrawalResponse,
76}
77
78pub struct CachingGuardianGrpc<S, L> {
79    inner: S,
80    l1: Mutex<LruCache<WithdrawalID, CacheEntry>>,
81    log: L,
82    /// Network the guardian validates requests against; needed to recompute
83    /// sighashes when verifying a log replay.
84    network: Network,
85    metrics: Arc<ProxyMetrics>,
86}
87
88impl<S, L> CachingGuardianGrpc<S, L> {
89    pub fn new(inner: S, log: L, network: Network, metrics: Arc<ProxyMetrics>) -> Self {
90        Self::with_capacity(
91            inner,
92            log,
93            network,
94            metrics,
95            NonZeroUsize::new(CACHE_CAPACITY).expect("CACHE_CAPACITY > 0"),
96        )
97    }
98
99    fn with_capacity(
100        inner: S,
101        log: L,
102        network: Network,
103        metrics: Arc<ProxyMetrics>,
104        capacity: NonZeroUsize,
105    ) -> Self {
106        Self {
107            inner,
108            l1: Mutex::new(LruCache::new(capacity)),
109            log,
110            network,
111            metrics,
112        }
113    }
114
115    // The critical section never spans an `.await`, so a sync `std::sync::Mutex`
116    // is the right tool and keeps the handler future `Send`. Panics abort the
117    // process (see `abort_on_panic` in main), so a poisoned lock is unreachable.
118    fn try_hit(
119        &self,
120        wid: &WithdrawalID,
121    ) -> Option<(u64, proto::SignedStandardWithdrawalResponse)> {
122        let mut cache = self.l1.lock().expect("cache mutex poisoned");
123        cache
124            .get(wid)
125            .map(|entry| (entry.consumed_seq, entry.response.clone()))
126    }
127
128    fn store(
129        &self,
130        wid: WithdrawalID,
131        consumed_seq: u64,
132        response: proto::SignedStandardWithdrawalResponse,
133    ) {
134        self.l1.lock().expect("cache mutex poisoned").put(
135            wid,
136            CacheEntry {
137                consumed_seq,
138                response,
139            },
140        );
141    }
142}

144impl<S, L> CachingGuardianGrpc<S, L>

149    /// Serve a wid from the guardian's S3 withdrawal log. `Ok(None)` means
150    /// "definitively not in the log: forward"; `Err` is the fail-closed path.
151    async fn replay_from_log(
152        &self,
153        wid: &WithdrawalID,
154        requested_seq: u64,
155        request: &proto::SignedStandardWithdrawalRequest,
156    ) -> Result<Option<proto::SignedStandardWithdrawalResponse>, Status> {
157        let found = match find_success_record(&self.log, wid, requested_seq, &self.metrics).await {
158            Ok(Some(found)) => found,
159            Ok(None) => return Ok(None),
160            Err(WidLogError::CapExceeded) => {
161                self.metrics.outcome(metrics::OUTCOME_UNAVAILABLE_SCAN_CAP);
162                error!(%wid, requested_seq, "Wid log scan hit its LIST cap; refusing to forward blind.");
163                return Err(unavailable());
164            }
165            Err(WidLogError::Store(e)) => {
166                self.metrics.outcome(metrics::OUTCOME_UNAVAILABLE_LOG_STORE);
167                error!(%wid, requested_seq, error = %e, "Wid log unavailable; refusing to forward blind.");
168                return Err(unavailable());
169            }
170        };
171
172        // Idempotency is by wid; the proxy does not gate on the limiter `seq`.
173        // A record the enclave signed but never committed (a lost-ack S3 write
174        // it then reverted) still replays here: the node's requested seq stays
175        // pinned to its mirror, and its stall reconcile snaps that mirror back
176        // to the guardian's authoritative seq (`hashi/src/guardian_limiter.rs`),
177        // so a served orphan self-heals — at worst a bounded one-withdrawal
178        // limiter under-count, which is the guardian's to close, not the proxy's.
179        //
180        // Integrity gate: never hand the node signatures that don't spend its
181        // transaction. Needs the guardian's live BTC key + master G (the enclave
182        // key rotates per instance, so it can't be cached across rotations).
183        let info = match self.guardian_info().await {
184            Ok(info) => info,
185            Err(e) => {
186                self.metrics
187                    .outcome(metrics::OUTCOME_UNAVAILABLE_GUARDIAN_INFO);
188                error!(%wid, error = %e, "Cannot fetch guardian info to verify a log replay.");
189                return Err(unavailable());
190            }
191        };
192        let (Some(enclave_btc_pubkey), Some(master_g)) =
193            (info.enclave_btc_pubkey, info.mpc_master_g)
194        else {
195            self.metrics
196                .outcome(metrics::OUTCOME_UNAVAILABLE_GUARDIAN_INFO);
197            error!(%wid, "Guardian info lacks BTC pubkey / master G; cannot verify a log replay.");
198            return Err(unavailable());
199        };
200
201        if let Err(e) = verify_recorded_signatures(
202            request,
203            &found.response.enclave_signatures,
204            &enclave_btc_pubkey,
205            &master_g,
206            self.network,
207        ) {
208            self.metrics
209                .outcome(metrics::OUTCOME_UNAVAILABLE_VERIFY_FAILED);
210            error!(
211                %wid,
212                error = %e,
213                "Log record signatures do not verify; possible bucket tampering or version skew."
214            );
215            return Err(unavailable());
216        }
217
218        let response = synthesize_response(&found);
219        self.store(*wid, found.consumed_seq, response.clone());
220        self.metrics.outcome(metrics::OUTCOME_S3_HIT);
221        info!(
222            %wid,
223            requested_seq,
224            consumed_seq = found.consumed_seq,
225            "Replaying StandardWithdrawal response from the guardian's S3 log (idempotent by wid)."
226        );
227        Ok(Some(response))
228    }
229
230    async fn guardian_info(&self) -> anyhow::Result<GuardianInfo> {
231        let info_pb = self
232            .inner
233            .get_guardian_info(Request::new(proto::GetGuardianInfoRequest::default()))
234            .await
235            .map_err(|s| anyhow::anyhow!("get_guardian_info: {s}"))?
236            .into_inner();
237        let response = GetGuardianInfoResponse::try_from(info_pb)
238            .map_err(|e| anyhow::anyhow!("parse guardian info: {e:?}"))?;
239        // The proxy talks to the enclave over its own direct channel; like the
240        // node, it does not re-verify the info envelope (worst case is liveness).
241        let (info, _) = response.into_info_unchecked();
242        Ok(info)
243    }
244}

246/// Verify recorded BTC signatures against the sighashes of the *incoming*
247/// request (the same immutable tx as the recorded one, by the wid invariant).
248fn verify_recorded_signatures(
249    request: &proto::SignedStandardWithdrawalRequest,
250    signatures: &[BitcoinSignature],
251    enclave_btc_pubkey: &BitcoinPubkey,
252    master_g: &HashiMasterG,
253    network: Network,
254) -> anyhow::Result<()> {
255    let wire = SignedStandardWithdrawalRequestWire::try_from(request.clone())
256        .map_err(|e| anyhow::anyhow!("parse request: {e:?}"))?;
257    let signed = HashiSigned::<StandardWithdrawalRequest>::validate_addr(wire, network)
258        .map_err(|e| anyhow::anyhow!("validate request: {e:?}"))?;
259    let (messages, _txid) = signed
260        .message()
261        .utxos()
262        .signing_messages_and_txid(enclave_btc_pubkey, master_g);
263    anyhow::ensure!(
264        messages.len() == signatures.len(),
265        "record has {} signatures for {} inputs",
266        signatures.len(),
267        messages.len()
268    );
269    for (i, (message, signature)) in messages.iter().zip(signatures).enumerate() {
270        BTC_LIB
271            .verify_schnorr(&signature.signature, message, enclave_btc_pubkey)
272            .map_err(|e| anyhow::anyhow!("input {i}: {e}"))?;
273    }
274    Ok(())
275}

277fn synthesize_response(found: &FoundSuccess) -> proto::SignedStandardWithdrawalResponse {
278    proto::SignedStandardWithdrawalResponse {
279        data: Some(proto::StandardWithdrawalResponseData {
280            enclave_signatures: found
281                .response
282                .enclave_signatures
283                .iter()
284                .map(|sig| sig.to_vec().into())
285                .collect(),
286        }),
287        timestamp_ms: Some(found.timestamp_ms),
288        // The record predates the response envelope (the enclave signs it after
289        // the S3 write). Nodes require a 64-byte value but never verify it
290        // (`into_data_unchecked`), so zeros can't pass for a real signature.
291        signature: Some(vec![0u8; 64].into()),
292    }
293}

295fn extract_wid_and_seq(
296    req: &proto::SignedStandardWithdrawalRequest,
297) -> Option<(WithdrawalID, u64)> {
298    let data = req.data.as_ref()?;
299    let wid_bytes = data.wid.as_ref()?;
300    let seq = data.seq?;
301    let wid = WithdrawalID::from_bytes(wid_bytes.as_ref()).ok()?;
302    Some((wid, seq))
303}

353    async fn standard_withdrawal(
354        &self,
355        request: Request<proto::SignedStandardWithdrawalRequest>,
356    ) -> Result<Response<proto::SignedStandardWithdrawalResponse>, Status> {
357        let Some((wid, seq)) = extract_wid_and_seq(request.get_ref()) else {
358            // No wid to key on; let the enclave produce the precise rejection.
359            return self.inner.standard_withdrawal(request).await;
360        };
361
362        if let Some((consumed_seq, cached)) = self.try_hit(&wid) {
363            self.metrics.outcome(metrics::OUTCOME_L1_HIT);
364            info!(
365                %wid,
366                requested_seq = seq,
367                consumed_seq,
368                "Cache hit; replaying stored StandardWithdrawal response (idempotent by wid)."
369            );
370            return Ok(Response::new(cached));
371        }
372
373        if let Some(replayed) = self.replay_from_log(&wid, seq, request.get_ref()).await? {
374            return Ok(Response::new(replayed));
375        }
376
377        self.metrics.outcome(metrics::OUTCOME_FORWARDED);
378        let response_inner = self.inner.standard_withdrawal(request).await?.into_inner();
379
380        self.store(wid, seq, response_inner.clone());
381        info!(%wid, seq, "Stored StandardWithdrawal response in cache");
382
383        Ok(Response::new(response_inner))
384    }
