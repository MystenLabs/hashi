72impl SecretSharingParams {
73    pub fn new(num_shares: usize, threshold: usize) -> GuardianResult<Self> {
74        if threshold < MIN_THRESHOLD {
75            return Err(InvalidInputs(format!(
76                "threshold {threshold} below minimum {MIN_THRESHOLD}"
77            )));
78        }
79        if num_shares < threshold {
80            return Err(InvalidInputs(format!(
81                "num_shares {num_shares} below threshold {threshold}"
82            )));
83        }
84        if num_shares > MAX_NUM_SHARES {
85            return Err(InvalidInputs(format!(
86                "{num_shares} must be at most u16::MAX"
87            )));
88        }
89        Ok(Self {
90            num_shares,
91            threshold,
92        })
93    }

96        self.num_shares
97    }

137/// Public description of the current BTC key's secret-sharing scheme.
138/// `sharing_seq` versions the instance: setup writes 0, each rotation bumps it by 1.
139#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]

141    commitments: ShareCommitments,
142    params: SecretSharingParams,
143    sharing_seq: u64,

146impl SecretSharingInstance {
147    pub fn new(
148        commitments: ShareCommitments,
149        num_shares: usize,
150        threshold: usize,
151        sharing_seq: u64,
152    ) -> GuardianResult<Self> {
153        let params = SecretSharingParams::new(num_shares, threshold)?;
154        if commitments.len() != params.num_shares() {
155            return Err(InvalidInputs(format!(
156                "expected {} commitments, got {}",
157                params.num_shares(),
158                commitments.len()
159            )));
160        }
161        let commitment_ids: Vec<u16> = commitments.iter().map(|c| c.id.get()).collect();
162        let expected_ids: Vec<u16> = (1..=params.num_shares() as u16).collect();
163        if commitment_ids != expected_ids {
164            return Err(InvalidInputs(format!(
165                "commitment ids are not exactly 1..={}: got {commitment_ids:?}",
166                params.num_shares()
167            )));
168        }
169        Ok(Self {
170            commitments,
171            params,
172            sharing_seq,
173        })
174    }

177        &self.commitments

181        self.params.num_shares()
182    }

184    pub fn threshold(&self) -> usize {
185        self.params.threshold()
186    }

188    pub fn sharing_seq(&self) -> u64 {

195        let bytes = bcs::to_bytes(&self.commitments).expect("serialization should work");
196        let commitments_hash = Blake2b::<U32>::digest(bytes);
197        let commitments_hash_hex = hex::encode(commitments_hash);
198        write!(
199            f,
200            "SecretSharingInstance(seq={},n={},t={},commitments_hash={})",
201            self.sharing_seq,
202            self.num_shares(),
203            self.threshold(),
204            commitments_hash_hex
205        )
206    }

209impl ShareCommitments {
210    pub fn new(commitments: Vec<ShareCommitment>) -> GuardianResult<Self> {
211        let mut map = BTreeMap::new();
212        for commitment in commitments {
213            if map.insert(commitment.id, commitment.digest).is_some() {
214                return Err(InvalidInputs("duplicate share id".into()));
215            }
216        }
217        Ok(Self(map))
218    }
219
220    pub fn from_shares(shares: &[Share]) -> GuardianResult<Self> {
221        Self::new(shares.iter().map(commit_share).collect())
222    }
223
224    pub fn len(&self) -> usize {
225        self.0.len()
226    }
227
228    pub fn is_empty(&self) -> bool {
229        self.0.is_empty()
230    }
231
232    pub fn contains(&self, commitment: &ShareCommitment) -> bool {
233        self.0
234            .get(&commitment.id)
235            .is_some_and(|digest| digest == &commitment.digest)
236    }
237
238    /// Verify `share`'s commitment is in this set.
239    pub fn verify_share(&self, share: &Share) -> GuardianResult<()> {
240        self.contains(&commit_share(share))
241            .then_some(())
242            .ok_or_else(|| InvalidInputs("No matching share found".into()))
243    }
244
245    pub fn iter(&self) -> impl Iterator<Item = ShareCommitment> + '_ {
246        self.0.iter().map(|(id, digest)| ShareCommitment {
247            id: *id,
248            digest: digest.clone(),
249        })
250    }

291impl KPEncryptedShares {
292    /// Verify that this share has exactly one ciphertext per expected cert and
293    /// that every ciphertext targets only keys owned by its keyed cert.
294    pub fn verify_recipients(&self, certs: &KpCerts) -> GuardianResult<()> {
295        let expected_fingerprints = certs.fingerprints();
296        let actual_fingerprints = self
297            .ciphertexts_by_fingerprint
298            .keys()
299            .cloned()
300            .collect::<Vec<_>>();
301        if actual_fingerprints != expected_fingerprints {
302            return Err(InvalidInputs(format!(
303                "encrypted share recipient roster differs for share id {}: expected {:?}, got \
304                 {:?}",
305                self.id.get(),
306                expected_fingerprints,
307                actual_fingerprints
308            )));
309        }
310
311        for cert in certs.pgp_certs() {
312            let fingerprint = cert.fingerprint().to_hex();
313            let ciphertext = self
314                .ciphertexts_by_fingerprint
315                .get(&fingerprint)
316                .expect("fingerprint rosters were checked for equality");
317            verify_pgp_ciphertext_recipient(self.id, &fingerprint, ciphertext, cert)?;
318        }
319        Ok(())
320    }
321}

323impl KPEncryptedSharesRoster {
324    pub fn new(mut shares: Vec<KPEncryptedShares>) -> GuardianResult<Self> {
325        shares.sort_by_key(|s| s.id);
326
327        if shares.len() > MAX_NUM_SHARES {
328            return Err(InvalidInputs(format!(
329                "{} shares must be at most u16::MAX",
330                shares.len()
331            )));
332        }
333        let ids: Vec<u16> = shares.iter().map(|s| s.id.get()).collect();
334        let expected: Vec<u16> = (1..=shares.len() as u16).collect();
335        if ids != expected {
336            return Err(InvalidInputs(format!(
337                "encrypted share ids are not exactly 1..={}: got {ids:?}",
338                shares.len()
339            )));
340        }
341
342        let mut seen_fingerprints = HashSet::new();
343        for share in &shares {
344            if share.ciphertexts_by_fingerprint.is_empty() {
345                return Err(InvalidInputs(format!(
346                    "encrypted share id {} must have at least one PGP ciphertext",
347                    share.id.get()
348                )));
349            }
350            for fingerprint in share.ciphertexts_by_fingerprint.keys() {
351                if !seen_fingerprints.insert(fingerprint.clone()) {
352                    return Err(InvalidInputs(format!(
353                        "duplicate encrypted share recipient fingerprint {}",
354                        fingerprint
355                    )));
356                }
357            }
358        }
359
360        Ok(Self(shares))
361    }
362
363    /// Number of secret shares, not the number of cert-wrapped
364    /// ciphertexts. A KP may have multiple ciphertexts for the same share id.
365    pub fn share_count(&self) -> usize {
366        self.0.len()
367    }
368
369    pub fn ciphertext_count(&self) -> usize {
370        self.0
371            .iter()
372            .map(|share| share.ciphertexts_by_fingerprint.len())
373            .sum()
374    }
375
376    pub fn iter(&self) -> impl Iterator<Item = &KPEncryptedShares> {
377        self.0.iter()
378    }
379
380    pub fn into_vec(self) -> Vec<KPEncryptedShares> {
381        self.0
382    }
383
384    pub fn find_by_fingerprint(&self, fingerprint: &str) -> Option<(&KPEncryptedShares, &str)> {
385        self.iter().find_map(|share| {
386            share
387                .ciphertexts_by_fingerprint
388                .get(fingerprint)
389                .map(|ciphertext| (share, ciphertext.as_str()))
390        })
391    }
392
393    /// Replace one certificate-specific ciphertext while preserving every
394    /// other KP/share entry and revalidating global fingerprint uniqueness.
395    pub fn replace_recipient(
396        self,
397        current_fingerprint: &str,
398        new_fingerprint: KPFingerprint,
399        new_ciphertext: String,
400    ) -> GuardianResult<(Self, KPEncryptedShares)> {
401        if new_fingerprint == current_fingerprint {
402            return Err(InvalidInputs(format!(
403                "replacement KP fingerprint {new_fingerprint} must differ from the current \
404                 fingerprint"
405            )));
406        }
407        if self.find_by_fingerprint(&new_fingerprint).is_some() {
408            return Err(InvalidInputs(format!(
409                "new KP fingerprint {new_fingerprint} is already present in the encrypted share \
410                 roster"
411            )));
412        }
413
414        let mut shares = self.0;
415        let share = shares
416            .iter_mut()
417            .find(|share| {
418                share
419                    .ciphertexts_by_fingerprint
420                    .contains_key(current_fingerprint)
421            })
422            .ok_or_else(|| {
423                InvalidInputs(format!(
424                    "current KP fingerprint {current_fingerprint} is not present in the \
425                     encrypted share roster"
426                ))
427            })?;
428        share
429            .ciphertexts_by_fingerprint
430            .remove(current_fingerprint)
431            .expect("the current fingerprint was located in this entry");
432        share
433            .ciphertexts_by_fingerprint
434            .insert(new_fingerprint, new_ciphertext);
435        let changed_share = share.clone();
436        Ok((Self::new(shares)?, changed_share))
437    }
438
439    /// Recipient PGP fingerprints grouped by share id.
440    pub fn recipient_roster(&self) -> Vec<Vec<KPFingerprint>> {
441        let mut grouped: Vec<Vec<KPFingerprint>> = Vec::with_capacity(self.share_count());
442        for share in &self.0 {
443            grouped.push(share.ciphertexts_by_fingerprint.keys().cloned().collect());
444        }
445        grouped
446    }
447
448    /// Verify every encrypted share against the certs assigned to its share id.
449    pub fn verify_recipients(&self, certs_roster: &KpCertsRoster) -> GuardianResult<()> {
450        if self.share_count() != certs_roster.num_kps() {
451            return Err(InvalidInputs(format!(
452                "expected {} KP cert roster entries, got {} encrypted shares",
453                certs_roster.num_kps(),
454                self.share_count()
455            )));
456        }
457
458        for share in self.iter() {
459            let certs = certs_roster
460                .certs_for_share(share.id)
461                .expect("encrypted share ids are exactly 1..=roster length");
462            share.verify_recipients(certs)?;
463        }
464        Ok(())
465    }
466}

478fn verify_pgp_ciphertext_recipient(
479    share_id: ShareID,
480    recipient_fingerprint: &str,
481    ciphertext: &str,
482    expected_cert: &PgpPublicCert,
483) -> GuardianResult<()> {
484    let recipients = pgp_message_recipients(ciphertext).map_err(|e| {
485        InvalidInputs(format!(
486            "failed to parse PGP recipients for share id {}: {e}",
487            share_id.get()
488        ))
489    })?;
490    if recipients.is_empty() {
491        return Err(InvalidInputs(format!(
492            "share id {} has no PGP recipients",
493            share_id.get()
494        )));
495    }
496    for handle in &recipients {
497        if !cert_owns_key_handle(expected_cert, handle) {
498            return Err(InvalidInputs(format!(
499                "share id {} (keyed by {}) is encrypted to key {handle}, which is not in that \
500                 cert",
501                share_id.get(),
502                recipient_fingerprint
503            )));
504        }
505    }
506    info!(
507        share_id = share_id.get(),
508        fingerprint = %recipient_fingerprint,
509        recipient_count = recipients.len(),
510        "verified encrypted share targets only its keyed recipient cert"
511    );
512    Ok(())
513}

523/// Encrypts plaintext. Returns InvalidInputs if plaintext / aad is extraordinarily long (~2^36).
524pub fn encrypt<R: CryptoRng + RngCore>(
525    bytes: &[u8],
526    pk: &EncPubKey,
527    aad: Option<&[u8; 32]>,
528    rng: &mut R,
529) -> GuardianResult<Ciphertext> {
530    let (encapsulated_key, aes_ciphertext) =
531        hpke::single_shot_seal::<AesGcm256, HkdfSha384, X25519HkdfSha256, _>(
532            &hpke::OpModeS::Base,
533            pk,
534            &[],
535            bytes,
536            aad.unwrap_or(&[0; 32]),
537            rng,
538        )
539        .map_err(|e| InvalidInputs(format!("Encryption failed: {}", e)))?;
540    Ok(Ciphertext {
541        encapsulated_key: encapsulated_key.to_bytes().to_vec(),
542        aes_ciphertext,
543    })
544}
545
546/// Decrypts ciphertext. Returns InvalidInputs if aad is invalid.
547pub fn decrypt(
548    ciphertext: &Ciphertext,
549    sk: &EncSecKey,
550    aad: Option<&[u8; 32]>,
551) -> GuardianResult<Vec<u8>> {
552    let encapsulated_key = EncapsulatedKey::from_bytes(&ciphertext.encapsulated_key)
553        .map_err(|e| InvalidInputs(format!("Failed to deserialize encapsulated key: {}", e)))?;
554    hpke::single_shot_open::<AesGcm256, HkdfSha384, X25519HkdfSha256>(
555        &hpke::OpModeR::Base,
556        sk,
557        &encapsulated_key,
558        &[],
559        &ciphertext.aes_ciphertext,
560        aad.unwrap_or(&[0; 32]),
561    )
562    .map_err(|e| InvalidInputs(format!("Decryption failed: {}", e)))
563}

569/// Split a k256 SecretKey into `params.num_shares()` shares using Shamir's
570/// secret-sharing with reconstruction threshold `params.threshold()`.
571pub fn split_secret<R: CryptoRng + RngCore>(
572    sk: &k256::SecretKey,
573    params: &SecretSharingParams,
574    rng: &mut R,
575) -> Vec<Share> {
576    let secret = *sk.to_nonzero_scalar().as_ref();
577    let mut coefficients = vec![secret];
578    for _ in 0..(params.threshold() - 1) {
579        coefficients.push(Scalar::random(&mut *rng))
580    }
581
582    (1..=params.num_shares())
583        .map(|i| NonZeroU16::new(i as u16).expect("validated num_shares fits in u16"))
584        .map(|i| Share {
585            id: i,
586            value: eval_poly(i, &coefficients),
587        })
588        .collect()
589}
590
591// Coefficients: [c0, c1, c2, c3]
592// Returns: c0 + c1 * x + c2 * x^2 + c3 * x^3
593pub fn eval_poly(pos: ShareID, coefficients: &[Scalar]) -> Scalar {
594    let x = to_scalar(pos);
595    let mut out = Scalar::ZERO;
596    let mut xpow = Scalar::ONE;
597    for c in coefficients {
598        out = out.add(&c.mul(&xpow));
599        xpow = xpow.mul(&x);
600    }
601    out
602}

604/// Combine secret shares into the reconstructed `k256::SecretKey` with
605/// threshold `t`. Errors on duplicate share IDs or fewer than `t` shares.
606/// Callers that need a `bitcoin::secp256k1::Keypair` should pass the result
607/// through `k256_sk_to_btc_keypair`.
608pub fn combine_shares(shares: &[Share], t: usize) -> GuardianResult<k256::SecretKey> {
609    // Validation: ensure no duplicates
610    let mut seen_ids = std::collections::HashSet::new();
611    for share in shares {
612        if !seen_ids.insert(share.id) {
613            return Err(InvalidInputs("Duplicate share ID".into()));
614        }
615    }
616    if seen_ids.len() < t {
617        return Err(InvalidInputs(format!(
618            "Received only {} out of {} shares",
619            seen_ids.len(),
620            t
621        )));
622    }
623
624    let ids = shares.iter().map(|s| to_scalar(s.id)).collect::<Vec<_>>();
625    let mut result = Scalar::ZERO;
626    for share in shares {
627        let cur_share_id = to_scalar(share.id);
628        let numerator: Scalar = ids
629            .iter()
630            .filter(|&id| cur_share_id != *id)
631            .map(|id| id.negate())
632            .product();
633        let denominator: Scalar = ids
634            .iter()
635            .filter(|&id| cur_share_id != *id)
636            .map(|id| cur_share_id.sub(id))
637            .product();
638
639        // Lagrange basis polynomial evaluated at x=0
640        // L_i(0) = product_{j != i} (-x_j) / (x_i - x_j)
641        let lagrange_basis = numerator.mul(
642            &denominator
643                .invert()
644                .expect("Denominator is never zero because share IDs are unique"),
645        );
646        result = result.add(&share.value.mul(&lagrange_basis));
647    }
648
649    info!("Bitcoin key created with fingerprint {:x}", exp_g(&result));
650
651    Ok(k256::SecretKey::from_slice(&result.to_bytes())
652        .expect("k256 scalar bytes are a valid k256 secret key"))
653}

655/// Convert a `k256::SecretKey` to a `bitcoin::secp256k1::Keypair`. Both libs
656/// use big-endian 32-byte scalars so the byte round-trip is value-preserving.
657/// We juggle between the two libraries because secp256k1 does not expose the
658/// arithmetic tools needed for secret-sharing.
659pub fn k256_sk_to_btc_keypair(sk: &k256::SecretKey) -> bitcoin::secp256k1::Keypair {
660    let btc_sk = bitcoin::secp256k1::SecretKey::from_slice(&sk.to_bytes())
661        .expect("k256 secret key bytes are a valid secp256k1 secret key");
662    bitcoin::secp256k1::Keypair::from_secret_key(&BTC_LIB, &btc_sk)
663}

672pub fn commit_share(share: &Share) -> ShareCommitment {
673    let commitment = ProjectivePoint::GENERATOR * share.value;
674    ShareCommitment {
675        id: share.id,
676        digest: commitment.to_bytes().to_vec(),
677    }
678}

680/// Encrypt a share with optional AAD
681pub fn encrypt_share<R: CryptoRng + RngCore>(
682    share: &Share,
683    pk: &EncPubKey,
684    aad: Option<&[u8; 32]>,
685    rng: &mut R,
686) -> GuardianEncryptedShare {
687    GuardianEncryptedShare {
688        id: share.id,
689        ciphertext: encrypt(&share.value.to_bytes(), pk, aad, rng)
690            .expect("neither plaintext nor aad are long"),
691    }
692}

694/// Split `sk` into `params.num_shares()` shares with reconstruction threshold
695/// `params.threshold()`, encrypt each share to every cert in the matching KP
696/// cert collection, and compute one commitment per share. The roster assigns
697/// its entry at position `i` to share ID `i + 1`.
698///
699/// # Panics
700///
701/// Panics if `kp_certs_roster.num_kps() != params.num_shares()`.
702pub fn split_and_encrypt_for_kps<R: CryptoRng + RngCore>(
703    sk: &k256::SecretKey,
704    kp_certs_roster: &KpCertsRoster,
705    params: &SecretSharingParams,
706    rng: &mut R,
707) -> (KPEncryptedSharesRoster, ShareCommitments) {
708    assert_eq!(
709        kp_certs_roster.num_kps(),
710        params.num_shares(),
711        "request validation ensures one KP cert collection per share",
712    );
713    let shares = split_secret(sk, params, rng);
714    let n = params.num_shares();
715    let mut encrypted_shares = Vec::with_capacity(n);
716    let mut commitments = Vec::with_capacity(n);
717    for (share, cert_set) in shares.iter().zip(kp_certs_roster.iter()) {
718        let ciphertexts_by_fingerprint = cert_set
719            .pgp_certs()
720            .iter()
721            .map(|cert| {
722                (
723                    cert.fingerprint().to_hex(),
724                    encrypt_share_for_provisioner(share, cert),
725                )
726            })
727            .collect();
728        encrypted_shares.push(KPEncryptedShares {
729            id: share.id,
730            ciphertexts_by_fingerprint,
731        });
732        commitments.push(commit_share(share));
733    }
734    let encrypted_shares = KPEncryptedSharesRoster::new(encrypted_shares)
735        .expect("split_secret produces share ids exactly 1..=n");
736    let commitments =
737        ShareCommitments::new(commitments).expect("share IDs 1..=n are unique by construction");
738    (encrypted_shares, commitments)
739}

741/// Encrypt a share for delivery to a key provisioner using OpenPGP ASCII armor.
742pub fn encrypt_share_for_provisioner(share: &Share, cert: &PgpPublicCert) -> String {
743    encrypt_armored(&share.value.to_bytes(), cert)
744        .expect("PgpPublicCert validation ensures OpenPGP encryption works")
745}

747/// Decrypt an encrypted share with optional AAD
748pub fn decrypt_share(
749    encrypted_share: &GuardianEncryptedShare,
750    sk: &EncSecKey,
751    aad: Option<&[u8; 32]>,
752) -> GuardianResult<Share> {
753    let serialized_share = decrypt(&encrypted_share.ciphertext, sk, aad)?;
754    let value = ScalarPrimitive::<Secp256k1>::from_slice(&serialized_share)
755        .map(Scalar::from)
756        .map_err(|_| InvalidInputs("Failed to deserialize share".into()))?;
757    Ok(Share {
758        id: encrypted_share.id,
759        value,
760    })
761}
762
763/// Decrypt each submission under optional `aad`, verify it against `commitments`,
764/// and reject duplicate share ids. Errors if fewer than `threshold` shares
765/// result — i.e. the set can't reconstruct the secret. Shared by
766/// `provisioner_init` (signature-bound, no AAD) and `rotate_kps` (state-bound
767/// AAD), which collect a batch of KP shares the same way.
768pub fn decrypt_verify_shares(
769    encrypted: &[GuardianEncryptedShare],
770    sk: &EncSecKey,
771    aad: Option<&[u8; 32]>,
772    commitments: &ShareCommitments,
773    threshold: usize,
774) -> GuardianResult<Vec<Share>> {
775    let mut shares: Vec<Share> = Vec::with_capacity(encrypted.len());
776    for enc in encrypted {
777        let share = decrypt_share(enc, sk, aad)?;
778        commitments.verify_share(&share)?;
779        if shares.iter().any(|s| s.id == share.id) {
780            return Err(InvalidInputs("Duplicate share ID".into()));
781        }
782        shares.push(share);
783    }
784    if shares.len() < threshold {
785        return Err(InvalidInputs(format!(
786            "need at least {threshold} shares, got {}",
787            shares.len()
788        )));
789    }
790    Ok(shares)
791}

793pub fn fingerprint(sk: &k256::SecretKey) -> CompressedPoint {
794    exp_g(&Scalar::from(sk.as_scalar_primitive()))
795}

797pub fn exp_g(scalar: &Scalar) -> CompressedPoint {
798    (ProjectivePoint::GENERATOR * scalar).to_bytes()
799}
