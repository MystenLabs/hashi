4use super::crypto::ShareID;
5use super::errors::GuardianError::InvalidInputs;
6use super::errors::GuardianResult;
7use crate::pgp::Fingerprint;
8use crate::pgp::PgpPublicCert;

10use std::collections::HashSet;

12/// A key-provisioner's PGP fingerprint as bare uppercase hex — the string
13/// form persisted in ceremony artifacts (ciphertext map keys, log rosters). For
14/// comparing fingerprints, prefer the canonical [`crate::pgp::Fingerprint`].
15pub type KPFingerprint = String;
16
17/// One key provisioner's accepted OpenPGP certs. A KP may have multiple
18/// certs for the same share id, e.g. independent yubikeys.
19/// Certificates must have unique fingerprints.
20#[derive(Serialize, Debug, Clone, PartialEq)]
21pub struct KpCerts(Vec<PgpPublicCert>);
22
23/// The ordered KP certificate roster for a sharing instance.
24///
25/// The cert collection at position `i` is assigned share id `i + 1`. This type
26/// preserves that caller-supplied order and requires every certificate
27/// fingerprint to occur in exactly one roster entry. Each [`KpCerts`]
28/// separately canonicalizes its certificate order.
29#[derive(Serialize, Debug, Clone, PartialEq)]
30pub struct KpCertsRoster(Vec<KpCerts>);
31

36impl KpCerts {
37    pub fn new(mut pgp_certs: Vec<PgpPublicCert>) -> GuardianResult<Self> {
38        if pgp_certs.is_empty() {
39            return Err(InvalidInputs(
40                "KP certs must contain at least one OpenPGP certificate".into(),
41            ));
42        }
43
44        pgp_certs.sort_by_key(|cert| cert.fingerprint().to_hex());
45        let mut seen = HashSet::with_capacity(pgp_certs.len());
46        for cert in &pgp_certs {
47            let fingerprint = cert.fingerprint();
48            if !seen.insert(fingerprint.clone()) {
49                return Err(InvalidInputs(format!(
50                    "duplicate OpenPGP certificate fingerprint {fingerprint}"
51                )));
52            }
53        }
54
55        Ok(Self(pgp_certs))
56    }
57
58    pub fn pgp_certs(&self) -> &[PgpPublicCert] {
59        &self.0
60    }
61
62    pub fn into_pgp_certs(self) -> Vec<PgpPublicCert> {
63        self.0
64    }
65
66    pub fn fingerprints(&self) -> Vec<KPFingerprint> {
67        self.0
68            .iter()
69            .map(|cert| cert.fingerprint().to_hex())
70            .collect()
71    }
72
73    /// Replace one certificate identified by its current primary-key
74    /// fingerprint, then revalidate and canonicalize this KP's certificate set.
75    pub fn replace_cert(
76        self,
77        current_fingerprint: &Fingerprint,
78        new_cert: PgpPublicCert,
79    ) -> GuardianResult<Self> {
80        let mut pgp_certs = self.0;
81        let cert = pgp_certs
82            .iter_mut()
83            .find(|cert| cert.fingerprint() == *current_fingerprint)
84            .ok_or_else(|| {
85                InvalidInputs(format!(
86                    "OpenPGP certificate fingerprint {current_fingerprint} is not in this KP \
87                    certificate set"
88                ))
89            })?;
90        let new_fingerprint = new_cert.fingerprint();
91        if new_fingerprint == *current_fingerprint {
92            return Err(InvalidInputs(format!(
93                "replacement OpenPGP certificate fingerprint {new_fingerprint} must differ from \
94                 the current fingerprint"
95            )));
96        }
97        *cert = new_cert;
98        Self::new(pgp_certs)
99    }
100}

102impl KpCertsRoster {
103    pub fn new(kp_certs: Vec<KpCerts>) -> GuardianResult<Self> {
104        let cert_count = kp_certs.iter().map(|certs| certs.pgp_certs().len()).sum();
105        let mut seen = HashSet::with_capacity(cert_count);
106        for cert in kp_certs.iter().flat_map(KpCerts::pgp_certs) {
107            let fingerprint = cert.fingerprint();
108            if !seen.insert(fingerprint.clone()) {
109                return Err(InvalidInputs(format!(
110                    "duplicate OpenPGP certificate fingerprint {fingerprint}"
111                )));
112            }
113        }
114
115        Ok(Self(kp_certs))
116    }
117
118    pub fn num_kps(&self) -> usize {
119        self.0.len()
120    }
121
122    pub fn iter(&self) -> impl Iterator<Item = &KpCerts> {
123        self.0.iter()
124    }
125
126    pub fn certs_for_share(&self, share_id: ShareID) -> Option<&KpCerts> {
127        self.0.get(usize::from(share_id.get()) - 1)
128    }
129
130    /// Return the complete KP cert set containing `fingerprint`.
131    pub fn certs_for_fingerprint(&self, fingerprint: &Fingerprint) -> Option<&KpCerts> {
132        self.0.iter().find(|certs| {
133            certs
134                .pgp_certs()
135                .iter()
136                .any(|cert| cert.fingerprint() == *fingerprint)
137        })
138    }
139
140    pub fn fingerprints(&self) -> Vec<Vec<KPFingerprint>> {
141        self.0.iter().map(KpCerts::fingerprints).collect()
142    }
143
144    pub fn into_vec(self) -> Vec<KpCerts> {
145        self.0
146    }
147
148    /// Replace one certificate in the roster while preserving the KP/share
149    /// ordering and global fingerprint-uniqueness invariant.
150    pub fn replace_cert(
151        self,
152        current_fingerprint: &Fingerprint,
153        new_cert: PgpPublicCert,
154    ) -> GuardianResult<Self> {
155        let mut kp_certs = self.0;
156        let certs = kp_certs
157            .iter_mut()
158            .find(|certs| {
159                certs
160                    .pgp_certs()
161                    .iter()
162                    .any(|cert| cert.fingerprint() == *current_fingerprint)
163            })
164            .ok_or_else(|| {
165                InvalidInputs(format!(
166                    "OpenPGP certificate fingerprint {current_fingerprint} is not in the KP \
167                     certificate roster"
168                ))
169            })?;
170        *certs = certs.clone().replace_cert(current_fingerprint, new_cert)?;
171        Self::new(kp_certs)
172    }
173}

175#[cfg(test)]
176mod tests {
177    use super::*;
178    use crate::pgp::test_utils::mock_pgp_keypair;
179
180    fn cert() -> PgpPublicCert {
181        let (public, _) = mock_pgp_keypair();
182        PgpPublicCert::new(public).unwrap()
183    }
184
185    #[test]
186    fn replace_cert_preserves_kp_grouping_and_rejects_fingerprint_collisions() {
187        let old = cert();
188        let sibling = cert();
189        let other_kp = cert();
190        let replacement = cert();
191        let roster = KpCertsRoster::new(vec![
192            KpCerts::new(vec![old.clone(), sibling.clone()]).unwrap(),
193            KpCerts::new(vec![other_kp.clone()]).unwrap(),
194        ])
195        .unwrap();
196
197        let rotated = roster
198            .clone()
199            .replace_cert(&old.fingerprint(), replacement.clone())
200            .unwrap();
201        assert_eq!(rotated.num_kps(), 2);
202        assert_eq!(
203            rotated.fingerprints(),
204            vec![
205                KpCerts::new(vec![sibling.clone(), replacement])
206                    .unwrap()
207                    .fingerprints(),
208                vec![other_kp.fingerprint().to_hex()],
209            ]
210        );
211
212        let err = roster
213            .clone()
214            .replace_cert(&old.fingerprint(), sibling)
215            .unwrap_err();
216        assert!(format!("{err}").contains("duplicate"), "{err}");
217
218        let err = roster.replace_cert(&old.fingerprint(), old).unwrap_err();
219        assert!(format!("{err}").contains("must differ"), "{err}");
220    }
221
222    #[test]
223    fn finds_complete_cert_set_by_member_fingerprint() {
224        let first = cert();
225        let sibling = cert();
226        let other_kp = cert();
227        let expected = KpCerts::new(vec![first, sibling.clone()]).unwrap();
228        let roster = KpCertsRoster::new(vec![
229            expected.clone(),
230            KpCerts::new(vec![other_kp]).unwrap(),
231        ])
232        .unwrap();
233
234        assert_eq!(
235            roster.certs_for_fingerprint(&sibling.fingerprint()),
236            Some(&expected)
237        );
238        assert!(
239            roster
240                .certs_for_fingerprint(&cert().fingerprint())
241                .is_none()
242        );
243    }
