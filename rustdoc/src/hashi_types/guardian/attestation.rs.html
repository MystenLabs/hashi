18/// Raw AWS Nitro attestation document bytes.
19#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
20pub struct NitroAttestation(#[serde(with = "crate::guardian::serde_utils::base64_bytes")] Vec<u8>);

22impl NitroAttestation {
23    pub fn new(bytes: Vec<u8>) -> Self {
24        Self(bytes)
25    }
26
27    pub fn into_bytes(self) -> Vec<u8> {
28        self.0
29    }
30
31    /// Verify a LIVE attestation document (COSE signature + AWS cert chain to the
32    /// Nitro root, chain validity checked at the current time), that it commits to
33    /// `signing_pubkey`, and that its PCR0 matches `build_pcrs`.
34    ///
35    /// In non-enclave dev/test builds the enclave emits a mock document, so the
36    /// attestation check is a no-op, mirroring `get_attestation` `non-enclave-dev`
37    /// stub behavior. Real verification runs only in enclave builds.
38    pub fn verify_live(
39        &self,
40        signing_pubkey: &GuardianPubKey,
41        build_pcrs: &BuildPcrs,
42    ) -> GuardianResult<()> {
43        self.verify_at(signing_pubkey, build_pcrs, VerifyTime::Now)
44    }
45
46    /// Verify a REPLAYED attestation document read back from the S3 audit log.
47    ///
48    /// The Nitro leaf certificate lives only a few hours, so a replayed document
49    /// can never chain-validate at the current time. Anchor the chain check at the
50    /// document's own timestamp instead: it is inside the COSE-signed payload (so
51    /// it cannot be altered without breaking the signature just verified), and a
52    /// genuine document's chain was valid at the instant NSM produced it.
53    pub fn verify_replay(
54        &self,
55        signing_pubkey: &GuardianPubKey,
56        build_pcrs: &BuildPcrs,
57    ) -> GuardianResult<()> {
58        self.verify_at(signing_pubkey, build_pcrs, VerifyTime::DocumentTimestamp)
59    }
60
61    fn verify_at(
62        &self,
63        signing_pubkey: &GuardianPubKey,
64        build_pcrs: &BuildPcrs,
65        verify_time: VerifyTime,
66    ) -> GuardianResult<()> {
67        #[cfg(any(test, feature = "non-enclave-dev"))]
68        {
69            let _ = (signing_pubkey, build_pcrs, verify_time);
70            // Real attestation is stubbed here — announce it loudly (once) so a
71            // `non-enclave-dev` binary can't be mistaken for a real enclave.
72            // Gated off `test` so unit tests stay quiet.
73            #[cfg(all(feature = "non-enclave-dev", not(test)))]
74            {
75                static WARNED: std::sync::Once = std::sync::Once::new();
76                WARNED.call_once(|| {
77                    tracing::warn!(
78                        "Nitro attestation verification DISABLED (non-enclave-dev build)"
79                    );
80                });
81            }
82            Ok(())
83        }
84        #[cfg(not(any(test, feature = "non-enclave-dev")))]
85        {
86            use fastcrypto::nitro_attestation::parse_nitro_attestation;
87            use fastcrypto::nitro_attestation::verify_nitro_attestation;
88
89            // Bools: (is_upgraded_parsing, include_all_nonzero_pcrs,
90            // always_include_required_pcrs). The last keeps PCR0 in `pcr_map` even if
91            // zero, so the pin below can't be bypassed by a missing entry.
92            let (signature, signed_message, doc) =
93                parse_nitro_attestation(&self.0, true, true, true)
94                    .map_err(|e| InvalidInputs(format!("attestation parse failed: {e}")))?;
95            let timestamp_ms = match verify_time {
96                VerifyTime::Now => now_timestamp_ms(),
97                VerifyTime::DocumentTimestamp => doc.timestamp,
98            };
99            verify_nitro_attestation(&signature, &signed_message, &doc, timestamp_ms)
100                .map_err(|e| InvalidInputs(format!("attestation verification failed: {e}")))?;
101
102            let attested = doc
103                .public_key
104                .ok_or_else(|| InvalidInputs("attestation has no public_key".into()))?;
105            if attested != signing_pubkey.to_bytes() {
106                return Err(InvalidInputs(
107                    "attestation public_key does not match the session signing pubkey".into(),
108                ));
109            }
110
111            // Pin PCR0 (the whole EIF image hash).
112            if doc.pcr_map.get(&0).map(Vec::as_slice) != Some(build_pcrs.pcr0()) {
113                return Err(InvalidInputs(
114                    "attestation PCR0 does not match the expected enclave image".into(),
115                ));
116            }
117            Ok(())
118        }
119    }
120}

122/// When the attestation's cert chain validity is checked: at the current time
123/// (live RPC responses) or at the document's own COSE-signed timestamp
124/// (replays from the S3 audit log, where the short-lived leaf has expired).
125#[derive(Clone, Copy)]
126enum VerifyTime {
127    Now,
128    DocumentTimestamp,
129}

140/// One enclave build: its git revision and known-good Nitro measurement. A build is
141/// identified by its revision and pinned by PCR0 - the hash of the whole enclave
142/// image (EIF), which uniquely identifies the build.
143///
144/// We record only PCR0: in a StageX (reproducible, single-binary) build it is
145/// the only measurement that carries signal - the others (kernel, bootloader, IAM
146/// role) are constant or irrelevant for our pinning.
147#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
148pub struct BuildPcrs {
149    git_revision: GitRevision,
150    pcr0: Vec<u8>,
151}
152
153impl BuildPcrs {
154    pub fn new(git_revision: &str, pcr0: Vec<u8>) -> Self {
155        Self {
156            git_revision: git_revision.to_string(),
157            pcr0,
158        }
159    }
160
161    pub fn git_revision(&self) -> &str {
162        &self.git_revision
163    }
164
165    pub fn pcr0(&self) -> &[u8] {
166        &self.pcr0
167    }
168}

170/// PCR pins for enclave builds that may appear in guardian attestations.
171/// Used by GuardianReader in s3_reader.rs.
172///
173/// `current_build` is the current/live build. `prev_builds` contains older
174/// builds that may still appear in persisted logs during an upgrade or replay.
175/// Verification matches the signature-verified `untrusted_git_revision` to one
176/// entry, then checks PCR0 against that entry. Callers use the resolved
177/// `BuildPcrs` to enforce the policy for their context.
178#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
179pub struct PcrAllowlist {
180    current_build: BuildPcrs,
181    prev_builds: Vec<BuildPcrs>,
182}
183
184impl PcrAllowlist {
185    pub fn new(
186        current_build: BuildPcrs,
187        prev_builds: impl IntoIterator<Item = BuildPcrs>,
188    ) -> GuardianResult<Self> {
189        let prev_builds = prev_builds.into_iter().collect::<Vec<_>>();
190        let mut seen = BTreeSet::new();
191        for build in std::iter::once(&current_build).chain(prev_builds.iter()) {
192            if !seen.insert(build.git_revision.clone()) {
193                return Err(InvalidInputs(format!(
194                    "duplicate PCR allowlist entry for build '{}'",
195                    build.git_revision
196                )));
197            }
198        }
199
200        Ok(Self {
201            current_build,
202            prev_builds,
203        })
204    }
205
206    pub fn current_build(&self) -> &BuildPcrs {
207        &self.current_build
208    }
209
210    pub fn prev_builds(&self) -> &[BuildPcrs] {
211        &self.prev_builds
212    }
213
214    /// The `BuildPcrs` whose revision is `git_revision`.
215    pub fn resolve(&self, git_revision: &str) -> GuardianResult<&BuildPcrs> {
216        if self.current_build.git_revision == git_revision {
217            return Ok(&self.current_build);
218        }
219        if let Some(prev_build) = self
220            .prev_builds
221            .iter()
222            .find(|build| build.git_revision == git_revision)
223        {
224            return Ok(prev_build);
225        }
226        Err(InvalidInputs(format!(
227            "enclave reports build '{git_revision}' not present in PCR allowlist"
228        )))
229    }
230
231    pub fn is_current_build(&self, build_pcrs: &BuildPcrs) -> bool {
232        self.current_build == *build_pcrs
233    }
234
235    pub fn require_current_build(&self, build_pcrs: &BuildPcrs) -> GuardianResult<()> {
236        if self.is_current_build(build_pcrs) {
237            Ok(())
238        } else {
239            Err(InvalidInputs(
240                "guardian attestation matched a non-current build, but current build is required"
241                    .into(),
242            ))
243        }
244    }
245}

247impl<'de> Deserialize<'de> for BuildPcrs {
248    fn deserialize<D: serde::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
249        #[derive(Deserialize)]
250        struct BuildPcrsWire {
251            git_revision: GitRevision,
252            pcr0: String,
253        }
254
255        let wire = BuildPcrsWire::deserialize(deserializer)?;
256        let pcr0 = hex::decode(wire.pcr0.trim_start_matches("0x")).map_err(|e| {
257            serde::de::Error::custom(format!(
258                "build '{}' pcr0 is not valid hex: {e}",
259                wire.git_revision
260            ))
261        })?;
262        Ok(BuildPcrs::new(&wire.git_revision, pcr0))
263    }
264}

266impl<'de> Deserialize<'de> for PcrAllowlist {
267    fn deserialize<D: serde::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
268        #[derive(Deserialize)]
269        struct PcrAllowlistWire {
270            current_build: BuildPcrs,
271            #[serde(default)]
272            prev_builds: Vec<BuildPcrs>,
273        }
274
275        let wire = PcrAllowlistWire::deserialize(deserializer)?;
276        PcrAllowlist::new(wire.current_build, wire.prev_builds).map_err(serde::de::Error::custom)
277    }
