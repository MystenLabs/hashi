54/// Length of the session ID prefix (hex chars) used in S3 keys. 16 hex =
55/// 64 bits of the signing pubkey, comfortably below any collision risk for
56/// realistic session counts.
57pub const SESSION_ID_HEX_LEN: usize = 16;
58
59/// Canonical guardian session ID — a short prefix of the hex-encoded signing
60/// public key. Used as a per-session tag in S3 object keys; full pubkey
61/// verification still happens via the signed log payload.
62pub fn session_id_from_signing_pubkey(signing_pub_key: &GuardianPubKey) -> SessionID {
63    let mut s = ::hex::encode(signing_pub_key.as_bytes());
64    s.truncate(SESSION_ID_HEX_LEN);
65    s
66}

139/// Full operator-supplied withdraw-mode config: the attested `WithdrawModeState`

152/// The withdraw-mode state KPs attest to. Its `digest()` is the `state_hash`:
153/// bound as HPKE AAD on each KP's share and exposed (as a hash) via `GuardianInfo`.
154/// These are exactly the fields whose agreement is enforced *only* via the digest.
155#[derive(Debug, Clone, PartialEq)]
156pub struct WithdrawModeState {
157    /// Current Hashi committee

163    /// Raw MPC verifying key (curve point with y-parity preserved). The
164    /// guardian uses this directly for `derive_verifying_key` so the
165    /// 2-of-2 child key in the leaf script matches the MPC signature.
166    hashi_btc_master_pubkey: HashiMasterG,
167}

488impl RotateKpsState {
489    pub fn new(
490        new_kp_pgp_certs: Vec<PgpPublicCert>,
491        new_num_shares: usize,
492        new_threshold: usize,
493    ) -> GuardianResult<Self> {
494        let new_params = SecretSharingParams::new(new_num_shares, new_threshold)?;
495        if new_kp_pgp_certs.len() != new_params.num_shares() {
496            return Err(InvalidInputs(format!(
497                "expected {} new KP certs, got {}",
498                new_params.num_shares(),
499                new_kp_pgp_certs.len()
500            )));
501        }
502        // Sort to a canonical order so the serialized state's digest (which all
503        // T old KPs must agree on) is independent of submission order. Sorting
504        // also makes duplicates adjacent.
505        let mut new_kp_pgp_certs = new_kp_pgp_certs;
506        new_kp_pgp_certs.sort();
507        for pair in new_kp_pgp_certs.windows(2) {
508            if pair[0] == pair[1] {
509                return Err(InvalidInputs("duplicate new KP cert".into()));
510            }
511        }
512        Ok(Self {
513            new_kp_pgp_certs,
514            new_params,
515        })
516    }
517
518    pub fn new_kp_pgp_certs(&self) -> &[PgpPublicCert] {
519        &self.new_kp_pgp_certs
520    }
521
522    pub fn new_params(&self) -> &SecretSharingParams {
523        &self.new_params
524    }
525
526    pub fn into_parts(self) -> (Vec<PgpPublicCert>, SecretSharingParams) {
527        (self.new_kp_pgp_certs, self.new_params)
528    }
529
530    pub fn digest(&self) -> [u8; 32] {
531        let bytes = bcs::to_bytes(self).expect("serialization should work");
532        Blake2b::<U32>::digest(bytes).into()
533    }
534}

654impl GetGuardianInfoResponse {
655    /// Verify the response end to end and return the trusted `GuardianInfo`: the
656    /// attestation anchors `signing_pub_key`, and `signed_info` must be signed by
657    /// it. Self-contained so any caller (KP init, hashi nodes) can turn an
658    /// untrusted response into a trusted `GuardianInfo` in one call.
659    pub fn verify(&self) -> GuardianResult<GuardianInfo> {
660        verify_enclave_attestation(&self.attestation, &self.signing_pub_key)?;
661        self.signed_info.clone().verify(&self.signing_pub_key)
662    }

702/// Converter from T -> Self that internally validates addresses
703pub trait AddressValidation<T>: Sized {
704    fn validate_addr(value: T, network: Network) -> GuardianResult<Self>;
705}
706
707impl AddressValidation<SignedStandardWithdrawalRequestWire>
708    for HashiSigned<StandardWithdrawalRequest>
709{
710    fn validate_addr(
711        wire_value: SignedStandardWithdrawalRequestWire,
712        network: Network,
713    ) -> GuardianResult<Self> {
714        HashiSigned::<StandardWithdrawalRequest>::new(
715            wire_value.signature.epoch,
716            StandardWithdrawalRequest::validate_addr(wire_value.data, network)?,
717            &wire_value.signature.signature,
718            &wire_value.signature.signers_bitmap,
719        )
720        .map_err(|e| InvalidInputs(format!("{:?}", e)))
721    }
722}
723
724impl AddressValidation<StandardWithdrawalRequestWire> for StandardWithdrawalRequest {
725    fn validate_addr(
726        value: StandardWithdrawalRequestWire,
727        network: Network,
728    ) -> GuardianResult<Self> {
729        Ok(Self {
730            wid: value.wid,
731            utxos: TxUTXOs::new(value.utxos.inputs, value.utxos.outputs, network)
732                .map_err(|e| InvalidInputs(e.to_string()))?,
733            timestamp_secs: value.timestamp_secs,
734            seq: value.seq,
735        })
736    }
737}

739impl From<StandardWithdrawalRequest> for StandardWithdrawalRequestWire {

742            wid: m.wid,
743            utxos: m.utxos.into(),
744            timestamp_secs: m.timestamp_secs,
745            seq: m.seq,
746        }
747    }
748}

790        let a = RotateKpsState::new(certs, 5, 3).unwrap();
791        let b = RotateKpsState::new(reversed, 5, 3).unwrap();
