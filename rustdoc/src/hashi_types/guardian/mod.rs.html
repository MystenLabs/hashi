60/// Length of the session ID prefix (hex chars) used in S3 keys. 16 hex =
61/// 64 bits of the signing pubkey, comfortably below any collision risk for
62/// realistic session counts.
63pub const SESSION_ID_HEX_LEN: usize = 16;
64
65/// Canonical guardian session ID — a short prefix of the hex-encoded signing
66/// public key. Used as a per-session tag in S3 object keys; full pubkey
67/// verification still happens via the signed log payload.
68pub fn session_id_from_signing_pubkey(signing_pub_key: &GuardianPubKey) -> SessionID {
69    let mut s = ::hex::encode(signing_pub_key.as_bytes());
70    s.truncate(SESSION_ID_HEX_LEN);
71    s
72}

313impl SetupNewKeyRequest {
314    pub fn new(
315        pgp_certs: Vec<PgpPublicCert>,
316        num_shares: usize,
317        threshold: usize,
318    ) -> GuardianResult<Self> {
319        let params = SecretSharingParams::new(num_shares, threshold)?;
320        if pgp_certs.len() != params.num_shares() {
321            return Err(InvalidInputs(format!(
322                "expected {} OpenPGP certificates, got {}",
323                params.num_shares(),
324                pgp_certs.len()
325            )));
326        }
327        ensure_unique_pgp_cert_fingerprints(&pgp_certs)?;
328        Ok(Self {
329            key_provisioner_pgp_certs: pgp_certs,
330            params,
331        })
332    }
333
334    pub fn pgp_certs(&self) -> &[PgpPublicCert] {
335        &self.key_provisioner_pgp_certs
336    }
337
338    pub fn params(&self) -> &SecretSharingParams {
339        &self.params
340    }
341
342    pub fn num_shares(&self) -> usize {
343        self.params.num_shares()
344    }
345
346    pub fn threshold(&self) -> usize {
347        self.params.threshold()
348    }
349}
350
351fn ensure_unique_pgp_cert_fingerprints(pgp_certs: &[PgpPublicCert]) -> GuardianResult<()> {
352    let mut seen = std::collections::HashSet::with_capacity(pgp_certs.len());
353    for cert in pgp_certs {
354        let fingerprint = cert.fingerprint();
355        if !seen.insert(fingerprint.clone()) {
356            return Err(InvalidInputs(format!(
357                "duplicate OpenPGP certificate fingerprint {fingerprint}"
358            )));
359        }
360    }
361    Ok(())
362}

521impl RotateKpsState {
522    pub fn new(
523        new_kp_pgp_certs: Vec<PgpPublicCert>,
524        new_num_shares: usize,
525        new_threshold: usize,
526    ) -> GuardianResult<Self> {
527        let new_params = SecretSharingParams::new(new_num_shares, new_threshold)?;
528        if new_kp_pgp_certs.len() != new_params.num_shares() {
529            return Err(InvalidInputs(format!(
530                "expected {} new KP certs, got {}",
531                new_params.num_shares(),
532                new_kp_pgp_certs.len()
533            )));
534        }
535        ensure_unique_pgp_cert_fingerprints(&new_kp_pgp_certs)?;
536        // Sort to a canonical order so the serialized state's digest (which all
537        // T old KPs must agree on) is independent of submission order.
538        let mut new_kp_pgp_certs = new_kp_pgp_certs;
539        new_kp_pgp_certs.sort();
540        Ok(Self {
541            new_kp_pgp_certs,
542            new_params,
543        })
544    }
545
546    pub fn new_kp_pgp_certs(&self) -> &[PgpPublicCert] {
547        &self.new_kp_pgp_certs
548    }
549
550    pub fn new_params(&self) -> &SecretSharingParams {
551        &self.new_params
552    }
553
554    pub fn into_parts(self) -> (Vec<PgpPublicCert>, SecretSharingParams) {
555        (self.new_kp_pgp_certs, self.new_params)
556    }
557
558    pub fn digest(&self) -> [u8; 32] {
559        let bytes = bcs::to_bytes(self).expect("serialization should work");
560        Blake2b::<U32>::digest(bytes).into()
561    }
562}

640impl GuardianInfo {
641    /// Destructure an operator-initialized `GuardianInfo`, erroring if any field
642    /// set at operator-init is absent. `enclave_btc_pubkey` stays `Option` — it
643    /// is only set later, at `provisioner_init`.
644    // Returns every info field by design; a tuple alias wouldn't read any clearer.
645    #[allow(clippy::type_complexity)]
646    pub fn into_parts(
647        self,
648    ) -> GuardianResult<(
649        SecretSharingInstance,
650        S3BucketInfo,
651        EncPubKeyBytes,
652        [u8; 32],
653        GitRevision,
654        Option<BitcoinPubkey>,
655        LimiterState,
656        LimiterConfig,
657        u64,
658        HashiMasterG,
659    )> {
660        Ok((
661            self.secret_sharing_instance
662                .ok_or(InvalidInputs("missing secret sharing instance".into()))?,
663            self.bucket_info
664                .ok_or(InvalidInputs("missing bucket info".into()))?,
665            self.encryption_pubkey,
666            self.state_hash
667                .ok_or(InvalidInputs("missing state hash".into()))?,
668            self.untrusted_git_revision,
669            self.enclave_btc_pubkey,
670            self.limiter_state
671                .ok_or(InvalidInputs("missing limiter state".into()))?,
672            self.limiter_config
673                .ok_or(InvalidInputs("missing limiter config".into()))?,
674            self.current_committee_epoch
675                .ok_or(InvalidInputs("missing current committee epoch".into()))?,
676            self.mpc_master_g
677                .ok_or(InvalidInputs("missing mpc master g".into()))?,
678        ))
679    }
680}

682impl GetGuardianInfoResponse {
683    pub fn new(
684        attestation: NitroAttestation,
685        signing_pub_key: GuardianPubKey,
686        signed_info: GuardianSigned<GuardianInfo>,
687        encrypted_shares: KPEncryptedShares,
688    ) -> Self {
689        Self {
690            attestation,
691            signing_pub_key,
692            signed_info,
693            encrypted_shares,
694        }
695    }
696
697    /// Verify the response end to end and return the trusted guardian payload:
698    /// the attestation anchors `signing_pub_key`, `signed_info` must be signed by
699    /// it, the signed git revision must match `expected_build`, and PCR0 must
700    /// match `expected_build`.
701    pub fn verify(&self, expected_build: &BuildPcrs) -> GuardianResult<VerifiedGuardianInfo> {
702        let info = self.signed_info.clone().verify(&self.signing_pub_key)?;
703        if info.untrusted_git_revision != expected_build.git_revision() {
704            return Err(InvalidInputs(format!(
705                "guardian info reports build '{}', expected current build '{}'",
706                info.untrusted_git_revision,
707                expected_build.git_revision()
708            )));
709        }
710        self.attestation
711            .verify(&self.signing_pub_key, expected_build)?;
712        Ok(VerifiedGuardianInfo {
713            info,
714            signing_pub_key: self.signing_pub_key,
715            session_id: session_id_from_signing_pubkey(&self.signing_pub_key),
716            encrypted_shares: self.encrypted_shares.clone(),
717        })
718    }
719
720    /// Verify only the signed `GuardianInfo` payload.
721    ///
722    /// This does not authenticate the enclave image, the Nitro attestation, or
723    /// PCRs. Prefer [`Self::verify`] whenever the caller has PCR config.
724    pub fn verify_signed_info_without_attestation(&self) -> GuardianResult<VerifiedGuardianInfo> {
725        let info = self.signed_info.clone().verify(&self.signing_pub_key)?;
726        Ok(VerifiedGuardianInfo {
727            info,
728            signing_pub_key: self.signing_pub_key,
729            session_id: session_id_from_signing_pubkey(&self.signing_pub_key),
730            encrypted_shares: self.encrypted_shares.clone(),
731        })
732    }
733
734    /// Extract the guardian's self-reported `GuardianInfo` WITHOUT verifying the
735    /// signature or attestation. The node uses this after authenticating the
736    /// guardian over TLS; withdrawals are gated by the on-chain BTC key, and the
737    /// ed25519 signing key is verified only by KPs/monitors on the S3 audit logs.
738    pub fn into_info_unchecked(self) -> GuardianInfo {
739        self.signed_info.into_data_unchecked()
740    }
741}

771/// Converter from T -> Self that internally validates addresses
772pub trait AddressValidation<T>: Sized {
773    fn validate_addr(value: T, network: Network) -> GuardianResult<Self>;
774}
775
776impl AddressValidation<SignedStandardWithdrawalRequestWire>
777    for HashiSigned<StandardWithdrawalRequest>
778{
779    fn validate_addr(
780        wire_value: SignedStandardWithdrawalRequestWire,
781        network: Network,
782    ) -> GuardianResult<Self> {
783        HashiSigned::<StandardWithdrawalRequest>::new(
784            wire_value.signature.epoch,
785            StandardWithdrawalRequest::validate_addr(wire_value.data, network)?,
786            &wire_value.signature.signature,
787            &wire_value.signature.signers_bitmap,
788        )
789        .map_err(|e| InvalidInputs(format!("{:?}", e)))
790    }
791}
792
793impl AddressValidation<StandardWithdrawalRequestWire> for StandardWithdrawalRequest {
794    fn validate_addr(
795        value: StandardWithdrawalRequestWire,
796        network: Network,
797    ) -> GuardianResult<Self> {
798        Ok(Self {
799            wid: value.wid,
800            utxos: TxUTXOs::new(value.utxos.inputs, value.utxos.outputs, network)
801                .map_err(|e| InvalidInputs(e.to_string()))?,
802            timestamp_secs: value.timestamp_secs,
803            seq: value.seq,
804        })
805    }
806}
