4//! The guardian-signed envelope and its intent-based domain separation.
5//!
6//! `IntentType` is a registry: each signable type maps to exactly one intent
7//! value, and `GuardianSigned::{new,verify}` mix that value into the signed
8//! bytes so a signature over one type can never be replayed as another.
9

24/// All possible signing intent types.
25/// Using an enum ensures no two types can accidentally share the same intent value.
26#[repr(u8)]
27#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
28pub enum IntentType {
29    /// Intent for all LogMessage's
30    LogMessage = 0,
31    /// Intent for SetupNewKeyResponse
32    SetupNewKeyResponse = 1,
33    /// Intent for StandardWithdrawalResponse
34    StandardWithdrawalResponse = 2,
35    /// Intent for GuardianInfo
36    GuardianInfo = 3,
37    /// Intent for RotateKpsResponse
38    RotateKpsResponse = 4,
39}
40
41/// Trait for types that can be signed, providing domain separation via an intent.
42pub trait SigningIntent {
43    const INTENT: IntentType;
44}
45
46impl SigningIntent for LogMessage {
47    const INTENT: IntentType = IntentType::LogMessage;
48}
49
50impl SigningIntent for SetupNewKeyResponse {
51    const INTENT: IntentType = IntentType::SetupNewKeyResponse;
52}
53
54impl SigningIntent for StandardWithdrawalResponse {
55    const INTENT: IntentType = IntentType::StandardWithdrawalResponse;
56}
57
58impl SigningIntent for GuardianInfo {
59    const INTENT: IntentType = IntentType::GuardianInfo;
60}
61
62impl SigningIntent for RotateKpsResponse {
63    const INTENT: IntentType = IntentType::RotateKpsResponse;
64}

66/// Guardian-signed wrapper - adds timestamp and signature to any data
67#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]
68pub struct GuardianSigned<T> {
69    pub data: T,
70    /// Milliseconds since Unix epoch.
71    pub timestamp_ms: UnixMillis,
72    pub signature: GuardianSignature,
73}
74
75/// Methods for `Signed<T>` wrapper - signing and verification
76impl<T: Serialize + SigningIntent> GuardianSigned<T> {
77    /// Create a new signed payload (used by enclave)
78    /// Includes intent byte for domain separation to prevent cross-type signature attacks
79    pub fn new(data: T, signing_key: &SigningKey, timestamp_ms: UnixMillis) -> Self {
80        let tuple = (T::INTENT, &data, timestamp_ms);
81        let signing_payload = bcs::to_bytes(&tuple).expect("serialization should not fail");
82        let signature = signing_key.sign(&signing_payload);
83        Self {
84            data,
85            timestamp_ms,
86            signature,
87        }
88    }
89
90    /// Verify signature and extract payload
91    /// Checks intent byte to ensure signature is for the correct type
92    pub fn verify(self, pub_key: &VerificationKey) -> GuardianResult<T> {
93        let tuple = (T::INTENT, &self.data, self.timestamp_ms);
94        let msg_bytes = bcs::to_bytes(&tuple).expect("serialization should not fail");
95        pub_key
96            .verify(&self.signature, &msg_bytes)
97            .map_err(|_| InvalidInputs("signature invalid".into()))?;
98        Ok(self.data)
99    }
100}
101
102impl<T> GuardianSigned<T> {
103    /// Move out the payload WITHOUT verifying the signature. The node uses this
104    /// on guardian responses it has already authenticated over TLS; the ed25519
105    /// signing key is verified only by KPs/monitors on the S3 audit logs.
106    pub fn into_data_unchecked(self) -> T {
107        self.data
108    }
109}
