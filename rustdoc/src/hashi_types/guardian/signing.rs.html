4//! The guardian-signed envelope and its intent-based domain separation.
5//!
6//! `IntentType` is a registry: each signable type maps to exactly one intent
7//! value, and `GuardianSigned::{new,verify}` mix that value into the signed
8//! bytes so a signature over one type can never be replayed as another.

32/// All possible signing intent types.
33/// Using an enum ensures no two types can accidentally share the same intent value.
34#[repr(u8)]
35#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
36pub enum IntentType {
37    /// Intent for key-bound guardian log records.
38    LogMessage = 0,
39    /// Intent for SetupNewKeyResponse
40    SetupNewKeyResponse = 1,
41    /// Intent for StandardWithdrawalResponse
42    StandardWithdrawalResponse = 2,
43    /// Intent for GuardianInfo
44    GuardianInfo = 3,
45    /// Intent for RotateKpsResponse
46    RotateKpsResponse = 4,
47    /// Intent for ProvisionerRotateCertResponse
48    ProvisionerRotateCertResponse = 5,
49}
50
51/// Trait for types that can be signed, providing domain separation via an intent.
52pub trait SigningIntent {
53    const INTENT: IntentType;
54
55    /// Canonical bytes covered by this guardian signature. Most payloads use
56    /// the standard `(intent, data, timestamp)` layout; routing-sensitive
57    /// payloads may override this while retaining the central intent registry.
58    fn signing_bytes(&self, timestamp_ms: UnixMillis) -> Vec<u8>
59    where
60        Self: Serialize,
61    {
62        bcs::to_bytes(&(Self::INTENT, self, timestamp_ms)).expect("serialization should not fail")
63    }
64}
65
66pub(crate) fn sign_intent<T: Serialize + SigningIntent>(
67    data: &T,
68    timestamp_ms: UnixMillis,
69    signing_key: &SigningKey,
70) -> GuardianSignature {
71    signing_key.sign(&data.signing_bytes(timestamp_ms))
72}
73
74pub(crate) fn verify_intent<T: Serialize + SigningIntent>(
75    data: &T,
76    timestamp_ms: UnixMillis,
77    signature: &GuardianSignature,
78    pub_key: &VerificationKey,
79) -> GuardianResult<()> {
80    pub_key
81        .verify(signature, &data.signing_bytes(timestamp_ms))
82        .map_err(|_| InvalidInputs("signature invalid".into()))
83}

85/// All possible KP signing intent types.
86///
87/// These signatures are detached OpenPGP signatures produced by KPs, not
88/// enclave ed25519 signatures. Each KP-submitted request type gets a stable
89/// intent so a signature for one request cannot be replayed as another request
90/// with the same BCS shape.
91#[repr(u8)]
92#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
93pub enum KpSigningIntentType {
94    /// One KP's share submission to the provisioning relay.
95    ProvisionerInitRelaySubmission = 0,
96    /// One KP's request to replace one certificate wrapping its committed share.
97    ProvisionerRotateCertRequest = 1,
98}
99
100/// Trait for KP-submitted request payloads that need detached signatures.
101pub trait KpSigningIntent {
102    const INTENT: KpSigningIntentType;
103}

133/// KP-signed wrapper - adds signer cert and detached OpenPGP signature to any
134/// KP-submitted request payload.
135#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]
136pub struct KpSigned<T> {
137    pub data: T,
138    pub signer_cert: PgpPublicCert,
139    pub signature: String,
140}
141
142/// Guardian-signed wrapper - adds timestamp and signature to any data
143#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]
144pub struct GuardianSigned<T> {
145    pub data: T,
146    /// Milliseconds since Unix epoch.
147    pub timestamp_ms: UnixMillis,
148    pub signature: GuardianSignature,
149}
150
151/// Methods for `Signed<T>` wrapper - signing and verification
152impl<T: Serialize + SigningIntent> GuardianSigned<T> {
153    /// Create a new signed payload (used by enclave)
154    /// Includes intent byte for domain separation to prevent cross-type signature attacks
155    pub fn new(data: T, signing_key: &SigningKey, timestamp_ms: UnixMillis) -> Self {
156        let signature = sign_intent(&data, timestamp_ms, signing_key);
157        Self {
158            data,
159            timestamp_ms,
160            signature,
161        }
162    }
163
164    /// Verify signature and extract payload
165    /// Checks intent byte to ensure signature is for the correct type
166    pub fn verify(self, pub_key: &VerificationKey) -> GuardianResult<T> {
167        verify_intent(&self.data, self.timestamp_ms, &self.signature, pub_key)?;
168        Ok(self.data)
169    }
170}
171
172impl<T: Serialize + KpSigningIntent> KpSigned<T> {
173    /// Create a new signed KP payload by invoking `gpg --detach-sign` for the
174    /// signer certificate's fingerprint. Includes the KP intent in the signed
175    /// bytes; payload types carry any request-specific replay-binding fields.
176    pub fn new(
177        data: T,
178        signer_cert: PgpPublicCert,
179        gpg_home: Option<&Path>,
180    ) -> GuardianResult<Self> {
181        let signing_payload = Self::signed_bytes(&data);
182        let signature =
183            sign_detached_via_gpg(&signing_payload, &signer_cert.fingerprint(), gpg_home)
184                .map_err(|e| InternalError(format!("KP signing failed: {e}")))?;
185        Ok(Self {
186            data,
187            signer_cert,
188            signature,
189        })
190    }
191
192    /// The exact bytes a key provisioner detached-signs for a typed guardian
193    /// request. Binds the request intent and request payload.
194    pub fn signed_bytes(data: &T) -> Vec<u8> {
195        let tuple = (T::INTENT, data);
196        bcs::to_bytes(&tuple).expect("serialization should not fail")
197    }
198
199    /// Verify the signature without consuming the signed request.
200    /// Checks the intent byte to ensure the signature is for this request type.
201    pub fn verify_signature(&self) -> GuardianResult<()> {
202        let msg_bytes = Self::signed_bytes(&self.data);
203        verify_detached_signature(&msg_bytes, &self.signature, &self.signer_cert)
204            .map_err(|e| InvalidInputs(format!("KP signature verification failed: {e}")))?;
205        Ok(())
206    }
207
208    /// Verify the signature and extract the payload.
209    pub fn verify(self) -> GuardianResult<T> {
210        self.verify_signature()?;
211        Ok(self.data)
212    }
213
214    pub fn signer_fingerprint(&self) -> Fingerprint {
215        self.signer_cert.fingerprint()
216    }
217}

219impl<T> GuardianSigned<T> {
220    /// Move out the payload WITHOUT verifying the signature. The node uses this
221    /// on guardian responses it has already authenticated over TLS; the ed25519
222    /// signing key is verified only by KPs/monitors on the S3 audit logs.
223    pub fn into_data_unchecked(self) -> T {
224        self.data
225    }
