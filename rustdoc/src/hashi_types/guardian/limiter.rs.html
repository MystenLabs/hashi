4use super::GuardianError::InvalidInputs;
5use super::GuardianError::RateLimitExceeded;
6use super::GuardianResult;
7use serde::Deserialize;
8use serde::Serialize;

10/// Immutable configuration for the token bucket rate limiter.
11#[derive(Debug, Copy, Clone, PartialEq, Serialize, Deserialize)]
12pub struct LimiterConfig {
13    /// Refill rate in sats per second.
14    pub refill_rate: u64,
15    /// Maximum bucket capacity in sats.
16    pub max_bucket_capacity: u64,
17}
18
19/// Serializable state for the token bucket rate limiter.
20/// Provisioners provide this when initializing the enclave; it is also embedded
21/// in each successful withdrawal log so the next KP can recover it.
22#[derive(Debug, Copy, Clone, PartialEq, Serialize, Deserialize)]
23pub struct LimiterState {
24    /// Available tokens in sats.
25    pub num_tokens_available: u64,
26    /// Last refill timestamp in unix seconds.
27    pub last_updated_at: u64,
28    /// Next expected withdrawal sequence number.
29    pub next_seq: u64,
30}
31
32impl LimiterState {
33    /// Genesis state for a freshly bootstrapped enclave: a full bucket, no
34    /// consumes yet, no refill timestamp.
35    pub fn genesis(config: &super::LimiterConfig) -> Self {
36        Self {
37            num_tokens_available: config.max_bucket_capacity,
38            last_updated_at: 0,
39            next_seq: 0,
40        }
41    }
42}
43
44/// Token bucket rate limiter. Tokens refill linearly over time.
45///
46/// Pure data structure — concurrency is handled by the caller via a Mutex.
47pub struct RateLimiter {
48    config: LimiterConfig,
49    state: LimiterState,
50}
51
52impl RateLimiter {
53    pub fn new(config: LimiterConfig, state: LimiterState) -> GuardianResult<Self> {
54        if state.num_tokens_available > config.max_bucket_capacity {
55            return Err(InvalidInputs(
56                "num_tokens_available exceeds max_bucket_capacity".into(),
57            ));
58        }
59        Ok(Self { config, state })
60    }
61
62    pub fn config(&self) -> &LimiterConfig {
63        &self.config
64    }
65
66    pub fn state(&self) -> &LimiterState {
67        &self.state
68    }
69
70    pub fn next_seq(&self) -> u64 {
71        self.state.next_seq
72    }
73
74    /// Consume tokens from the bucket. Validates seq and timestamp ordering,
75    /// refills based on elapsed time, then debits the requested amount.
76    pub fn consume(&mut self, seq: u64, timestamp: u64, amount_sats: u64) -> GuardianResult<()> {
77        if seq != self.state.next_seq {
78            return Err(InvalidInputs(format!(
79                "seq mismatch: expected {}, got {}",
80                self.state.next_seq, seq
81            )));
82        }
83        if timestamp < self.state.last_updated_at {
84            return Err(InvalidInputs(format!(
85                "timestamp {} < last_updated_at {}",
86                timestamp, self.state.last_updated_at
87            )));
88        }
89
90        // Refill tokens based on elapsed time.
91        let elapsed = timestamp
92            .checked_sub(self.state.last_updated_at)
93            .expect("timestamp checked above");
94        let refilled = elapsed.saturating_mul(self.config.refill_rate);
95        let capacity = self
96            .state
97            .num_tokens_available
98            .saturating_add(refilled)
99            .min(self.config.max_bucket_capacity);
100
101        if capacity < amount_sats {
102            return Err(RateLimitExceeded);
103        }
104
105        self.state.last_updated_at = timestamp;
106        self.state.num_tokens_available = capacity - amount_sats;
107        self.state.next_seq += 1;
108        Ok(())
109    }
110}

112#[cfg(test)]
113mod test {
114    use super::*;
115
116    fn make_limiter() -> (LimiterConfig, LimiterState) {
117        let config = LimiterConfig {
118            refill_rate: 1_000,
119            max_bucket_capacity: 2_000_000,
120        };
121        let state = LimiterState {
122            num_tokens_available: 0,
123            last_updated_at: 0,
124            next_seq: 0,
125        };
126
127        (config, state)
128    }
129
130    #[test]
131    fn test_basic() {
132        let (config, state) = make_limiter();
133        let mut limiter = RateLimiter::new(config, state).unwrap();
134        assert!(limiter.consume(0, 1, config.refill_rate).is_ok());
135
136        let target_amount = 1_000_000u64;
137        let num_secs_required = target_amount.div_ceil(config.refill_rate);
138        assert!(
139            limiter
140                .consume(1, num_secs_required, target_amount)
141                .is_err()
142        );
143        assert!(
144            limiter
145                .consume(1, 1 + num_secs_required, target_amount)
146                .is_ok()
147        );
148    }
149
150    #[test]
151    fn test_limits() {
152        let (config, state) = make_limiter();
153        let mut limiter = RateLimiter::new(config, state).unwrap();
154        assert!(
155            limiter
156                .consume(0, u64::MAX, config.max_bucket_capacity + 1)
157                .is_err()
158        );
159        assert!(
160            limiter
161                .consume(0, u64::MAX, config.max_bucket_capacity)
162                .is_ok()
163        );
164    }
165
166    #[test]
167    fn test_rejects_wrong_seq_and_old_timestamp() {
168        let (config, state) = make_limiter();
169        let mut limiter = RateLimiter::new(config, state).unwrap();
170        // Wrong seq.
171        assert!(limiter.consume(1, 0, 0).is_err());
172        // Advance state.
173        limiter.consume(0, 100, 1_000).unwrap();
174        // Old timestamp.
175        assert!(limiter.consume(1, 50, 1_000).is_err());
176    }
177}
