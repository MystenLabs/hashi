16/// Rate limiter state: Amount withdrawn in the last N epochs.
17/// RateLimiterState and CommitteeStore have the same window & entries size. This is enforced in
18/// ProvisionerInitState::new() and Enclave::register_new_epoch().
19#[derive(Debug, Clone, PartialEq, Serialize)]
20pub struct RateLimiter {
21    state: ConsecutiveEpochStore<Amount>,
22    max_withdrawable_per_epoch: Amount,
23}
24
25#[derive(Debug, Clone, PartialEq, Serialize)]
26pub struct WithdrawalState {
27    limiter: RateLimiter,
28}

30/// A store for last N committees. Needs to be initialized with at least one committee.
31#[derive(Debug, Clone, PartialEq)]
32pub struct CommitteeStore(ConsecutiveEpochStore<HashiCommittee>);
33
34impl WithdrawalState {
35    pub fn new(limiter: RateLimiter) -> Self {
36        Self { limiter }
37    }
38
39    pub fn rate_limiter(&self) -> &RateLimiter {
40        &self.limiter
41    }
42
43    /// Consume amount units from the given epoch's rate limit
44    pub fn consume_from_limiter(&mut self, epoch: u64, amount: Amount) -> GuardianResult<()> {
45        self.limiter.consume(epoch, amount)
46    }
47
48    /// Adds a new epoch and prunes an old epoch
49    pub fn add_epoch_to_limiter(&mut self, epoch: u64) -> GuardianResult<()> {
50        self.limiter.add_epoch(epoch)
51    }
52
53    /// Reverse of consume_from_limiter
54    pub fn revert_limiter(&mut self, epoch: u64, amount: Amount) -> GuardianResult<()> {
55        self.limiter.revert(epoch, amount)
56    }
57}

59impl RateLimiter {
60    pub fn new(
61        epoch_window: EpochWindow,
62        amounts_withdrawn_per_epoch: Vec<Amount>,
63        max_withdrawable_per_epoch: Amount,
64    ) -> GuardianResult<Self> {
65        // Note: instead of erring out, we could use zero as default
66        if amounts_withdrawn_per_epoch.is_empty() {
67            return Err(InvalidInputs("amounts empty".into()));
68        }
69        Ok(Self {
70            state: ConsecutiveEpochStore::<Amount>::new(epoch_window, amounts_withdrawn_per_epoch)?,
71            max_withdrawable_per_epoch,
72        })
73    }
74
75    pub fn max_withdrawable_per_epoch(&self) -> Amount {
76        self.max_withdrawable_per_epoch
77    }
78
79    pub fn state(&self) -> &ConsecutiveEpochStore<Amount> {
80        &self.state
81    }
82
83    pub fn epoch_window(&self) -> EpochWindow {
84        self.state.epoch_window()
85    }
86
87    pub fn num_entries(&self) -> usize {
88        self.state.num_entries()
89    }
90
91    /// Consume amount units from the given epoch's rate limit.
92    /// Stored values are the amount withdrawn so far in that epoch.
93    pub fn consume(&mut self, epoch: u64, amount: Amount) -> GuardianResult<()> {
94        let cur_sum = *self.state.get_checked(epoch)?;
95
96        let new_sum = cur_sum
97            .checked_add(amount)
98            .ok_or(InvalidInputs("Overflow when computing sum".into()))?;
99
100        if new_sum > self.max_withdrawable_per_epoch {
101            return Err(RateLimitExceeded);
102        }
103
104        *self.state.get_mut_checked(epoch)? = new_sum;
105        Ok(())
106    }
107
108    /// Add back consumed units to the limiter
109    pub fn revert(&mut self, epoch: u64, amount: Amount) -> GuardianResult<()> {
110        let cur_sum = *self.state.get_checked(epoch)?;
111
112        debug_assert!(cur_sum >= amount);
113        let new_sum = cur_sum
114            .checked_sub(amount)
115            .ok_or(InternalError("Underflow when computing sub".into()))?; // this should be unreachable
116
117        *self.state.get_mut_checked(epoch)? = new_sum;
118        Ok(())
119    }
120
121    /// Adds a new epoch (must be the next consecutive epoch). Old epochs are pruned automatically.
122    pub fn add_epoch(&mut self, epoch: u64) -> GuardianResult<()> {
123        self.state.insert(epoch, Amount::from_sat(0))?;
124        Ok(())
125    }
126}

128impl CommitteeStore {
129    pub fn new(epoch_window: EpochWindow, committees: Vec<HashiCommittee>) -> GuardianResult<Self> {
130        // Err if input has no committees
131        if committees.is_empty() {
132            return Err(InvalidInputs("No committee set".into()));
133        }
134
135        // Match window.epoch() against committee.epoch()
136        let mut base_epoch = epoch_window.first_epoch;
137        for committee in &committees {
138            if committee.epoch() != base_epoch {
139                return Err(InvalidInputs("epoch doesn't match".into()));
140            }
141            base_epoch += 1;
142        }
143
144        Ok(Self(ConsecutiveEpochStore::<HashiCommittee>::new(
145            epoch_window,
146            committees,
147        )?))
148    }
149
150    pub fn num_entries(&self) -> usize {
151        self.0.num_entries()
152    }
153
154    pub fn epoch_window(&self) -> EpochWindow {
155        self.0.epoch_window()
156    }
157
158    pub fn iter(&self) -> impl Iterator<Item = (u64, &HashiCommittee)> {
159        self.0.iter()
160    }
161
162    pub fn into_owned_iter(self) -> impl Iterator<Item = (u64, HashiCommittee)> {
163        self.0.into_owned_iter()
164    }
165}

167#[derive(Serialize, Deserialize)]
168pub struct CommitteeStoreRepr(pub ConsecutiveEpochStoreRepr<crate::move_types::Committee>);
169
170impl CommitteeStoreRepr {
171    pub fn epoch_window(&self) -> EpochWindow {
172        self.0.window
173    }
174}
175
176impl From<CommitteeStore> for CommitteeStoreRepr {
177    fn from(store: CommitteeStore) -> Self {
178        CommitteeStoreRepr(ConsecutiveEpochStoreRepr::<crate::move_types::Committee> {
179            window: store.0.epoch_window(),
180            entries: store.0.iter().map(|(_, c)| c.into()).collect(),
181        })
182    }
183}
184
185impl TryFrom<CommitteeStoreRepr> for CommitteeStore {
186    type Error = anyhow::Error;
187
188    fn try_from(value: CommitteeStoreRepr) -> Result<Self, Self::Error> {
189        let window = value.0.window;
190        let committees = value
191            .0
192            .entries
193            .into_iter()
194            .map(crate::committee::Committee::try_from)
195            .collect::<Result<Vec<_>, _>>()?;
196        CommitteeStore::new(window, committees)
197            .map_err(|e| anyhow::anyhow!("invalid committee store repr: {e}"))
198    }
199}

201#[cfg(test)]
202mod tests {
203    use super::super::epoch_store::EpochWindow;
204    use super::*;
205    use std::num::NonZeroU16;
206
207    fn nz(v: u16) -> NonZeroU16 {
208        NonZeroU16::new(v).expect("non-zero")
209    }
210
211    #[test]
212    fn rate_limiter_consume_tests() {
213        let window = EpochWindow::new(0, nz(1));
214        let mut limiter =
215            RateLimiter::new(window, vec![Amount::from_sat(0)], Amount::from_sat(100)).unwrap();
216
217        limiter.consume(0, Amount::from_sat(60)).unwrap();
218        let err = limiter.consume(0, Amount::from_sat(50)).unwrap_err();
219        assert_eq!(err, RateLimitExceeded);
220
221        // Only epoch 0 is present.
222        let err = limiter.consume(1, Amount::from_sat(1)).unwrap_err();
223        assert!(matches!(err, InvalidInputs(_)));
224    }
225
226    #[test]
227    fn rate_limiter_revert_tests() {
228        let window = EpochWindow::new(0, nz(1));
229        let mut limiter =
230            RateLimiter::new(window, vec![Amount::from_sat(0)], Amount::from_sat(100)).unwrap();
231
232        limiter.consume(0, Amount::from_sat(60)).unwrap();
233        limiter.revert(0, Amount::from_sat(60)).unwrap();
234
235        // If revert worked, we should be able to consume the full max again.
236        limiter.consume(0, Amount::from_sat(100)).unwrap();
237        let err = limiter.consume(0, Amount::from_sat(1)).unwrap_err();
238        assert_eq!(err, RateLimitExceeded);
239    }
240
241    #[test]
242    fn rate_limiter_add_epoch_tests() {
243        let window = EpochWindow::new(0, nz(2));
244        let mut limiter = RateLimiter::new(
245            window,
246            vec![Amount::from_sat(0), Amount::from_sat(0)],
247            Amount::from_sat(100),
248        )
249        .unwrap();
250
251        assert_eq!(limiter.state.next_epoch_to_be_inserted(), 2);
252        limiter.add_epoch(2).unwrap(); // prunes epoch 0
253        assert_eq!(limiter.state.next_epoch_to_be_inserted(), 3);
254
255        // Epoch 0 is now too old.
256        let err = limiter.consume(0, Amount::from_sat(1)).unwrap_err();
257        assert!(matches!(err, InvalidInputs(_)));
258
259        // Epoch 1 and 2 are present.
260        limiter.consume(1, Amount::from_sat(70)).unwrap();
261        limiter.consume(2, Amount::from_sat(70)).unwrap();
262
263        // Non-consecutive epoch insert should fail.
264        let err = limiter.add_epoch(4).unwrap_err();
265        assert!(matches!(err, InvalidInputs(_)));
266    }
