4//! The versioned `LogMessage` family the enclave emits and its per-message S3 object-key
5//! rules. The `LogRecord` wrapper that carries these to S3 lives in
6//! `super::envelope`.
7//!
8//! Every message type exposes `object_key_pattern()`.
9//! Types with deterministic keys also expose `object_key()`, which returns the
10//! complete bucket-relative key. Types supporting batch reads may additionally
11//! expose `object_key_dir()`, a slash-terminated S3 key prefix.
12//!
13//! Writers call `LogRecord::new()`, which finalizes the pattern exactly once,
14//! stores the resulting key, and uses it for signing and upload.
15//! Readers either fetch a deterministic record using `object_key()` or list
16//! records in `object_key_dir()`. In both read paths, the S3 client rejects a
17//! record unless its signed key matches the actual key returned by S3.

49/// The wire message stored in a [`super::LogRecord`]. Its version is serialized
50/// as the record's sibling `schema_version` field rather than as an additional
51/// JSON enum layer.
52#[derive(Debug)]
53pub enum VersionedLogMessage {
54    V1(LogMessageV1),
55    V2(LogMessageV2),
56}
57
58impl From<LogMessageV1> for VersionedLogMessage {
59    fn from(message: LogMessageV1) -> Self {
60        Self::V1(message)
61    }
62}
63
64impl From<LogMessageV2> for VersionedLogMessage {
65    fn from(message: LogMessageV2) -> Self {
66        Self::V2(message)
67    }
68}
69
70impl Serialize for VersionedLogMessage {
71    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
72    where
73        S: serde::Serializer,
74    {
75        match self {
76            Self::V1(message) => message.serialize(serializer),
77            Self::V2(message) => message.serialize(serializer),
78        }
79    }
80}

139impl ObjectKeyPattern {
140    /// Finalizes the pattern into the complete S3 object key.
141    pub(super) fn finalize(self) -> String {
142        match self {
143            Self::Fixed(key) => key,
144            Self::RandomSuffix(prefix) => {
145                format!("{prefix}{:08x}.json", rand::random::<u32>())
146            }
147        }
148    }
149}

222#[derive(Debug, Serialize, Clone, PartialEq)]
223pub struct KPEncryptedSharesV1(Vec<KPEncryptedShareV1>);

225impl KPEncryptedSharesV1 {
226    fn new(mut shares: Vec<KPEncryptedShareV1>) -> Result<Self, GuardianError> {
227        if shares.len() > crate::guardian::crypto::MAX_NUM_SHARES {
228            return Err(GuardianError::InvalidInputs(format!(
229                "{} encrypted shares must be at most u16::MAX",
230                shares.len()
231            )));
232        }
233
234        shares.sort_by_key(|share| share.id);
235        let ids = shares
236            .iter()
237            .map(|share| share.id.get())
238            .collect::<Vec<_>>();
239        let expected = (1..=shares.len() as u16).collect::<Vec<_>>();
240        if ids != expected {
241            return Err(GuardianError::InvalidInputs(format!(
242                "encrypted share ids are not exactly 1..={}: got {ids:?}",
243                shares.len()
244            )));
245        }
246
247        Ok(Self(shares))
248    }
249}
250
251impl<'de> Deserialize<'de> for KPEncryptedSharesV1 {
252    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
253    where
254        D: serde::Deserializer<'de>,
255    {
256        let shares = Vec::<KPEncryptedShareV1>::deserialize(deserializer)?;
257        Self::new(shares).map_err(serde::de::Error::custom)
258    }
259}

325/// The authoritative secret-sharing instance, written to `ceremony/` after each
326/// ceremony. Carries the commitments + n/t/seq; encrypted KP shares live in
327/// `kp-shares/`. A rotation records the `old_instance` it consumed so the chain
328/// is auditable from the log alone.
329#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
330pub enum CeremonyLogMessage {
331    /// Initial key setup (`setup_new_key`); `instance` has `sharing_seq` 0.
332    NewKey {
333        instance: SecretSharingInstance,
334        /// The x-only BTC master pubkey this ceremony produced; lets KPs and
335        /// monitors cross-check it against the on-chain `guardian_btc_public_key`.
336        btc_master_pubkey: BitcoinPubkey,
337    },
338    /// Key rotation (`rotate_kps`) from `old_instance` to `new_instance`.
339    Rotate {
340        old_instance: SecretSharingInstance,
341        new_instance: SecretSharingInstance,
342        /// See [`Self::NewKey`]; invariant across rotations (the same key is re-shared).
343        btc_master_pubkey: BitcoinPubkey,
344    },
345}
346
347impl CeremonyLogMessage {
348    /// Consume the ceremony result. `NewKey` yields its initial instance;
349    /// `Rotate` yields the new instance after verifying that it advances exactly
350    /// one `sharing_seq` from the consumed instance.
351    pub fn into_instance_and_pubkey(self) -> (SecretSharingInstance, BitcoinPubkey) {
352        match self {
353            Self::NewKey {
354                instance,
355                btc_master_pubkey,
356            } => (instance, btc_master_pubkey),
357            Self::Rotate {
358                old_instance,
359                new_instance,
360                btc_master_pubkey,
361            } => {
362                let expected = old_instance
363                    .sharing_seq()
364                    .checked_add(1)
365                    .expect("Rotate old sharing_seq must not be u64::MAX");
366                assert_eq!(
367                    new_instance.sharing_seq(),
368                    expected,
369                    "Rotate must advance sharing_seq by exactly one"
370                );
371                (new_instance, btc_master_pubkey)
372            }
373        }
374    }
375
376    /// The resulting instance's `sharing_seq` — used as the `ceremony/` object key.
377    pub fn sharing_seq(&self) -> u64 {
378        match self {
379            CeremonyLogMessage::NewKey { instance, .. } => instance.sharing_seq(),
380            CeremonyLogMessage::Rotate { new_instance, .. } => new_instance.sharing_seq(),
381        }
382    }
383
384    pub fn object_key(&self, session_id: &str) -> String {
385        format!(
386            "{S3_DIR_CEREMONY}/{:020}-{session_id}.json",
387            self.sharing_seq(),
388        )
389    }
390
391    fn object_key_pattern(&self, session_id: &str) -> ObjectKeyPattern {
392        ObjectKeyPattern::Fixed(self.object_key(session_id))
393    }
394}

450pub enum CommitteeUpdateLogMessage {

470impl InitLogMessage {
471    pub const OI_ATTEST_UNSIGNED: &'static str = "01-oi-attestation-unsigned";
472    pub const OI_GUARDIAN_INFO: &'static str = "02-oi-guardian-info";

504impl WithdrawalLogMessage {
505    /// Success keys lead with `success-{seq:020}` so that lexicographic listing
506    /// within an hour bucket is also seq-sorted — the last key is the max-seq
507    /// log, which the KP reads to recover limiter state. Failures don't have a
508    /// meaningful seq (the request's seq may be stale), so they use a random
509    /// suffix for dedup.
510    fn object_key_pattern(&self, session_id: &str, timestamp_ms: UnixMillis) -> ObjectKeyPattern {
511        let directory =
512            S3HourScopedDirectory::new(S3_DIR_WITHDRAW, unix_millis_to_seconds(timestamp_ms));
513        match self {
514            Self::Success { request_data, .. } => ObjectKeyPattern::Fixed(format!(
515                "{directory}success-{:020}-{session_id}-wid{}.json",
516                request_data.seq, request_data.wid,
517            )),
518            Self::Failure { request_data, .. } => ObjectKeyPattern::RandomSuffix(format!(
519                "{directory}failure-{session_id}-wid{}-",
520                request_data.wid,
521            )),
522        }
523    }
524
525    pub fn wid(&self) -> WithdrawalID {
526        match self {
527            WithdrawalLogMessage::Success { request_data, .. } => request_data.wid,
528            WithdrawalLogMessage::Failure { request_data, .. } => request_data.wid,
529        }
530    }
531}

533impl CommitteeUpdateLogMessage {
534    /// Success keys lead with the new epoch (zero-padded) so a lex listing
535    /// is epoch-sorted; failures lead with `failure-` so they sort after
536    /// all successes, leaving the lex-last success key as the latest
537    /// successfully-applied epoch.
538    fn object_key_pattern(&self, session_id: &str) -> ObjectKeyPattern {
539        match self {
540            Self::Success { new_committee, .. } => ObjectKeyPattern::Fixed(format!(
541                "{S3_DIR_COMMITTEE_UPDATE}/{:020}-{session_id}.json",
542                new_committee.epoch,
543            )),
544            Self::Failure { new_committee, .. } => ObjectKeyPattern::RandomSuffix(format!(
545                "{S3_DIR_COMMITTEE_UPDATE}/failure-{:020}-{session_id}-",
546                new_committee.epoch,
547            )),
548        }
549    }
550}

552macro_rules! impl_log_message_schema {
553    ($schema:ty) => {
554        impl LogMessageSchema for $schema {
555            fn is_allowed_unsigned(&self) -> bool {
556                matches!(
557                    self,
558                    Self::Init(init_message)
559                        if matches!(
560                            **init_message,
561                            InitLogMessage::OIAttestationUnsigned { .. }
562                        )
563                )
564            }
565
566            fn log_type(&self) -> LogType {
567                match self {
568                    Self::Heartbeat(..) => LogType::Heartbeat,
569                    Self::Init(..) => LogType::Init,
570                    Self::Withdrawal(..) => LogType::Withdrawal,
571                    Self::Ceremony(..) => LogType::Ceremony,
572                    Self::KpShareState(..) => LogType::KpShareState,
573                    Self::CommitteeUpdate(..) => LogType::CommitteeUpdate,
574                    Self::Genesis(..) => LogType::Genesis,
575                }
576            }
577
578            fn object_key_pattern(
579                &self,
580                session_id: &str,
581                timestamp_ms: UnixMillis,
582            ) -> ObjectKeyPattern {
583                match self {
584                    Self::Heartbeat(message) => {
585                        message.object_key_pattern(session_id, timestamp_ms)
586                    }
587                    Self::Init(message) => message.object_key_pattern(session_id),
588                    Self::Withdrawal(message) => {
589                        message.object_key_pattern(session_id, timestamp_ms)
590                    }
591                    Self::Ceremony(message) => message.object_key_pattern(session_id),
592                    Self::KpShareState(message) => message.object_key_pattern(session_id),
593                    Self::CommitteeUpdate(message) => message.object_key_pattern(session_id),
594                    Self::Genesis(message) => message.object_key_pattern(),
595                }
596            }
597        }
598    };
599}

629impl VersionedLogMessage {

633    pub fn schema_version(&self) -> u64 {
634        match self {
635            Self::V1(_) => Self::SCHEMA_VERSION_V1,
636            Self::V2(_) => Self::SCHEMA_VERSION_V2,
637        }

640    pub fn is_allowed_unsigned(&self) -> bool {
