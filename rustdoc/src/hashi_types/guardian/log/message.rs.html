34/// All log messages emitted by the guardian enclave.
35/// Uses enum discriminator for automatic domain separation between variants.
36#[derive(Debug, Serialize, Deserialize)]
37pub enum LogMessage {
38    Heartbeat { seq: u64 },
39    Init(Box<InitLogMessage>),
40    Withdrawal(Box<WithdrawalLogMessage>),
41    Ceremony(Box<CeremonyLogMessage>),
42    Shares(Box<SharesLogMessage>),
43    CommitteeUpdate(Box<CommitteeUpdateLogMessage>),
44}
45
46/// Encrypted KP shares persisted for recovery, written to `shares/` after each
47/// ceremony. Keyed by `sharing_seq` so it pairs with the matching `ceremony/`
48/// instance; carries the ciphertexts the `ceremony/` log deliberately omits.
49#[derive(Debug, Serialize, Deserialize)]
50pub struct SharesLogMessage {
51    pub sharing_seq: u64,
52    pub encrypted_shares: KPEncryptedShares,
53}
54
55impl SharesLogMessage {
56    /// `shares/{sharing_seq:020}-{session_id}.json` — the object key a reader
57    /// fetches for `(session_id, sharing_seq)`.
58    pub fn object_key(session_id: &str, sharing_seq: u64) -> String {
59        super::seq_scoped_object_key(S3_DIR_SHARES, sharing_seq, session_id)
60    }
61}

63/// The authoritative share state, written to `ceremony/` after each ceremony.
64/// Carries the instance (commitments + n/t/seq) plus the recipient roster; the
65/// ciphertexts live in `shares/`. A rotation records the `old_instance` it
66/// consumed so the chain is auditable from the log alone.
67#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
68pub enum CeremonyLogMessage {
69    /// Initial key setup (`setup_new_key`); `instance` has `sharing_seq` 0.
70    NewKey {
71        instance: SecretSharingInstance,
72        /// Recipient fingerprints ordered by share id; lets a KP check the full
73        /// roster against the agreed set from the immutable log, not just the
74        /// operator-served shares.
75        roster: Vec<KPFingerprint>,
76    },
77    /// Key rotation (`rotate_kps`) from `old_instance` to `new_instance`.
78    Rotate {
79        old_instance: SecretSharingInstance,
80        new_instance: SecretSharingInstance,
81        /// Recipient fingerprints for `new_instance`; see [`Self::NewKey`].
82        roster: Vec<KPFingerprint>,
83    },
84}
85
86impl CeremonyLogMessage {
87    /// The resulting instance's `sharing_seq` — used as the `ceremony/` object key.
88    pub fn sharing_seq(&self) -> u64 {
89        match self {
90            CeremonyLogMessage::NewKey { instance, .. } => instance.sharing_seq(),
91            CeremonyLogMessage::Rotate { new_instance, .. } => new_instance.sharing_seq(),
92        }
93    }
94}

96/// OI: operator_init
97/// PI: provisioner_init
98/// Init messages are expected to be logged in the following order:
99/// OIAttestationUnsigned -> OIGuardianInfo -> PIEnclaveFullyInitialized.
100#[derive(Debug, Serialize, Deserialize)]
101pub enum InitLogMessage {
102    /// Attestation and signing public key posted in /operator_init
103    OIAttestationUnsigned {
104        attestation: NitroAttestation,
105        signing_public_key: GuardianPubKey,
106    },
107    /// Signed GuardianInfo logged in /operator_init (secret-sharing instance,
108    /// state_hash, encryption/BTC pubkeys). Boxed: much larger than the other
109    /// variants (`clippy::large_enum_variant`).
110    OIGuardianInfo(Box<GuardianInfo>),
111    /// Threshold reached — enclave fully initialized (happens once). Records the
112    /// ids of the shares that were combined.
113    PIEnclaveFullyInitialized { share_ids: Vec<ShareID> },
114}

116#[derive(Debug, Serialize, Deserialize)]
117pub enum WithdrawalLogMessage {
118    Success {
119        txid: Txid,
120        request_data: StandardWithdrawalRequestWire,
121        request_sign: CommitteeSignature,
122        response: StandardWithdrawalResponse,
123        /// Limiter state after this withdrawal was consumed. The KP rotating in
124        /// the next enclave reads the max-seq Success log and uses its
125        /// `post_state` as the new enclave's initial limiter state.
126        post_state: LimiterState,
127    },
128    Failure {
129        request_data: StandardWithdrawalRequestWire,
130        request_sign: CommitteeSignature,
131        error: GuardianError,
132    },
133}

135#[derive(Debug, Serialize, Deserialize)]
136pub enum CommitteeUpdateLogMessage {
137    /// `from_epoch` is the guardian's current epoch at the time; the
138    /// applied epoch is `new_committee.epoch`. Both are recorded because
139    /// hashi reconfig is sparse — `new_committee.epoch` is not
140    /// necessarily `from_epoch + 1`.
141    Success {
142        from_epoch: u64,
143        new_committee: crate::move_types::Committee,
144        request_sign: CommitteeSignature,
145    },
146    /// `from_epoch` is the guardian's current epoch at the time;
147    /// `new_committee` is what was proposed (and rejected).
148    Failure {
149        from_epoch: u64,
150        new_committee: crate::move_types::Committee,
151        request_sign: CommitteeSignature,
152        error: GuardianError,
153    },
154}

156impl InitLogMessage {
157    pub const OI_ATTEST_UNSIGNED: &'static str = "oi-attestation-unsigned";
158    pub const OI_GUARDIAN_INFO: &'static str = "oi-guardian-info";
159    pub const PI_FULLY_INITIALIZED: &'static str = "pi-enclave-fully-initialized";
160
161    pub fn log_name(&self, prefix: &str) -> String {
162        let suffix = match self {
163            InitLogMessage::OIAttestationUnsigned { .. } => Self::OI_ATTEST_UNSIGNED.to_string(),
164            InitLogMessage::OIGuardianInfo(_) => Self::OI_GUARDIAN_INFO.to_string(),
165            InitLogMessage::PIEnclaveFullyInitialized { .. } => {
166                Self::PI_FULLY_INITIALIZED.to_string()
167            }
168        };
169
170        format!("{}-{}.json", prefix, suffix)
171    }
172
173    pub fn attestation_object_key(session_id: &str) -> String {
174        format!(
175            "{}/{}-{}.json",
176            S3_DIR_INIT,
177            session_id,
178            Self::OI_ATTEST_UNSIGNED
179        )
180    }
181
182    pub fn guardian_info_object_key(session_id: &str) -> String {
183        format!(
184            "{}/{}-{}.json",
185            S3_DIR_INIT,
186            session_id,
187            Self::OI_GUARDIAN_INFO
188        )
189    }
190}

192impl WithdrawalLogMessage {
193    /// Success keys lead with `success-{seq:020}` so that lexicographic listing
194    /// within an hour bucket is also seq-sorted — the last key is the max-seq
195    /// log, which the KP reads to recover limiter state. Failures don't have a
196    /// meaningful seq (the request's seq may be stale), so they use a random
197    /// suffix for dedup.
198    pub fn log_name(&self, prefix: &str) -> String {
199        match self {
200            WithdrawalLogMessage::Success { request_data, .. } => format!(
201                "success-{:020}-{}-wid{}.json",
202                request_data.seq,
203                prefix,
204                self.wid()
205            ),
206            WithdrawalLogMessage::Failure { .. } => {
207                let random_suffix = rand::random::<u32>();
208                format!(
209                    "failure-{}-wid{}-{:08x}.json",
210                    prefix,
211                    self.wid(),
212                    random_suffix
213                )
214            }
215        }
216    }
217
218    pub fn wid(&self) -> WithdrawalID {
219        match self {
220            WithdrawalLogMessage::Success { request_data, .. } => request_data.wid,
221            WithdrawalLogMessage::Failure { request_data, .. } => request_data.wid,
222        }
223    }
224}
225
226impl CommitteeUpdateLogMessage {
227    /// Success keys lead with the new epoch (zero-padded) so a lex listing
228    /// is epoch-sorted; failures lead with `failure-` so they sort after
229    /// all successes, leaving the lex-last success key as the latest
230    /// successfully-applied epoch.
231    pub fn log_name(&self, prefix: &str) -> String {
232        match self {
233            CommitteeUpdateLogMessage::Success { new_committee, .. } => {
234                format!("{:020}-{}.json", new_committee.epoch, prefix)
235            }
236            CommitteeUpdateLogMessage::Failure { new_committee, .. } => {
237                let random_suffix = rand::random::<u32>();
238                format!(
239                    "failure-{:020}-{}-{:08x}.json",
240                    new_committee.epoch, prefix, random_suffix
241                )
242            }
243        }
244    }
245}

247impl LogMessage {
248    pub fn is_allowed_unsigned(&self) -> bool {
249        if let LogMessage::Init(init_message) = self {
250            matches!(**init_message, InitLogMessage::OIAttestationUnsigned { .. })
251        } else {
252            false
253        }
254    }
255
256    pub fn must_be_signed(&self) -> bool {
257        !self.is_allowed_unsigned()
258    }
259
260    /// The directory under which logs are written. Ends with a slash.
261    pub fn log_dir(&self, timestamp_ms: UnixMillis) -> String {
262        match self {
263            LogMessage::Init(_) => format!("{}/", S3_DIR_INIT),
264            LogMessage::Heartbeat { .. } => {
265                S3HourScopedDirectory::new(S3_DIR_HEARTBEAT, unix_millis_to_seconds(timestamp_ms))
266                    .to_string()
267            }
268            LogMessage::Withdrawal(..) => {
269                S3HourScopedDirectory::new(S3_DIR_WITHDRAW, unix_millis_to_seconds(timestamp_ms))
270                    .to_string()
271            }
272            LogMessage::Ceremony(..) => format!("{}/", S3_DIR_CEREMONY),
273            LogMessage::Shares(..) => format!("{}/", S3_DIR_SHARES),
274            LogMessage::CommitteeUpdate(..) => format!("{}/", S3_DIR_COMMITTEE_UPDATE),
275        }
276    }
277
278    /// The name of the log.
279    pub fn log_name(&self, prefix: &str) -> String {
280        match self {
281            LogMessage::Init(init_message) => init_message.log_name(prefix),
282            LogMessage::Heartbeat { seq } => format!("{}-{:020}.json", prefix, seq),
283            LogMessage::Withdrawal(withdrawal_message) => withdrawal_message.log_name(prefix),
284            LogMessage::Ceremony(ss) => format!("{:020}-{}.json", ss.sharing_seq(), prefix),
285            LogMessage::Shares(s) => format!("{:020}-{}.json", s.sharing_seq, prefix),
286            LogMessage::CommitteeUpdate(committee_message) => committee_message.log_name(prefix),
287        }
288    }
289
290    pub fn into_init_log(self) -> Option<InitLogMessage> {
291        match self {
292            LogMessage::Init(init_message) => Some(*init_message),
293            _ => None,
294        }
295    }
