4//! The `LogRecord` envelope written to S3: it wraps a `super::message::LogMessage`
5//! with the session id, timestamp, and (for signed logs) the guardian signature,
6//! and derives the object key + lock duration from the wrapped message.
7
8use super::S3_OBJECT_LOCK_DURATION_CEREMONY;
9use super::S3_OBJECT_LOCK_DURATION_COMMITTEE_UPDATE;
10use super::S3_OBJECT_LOCK_DURATION_HEARTBEAT;
11use super::S3_OBJECT_LOCK_DURATION_INIT;
12use super::S3_OBJECT_LOCK_DURATION_WITHDRAW;
13use super::message::LogMessage;
14use crate::guardian::GuardianError::InvalidInputs;
15use crate::guardian::GuardianPubKey;
16use crate::guardian::GuardianResult;
17use crate::guardian::GuardianSignKeyPair;
18use crate::guardian::GuardianSignature;
19use crate::guardian::GuardianSigned;
20use crate::guardian::SessionID;
21use crate::guardian::UnixMillis;
22use crate::guardian::now_timestamp_ms;
23use serde::Deserialize;
24use serde::Serialize;
25use std::time::Duration;

27/// Canonical log record written to S3.
28#[derive(Serialize, Deserialize, Debug)]
29pub struct LogRecord {
30    pub session_id: SessionID,
31    pub timestamp_ms: UnixMillis,
32    pub message: LogMessage,
33    /// Present for signed logs; omitted for unsigned logs (currently only OIAttestationUnsigned).
34    pub signature: Option<GuardianSignature>,
35}
36
37/// A verified log record where message authenticity has been checked.
38#[derive(Debug)]
39pub struct VerifiedLogRecord {
40    pub session_id: SessionID,
41    pub timestamp_ms: UnixMillis,
42    pub message: LogMessage,
43}
44
45impl LogRecord {
46    pub fn new(
47        session_id: SessionID,
48        message: LogMessage,
49        signing_key: &GuardianSignKeyPair,
50    ) -> Self {
51        let timestamp_ms = now_timestamp_ms();
52        if message.is_allowed_unsigned() {
53            Self::unsigned(session_id, message, timestamp_ms)
54        } else {
55            Self::signed(session_id, message, signing_key, timestamp_ms)
56        }
57    }
58
59    /// Full S3 object key for this log record (directory + file name). See the
60    /// `hashi-guardian` README for the canonical per-log-type key layout.
61    pub fn object_key(&self) -> String {
62        let dir = self.message.log_dir(self.timestamp_ms);
63        let log_name = self.message.log_name(&self.session_id);
64        format!("{}{}", dir, log_name)
65    }
66
67    pub fn object_lock_duration(&self) -> Duration {
68        match &self.message {
69            LogMessage::Init(..) => S3_OBJECT_LOCK_DURATION_INIT,
70            LogMessage::Heartbeat { .. } => S3_OBJECT_LOCK_DURATION_HEARTBEAT,
71            LogMessage::Withdrawal(..) => S3_OBJECT_LOCK_DURATION_WITHDRAW,
72            LogMessage::Ceremony(..) => S3_OBJECT_LOCK_DURATION_CEREMONY,
73            LogMessage::CommitteeUpdate(..) => S3_OBJECT_LOCK_DURATION_COMMITTEE_UPDATE,
74        }
75    }
76
77    fn signed(
78        session_id: SessionID,
79        message: LogMessage,
80        signing_key: &GuardianSignKeyPair,
81        timestamp_ms: UnixMillis,
82    ) -> Self {
83        let signed = GuardianSigned::new(message, signing_key, timestamp_ms);
84        Self {
85            session_id,
86            timestamp_ms: signed.timestamp_ms,
87            message: signed.data,
88            signature: Some(signed.signature),
89        }
90    }
91
92    fn unsigned(session_id: SessionID, message: LogMessage, timestamp_ms: UnixMillis) -> Self {
93        assert!(
94            message.is_allowed_unsigned(),
95            "message must be Init(OIAttestationUnsigned)"
96        );
97        Self {
98            session_id,
99            timestamp_ms,
100            message,
101            signature: None,
102        }
103    }
104
105    pub fn verify(self, pub_key: &GuardianPubKey) -> GuardianResult<VerifiedLogRecord> {
106        let session_id = self.session_id;
107        let timestamp_ms = self.timestamp_ms;
108        let message = self.message;
109
110        let message = if message.is_allowed_unsigned() {
111            message
112        } else {
113            let signature = self
114                .signature
115                .ok_or_else(|| InvalidInputs("missing log signature".into()))?;
116            GuardianSigned {
117                data: message,
118                timestamp_ms,
119                signature,
120            }
121            .verify(pub_key)?
122        };
123
124        Ok(VerifiedLogRecord {
125            session_id,
126            timestamp_ms,
127            message,
128        })
129    }
130
131    pub fn verify_unsigned(self) -> GuardianResult<VerifiedLogRecord> {
132        if !self.message.is_allowed_unsigned() {
133            return Err(InvalidInputs(
134                "expected unsigned log record but message requires a signature".into(),
135            ));
136        }
137        Ok(VerifiedLogRecord {
138            session_id: self.session_id,
139            timestamp_ms: self.timestamp_ms,
140            message: self.message,
141        })
142    }
143}

145#[cfg(test)]
146mod tests {
147    use super::*;
148    use crate::guardian::InitLogMessage;
149    use crate::guardian::LimiterState;
150    use crate::guardian::StandardWithdrawalRequest;
151    use crate::guardian::StandardWithdrawalRequestWire;
152    use crate::guardian::StandardWithdrawalResponse;
153    use crate::guardian::WithdrawalID;
154    use crate::guardian::WithdrawalLogMessage;
155    use bitcoin::Network;
156    use bitcoin::Txid;
157    use bitcoin::hashes::Hash as _;
158
159    fn set_timestamp(log: &mut LogRecord, timestamp_ms: UnixMillis) {
160        log.timestamp_ms = timestamp_ms;
161    }
162
163    #[test]
164    fn object_key_for_init_attestation_unsigned() {
165        let session_id = "session-a".to_string();
166        let signing_key = GuardianSignKeyPair::from([7u8; 32]);
167        let mut log = LogRecord::new(
168            session_id.clone(),
169            LogMessage::Init(Box::new(InitLogMessage::OIAttestationUnsigned {
170                attestation: vec![1, 2, 3],
171                signing_public_key: signing_key.verification_key(),
172            })),
173            &signing_key,
174        );
175        set_timestamp(&mut log, 1_700_000_000_000);
176
177        assert_eq!(
178            log.object_key(),
179            "init/session-a-oi-attestation-unsigned.json"
180        );
181    }
182
183    #[test]
184    fn object_key_for_heartbeat() {
185        let session_id = "session-b".to_string();
186        let signing_key = GuardianSignKeyPair::from([8u8; 32]);
187        let seq = 42_u64;
188        let timestamp_ms = 1_700_000_000_000;
189
190        let mut log = LogRecord::new(
191            session_id.clone(),
192            LogMessage::Heartbeat { seq },
193            &signing_key,
194        );
195        set_timestamp(&mut log, timestamp_ms);
196
197        assert_eq!(
198            log.object_key(),
199            "heartbeat/2023/11/14/22/session-b-00000000000000000042.json"
200        );
201    }
202
203    #[test]
204    fn object_key_for_withdrawal_success() {
205        let session_id = "session-c".to_string();
206        let signing_key = GuardianSignKeyPair::from([9u8; 32]);
207        let timestamp_ms = 1_700_000_000_000;
208        let wid = WithdrawalID::new([0xcd; 32]);
209        let signed_request =
210            StandardWithdrawalRequest::mock_signed_for_testing_with_wid(Network::Regtest, wid);
211        let (request_sign, request_data) = signed_request.into_parts();
212        let request_data: StandardWithdrawalRequestWire = request_data.into();
213        let seq = request_data.seq;
214
215        let mut log = LogRecord::new(
216            session_id.clone(),
217            LogMessage::Withdrawal(Box::new(WithdrawalLogMessage::Success {
218                txid: Txid::from_slice(&[3u8; 32]).expect("valid txid"),
219                request_data,
220                request_sign,
221                response: GuardianSigned::<StandardWithdrawalResponse>::mock_for_testing().data,
222                post_state: LimiterState {
223                    num_tokens_available: 0,
224                    last_updated_at: 0,
225                    next_seq: seq + 1,
226                },
227            })),
228            &signing_key,
229        );
230        set_timestamp(&mut log, timestamp_ms);
231
232        assert_eq!(
233            log.object_key(),
234            format!("withdraw/2023/11/14/22/success-{seq:020}-session-c-wid{wid}.json"),
235        );
236    }
