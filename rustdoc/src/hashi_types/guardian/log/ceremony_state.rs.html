4//! Combined ceremony and KP share state derived from guardian log messages.
5
6use super::message::CeremonyLogMessage;
7use super::message::KpShareStateLogMessage;
8use crate::bitcoin::BitcoinPubkey;
9use crate::guardian::GuardianError;
10use crate::guardian::GuardianResult;
11use crate::guardian::KPEncryptedSharesRoster;
12use crate::guardian::SecretSharingInstance;
13use crate::guardian::SetupNewKeyResponse;
14use serde::Serialize;
15
16/// The current ceremony result together with its latest encrypted KP share state.
17#[derive(Clone, Debug, PartialEq, Serialize)]
18pub struct CeremonyState {
19    pub secret_sharing_instance: SecretSharingInstance,
20    pub btc_master_pubkey: BitcoinPubkey,
21    pub cert_seq: u64,
22    pub encrypted_shares: KPEncryptedSharesRoster,
23}
24
25impl CeremonyState {
26    /// Combine a ceremony log message with the KP share-state message for its
27    /// resulting secret-sharing instance.
28    pub fn new(
29        ceremony: CeremonyLogMessage,
30        kp_share_state: KpShareStateLogMessage,
31    ) -> GuardianResult<Self> {
32        let (secret_sharing_instance, btc_master_pubkey) = ceremony.into_instance_and_pubkey();
33        if kp_share_state.sharing_seq != secret_sharing_instance.sharing_seq() {
34            return Err(GuardianError::InternalError(format!(
35                "kp-shares sharing_seq ({}) differs from ceremony sharing_seq ({})",
36                kp_share_state.sharing_seq,
37                secret_sharing_instance.sharing_seq()
38            )));
39        }
40        Self::from_parts(
41            secret_sharing_instance,
42            btc_master_pubkey,
43            kp_share_state.cert_seq,
44            kp_share_state.encrypted_shares,
45        )
46    }
47
48    fn from_parts(
49        secret_sharing_instance: SecretSharingInstance,
50        btc_master_pubkey: BitcoinPubkey,
51        cert_seq: u64,
52        encrypted_shares: KPEncryptedSharesRoster,
53    ) -> GuardianResult<Self> {
54        if encrypted_shares.share_count() != secret_sharing_instance.num_shares() {
55            return Err(GuardianError::InternalError(format!(
56                "encrypted share count ({}) differs from ceremony num_shares ({})",
57                encrypted_shares.share_count(),
58                secret_sharing_instance.num_shares()
59            )));
60        }
61        Ok(Self {
62            secret_sharing_instance,
63            btc_master_pubkey,
64            cert_seq,
65            encrypted_shares,
66        })
67    }
68
69    /// Confirm the ceremony uses the expected secret-sharing parameters.
70    pub fn validate_sharing_params(
71        &self,
72        expected_n: usize,
73        expected_t: usize,
74    ) -> GuardianResult<()> {
75        if self.secret_sharing_instance.num_shares() != expected_n {
76            return Err(GuardianError::InvalidInputs(format!(
77                "ceremony num_shares ({}) differs from expected ({expected_n})",
78                self.secret_sharing_instance.num_shares()
79            )));
80        }
81        if self.secret_sharing_instance.threshold() != expected_t {
82            return Err(GuardianError::InvalidInputs(format!(
83                "ceremony threshold ({}) differs from expected ({expected_t})",
84                self.secret_sharing_instance.threshold()
85            )));
86        }
87        Ok(())
88    }
89}
90
91impl From<SetupNewKeyResponse> for CeremonyState {
92    fn from(response: SetupNewKeyResponse) -> Self {
93        let SetupNewKeyResponse {
94            encrypted_shares,
95            secret_sharing_instance,
96            btc_master_pubkey,
97        } = response;
98        Self::from_parts(
99            secret_sharing_instance,
100            btc_master_pubkey,
101            0,
102            encrypted_shares,
103        )
104        .expect("SetupNewKeyResponse must contain one encrypted share per participant")
105    }
106}

108#[cfg(test)]
109mod tests {
110    use super::*;
111    use crate::guardian::GuardianSigned;
112
113    #[test]
114    fn serialization_includes_all_encrypted_shares() {
115        let response = GuardianSigned::<SetupNewKeyResponse>::mock_for_testing().data;
116        let expected_share_count = response.encrypted_shares.share_count();
117        let state = CeremonyState::from(response);
118
119        let json = serde_json::to_value(&state).unwrap();
120
121        assert_eq!(
122            json["encrypted_shares"].as_array().unwrap().len(),
123            expected_share_count
124        );
125    }
126
127    #[test]
128    fn new_rejects_mismatched_sharing_seq() {
129        let response = GuardianSigned::<SetupNewKeyResponse>::mock_for_testing().data;
130        let sharing_seq = response.secret_sharing_instance.sharing_seq();
131        let err = CeremonyState::new(
132            CeremonyLogMessage::NewKey {
133                instance: response.secret_sharing_instance,
134                btc_master_pubkey: response.btc_master_pubkey,
135            },
136            KpShareStateLogMessage::new(sharing_seq + 1, 0, response.encrypted_shares),
137        )
138        .unwrap_err();
139        assert!(format!("{err}").contains("kp-shares sharing_seq"), "{err}");
140    }
141
142    #[test]
143    fn new_rejects_mismatched_share_count() {
144        let response = GuardianSigned::<SetupNewKeyResponse>::mock_for_testing().data;
145        let sharing_seq = response.secret_sharing_instance.sharing_seq();
146        let err = CeremonyState::new(
147            CeremonyLogMessage::NewKey {
148                instance: response.secret_sharing_instance,
149                btc_master_pubkey: response.btc_master_pubkey,
150            },
151            KpShareStateLogMessage::new(
152                sharing_seq,
153                0,
154                KPEncryptedSharesRoster::new(vec![]).unwrap(),
155            ),
156        )
157        .unwrap_err();
158        assert!(format!("{err}").contains("encrypted share count"), "{err}");
159    }
