4//! Guardian S3 logs. `message` holds the `LogMessage` family the enclave emits;
5//! `envelope` holds the `LogRecord` wrapper written to S3 and its keying/signing.
6//! The S3 layout constants (stream prefixes + object-lock durations) live here so
7//! the whole key/lock scheme reads from one place. See
8//! `crates/hashi-guardian/README.md` for the canonical key layout.
9
10pub mod envelope;
11pub mod message;
12
13pub use envelope::*;
14pub use message::*;

16use std::time::Duration;
17
18/// Object lock durations used for S3 log objects.
19///
20/// These are public so that external verifiers/monitors can apply the same expectations.
21///
22/// TODO: Uniform 7 days is a coarse placeholder. Revisit per-log-type
23/// against the SLO we actually want to defend — heartbeats could be shorter,
24/// ceremony/withdraws likely want years.
25const ONE_WEEK: Duration = Duration::from_secs(7 * 24 * 60 * 60);
26pub const S3_OBJECT_LOCK_DURATION_INIT: Duration = ONE_WEEK;
27pub const S3_OBJECT_LOCK_DURATION_WITHDRAW: Duration = ONE_WEEK;
28pub const S3_OBJECT_LOCK_DURATION_HEARTBEAT: Duration = ONE_WEEK;
29pub const S3_OBJECT_LOCK_DURATION_CEREMONY: Duration = ONE_WEEK;
30pub const S3_OBJECT_LOCK_DURATION_COMMITTEE_UPDATE: Duration = ONE_WEEK;
31
32/// S3 sub-prefixes used for guardian log streams.
33/// See `crates/hashi-guardian/README.md` for canonical key layout.
34pub const S3_DIR_INIT: &str = "init";
35pub const S3_DIR_WITHDRAW: &str = "withdraw";
36pub const S3_DIR_HEARTBEAT: &str = "heartbeat";
37pub const S3_DIR_CEREMONY: &str = "ceremony";
38pub const S3_DIR_COMMITTEE_UPDATE: &str = "committee-update";
