23const ONE_WEEK: Duration = Duration::from_secs(7 * 24 * 60 * 60);
24const THIRTY_DAYS: Duration = Duration::from_secs(30 * 24 * 60 * 60);
25const NINETY_DAYS: Duration = Duration::from_secs(90 * 24 * 60 * 60);
26const TEN_YEARS: Duration = Duration::from_secs(10 * 365 * 24 * 60 * 60);
27
28/// Object-lock retention policy for Guardian S3 logs.
29///
30/// Long-lived records include initialization, withdrawals, ceremonies,
31/// committee updates, and genesis. They remain usable only while both the
32/// record and its initialization chain are protected from replacement.
33///
34/// Object Lock expiry permits deletion; it does not trigger it. Heartbeats may
35/// be removed by a separate lifecycle policy. Encrypted KP shares use the same
36/// short-lived lock so stale shares can be explicitly deleted after their
37/// recovery window, for example when a KP encryption key is retired or lost.
38#[derive(Clone, Copy, Debug, Eq, PartialEq)]
39pub struct S3ObjectLockPolicy {
40    pub long_lived: Duration,
41    pub short_lived: Duration,
42}
43
44// TODO(s3-retention): Retire this enum once the repository has a canonical
45// Hashi network enum from which the retention policy can be derived.
46/// Hashi deployment class used to select an S3 object-lock policy.
47///
48/// This is intentionally independent of `bitcoin::Network`: a Hashi
49/// deployment comprises both a Sui chain and a Bitcoin chain, and neither one
50/// alone is the deployment identity.
51#[derive(Clone, Copy, Debug, Deserialize, Eq, PartialEq, Serialize)]
52#[serde(rename_all = "snake_case")]
53pub enum S3RetentionEnvironment {
54    Devnet,
55    Mainnet,
56    Testnet,
57}
58
59/// Devnet records balance useful debugging history with rapid iteration.
60pub const DEVNET_S3_OBJECT_LOCK_POLICY: S3ObjectLockPolicy = S3ObjectLockPolicy {
61    long_lived: NINETY_DAYS,
62    short_lived: ONE_WEEK,
63};
64
65/// Mainnet records are durable audit and recovery inputs.
66pub const MAINNET_S3_OBJECT_LOCK_POLICY: S3ObjectLockPolicy = S3ObjectLockPolicy {
67    long_lived: TEN_YEARS,
68    short_lived: THIRTY_DAYS,
69};
70
71/// Testnet is a durable deployment and uses the mainnet retention policy.
72pub const TESTNET_S3_OBJECT_LOCK_POLICY: S3ObjectLockPolicy = S3ObjectLockPolicy {
73    long_lived: TEN_YEARS,
74    short_lived: THIRTY_DAYS,
75};
76
77impl S3ObjectLockPolicy {
78    pub const fn for_environment(environment: S3RetentionEnvironment) -> Self {
79        match environment {
80            S3RetentionEnvironment::Devnet => DEVNET_S3_OBJECT_LOCK_POLICY,
81            S3RetentionEnvironment::Mainnet => MAINNET_S3_OBJECT_LOCK_POLICY,
82            S3RetentionEnvironment::Testnet => TESTNET_S3_OBJECT_LOCK_POLICY,
83        }
84    }
85
86    const fn duration_for(self, log_type: message::LogType) -> Duration {
87        match log_type {
88            message::LogType::Heartbeat | message::LogType::KpShareState => self.short_lived,
89            message::LogType::Init
90            | message::LogType::Withdrawal
91            | message::LogType::Ceremony
92            | message::LogType::CommitteeUpdate
93            | message::LogType::Genesis => self.long_lived,
94        }
95    }
96}
