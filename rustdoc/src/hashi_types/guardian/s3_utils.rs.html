5use anyhow::Context;
6use std::convert::TryFrom;
7use std::fmt;

9use time::Date;
10use time::OffsetDateTime;
11use time::PrimitiveDateTime;
12use time::Time;

14pub const SECONDS_PER_HOUR: UnixSeconds = 60 * 60;
15const DIR_WRITES_COMPLETION_DELAY: Duration = Duration::from_mins(10);
16
17type Year = i32;
18type Month = u8;
19type Day = u8;
20type Hour = u8;
21
22/// An S3 directory: prefix/YYYY/MM/DD/HH.
23/// All logs emitted within an hour are stored in the same directory, e.g., logs emitted between 12-1 PM are in `<prefix>`/12 directory.
24#[derive(Clone, Debug, Eq, PartialEq)]
25pub struct S3HourScopedDirectory {
26    prefix: String,
27    year: Year,
28    month: Month,
29    day: Day,
30    hour: Hour,
31}
32
33impl S3HourScopedDirectory {
34    pub fn new(prefix: &str, t: UnixSeconds) -> Self {
35        let unix_seconds = i64::try_from(t).expect("timestamp should fit i64");
36        let datetime =
37            OffsetDateTime::from_unix_timestamp(unix_seconds).expect("timestamp should be valid");
38        Self {
39            prefix: prefix.to_string(),
40            year: datetime.year(),
41            month: u8::from(datetime.month()),
42            day: datetime.day(),
43            hour: datetime.hour(),
44        }
45    }
46
47    pub fn next_dir(&self) -> Self {
48        Self::new(
49            &self.prefix,
50            self.to_unix_seconds().saturating_add(SECONDS_PER_HOUR),
51        )
52    }
53
54    /// Returns the directory for the previous hour. Saturates at the Unix epoch.
55    pub fn prev_dir(&self) -> Self {
56        Self::new(
57            &self.prefix,
58            self.to_unix_seconds().saturating_sub(SECONDS_PER_HOUR),
59        )
60    }
61
62    /// The time at which writes to current S3 directory finish.
63    /// DIR_WRITES_COMPLETION_DELAY accounts for any in-flight retries and clock skew.
64    pub fn write_completion_time(&self) -> UnixSeconds {
65        self.next_dir()
66            .to_unix_seconds()
67            .saturating_add(DIR_WRITES_COMPLETION_DELAY.as_secs())
68    }
69
70    pub fn to_unix_seconds(&self) -> UnixSeconds {
71        let (date, time) = parse_calendar(self.year, self.month, self.day, self.hour)
72            .expect("invariants validated at construction");
73        let ts = PrimitiveDateTime::new(date, time)
74            .assume_utc()
75            .unix_timestamp();
76        UnixSeconds::try_from(ts).expect("timestamp should be non-negative")
77    }
78
79    /// Parses a directory path of the form `{prefix}/{yyyy}/{mm}/{dd}/{hh}/`
80    /// (with or without the trailing slash) back into a directory value.
81    /// Inverse of the `Display` impl.
82    pub fn from_path(path: &str) -> anyhow::Result<Self> {
83        let parts: Vec<&str> = path.trim_end_matches('/').split('/').collect();
84        anyhow::ensure!(
85            parts.len() == 5,
86            "expected `{{prefix}}/YYYY/MM/DD/HH/` in {path}"
87        );
88        let prefix = parts[0];
89        let year: Year = parts[1]
90            .parse()
91            .with_context(|| format!("invalid year in {path}"))?;
92        let month: Month = parts[2]
93            .parse()
94            .with_context(|| format!("invalid month in {path}"))?;
95        let day: Day = parts[3]
96            .parse()
97            .with_context(|| format!("invalid day in {path}"))?;
98        let hour: Hour = parts[4]
99            .parse()
100            .with_context(|| format!("invalid hour in {path}"))?;
101        parse_calendar(year, month, day, hour).with_context(|| format!("invalid path {path}"))?;
102        Ok(Self {
103            prefix: prefix.to_string(),
104            year,
105            month,
106            day,
107            hour,
108        })
109    }
110}
111
112/// Validates the (year, month, day, hour) tuple and returns the corresponding
113/// `(Date, Time)` if every component is in range. Shared by [`S3HourScopedDirectory::from_path`]
114/// (which uses it to validate before construction) and
115/// [`S3HourScopedDirectory::to_unix_seconds`] (which is infallible because the
116/// struct invariant guarantees validity).
117fn parse_calendar(year: Year, month: Month, day: Day, hour: Hour) -> anyhow::Result<(Date, Time)> {
118    let month_enum =
119        time::Month::try_from(month).map_err(|e| anyhow::anyhow!("invalid month {month}: {e}"))?;
120    let date = Date::from_calendar_date(year, month_enum, day)
121        .map_err(|e| anyhow::anyhow!("invalid date {year}-{month:02}-{day:02}: {e}"))?;
122    let time =
123        Time::from_hms(hour, 0, 0).map_err(|e| anyhow::anyhow!("invalid hour {hour}: {e}"))?;
124    Ok((date, time))
125}
126
127impl fmt::Display for S3HourScopedDirectory {
128    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
129        write!(
130            f,
131            "{}/{:04}/{:02}/{:02}/{:02}/",
132            self.prefix, self.year, self.month, self.day, self.hour
133        )
134    }
135}

137#[cfg(test)]
138mod tests {
139    use super::DIR_WRITES_COMPLETION_DELAY;
140    use super::S3HourScopedDirectory;
141
142    #[test]
143    fn test_epoch_directory_format() {
144        let dir = S3HourScopedDirectory::new("heartbeat", 0);
145        assert_eq!(dir.to_string(), "heartbeat/1970/01/01/00/");
146    }
147
148    #[test]
149    fn test_hour_and_day_rollover_format() {
150        let before_hour_boundary = S3HourScopedDirectory::new("withdraw", 3_599);
151        assert_eq!(before_hour_boundary.to_string(), "withdraw/1970/01/01/00/");
152
153        let next_hour = S3HourScopedDirectory::new("withdraw", 3_600);
154        assert_eq!(next_hour.to_string(), "withdraw/1970/01/01/01/");
155
156        let next_day = S3HourScopedDirectory::new("withdraw", 86_400);
157        assert_eq!(next_day.to_string(), "withdraw/1970/01/02/00/");
158    }
159
160    #[test]
161    fn test_prev_dir_walks_back_and_saturates_at_epoch() {
162        let mut dir = S3HourScopedDirectory::new("withdraw", 86_400 + 3_600);
163        assert_eq!(dir.to_string(), "withdraw/1970/01/02/01/");
164        dir = dir.prev_dir();
165        assert_eq!(dir.to_string(), "withdraw/1970/01/02/00/");
166        dir = dir.prev_dir();
167        assert_eq!(dir.to_string(), "withdraw/1970/01/01/23/");
168
169        // Saturates at epoch.
170        let epoch = S3HourScopedDirectory::new("withdraw", 0);
171        assert_eq!(epoch.prev_dir(), epoch);
172    }
173
174    #[test]
175    fn test_from_path_roundtrips_with_display() {
176        let dir = S3HourScopedDirectory::new("withdraw", 1_700_000_000);
177        let displayed = dir.to_string();
178        let parsed = S3HourScopedDirectory::from_path(&displayed).expect("roundtrip");
179        assert_eq!(parsed, dir);
180        // Also accept the trailing-slash-stripped form.
181        let parsed_nopfx = S3HourScopedDirectory::from_path(displayed.trim_end_matches('/'))
182            .expect("roundtrip without trailing slash");
183        assert_eq!(parsed_nopfx, dir);
184    }
185
186    #[test]
187    fn test_from_path_rejects_wrong_shape() {
188        assert!(S3HourScopedDirectory::from_path("withdraw/2024/03/15/").is_err()); // missing hour
189        assert!(S3HourScopedDirectory::from_path("withdraw/2024/03/15/14/extra/").is_err()); // too many parts
190        assert!(S3HourScopedDirectory::from_path("withdraw/2024/13/15/14/").is_err()); // invalid month
191        assert!(S3HourScopedDirectory::from_path("withdraw/2024/02/30/14/").is_err()); // invalid day
192        assert!(S3HourScopedDirectory::from_path("withdraw/2024/02/15/24/").is_err()); // invalid hour
193        assert!(S3HourScopedDirectory::from_path("withdraw/notayear/03/15/14/").is_err()); // non-numeric
194    }
195
196    #[test]
197    fn test_next_dir_and_completion_time() {
198        let mut dir = S3HourScopedDirectory::new("withdraw", 3_599);
199        assert_eq!(dir.to_string(), "withdraw/1970/01/01/00/");
200        assert_eq!(dir.to_unix_seconds(), 0);
201        assert_eq!(
202            dir.write_completion_time(),
203            3_600 + DIR_WRITES_COMPLETION_DELAY.as_secs()
204        );
205
206        for i in 0..24 {
207            assert_eq!(dir.to_string(), format!("withdraw/1970/01/01/{:02}/", i));
208            dir = dir.next_dir();
209        }
210        assert_eq!(dir.to_string(), "withdraw/1970/01/02/00/");
211    }
