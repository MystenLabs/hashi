4//! UTXO and transaction types for Hashi Bitcoin withdrawals.
5//!
6//! Two classes of types exist here:
7//! - Domain types with checked addresses that implement Serialize but not Deserialize
8//! - `*Wire` types with unchecked addresses that implement both Serialize and Deserialize
9//!
10//! Untrusted input arrives as the `*Wire` form; `TxUTXOs::new` is the single gate
11//! that validates it (address-network, amounts, duplicates, fees) and converts it
12//! into the checked domain types.
13//!
14//! Internal-output addresses are derived via `super::taproot`.
15

87/// Withdrawal destination and amount. Internal to `TxUTXOs`; callers build the
88/// `OutputUTXOWire` form and let `TxUTXOs::new` validate and convert.
89/// External amounts count towards rate limits whereas internal amounts don't.
90/// Internal address is derived inside the enclave to ensure that it is actually internal.
91#[derive(Serialize, Debug, Clone, PartialEq)]
92enum OutputUTXO {
93    External(ExternalOutputUTXO),
94    Internal(InternalOutputUTXO),
95}

97/// Copy of OutputUTXO with unchecked address
98#[derive(Debug, Clone, Serialize, Deserialize)]
99pub enum OutputUTXOWire {
100    External(ExternalOutputUTXOWire),
101    Internal(InternalOutputUTXO),
102}

126/// Represents an input to be spent.
127///
128/// All inputs are expected to be P2TR (Pay-to-Taproot) since spending is done via taproot script path.
129impl InputUTXO {
130    pub fn new(outpoint: OutPoint, amount: Amount, derivation_path: DerivationPath) -> Self {
131        Self {
132            outpoint,
133            amount,
134            derivation_path,
135        }
136    }
137
138    /// Returns a `TxIn` for this UTXO with placeholder witness data.
139    ///
140    /// The witness will be populated later after signing.
141    pub fn txin(&self) -> TxIn {
142        TxIn {
143            previous_output: self.outpoint,
144            // No script sig needed for taproot
145            script_sig: ScriptBuf::default(),
146            // Enables RBF, disables relative lock time, allows absolute lock time
147            sequence: Sequence::ENABLE_RBF_NO_LOCKTIME,
148            // Witness will be set later
149            witness: Witness::default(),
150        }
151    }
152
153    /// Prevout `TxOut` and tap leaf hash for sighash, derived from the path.
154    fn prevout_and_leaf_hash(
155        &self,
156        enclave_pubkey: &BitcoinPubkey,
157        hashi_master_g: &HashiMasterG,
158    ) -> (TxOut, TapLeafHash) {
159        let (script_pubkey, leaf_hash) = taproot_script_pubkey_and_leaf_hash(
160            enclave_pubkey,
161            hashi_master_g,
162            &self.derivation_path,
163        );
164        (
165            TxOut {
166                value: self.amount,
167                script_pubkey,
168            },
169            leaf_hash,
170        )
171    }
172}

209/// Represents an output destination for a withdrawal.
210///
211/// Outputs can be **external** (to a user-provided address) or **internal** (change, derived inside enclave).
212impl OutputUTXO {
213    /// Returns the output amount in satoshis.
214    fn amount(&self) -> Amount {
215        match self {
216            OutputUTXO::External(ExternalOutputUTXO { amount, .. }) => *amount,
217            OutputUTXO::Internal(InternalOutputUTXO { amount, .. }) => *amount,
218        }
219    }
220
221    /// Builds the `TxOut`: external uses the checked address, internal derives
222    /// its taproot script via `super::taproot`.
223    fn to_txout(&self, enclave_pubkey: &BitcoinPubkey, hashi_master_g: &HashiMasterG) -> TxOut {
224        match self {
225            OutputUTXO::External(ExternalOutputUTXO { address, amount }) => TxOut {
226                value: *amount,
227                script_pubkey: address.script_pubkey(),
228            },
229            OutputUTXO::Internal(InternalOutputUTXO {
230                derivation_path,
231                amount,
232            }) => {
233                let scripts = taproot_script_pubkey_and_leaf_hash(
234                    enclave_pubkey,
235                    hashi_master_g,
236                    derivation_path,
237                );
238                TxOut {
239                    value: *amount,
240                    script_pubkey: scripts.0,
241                }
242            }
243        }
244    }
245}

247/// Validates that an unchecked address is valid for `network` and returns a checked address.
248fn validate_address_for_network(
249    address: &BitcoinAddress<NetworkUnchecked>,
250    network: Network,
251) -> anyhow::Result<BitcoinAddress<NetworkChecked>> {
252    // Prefer the library's checked conversion to avoid accidentally assuming correctness.
253    address
254        .clone()
255        .require_network(network)
256        .map_err(|_| anyhow!("invalid address {:?} for network {}", address, network))
257}

259impl TxUTXOs {
260    /// Constructs a `TxUTXOs`, validating every invariant in one place: external
261    /// output addresses must be valid for `network`, amounts must be non-zero,
262    /// inputs must be unique, and fees must be positive. The single gate for both
263    /// locally-built and wire-parsed UTXO sets.
264    pub fn new(
265        inputs: Vec<InputUTXO>,
266        outputs: Vec<OutputUTXOWire>,
267        network: Network,
268    ) -> anyhow::Result<Self> {
269        if inputs.is_empty() {
270            anyhow::bail!("input utxos must not be empty");
271        }
272        if outputs.is_empty() {
273            anyhow::bail!("output utxos must not be empty");
274        }
275
276        // Validate each external address against the network, turning untrusted
277        // unchecked addresses into checked ones. Internal outputs carry no address.
278        let mut checked = Vec::with_capacity(outputs.len());
279        for output in outputs {
280            checked.push(match output {
281                OutputUTXOWire::External(ext) => OutputUTXO::External(ExternalOutputUTXO {
282                    address: validate_address_for_network(&ext.address, network)?,
283                    amount: ext.amount,
284                }),
285                OutputUTXOWire::Internal(int) => OutputUTXO::Internal(int),
286            });
287        }
288        let outputs = checked;
289
290        // Reject zero amounts on both sides: a 0-value input is meaningless and a
291        // 0-value output is invalid on Bitcoin.
292        for utxo in &inputs {
293            if utxo.amount == Amount::ZERO {
294                anyhow::bail!("input amount must be > 0");
295            }
296        }
297        for utxo in &outputs {
298            if utxo.amount() == Amount::ZERO {
299                anyhow::bail!("output amount must be > 0");
300            }
301        }
302
303        // Disallow duplicate inputs (same txid,vout), which would result in an invalid transaction.
304        let mut seen_inputs: HashSet<OutPoint> = HashSet::with_capacity(inputs.len());
305        for utxo in &inputs {
306            if !seen_inputs.insert(utxo.outpoint) {
307                anyhow::bail!("duplicate input outpoint: {}", utxo.outpoint);
308            }
309        }
310
311        let tx_info = Self { inputs, outputs };
312
313        // Enforce the intended invariant: fees > 0.
314        tx_info.assert_positive_fees()?;
315
316        Ok(tx_info)
317    }

319    /// Constructs all outputs (both external and internal).
320    ///
321    /// For `External` outputs, uses the user-provided address. For `Internal` outputs,
322    /// derives a taproot address using the enclave and hashi keys.
323    pub fn compute_all_outputs(
324        &self,
325        enclave_pubkey: &BitcoinPubkey,
326        hashi_master_g: &HashiMasterG,
327    ) -> Vec<TxOut> {
328        self.outputs
329            .iter()
330            .map(|utxo| utxo.to_txout(enclave_pubkey, hashi_master_g))
331            .collect()
332    }
333
334    /// BTC that leaves the pool when this txn broadcasts: `inputs - change`,
335    /// equivalent to `external outputs + miner_fee`. The amount that consumes
336    /// the rate-limiter (miner fee leaves the pool too; change flows back).
337    pub fn gross_outflow_amount(&self) -> Amount {
338        let inputs: Amount = self.inputs.iter().map(|i| i.amount).sum();
339        let internal: Amount = self
340            .outputs
341            .iter()
342            .filter_map(|utxo| match utxo {
343                OutputUTXO::Internal(x) => Some(x.amount),
344                OutputUTXO::External(_) => None,
345            })
346            .sum();
347        inputs - internal
348    }
349
350    /// Constructs an unsigned Bitcoin transaction for these UTXOs.
351    fn unsigned_tx(
352        &self,
353        enclave_pubkey: &BitcoinPubkey,
354        hashi_master_g: &HashiMasterG,
355    ) -> Transaction {
356        let all_outputs = self.compute_all_outputs(enclave_pubkey, hashi_master_g);
357        construct_tx(
358            self.inputs.iter().map(|input| input.txin()).collect(),
359            all_outputs,
360        )
361    }
362
363    /// Constructs sighash messages for each input, ready for signing.
364    ///
365    /// Uses `taproot_script_spend_signature_hash` for script-path spending.
366    pub fn signing_messages_and_txid(
367        &self,
368        enclave_pubkey: &BitcoinPubkey,
369        hashi_master_g: &HashiMasterG,
370    ) -> (Vec<Message>, Txid) {
371        let tx = self.unsigned_tx(enclave_pubkey, hashi_master_g);
372        // Derive each input's prevout + tap leaf hash from its path.
373        let artifacts: Vec<(TxOut, TapLeafHash)> = self
374            .inputs
375            .iter()
376            .map(|input| input.prevout_and_leaf_hash(enclave_pubkey, hashi_master_g))
377            .collect();
378        let prevouts: Vec<TxOut> = artifacts.iter().map(|(p, _)| p.clone()).collect();
379        let leaf_hashes: Vec<TapLeafHash> = artifacts.iter().map(|(_, h)| *h).collect();
380
381        let messages = taproot_script_spend_sighashes(&tx, &prevouts, &leaf_hashes)
382            .into_iter()
383            .map(Message::from_digest)
384            .collect();
385        (messages, tx.compute_txid())
386    }
387
388    fn assert_positive_fees(&self) -> anyhow::Result<()> {
389        let input_sum: Amount = self.inputs.iter().map(|utxo| utxo.amount).sum();
390        let output_sum: Amount = self.outputs.iter().map(|utxo| utxo.amount()).sum();
391        if input_sum <= output_sum {
392            anyhow::bail!(
393                "fees must be greater than zero: input_sum={} output_sum={}",
394                input_sum,
395                output_sum
396            );
397        }
398        Ok(())
399    }
400}

406/// Signs messages using Schnorr signatures (suitable for taproot script-spend).
407///
408/// Each message is signed and wrapped in a `Signature` with `TapSighashType::Default`.
409pub fn sign_btc_tx(messages: &[Message], kp: &BitcoinKeypair) -> Vec<BitcoinSignature> {
410    messages
411        .iter()
412        // Not using aux randomness which only provides side-channel protection
413        .map(|m| BTC_LIB.sign_schnorr_no_aux_rand(m, kp))
414        .map(|s| Signature {
415            signature: s,
416            sighash_type: TapSighashType::Default,
417        })
418        .collect()
419}

421/// Constructs a Bitcoin transaction with the given inputs and outputs.
422///
423/// Uses BTC tx version 2 and disables lock time.
424pub fn construct_tx(inputs: Vec<TxIn>, outputs: Vec<TxOut>) -> Transaction {
425    Transaction {
426        // The latest BTC tx version
427        version: Version::TWO,
428        // Disable absolute lock time (i.e., can be mined immediately)
429        lock_time: LockTime::ZERO,
430        input: inputs,
431        output: outputs,
432    }
433}
