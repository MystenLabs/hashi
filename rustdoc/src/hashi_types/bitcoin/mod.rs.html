40/// A Hashi key-derivation path: the 32-byte Sui address of the deposit
41/// recipient. Converted to fastcrypto's raw `[u8; 32]` form only at the
42/// `derive_verifying_key` boundary in `taproot`.
43pub type DerivationPath = sui_sdk_types::Address;
44
45pub fn witness_program_from_address(address: &BitcoinAddress) -> anyhow::Result<Vec<u8>> {
46    let script = address.script_pubkey();
47    let bytes = script.as_bytes();
48    match bytes {
49        [0x00, 0x14, rest @ ..] if rest.len() == 20 => Ok(rest.to_vec()),
50        [0x51, 0x20, rest @ ..] if rest.len() == 32 => Ok(rest.to_vec()),
51        _ => anyhow::bail!("Unsupported script pubkey for withdrawal output: {script}"),
52    }
53}

55/// Convert raw witness program bytes to a `ScriptBuf`.
56/// 32-byte addresses are P2TR (witness v1), 20-byte addresses are P2WPKH (witness v0).
57pub fn script_pubkey_from_witness_program(address_bytes: &[u8]) -> anyhow::Result<ScriptBuf> {
58    let version = match address_bytes.len() {
59        32 => WitnessVersion::V1,
60        20 => WitnessVersion::V0,
61        len => anyhow::bail!("Unsupported bitcoin address length: {len}"),
62    };
63    let program = WitnessProgram::new(version, address_bytes)
64        .map_err(|e| anyhow!("Invalid witness program: {e}"))?;
65    Ok(ScriptBuf::new_witness_program(&program))
66}
67
68/// Convert raw witness program bytes to a human-readable Bitcoin address string.
69pub fn address_string_from_witness_program(
70    address_bytes: &[u8],
71    network: Network,
72) -> anyhow::Result<String> {
73    let script = script_pubkey_from_witness_program(address_bytes)?;
74    let address = BitcoinAddress::from_script(&script, network)
75        .map_err(|e| anyhow!("Failed to convert script to address: {e}"))?;
76    Ok(address.to_string())
77}

79/// Full input weight (WU) for an immediate 2-of-2 taproot script-path spend.
80/// TXIN_BASE_WEIGHT (164 WU) + satisfaction (268 WU) = 432 WU (108 vB).
81pub const SCRIPT_PATH_2OF2_TXIN_WEIGHT: u64 = 164 + 268;
82
83/// Non-witness fixed overhead for a segwit transaction:
84/// nVersion(4x4) + nLockTime(4x4) = 32 WU, plus the segwit marker/flag (2 WU).
85pub const TX_FIXED_WEIGHT_WU: u64 = 34;
86
87/// P2TR output weight: TXOUT_BASE(36) + OP_1 OP_PUSHBYTES_32 <32 bytes>(136) = 172 WU.
88pub const P2TR_OUTPUT_WEIGHT_WU: u64 = 172;
89
90/// P2WPKH output weight: TXOUT_BASE(36) + OP_0 OP_PUSHBYTES_20 <20 bytes>(88) = 124 WU.
91pub const P2WPKH_OUTPUT_WEIGHT_WU: u64 = 124;
92
93pub fn output_weight_for_witness_program(bitcoin_address: &[u8]) -> anyhow::Result<u64> {
94    match bitcoin_address.len() {
95        32 => Ok(P2TR_OUTPUT_WEIGHT_WU),
96        20 => Ok(P2WPKH_OUTPUT_WEIGHT_WU),
97        len => anyhow::bail!("Unsupported bitcoin address length: {len}"),
98    }
99}

101/// Deterministic Bitcoin keypair helper for tests.
102pub fn create_btc_keypair_for_test(sk: &[u8; 32]) -> BitcoinKeypair {
103    let secret_key = SecretKey::from_slice(sk).expect("valid secret key");
104    BitcoinKeypair::from_secret_key(&BTC_LIB, &secret_key)
105}
