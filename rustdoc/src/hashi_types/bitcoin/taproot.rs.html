31/// Fixed nothing-up-my-sleeve (NUMS) point used as the taproot internal key. Copied from BIP-341.
32static NUMS_INTERNAL_KEY: LazyLock<BitcoinPubkey> = LazyLock::new(|| {
33    BitcoinPubkey::from_str("50929b74c1a04954b78b4b6035e97a5e078a5a0f28ec96d547bfee9ace803ac0")
34        .expect("valid nums key")
35});

37/// Initial MPC-only recovery delay for Hashi-controlled UTXOs.
38///
39/// WARNING: this value is part of the taproot script, so changing it changes
40/// deposit/change addresses. Future governance-controlled changes need a
41/// policy/versioning and grace-period mechanism so deposits broadcast under the
42/// previous delay are still accepted and spendable.
43pub const HASHI_MPC_RECOVERY_DELAY_SECONDS: u32 = 60 * 24 * 60 * 60;

45/// Leaf indices into the taproot tree built by [`compute_taproot_descriptor`].
46/// These must match the leaf order in its descriptor string.
47const IMMEDIATE_2OF2_LEAF_INDEX: usize = 0;
48const MPC_RECOVERY_LEAF_INDEX: usize = 1;

50/// scriptPubKey and immediate 2-of-2 tap leaf hash for the output at
51/// `derivation_path`.
52pub(super) fn taproot_script_pubkey_and_leaf_hash(
53    enclave_pubkey: &BitcoinPubkey,
54    hashi_master_g: &HashiMasterG,
55    hashi_derivation_path: &DerivationPath,
56) -> (ScriptBuf, TapLeafHash) {
57    let desc = compute_taproot_descriptor(enclave_pubkey, hashi_master_g, hashi_derivation_path);
58    let address_script = desc.script_pubkey();
59
60    // Keep the named index visible here instead of using .next()
61    #[allow(clippy::iter_nth_zero)]
62    let leaf_hash = desc
63        .leaves()
64        .nth(IMMEDIATE_2OF2_LEAF_INDEX)
65        .expect("tap tree should have the immediate 2-of-2 leaf")
66        .compute_tap_leaf_hash();
67
68    (address_script, leaf_hash)
69}

82/// Immediate 2-of-2 spend artifacts for the Hashi taproot tree.
83pub fn taproot_2of2_witness_artifacts(
84    enclave_pubkey: &BitcoinPubkey,

92/// Delayed MPC-only recovery spend artifacts for the Hashi taproot tree.
93pub fn taproot_mpc_recovery_witness_artifacts(
94    enclave_pubkey: &BitcoinPubkey,
95    hashi_master_g: &HashiMasterG,
96    hashi_derivation_path: &DerivationPath,

102fn taproot_witness_artifacts_at_leaf(
103    desc: &Tr<BitcoinPubkey>,
104    leaf_index: usize,
105) -> (ScriptBuf, ControlBlock, TapLeafHash) {
106    let leaf = desc
107        .leaves()
108        .nth(leaf_index)
109        .expect("tap tree should have requested leaf");
110    let tap_script = leaf.compute_script();
111    let leaf_hash = leaf.compute_tap_leaf_hash();
112
113    let control_block = desc
114        .spend_info()
115        .leaves()
116        .nth(leaf_index)
117        .expect("spend info should have requested leaf")
118        .into_control_block();
119
120    (tap_script, control_block, leaf_hash)
121}

123/// Per-input taproot script-spend sighashes for an unsigned tx. `prevouts` and
124/// `leaf_hashes` run parallel to `tx.input`. Builds one `SighashCache` and reuses
125/// it across inputs, since the whole-tx components are input-independent.
126pub fn taproot_script_spend_sighashes(
127    tx: &Transaction,
128    prevouts: &[TxOut],
129    leaf_hashes: &[TapLeafHash],
130) -> Vec<[u8; 32]> {
131    let prevouts = Prevouts::All(prevouts);
132    let mut sighasher = SighashCache::new(tx);
133    (0..tx.input.len())
134        .map(|index| {
135            let sighash = sighasher
136                .taproot_script_spend_signature_hash(
137                    index,
138                    &prevouts,
139                    leaf_hashes[index],
140                    TapSighashType::Default,
141                )
142                .expect("taproot script-spend sighash failed");
143            *sighash.as_byte_array()
144        })
145        .collect()
146}

148/// Creates a taproot descriptor for the Hashi taproot tree:
149///
150/// - Immediate 2-of-2 leaf: guardian/enclave + derived Hashi MPC child key.
151/// - Delayed recovery leaf: after `HASHI_MPC_RECOVERY_DELAY_SECONDS`, derived
152///   Hashi MPC child key only.
153///
154/// Both leaves are committed under a NUMS internal key, disabling meaningful
155/// key path spends.
156fn compute_taproot_descriptor(
157    enclave_pubkey: &BitcoinPubkey,
158    hashi_master_g: &HashiMasterG,
159    hashi_derivation_path: &DerivationPath,
160) -> Tr<BitcoinPubkey> {
161    let derived_hashi_pubkey = derive_hashi_child_pubkey(hashi_master_g, hashi_derivation_path);
162
163    let internal = *NUMS_INTERNAL_KEY;
164    let recovery_delay = mpc_recovery_delay_sequence().to_consensus_u32();
165
166    // Taproot descriptor with two leaves: immediate 2-of-2 checksigadd-style
167    // multisig, plus delayed MPC-only recovery.
168    // Descriptor docs: https://github.com/bitcoin/bitcoin/blob/master/doc/descriptors.md
169    let desc_str = format!(
170        "tr({},{{multi_a(2,{},{}),and_v(v:older({}),pk({}))}})",
171        internal, enclave_pubkey, derived_hashi_pubkey, recovery_delay, derived_hashi_pubkey,
172    );
173
174    match Descriptor::<BitcoinPubkey>::from_str(&desc_str).expect("valid descriptor") {
175        Descriptor::Tr(tr) => tr,
176        _ => panic!("unexpected descriptor"),
177    }
178}

180/// BIP68 sequence encoding the [`HASHI_MPC_RECOVERY_DELAY_SECONDS`] relative
181/// timelock. Used both in the recovery leaf's `older(...)` and as the input
182/// sequence of any transaction spending through that leaf.
183pub fn mpc_recovery_delay_sequence() -> Sequence {
184    relative::LockTime::from_seconds_ceil(HASHI_MPC_RECOVERY_DELAY_SECONDS)
185        .expect("60 days fits in BIP68 time-based relative locktime")
186        .to_sequence()
187}

189/// Derives the hashi child pubkey at `derivation_path` from `hashi_master_g`.
190///
191/// `hashi_master_g` must be the raw MPC verifying key with y-parity preserved:
192/// `derive_verifying_key` consumes the full `G`, so the x-only/even-y projection
193/// would derive a different child for ~half of all vks and break the 2-of-2 leaf
194/// script. The returned x-only key is exactly what the MPC protocol signs against.
195fn derive_hashi_child_pubkey(
196    hashi_master_g: &HashiMasterG,
197    hashi_derivation_path: &DerivationPath,
198) -> BitcoinPubkey {
199    let derived = derive_verifying_key(hashi_master_g, &hashi_derivation_path.into_inner())
200        .expect("deriving a child verifying key from a valid master key should not fail");
201    BitcoinPubkey::from_slice(&derived.to_byte_array()).expect("derived schnorr key is x-only")
202}

245    fn construct_witness(
246        hashi_signature: &Signature,
247        enclave_signature: &Signature,
248        script: &ScriptBuf,
249        control_block: &ControlBlock,
250    ) -> Witness {
251        // Witness stack order: [sig_for_pk2, sig_for_pk1, script, control_block]
252        // Since our script is <pk1> OP_CHECKSIG <pk2> OP_CHECKSIGADD ...
253        // And stack is LIFO, we need: [hashi_sig, enclave_sig, script, control]
254        let hashi_sig_vec = hashi_signature.to_vec();
255        let enclave_sig_vec = enclave_signature.to_vec();
256        let control_block_vec = control_block.serialize();
257        let witness_elements: Vec<Vec<u8>> = vec![
258            hashi_sig_vec,     // sig for pk2 (hashi)
259            enclave_sig_vec,   // sig for pk1 (enclave)
260            script.to_bytes(), // script
261            control_block_vec, // control block
262        ];
263        Witness::from_slice(&witness_elements)
264    }

308    fn mpc_recovery_delay_is_time_based_csv() {
309        let sequence = mpc_recovery_delay_sequence();
310        // BIP-68 time-based encoding of the 60-day recovery delay: type bit
311        // (1 << 22) OR ceil(60 * 24 * 60 * 60 / 512) = 10_125, i.e.
312        // 0x0040278D = 4_204_429. This literal is baked into every
313        // deposit/change address's recovery leaf; changing it changes all
314        // addresses.
315        assert_eq!(sequence.to_consensus_u32(), (1 << 22) | 10_125);
316        assert!(
317            bitcoin::relative::LockTime::from_sequence(sequence)
318                .expect("sequence should encode a relative locktime")
319                .is_block_time()
320        );
321    }

415    // Regression: an odd-y MPC master key used to silently disagree between
416    // the 2-of-2 leaf script (which reconstructed an even-y parent from the
417    // x-only bytes) and the MPC signature (which signs against
418    // `derive_verifying_key(raw_g, path)`), giving "Invalid Schnorr signature"
419    // at Bitcoin. This builds the real leaf for an odd-y master and asserts it
420    // embeds the MPC-signed child key, not the even-y-forced one.
421    #[test]
422    fn two_of_two_leaf_embeds_mpc_signed_child_for_odd_y_master() {
423        use fastcrypto::groups::GroupElement;
424        use fastcrypto_tbls::threshold_schnorr::S;
425
426        // Find a master G with ODD y so we exercise the formerly-broken branch.
427        let raw_g = loop {
428            let mut bytes = [0u8; 32];
429            rand::Rng::fill(&mut rand::thread_rng(), &mut bytes);
430            let g = HashiMasterG::generator() * S::from_bytes_mod_order(&bytes);
431            if !g.has_even_y().unwrap() {
432                break g;
433            }
434        };
435
436        let enclave_pubkey = create_btc_keypair_for_test(&[7u8; 32])
437            .x_only_public_key()
438            .0;
439        let path = [42u8; 32];
440
441        // The child key the MPC actually produces signatures against.
442        let mpc_child = derive_verifying_key(&raw_g, &path)
443            .expect("derive child key")
444            .to_byte_array();
445        // The child the old even-y-forcing code would have embedded instead.
446        let raw_g_xonly = SchnorrPublicKey::try_from(&raw_g)
447            .expect("valid schnorr key")
448            .to_byte_array();
449        let forced_even =
450            HashiMasterG::with_even_y_from_x_be_bytes(&raw_g_xonly).expect("valid x coordinate");
451        let buggy_child = derive_verifying_key(&forced_even, &path)
452            .expect("derive child key")
453            .to_byte_array();
454        assert_ne!(
455            buggy_child, mpc_child,
456            "odd-y parent must change the child, else this test can't catch the regression"
457        );
458
459        // The production 2-of-2 leaf must embed the MPC-signed child.
460        let (leaf_script, _, _) =
461            taproot_2of2_witness_artifacts(&enclave_pubkey, &raw_g, &DerivationPath::from(path));
462        let script = leaf_script.as_bytes();
463        assert!(
464            script.windows(32).any(|w| w == mpc_child.as_slice()),
465            "2-of-2 leaf must embed the MPC-signed child key"
466        );
467        assert!(
468            !script.windows(32).any(|w| w == buggy_child.as_slice()),
469            "2-of-2 leaf must not embed the even-y-forced child key"
470        );
471    }

607    // Odd-y companion to `cross_lang_2of2_test_vectors`: exercises the path
608    // #609 fixed, where `derive_hashi_child_pubkey` diverges from the legacy
609    // even-y-forcing flow. Seed [4u8;32] is the first scalar in 3..=255 whose
610    // `s·G` has odd y; Rust and the SDK both derive against the raw G.
611    #[test]
612    fn cross_lang_2of2_test_vectors_odd_y() {

614        use fastcrypto::groups::GroupElement;

616
617        const TEST_HASHI_BTC_SK_ODD_Y: [u8; 32] = [4u8; 32];
618
619        let sk = S::from_bytes_mod_order(&TEST_HASHI_BTC_SK_ODD_Y);
620        let master_g = HashiMasterG::generator() * sk;
621        assert!(
622            !master_g.has_even_y().unwrap(),
623            "seed [4u8;32] must land on odd y — if this fires, the upstream curve impl changed",
624        );
625
626        let (enclave_keypair, _) = gen_keypair_and_address(Some(TEST_ENCLAVE_BTC_SK), Regtest);
627        let enclave_pk = enclave_keypair.x_only_public_key().0;

629        let path = DerivationPath::from([1u8; 32]);
630        const EXPECTED_DERIVED: &str =
