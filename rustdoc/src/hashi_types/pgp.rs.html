52impl PgpPublicCert {
53    pub fn new(armored: String) -> Result<Self> {
54        let cert =
55            openpgp::Cert::from_bytes(armored.as_bytes()).context("invalid OpenPGP certificate")?;
56        validate_pgp_cert(&cert)?;
57        Ok(Self { armored, cert })
58    }

60    pub fn armored(&self) -> &str {
61        &self.armored
62    }

64    pub fn fingerprint(&self) -> Fingerprint {
65        self.cert.fingerprint()
66    }
67}

116/// Load and validate each armored OpenPGP cert at `paths`, logging the
117/// fingerprint + path of each. Returns the certs in input order.
118pub fn load_certs(paths: &[PathBuf]) -> Result<Vec<PgpPublicCert>> {
119    let mut certs = Vec::with_capacity(paths.len());
120    for path in paths {
121        let armored = std::fs::read_to_string(path)
122            .with_context(|| format!("failed to read PGP cert at {}", path.display()))?;
123        let cert = PgpPublicCert::new(armored)
124            .with_context(|| format!("invalid PGP cert at {}", path.display()))?;
125        info!(fingerprint = %cert, path = %path.display(), "loaded PGP cert");
126        certs.push(cert);
127    }
128    Ok(certs)
129}

131fn validate_pgp_cert(cert: &openpgp::Cert) -> Result<()> {
132    if cert.keys().secret().next().is_some() {
133        anyhow::bail!("OpenPGP backup certificate must not contain secret key material")
134    }
135
136    cert.keys()
137        .with_policy(&*POLICY, None)
138        .supported()
139        .alive()
140        .revoked(false)
141        .for_transport_encryption()
142        .next()
143        .ok_or_else(|| anyhow::anyhow!("OpenPGP certificate has no usable encryption key"))?;
144
145    cert.keys()
146        .with_policy(&*POLICY, None)
147        .supported()
148        .alive()
149        .revoked(false)
150        .for_signing()
151        .next()
152        .ok_or_else(|| anyhow::anyhow!("OpenPGP certificate has no usable signing key"))?;
153    Ok(())
154}

156pub fn encrypt_armored(plaintext: &[u8], cert: &PgpPublicCert) -> Result<String> {
157    let mut ciphertext = Vec::new();
158    let mut writer = armored_encrypt_writer(&mut ciphertext, cert)?;
159    writer
160        .write_all(plaintext)
161        .context("OpenPGP encryption failed")?;
162    writer
163        .finalize()
164        .context("OpenPGP encryption finalization failed")?;
165    String::from_utf8(ciphertext).context("OpenPGP ASCII armor was not valid UTF-8")
166}

168pub fn armored_encrypt_writer<'a, W>(output: W, cert: &'a PgpPublicCert) -> Result<Message<'a>>
169where
170    W: 'a + Write + Send + Sync,
171{
172    let recipients = cert
173        .cert
174        .keys()
175        .with_policy(&*POLICY, None)
176        .supported()
177        .alive()
178        .revoked(false)
179        .for_transport_encryption();
180    let message = Message::new(output);
181    let message = Armorer::new(message)
182        .kind(openpgp::armor::Kind::Message)
183        .build()
184        .context("OpenPGP armor setup failed")?;
185    let message = Encryptor::for_recipients(message, recipients)
186        .build()
187        .context("OpenPGP encryption setup failed")?;
188    // Compress inside the encryption layer so the archive is stored as a
189    // single OpenPGP message; the decryptor inflates this transparently.
190    let message = Compressor::new(message)
191        .build()
192        .context("OpenPGP compression setup failed")?;
193    LiteralWriter::new(message)
194        .build()
195        .context("OpenPGP literal data setup failed")
196}

198/// Recipient key handles found in an armored OpenPGP message's PKESK packets,
199/// parsed WITHOUT decrypting. One entry per recipient encryption key. Use to
200/// confirm a ciphertext is addressed to the expected cert without holding its
201/// secret key (e.g. a yubikey-bound key, which can't be inspected in memory).
202///
203/// Rejects anonymous/hidden recipients: if the recipient key handle is omitted,
204/// callers cannot prove the ciphertext is addressed only to the expected cert.
205pub fn pgp_message_recipients(armored: &str) -> Result<Vec<openpgp::KeyHandle>> {
206    use openpgp::parse::PacketParser;
207    use openpgp::parse::PacketParserResult;
208    use openpgp::parse::Parse;
209
210    let mut handles = Vec::new();
211    let mut ppr =
212        PacketParser::from_bytes(armored.as_bytes()).context("invalid OpenPGP message")?;
213    while let PacketParserResult::Some(pp) = ppr {
214        // `recurse` extracts the current packet (owned) and yields the next
215        // parser result; it also descends into nested structures. For an
216        // encrypted message the PKESKs sit at the top level and the SEIP body
217        // is opaque (still ciphertext), so only the recipients are visible.
218        let (packet, next_ppr) = pp.recurse().context("parsing OpenPGP packet stream")?;
219        if let openpgp::Packet::PKESK(pkesk) = packet {
220            let handle = pkesk
221                .recipient()
222                .ok_or_else(|| anyhow::anyhow!("OpenPGP message has an anonymous recipient"))?;
223            handles.push(handle);
224        }
225        ppr = next_ppr;
226    }
227    Ok(handles)
228}

230/// True if `cert` owns `handle` — i.e. `handle` aliases one of the cert's
231/// (primary or sub) keys. Handles the KeyID-vs-Fingerprint suffix matching so a
232/// PKESK's 8-byte KeyID matches a cert key's full fingerprint.
233pub fn cert_owns_key_handle(cert: &PgpPublicCert, handle: &openpgp::KeyHandle) -> bool {
234    cert.cert
235        .keys()
236        .any(|k| key_handles_alias(&k.key().key_handle(), handle))
237}
238
239fn key_handles_alias(a: &openpgp::KeyHandle, b: &openpgp::KeyHandle) -> bool {
240    a.aliases(b) || b.aliases(a)
241}

243pub fn decrypt_with_secret_key<R>(input: R, secret_key: &[u8]) -> Result<impl io::Read + use<R>>
244where
245    R: Read + Send + Sync + 'static,
246{
247    let cert = openpgp::Cert::from_bytes(secret_key).context("invalid OpenPGP secret key")?;
248    ensure_usable_unencrypted_secret_key(&cert)?;
249    let helper = LocalSecretKeyDecryptHelper { cert };
250    DecryptorBuilder::from_reader(input)
251        .context("invalid OpenPGP message")?
252        .with_policy(&*POLICY, None, helper)
253        .context("OpenPGP decryption failed")
254}

256/// Decrypt `backup` by shelling out to `gpg --decrypt`, returning a reader that
257/// streams the plaintext.
258///
259/// The spawned `gpg` process is owned by the returned reader: reaching EOF waits
260/// for `gpg` and turns a non-zero exit into an `io::Error`, and dropping the
261/// reader early kills and reaps `gpg` so it can never be orphaned or deadlock on
262/// a full pipe. Callers only deal with `io::Read`.
263pub fn decrypt_with_gpg(backup: &Path, homedir: Option<&Path>) -> Result<impl io::Read + use<>> {
264    let mut command = Command::new("gpg");
265    command.arg("--decrypt").arg("--").arg(backup);
266    if let Some(homedir) = homedir {
267        command.env("GNUPGHOME", homedir);
268    }
269    command.stdin(Stdio::null());
270    command.stdout(Stdio::piped());
271    command.stderr(Stdio::inherit());
272    let mut child = command
273        .spawn()
274        .with_context(|| "Failed to run `gpg --decrypt`; is gpg installed and on PATH?")?;
275    let stdout = child
276        .stdout
277        .take()
278        .ok_or_else(|| anyhow::anyhow!("Failed to capture gpg stdout"))?;
279    Ok(GpgDecryptReader {
280        child,
281        stdout,
282        writer: None,
283        finished: false,
284    })
285}

287/// Like [`decrypt_with_gpg`], but feeds `ciphertext` to gpg over its stdin
288/// instead of a file path. A background writer thread writes the ciphertext so
289/// reading decrypted bytes off stdout can proceed concurrently without
290/// deadlocking when the ciphertext exceeds the OS pipe buffer (~64 KB on
291/// Linux). The thread owns its `ChildStdin` and drops it on completion,
292/// signalling EOF to gpg. The reader joins the writer in `Drop` so it can't
293/// outlive it.
294pub fn decrypt_with_gpg_stdin(
295    ciphertext: &[u8],
296    homedir: Option<&Path>,
297) -> Result<impl io::Read + use<>> {
298    let mut command = Command::new("gpg");
299    command.arg("--decrypt").arg("--");
300    if let Some(homedir) = homedir {
301        command.env("GNUPGHOME", homedir);
302    }
303    command.stdin(Stdio::piped());
304    command.stdout(Stdio::piped());
305    command.stderr(Stdio::inherit());
306    let mut child = command
307        .spawn()
308        .with_context(|| "Failed to run `gpg --decrypt`; is gpg installed and on PATH?")?;
309    let mut stdin = child
310        .stdin
311        .take()
312        .ok_or_else(|| anyhow::anyhow!("Failed to open gpg stdin"))?;
313    let stdout = child
314        .stdout
315        .take()
316        .ok_or_else(|| anyhow::anyhow!("Failed to capture gpg stdout"))?;
317
318    // Copy the ciphertext onto the writer thread so the caller does not need to
319    // keep its buffer alive while gpg runs.
320    let ciphertext_owned = ciphertext.to_vec();
321    let writer = std::thread::Builder::new()
322        .name("gpg-stdin-writer".into())
323        .spawn(move || {
324            let _ = stdin.write_all(&ciphertext_owned);
325            // Drop stdin so gpg sees EOF and finishes its stdout stream.
326            drop(stdin);
327        })
328        .context("spawn gpg stdin writer thread")?;
329
330    Ok(GpgDecryptReader {
331        child,
332        stdout,
333        writer: Some(writer),
334        finished: false,
335    })
336}

338/// Decrypt an armored OpenPGP message string via `gpg --decrypt`, returning the
339/// plaintext bytes. The ciphertext is piped to gpg over its stdin (a background
340/// writer thread) and the plaintext streams back over gpg's stdout pipe into
341/// memory — nothing touches disk.
342pub fn decrypt_armored_via_gpg(armored: &str, homedir: Option<&Path>) -> Result<Vec<u8>> {
343    let mut decryptor = decrypt_with_gpg_stdin(armored.as_bytes(), homedir)?;
344    let mut plaintext = Vec::new();
345    decryptor
346        .read_to_end(&mut plaintext)
347        .context("read decrypted bytes from gpg")?;
348    Ok(plaintext)
349}

351/// Produce an armored detached OpenPGP signature over `payload` with the local
352/// gpg key selected by `signer_fingerprint` (`gpg --local-user`) — in
353/// production the KP's offline key (e.g. a yubikey).
354pub fn sign_detached_via_gpg(
355    payload: &[u8],
356    signer_fingerprint: &Fingerprint,
357    homedir: Option<&Path>,
358) -> Result<String> {
359    let mut command = Command::new("gpg");
360    command
361        .arg("--local-user")
362        .arg(signer_fingerprint.to_hex())
363        .arg("--armor")
364        .arg("--detach-sign");
365    if let Some(homedir) = homedir {
366        command.env("GNUPGHOME", homedir);
367    }
368    command.stdin(Stdio::piped());
369    command.stdout(Stdio::piped());
370    command.stderr(Stdio::inherit());
371    let mut child = command
372        .spawn()
373        .context("Failed to run `gpg --detach-sign`; is gpg installed and on PATH?")?;
374    // Dropping the `ChildStdin` below signals EOF, making gpg emit the signature.
375    let write_result = match child.stdin.take() {
376        Some(mut stdin) => stdin.write_all(payload),
377        None => Err(io::Error::other("failed to open gpg stdin")),
378    };
379    // Reap gpg even if the write failed (EPIPE = gpg died early); its exit
380    // status is the more useful error, so check it first.
381    let output = child
382        .wait_with_output()
383        .context("wait for `gpg --detach-sign`")?;
384    if !output.status.success() {
385        anyhow::bail!("`gpg --detach-sign` exited with status {}", output.status);
386    }
387    write_result.context("write payload to gpg stdin")?;
388    String::from_utf8(output.stdout).context("gpg produced a non-UTF8 armored signature")
389}

391/// A `VerificationHelper` that accepts exactly one signer cert.
392struct DetachedSigVerifier<'a> {
393    cert: &'a openpgp::Cert,
394}
395
396impl VerificationHelper for DetachedSigVerifier<'_> {
397    fn get_certs(&mut self, _ids: &[openpgp::KeyHandle]) -> openpgp::Result<Vec<openpgp::Cert>> {
398        Ok(vec![self.cert.clone()])
399    }
400
401    fn check(&mut self, structure: MessageStructure) -> openpgp::Result<()> {
402        for layer in structure.into_iter() {
403            if let MessageLayer::SignatureGroup { results } = layer
404                && results.iter().any(|r| r.is_ok())
405            {
406                return Ok(());
407            }
408        }
409        anyhow::bail!("no valid signature from the expected signer cert")
410    }
411}

413/// Verify an armored detached OpenPGP `signature` over `payload` was produced
414/// by `cert`. Pure Rust (no gpg subprocess), so it runs inside the distroless
415/// proxy image.
416pub fn verify_detached_signature(
417    payload: &[u8],
418    signature: &str,
419    cert: &PgpPublicCert,
420) -> Result<()> {
421    let helper = DetachedSigVerifier { cert: &cert.cert };
422    let mut verifier = DetachedVerifierBuilder::from_bytes(signature.as_bytes())
423        .context("parse detached signature")?
424        .with_policy(&*POLICY, None, helper)
425        .context("initialize detached verifier")?;
426    verifier
427        .verify_bytes(payload)
428        .context("detached signature verification failed")
429}

431struct GpgDecryptReader {
432    child: Child,
433    stdout: ChildStdout,
434    /// Writer thread feeding gpg's stdin, present only for the stdin variant.
435    writer: Option<std::thread::JoinHandle<()>>,
436    finished: bool,
437}
438
439impl io::Read for GpgDecryptReader {
440    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
441        let n = self.stdout.read(buf)?;
442        if n == 0 && !self.finished {
443            self.finished = true;
444            let status = self.child.wait()?;
445            if !status.success() {
446                return Err(io::Error::other(format!("gpg failed to decrypt: {status}")));
447            }
448        }
449        Ok(n)
450    }
451}
452
453impl Drop for GpgDecryptReader {
454    fn drop(&mut self) {
455        if !self.finished {
456            // The consumer stopped early (e.g. a tar/gzip parse error on
457            // truncated output). Kill and reap gpg so it cannot linger writing
458            // into a pipe nobody is reading. This also unblocks the writer
459            // thread's `write_all` (gpg closing stdin returns EPIPE).
460            let _ = self.child.kill();
461            let _ = self.child.wait();
462        }
463        if let Some(writer) = self.writer.take() {
464            // Make sure the stdin writer has flushed or observed gpg closing
465            // its pipe before the reader is fully torn down.
466            let _ = writer.join();
467        }
468    }
469}

471fn ensure_usable_unencrypted_secret_key(cert: &openpgp::Cert) -> Result<()> {
472    let mut has_transport_secret_key = false;
473    let mut has_unencrypted_transport_secret_key = false;
474
475    for key in cert
476        .keys()
477        .secret()
478        .with_policy(&*POLICY, None)
479        .supported()
480        .alive()
481        .revoked(false)
482        .for_transport_encryption()
483    {
484        has_transport_secret_key = true;
485        if key.key().has_unencrypted_secret() {
486            has_unencrypted_transport_secret_key = true;
487            break;
488        }
489    }
490
491    if has_unencrypted_transport_secret_key {
492        Ok(())
493    } else if has_transport_secret_key {
494        anyhow::bail!(
495            "OpenPGP secret key has no unencrypted transport encryption key. Passphrase-protected secret keys are not supported with --backup-pgp-secret-key; use --use-gpg-agent instead."
496        )
497    } else {
498        anyhow::bail!("OpenPGP secret key has no usable transport encryption key")
499    }
500}

508        Ok(Vec::new())
509    }

512        Ok(())

515
516impl DecryptionHelper for LocalSecretKeyDecryptHelper {
517    fn decrypt(
518        &mut self,
519        pkesks: &[openpgp::packet::PKESK],
520        _skesks: &[openpgp::packet::SKESK],
521        sym_algo: Option<SymmetricAlgorithm>,
522        decrypt: &mut dyn FnMut(Option<SymmetricAlgorithm>, &SessionKey) -> bool,
523    ) -> openpgp::Result<Option<openpgp::Cert>> {
524        for key in self
525            .cert
526            .keys()
527            .secret()
528            .with_policy(&*POLICY, None)
529            .supported()
530            .alive()
531            .revoked(false)
532            .for_transport_encryption()
533        {
534            let key = key.key().clone();
535            if !key.has_unencrypted_secret() {
536                continue;
537            }
538            let mut keypair = key.into_keypair()?;
539            for pkesk in pkesks {
540                if pkesk
541                    .decrypt(&mut keypair, sym_algo)
542                    .is_some_and(|(algo, session_key)| decrypt(algo, &session_key))
543                {
544                    return Ok(None);
545                }
546            }
547        }
548        Ok(None)
549    }
550}

675    /// What `sign_detached_via_gpg` produces, minus the gpg keyring.
676    pub fn sign_detached_in_process(secret_armored: &str, payload: &[u8]) -> String {
677        use openpgp::parse::Parse;
678        use openpgp::policy::StandardPolicy;
679        use openpgp::serialize::stream::Armorer;
680        use openpgp::serialize::stream::Message;
681        use openpgp::serialize::stream::Signer;
682        use std::io::Write;
683
684        let policy = StandardPolicy::new();
685        let cert = openpgp::Cert::from_bytes(secret_armored.as_bytes()).unwrap();
686        let keypair = cert
687            .keys()
688            .secret()
689            .with_policy(&policy, None)
690            .for_signing()
691            .next()
692            .unwrap()
693            .key()
694            .clone()
695            .into_keypair()
696            .unwrap();
697
698        let mut sig = Vec::new();
699        let message = Message::new(&mut sig);
700        let message = Armorer::new(message)
701            .kind(openpgp::armor::Kind::Signature)
702            .build()
703            .unwrap();
704        let mut signer = Signer::new(message, keypair)
705            .unwrap()
706            .detached()
707            .build()
708            .unwrap();
709        signer.write_all(payload).unwrap();
710        signer.finalize().unwrap();
711        String::from_utf8(sig).unwrap()
712    }
