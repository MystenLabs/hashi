19/// Load an Ed25519 private key from a file path or inline PEM string.
20///
21/// Supported formats:
22/// - DER-encoded binary file
23/// - PEM-encoded text file
24/// - Inline PEM string (if not a valid file path)
25pub fn load_ed25519_private_key(path_or_pem: &str) -> anyhow::Result<Ed25519PrivateKey> {
26    load_ed25519_private_key_from_path(Path::new(path_or_pem))
27        .or_else(|_| {
28            Ed25519PrivateKey::from_pem(path_or_pem)
29                .map_err(|e| anyhow::anyhow!("PEM parse error: {}", e))
30        })
31        .map_err(|_: anyhow::Error| {
32            anyhow::anyhow!("unable to load Ed25519 private key from '{}'", path_or_pem)
33        })
34}
35
36/// Load an Ed25519 private key from a file path.
37///
38/// Supported formats:
39/// - DER-encoded binary file
40/// - PEM-encoded text file
41pub fn load_ed25519_private_key_from_path(path: &Path) -> anyhow::Result<Ed25519PrivateKey> {
42    let contents = std::fs::read(path)?;
43
44    if let Ok(pk) = Ed25519PrivateKey::from_der(&contents) {
45        return Ok(pk);
46    }
47
48    if let Ok(contents_str) = std::str::from_utf8(&contents)
49        && let Ok(pk) = Ed25519PrivateKey::from_pem(contents_str)
50    {
51        return Ok(pk);
52    }
53
54    anyhow::bail!("unsupported key format in '{}'", path.display())
55}

264    pub fn tls_private_key(&self) -> Result<ed25519_dalek::SigningKey, anyhow::Error> {
265        use ed25519_dalek::pkcs8::DecodePrivateKey;
266
267        let raw = self
268            .tls_private_key
269            .as_ref()
270            .ok_or_else(|| anyhow::anyhow!("no tls_private_key configured"))?;
271
272        if let Ok(private_key) = ed25519_dalek::SigningKey::read_pkcs8_pem_file(raw) {
273            Ok(private_key)
274        } else if let Ok(private_key) = ed25519_dalek::SigningKey::read_pkcs8_der_file(raw) {
275            Ok(private_key)
276        } else if let Ok(private_key) = ed25519_dalek::SigningKey::from_pkcs8_pem(raw) {
277            Ok(private_key)
278        } else {
279            // maybe some other format?
280            Err(anyhow::anyhow!("unable to load tls_private_key"))
281        }
282    }

350    /// Parse configured Bitcoin peer strings into kyoto trusted peers.
351    /// Hostnames are not resolved here — kyoto resolves them at connection
352    /// time, and the monitor's supervisor rebuilds the node on disconnect
353    /// to re-resolve and follow IP changes (e.g., Kubernetes pod rotation).
354    pub fn bitcoin_trusted_peers(&self) -> anyhow::Result<Vec<kyoto::TrustedPeer>> {
355        let Some(peer_strs) = self.bitcoin_trusted_peers.as_ref() else {
356            return Ok(Vec::new());
357        };
358
359        let mut peers = Vec::new();
360        for s in peer_strs {
361            let (host, port_str) = s.rsplit_once(':').ok_or_else(|| {
362                anyhow::anyhow!("Invalid bitcoin peer '{s}': expected 'host:port' format")
363            })?;
364            let port = port_str
365                .parse::<u16>()
366                .map_err(|e| anyhow::anyhow!("Invalid port in bitcoin peer '{s}': {e}"))?;
367            peers.push(kyoto::TrustedPeer::from_hostname(host, port));
368        }
369        Ok(peers)
370    }

440    // Creates a new config suitable for testing. In particular this config will:
441    // - have randomly generated private key material
442    // - localhost only listen addresses using available ports
443    pub fn new_for_testing() -> Self {
444        use ed25519_dalek::pkcs8::EncodePrivateKey;
445        use std::ops::Deref;
446
447        let mut config = Config::default();
448
449        let tls_private_key = ed25519_dalek::SigningKey::generate(&mut rand_core::OsRng);
450
451        config.tls_private_key = Some(
452            tls_private_key
453                .to_pkcs8_pem(ed25519_dalek::pkcs8::spki::der::pem::LineEnding::LF)
454                .unwrap()
455                .deref()
456                .to_owned(),
457        );

459        let listen_addr = SocketAddr::from(([127, 0, 0, 1], get_available_port()));

461        config.endpoint_url = Some(format!("https://{listen_addr}"));
462        config.metrics_http_address =
463            Some(SocketAddr::from(([127, 0, 0, 1], get_available_port())));
464

479/// Return an ephemeral, available port. On unix systems, the port returned will be in the
480/// TIME_WAIT state ensuring that the OS won't hand out this port for some grace period.
481/// Callers should be able to bind to this port given they use SO_REUSEADDR.
482pub fn get_available_port() -> u16 {
483    const MAX_PORT_RETRIES: u32 = 1000;
484
485    for _ in 0..MAX_PORT_RETRIES {
486        if let Ok(port) = get_ephemeral_port() {
487            return port;
488        }
489    }
490
491    panic!("Error: could not find an available port on localhost");
492}
493
494fn get_ephemeral_port() -> std::io::Result<u16> {
495    use std::net::TcpListener;
496    use std::net::TcpStream;
497
498    // Request a random available port from the OS
499    let listener = TcpListener::bind(SocketAddr::from(([127, 0, 0, 1], 0)))?;
500    let addr = listener.local_addr()?;
501
502    // Create and accept a connection (which we'll promptly drop) in order to force the port
503    // into the TIME_WAIT state, ensuring that the port will be reserved from some limited
504    // amount of time (roughly 60s on some Linux systems)
505    let _sender = TcpStream::connect(addr)?;
506    let _incoming = listener.accept()?;
507
508    Ok(addr.port())
509}

516    fn test_new_for_testing() {
517        let config = Config::new_for_testing();
518        let localhost = std::net::Ipv4Addr::new(127, 0, 0, 1);

520        // Test addresses use localhost
521        assert_eq!(config.listen_address().ip(), localhost);
522        assert_eq!(config.metrics_http_address().ip(), localhost);

525        let listen_port = config.listen_address().port();

529        // Test endpoint_url is derived from listen_address
530        let endpoint_url = config.endpoint_url().unwrap();

535        let tls_key = config.tls_private_key.as_ref().unwrap();
