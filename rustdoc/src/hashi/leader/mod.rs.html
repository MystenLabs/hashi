154    /// Invoke a peer's bridge-service signing RPC, retrying on transient
155    /// transport failures. `call` is handed a freshly-fetched client each
156    /// attempt so the retry reconnects (tonic reconnects lazily) rather than
157    /// reusing the connection the peer just tore down. Returns `None` once
158    /// attempts are exhausted or on a non-transport error.
159    async fn call_peer_with_retry<Resp, F, Fut>(
160        inner: &Arc<Hashi>,
161        validator: Address,
162        what: &str,
163        mut call: F,
164    ) -> Option<tonic::Response<Resp>>
165    where
166        F: FnMut(BridgeServiceClient<BoxedChannel>) -> Fut,
167        Fut: Future<Output = Result<tonic::Response<Resp>, tonic::Status>>,
168    {
169        const MAX_ATTEMPTS: u32 = 3;
170        for attempt in 1..=MAX_ATTEMPTS {
171            let Some(client) = inner.onchain_state().bridge_service_client(&validator) else {
172                error!("Cannot find bridge-service client for validator: {validator:?}");
173                return None;
174            };
175            match call(client).await {
176                Ok(response) => return Some(response),
177                Err(status) if attempt < MAX_ATTEMPTS && is_retriable_transport(&status) => {
178                    warn!(
179                        "Failed to get {what} from {validator} (attempt {attempt}/{MAX_ATTEMPTS}): \
180                         {status}; retrying on a fresh connection"
181                    );
182                    tokio::time::sleep(Duration::from_millis(50 * u64::from(attempt))).await;
183                }
184                Err(status) => {
185                    error!("Failed to get {what} from {validator}: {status}");
186                    return None;
187                }
188            }
189        }
190        None
191    }

201    #[tracing::instrument(name = "leader", skip_all)]
202    async fn run(mut self) {
203        info!("Starting leader service");
204
205        // Wait for DKG to complete before processing any checkpoints.
206        let mpc_handle = self.inner.mpc_handle().expect("MpcHandle not initialized");
207        info!("Waiting for MPC key to become available...");
208        mpc_handle.wait_for_key_ready().await;
209        info!("MPC key is ready, starting leader loop");
210
211        let mut checkpoint_rx = self.inner.onchain_state().subscribe_checkpoint();
212        let mut btc_block_rx = self.inner.btc_monitor().subscribe_block_height();
213
214        loop {
215            trace!("Waiting for next checkpoint or task completion...");
216            tokio::select! {
217                wait_result = checkpoint_rx.changed() => {
218                    if let Err(e) = wait_result {
219                        error!("Error waiting for checkpoint change: {e}");
220                        break;
221                    }
222                    let (checkpoint_height, checkpoint_timestamp_ms) = {
223                        let checkpoint_info = checkpoint_rx.borrow_and_update();
224                        (checkpoint_info.height, checkpoint_info.timestamp_ms)
225                    };
226
227                    let is_leader = self.is_current_leader(checkpoint_height);
228                    self.inner.metrics.is_leader.set(i64::from(is_leader));
229                    if is_leader {
230                        debug!("Checkpoint {checkpoint_height}: We are the leader node");
231                    } else {
232                        trace!("We are not the leader node");
233                        continue;
234                    }
235
236                    self.check_reconcile_guardian_committee();
237                    self.process_unapproved_withdrawal_requests(checkpoint_timestamp_ms);
238                    self.process_approved_withdrawal_requests(checkpoint_timestamp_ms);
239                    self.process_unsigned_withdrawal_txns();
240                    self.process_signed_withdrawal_txns();
241                    self.check_delete_expired_deposit_requests(checkpoint_timestamp_ms);
242                    self.check_delete_proposals(checkpoint_timestamp_ms);
243                    self.check_cleanup_spent_utxos();
244                    self.process_approved_deposit_requests();
245                }
246                wait_result = btc_block_rx.changed() => {
247                    if let Err(e) = wait_result {
248                        error!("Error waiting for Bitcoin block height change: {e}");
249                        break;
250                    }
251
252                    let block_height = *btc_block_rx.borrow_and_update();
253                    let checkpoint_height = checkpoint_rx.borrow().height;
254
255                    self.schedule_withdrawal_checks_for_btc_block();
256
257                    if self.is_current_leader(checkpoint_height) {
258                        self.process_deposits_on_bitcoin_block(block_height);
259                    }
260                }
261                Some(result) = self.unapproved_deposit_tasks.join_next() => {
262                    self.handle_completed_unapproved_deposit_task(result);
263                    while let Some(result) = self.unapproved_deposit_tasks.try_join_next() {
264                        self.handle_completed_unapproved_deposit_task(result);
265                    }
266                }
267                Some(result) = self.approved_deposit_tasks.join_next() => {
268                    self.handle_completed_approved_deposit_task(result);
269                    while let Some(result) = self.approved_deposit_tasks.try_join_next() {
270                        self.handle_completed_approved_deposit_task(result);
271                    }
272                }
273                Some(result) = self.withdrawal_signing_tasks.join_next() => {
274                    self.handle_completed_withdrawal_signing_task(result);
275                }
276                Some(result) = self.withdrawal_broadcast_tasks.join_next() => {
277                    self.handle_completed_withdrawal_broadcast_task(result);
278                }
279                Some(result) = self.withdrawal_btc_block_check_tasks.join_next() => {
280                    self.handle_completed_withdrawal_btc_block_check_task(result);
281                }
282                Some(result) = OptionFuture::from(self.withdrawal_approval_task.as_mut()) => {
283                    self.withdrawal_approval_task = None;
284                    Self::log_task_result("withdrawal_approval", result);
285                }
286                Some(result) = OptionFuture::from(self.withdrawal_commitment_task.as_mut()) => {
287                    self.withdrawal_commitment_task = None;
288                    Self::log_task_result("withdrawal_commitment", result);
289                }
290                Some(result) = OptionFuture::from(self.deposit_gc_task.as_mut()) => {
291                    self.deposit_gc_task = None;
292                    Self::log_task_result("deposit_gc", result);
293                }
294                Some(result) = OptionFuture::from(self.proposal_gc_task.as_mut()) => {
295                    self.proposal_gc_task = None;
296                    Self::log_task_result("proposal_gc", result);
297                }
298                Some(result) = OptionFuture::from(self.utxo_cleanup_gc_task.as_mut()) => {
299                    self.utxo_cleanup_gc_task = None;
300                    Self::log_task_result("utxo_cleanup_gc", result);
301                    self.utxo_cleanup_scan_needed = true;
302                    self.check_cleanup_spent_utxos();
303                }
304                Some(result) = OptionFuture::from(self.guardian_committee_reconcile_task.as_mut()) => {
305                    self.guardian_committee_reconcile_task = None;
306                    // On failure, clear the epoch gate so the next tick retries
307                    // (e.g. transient guardian downtime); success holds the gate
308                    // until the hashi epoch advances again.
309                    if !matches!(&result, Ok(Ok(()))) {
310                        self.last_guardian_reconcile_epoch = None;
311                    }
312                    Self::log_task_result("guardian_committee_reconcile", result);
313                }
314
315            }
316        }
317    }

319    fn log_task_result(label: &str, result: Result<anyhow::Result<()>, tokio::task::JoinError>) {
320        match result {
321            Ok(Ok(())) => {}
322            Ok(Err(err)) => error!("{label} task failed: {err:#?}"),
323            Err(err) if err.is_panic() => error!("{label} task panicked: {err}"),
324            Err(err) => error!("{label} task failed to join: {err}"),
325        }
326    }

338    fn is_current_leader(&self, checkpoint_height: u64) -> bool {
339        if self.inner.onchain_state().state().hashi().config.paused() {
340            debug!("Bridge is paused, not acting as leader");
341            return false;
342        }
343
344        match self.inner.config.force_run_as_leader() {
345            ForceRunAsLeader::Always => return true,
346            ForceRunAsLeader::Never => return false,
347            ForceRunAsLeader::Default => (),
348        }
349
350        let Some(committee) = self.inner.onchain_state().current_committee() else {
351            // TODO: do we need to do anything when bootstrapping? At genesis there is no committee.
352            return false;
353        };
354        let this_validator_address = self
355            .inner
356            .config
357            .validator_address()
358            .expect("No configured validator address");
359        let Some(this_validator_idx) = committee
360            .index_of(&this_validator_address)
361            .map(|i| i as u64)
362        else {
363            // We are not in the committee yet, so we cannot be the leader
364            return false;
365        };
366        let num_validators = committee.members().len() as u64;
367
368        let current_turn = checkpoint_height / NUM_CONSECUTIVE_LEADER_CHECKPOINTS;
369        let is_leader = (current_turn % num_validators) == this_validator_idx;
370
371        trace!("Node index {this_validator_idx} is leader node: {is_leader}");
372        is_leader
373    }

376fn parse_member_signature(
377    member_signature: hashi_types::proto::MemberSignature,
378) -> anyhow::Result<MemberSignature> {
379    let epoch = member_signature
380        .epoch
381        .ok_or(anyhow::anyhow!("No epoch in MemberSignature"))?;
382    let address_string = member_signature
383        .address
384        .ok_or(anyhow::anyhow!("No address in MemberSignature"))?;
385    let address = address_string
386        .parse::<Address>()
387        .map_err(|e| anyhow::anyhow!("Unable to parse Address: {}", e))?;
388    let signature = BLS12381Signature::from_bytes(
389        member_signature
390            .signature
391            .ok_or(anyhow::anyhow!("No signature in MemberSignature"))?
392            .as_bytes(),
393    )?;
394    Ok(MemberSignature::new(epoch, address, signature))
395}

397/// Whether a failed peer RPC is worth retrying. Under sustained load a peer's
398/// HTTP/2 server tears the whole multiplexed connection down — `GoAway`
399/// (surfaced as `Internal`), broken pipe (`Unknown`), or the usual
400/// `Unavailable` — failing every in-flight request to that peer at once. The
401/// peer signing RPCs are idempotent, so retrying these transport-class codes is
402/// safe and lets the request land on a fresh connection.
403fn is_retriable_transport(status: &tonic::Status) -> bool {
404    matches!(
405        status.code(),
406        tonic::Code::Unavailable
407            | tonic::Code::Unknown
408            | tonic::Code::Internal
409            | tonic::Code::Cancelled
410            | tonic::Code::DeadlineExceeded
411    )
412}

414#[cfg(test)]
415mod tests {
416    use super::is_retriable_transport;
417    use tonic::Code;
418    use tonic::Status;
419
420    #[test]
421    fn classifies_transport_errors_as_retriable() {
422        // A peer tearing the connection down surfaces as these codes: GoAway ->
423        // Internal, broken pipe -> Unknown, plus the usual transient codes.
424        for code in [
425            Code::Unavailable,
426            Code::Unknown,
427            Code::Internal,
428            Code::Cancelled,
429            Code::DeadlineExceeded,
430        ] {
431            assert!(
432                is_retriable_transport(&Status::new(code, "boom")),
433                "{code:?} should be retried"
434            );
435        }
436        // Genuine application rejections from the peer must not be retried.
437        for code in [
438            Code::InvalidArgument,
439            Code::FailedPrecondition,
440            Code::PermissionDenied,
441            Code::NotFound,
442            Code::AlreadyExists,
443        ] {
444            assert!(
445                !is_retriable_transport(&Status::new(code, "nope")),
446                "{code:?} should not be retried"
447            );
448        }
449    }
