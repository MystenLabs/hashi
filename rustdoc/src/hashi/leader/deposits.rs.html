30enum UnapprovedDepositReloadMode {
31    All,
32    StaleEpochApprovalOnly,
33}

35impl LeaderService {
36    pub(super) fn process_deposits_on_bitcoin_block(&mut self) {
37        self.reload_pending_unapproved_deposit_requests(UnapprovedDepositReloadMode::All);
38        self.process_unapproved_deposit_requests();
39    }

41    pub(super) fn process_stale_unapproved_deposits_if_new_epoch(&mut self) {
42        let current_epoch = self.inner.onchain_state().epoch();
43        // This is a local-view optimization, not a correctness check: a stale
44        // view can miss an in-flight reconfig, but Move rejects approval during
45        // reconfig. Avoid doing obviously doomed work when we already know.
46        if !self.is_reconfiguring() && self.last_unapproved_deposit_epoch != Some(current_epoch) {
47            self.reload_pending_unapproved_deposit_requests(
48                UnapprovedDepositReloadMode::StaleEpochApprovalOnly,
49            );
50            self.process_unapproved_deposit_requests();
51            self.last_unapproved_deposit_epoch = Some(current_epoch);
52        }
53    }

55    fn reload_pending_unapproved_deposit_requests(&mut self, mode: UnapprovedDepositReloadMode) {
56        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
57        deposit_requests.sort_by_key(|r| r.created_timestamp_ms);
58        let deposit_ids: HashSet<Address> =
59            deposit_requests.iter().map(|request| request.id).collect();
60        self.inflight_deposits
61            .retain(|deposit_id| deposit_ids.contains(deposit_id));
62        self.never_retry_deposit_ids
63            .retain(|deposit_id| deposit_ids.contains(deposit_id));
64        self.inner
65            .metrics
66            .never_retry_deposit_ids
67            .set(self.never_retry_deposit_ids.len() as i64);
68
69        let current_epoch = self.inner.onchain_state().epoch();
70        self.pending_unapproved_deposit_requests = deposit_requests
71            .into_iter()
72            .filter(|request| !self.never_retry_deposit_ids.contains(&request.id))
73            .filter(|request| match mode {
74                UnapprovedDepositReloadMode::All => request
75                    .approval_cert
76                    .as_ref()
77                    .is_none_or(|cert| cert.epoch != current_epoch),
78                UnapprovedDepositReloadMode::StaleEpochApprovalOnly => request
79                    .approval_cert
80                    .as_ref()
81                    .is_some_and(|cert| cert.epoch != current_epoch),
82            })
83            .collect();
84        debug!(
85            reload_mode = ?mode,
86            pending_unapproved_deposits = self.pending_unapproved_deposit_requests.len(),
87            never_retry_deposits = self.never_retry_deposit_ids.len(),
88            "Reloaded pending unapproved deposit worklist"
89        );
90    }

92    fn process_unapproved_deposit_requests(&mut self) {
93        if self.check_halt_deposit_processing() {
94            return;
95        }
96
97        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
98        for deposit_request in self.pending_unapproved_deposit_requests.clone() {
99            if self.unapproved_deposit_tasks.len() >= max_concurrent {
100                break;
101            }
102            let deposit_id = deposit_request.id;
103            if self.inflight_deposits.contains(&deposit_id) {
104                continue;
105            }
106
107            let inner = self.inner.clone();
108
109            self.inflight_deposits.insert(deposit_id);
110            let task = async move {
111                let task = Self::process_unapproved_deposit(inner, deposit_request);
112                let result = match tokio::time::timeout(LEADER_TASK_TIMEOUT, task).await {
113                    Ok(result) => result,
114                    Err(_) => Err(UnapprovedDepositError::TimedOut(LEADER_TASK_TIMEOUT)),
115                };
116
117                (deposit_id, result)
118            };
119            self.unapproved_deposit_tasks.spawn(task);
120        }
121    }

123    pub(super) fn process_approved_deposit_requests(&mut self) {
124        if self.check_halt_deposit_processing() {
125            return;
126        }
127
128        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
129        let now_ms = self.inner.onchain_state().latest_checkpoint_timestamp_ms();
130        let delay_ms = self.inner.onchain_state().bitcoin_deposit_time_delay_ms();
131        let current_epoch = self.inner.onchain_state().epoch();
132
133        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
134        deposit_requests.sort_by_key(|r| r.created_timestamp_ms);
135        let approved_deposit_requests: Vec<_> = deposit_requests
136            .into_iter()
137            .filter(|request| !self.never_retry_deposit_ids.contains(&request.id))
138            .filter(|request| {
139                request
140                    .approval_cert
141                    .as_ref()
142                    .is_some_and(|cert| cert.epoch == current_epoch)
143            })
144            .collect();
145
146        let approved_deposit_ids: Vec<Address> = approved_deposit_requests
147            .iter()
148            .map(|request| request.id)
149            .collect();
150        self.approved_deposit_retry_tracker
151            .prune(&approved_deposit_ids);
152        self.inner
153            .metrics
154            .leader_items_in_backoff
155            .with_label_values(&["approved_deposit_confirmation"])
156            .set(self.approved_deposit_retry_tracker.in_backoff_count(now_ms) as i64);
157
158        for deposit_request in approved_deposit_requests {
159            let deposit_id = deposit_request.id;
160            if self.inflight_deposits.contains(&deposit_id) {
161                continue;
162            }
163            if self
164                .approved_deposit_retry_tracker
165                .should_skip(&deposit_id, now_ms)
166            {
167                continue;
168            }
169
170            let Some(approved_ms) = deposit_request.approved_timestamp_ms else {
171                warn!(
172                    deposit_id = %deposit_id,
173                    "Skipping deposit confirmation: approval timestamp is missing",
174                );
175                continue;
176            };
177            if approved_ms.saturating_add(delay_ms) > now_ms {
178                trace!(
179                    deposit_id = %deposit_id,
180                    approved_ms,
181                    delay_ms,
182                    now_ms,
183                    "Skipping deposit confirmation: time-delay has not elapsed",
184                );
185                continue;
186            }
187
188            if self.approved_deposit_tasks.len() >= max_concurrent {
189                break;
190            }
191
192            let inner = self.inner.clone();
193            self.inflight_deposits.insert(deposit_id);
194            self.approved_deposit_tasks.spawn(async move {
195                let task = Self::process_approved_deposit(inner, deposit_request);
196                let result = match tokio::time::timeout(LEADER_TASK_TIMEOUT, task).await {
197                    Ok(result) => result,
198                    Err(_) => Err(ApprovedDepositError::TimedOut(LEADER_TASK_TIMEOUT)),
199                };
200
201                (deposit_id, result)
202            });
203        }
204    }

206    pub(super) fn handle_completed_unapproved_deposit_task(
207        &mut self,
208        result: Result<(Address, Result<(), UnapprovedDepositError>), tokio::task::JoinError>,
209    ) {
210        match result {
211            Ok((deposit_id, result)) => {
212                self.inflight_deposits.remove(&deposit_id);
213                self.pending_unapproved_deposit_requests
214                    .retain(|request| request.id != deposit_id);
215                match result {
216                    Ok(()) => {
217                        info!(deposit_id = %deposit_id, "Deposit processed successfully");
218                    }
219                    Err(err) => match err.kind() {
220                        UnapprovedDepositErrorKind::RetryOnNextBlock => {
221                            warn!(deposit_id = %deposit_id, "Deferring deposit retry: {err:#}");
222                        }
223                        UnapprovedDepositErrorKind::NeverRetry => {
224                            self.never_retry_deposit_ids.insert(deposit_id);
225                            self.inner
226                                .metrics
227                                .never_retry_deposit_ids
228                                .set(self.never_retry_deposit_ids.len() as i64);
229                            warn!(deposit_id = %deposit_id, "Marking deposit as never retry: {err:#}");
230                        }
231                    },
232                }
233                self.process_unapproved_deposit_requests();
234            }
235            Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
236            Err(err) => error!("deposit task failed to join: {err}"),
237        }
238    }

240    pub(super) fn handle_completed_approved_deposit_task(
241        &mut self,
242        result: Result<(Address, Result<(), ApprovedDepositError>), tokio::task::JoinError>,
243    ) {
244        match result {
245            Ok((deposit_id, result)) => {
246                self.inflight_deposits.remove(&deposit_id);
247                match result {
248                    Ok(()) => {
249                        self.approved_deposit_retry_tracker.clear(&deposit_id);
250                        info!(deposit_id = %deposit_id, "Deposit processed successfully");
251                    }
252                    Err(err) => {
253                        if !self.inner.onchain_state().has_deposit_request(&deposit_id) {
254                            self.approved_deposit_retry_tracker.clear(&deposit_id);
255                            info!(deposit_id = %deposit_id, "Deposit confirmation task failed after request left the queue");
256                            return;
257                        }
258
259                        if err.to_string().contains("checkpoint wait timed out") {
260                            self.approved_deposit_retry_tracker.clear(&deposit_id);
261                            warn!(deposit_id = %deposit_id, "Deposit confirmation checkpoint wait timed out; retrying without backoff");
262                            return;
263                        }
264
265                        let kind = err.kind();
266                        self.inner
267                            .metrics
268                            .leader_retries_total
269                            .with_label_values(&[
270                                "approved_deposit_confirmation",
271                                &format!("{kind:?}"),
272                            ])
273                            .inc();
274                        self.approved_deposit_retry_tracker.record_failure(
275                            kind,
276                            deposit_id,
277                            self.inner.onchain_state().latest_checkpoint_timestamp_ms(),
278                        );
279                    }
280                }
281            }
282            Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
283            Err(err) => error!("deposit task failed to join: {err}"),
284        }
285    }

287    fn check_halt_deposit_processing(&mut self) -> bool {
288        if !(self.inner.onchain_state().state().hashi().config.paused() || self.is_reconfiguring())
289        {
290            return false;
291        }
292
293        self.unapproved_deposit_tasks.abort_all();
294        self.approved_deposit_tasks.abort_all();
295        self.pending_unapproved_deposit_requests.clear();
296        self.inflight_deposits.clear();
297        true
298    }

300    async fn process_unapproved_deposit(
301        inner: Arc<Hashi>,
302        deposit_request: DepositRequest,
303    ) -> Result<(), UnapprovedDepositError> {
304        info!("Approving deposit request");
305
306        // Validate deposit_request before asking for signatures
307        inner
308            .validate_deposit_request(&deposit_request)
309            .await
310            .inspect_err(|err| debug!("Deposit validation failed: {err}"))?;
311
312        info!("Deposit request validated successfully");
313
314        let proto_request = deposit_request_to_proto(&deposit_request);
315        let members = inner
316            .onchain_state()
317            .current_committee_members()
318            .expect("No current committee members");
319
320        let committee = inner
321            .onchain_state()
322            .current_committee()
323            .expect("No current committee");
324
325        let required_weight = certificate_threshold(committee.total_weight());
326
327        // Fan out signature requests to all members in parallel.
328        let mut sig_tasks = JoinSet::new();
329        for member in members {
330            let inner = inner.clone();
331            let proto_request = proto_request.clone();
332            sig_tasks.spawn(async move {
333                Self::request_deposit_confirmation_signature(&inner, proto_request, &member).await
334            });
335        }
336
337        // Collect signatures, stopping once we reach quorum.
338        let confirmation_message = DepositConfirmationMessage {
339            request_id: deposit_request.id,
340            utxo: deposit_request.utxo.clone(),
341        };
342        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation_message);
343        while let Some(result) = sig_tasks.join_next().await {
344            let Ok(Some(sig)) = result else { continue };
345            if let Err(e) = aggregator.add_signature(sig) {
346                error!("Failed to add deposit signature: {e}");
347            }
348            if aggregator.weight() >= required_weight {
349                break;
350            }
351        }
352
353        if aggregator.weight() < required_weight {
354            return Err(UnapprovedDepositError::FailedQuorum {
355                weight: aggregator.weight(),
356                required_weight,
357            });
358        }
359
360        let signed_message = match aggregator.finish() {
361            Ok(signed_message) => signed_message,
362            Err(err) => return Err(UnapprovedDepositError::CertificateBuildFailed(err.into())),
363        };
364        let mut executor = match SuiTxExecutor::from_hashi(inner.clone()) {
365            Ok(executor) => executor,
366            Err(err) => return Err(UnapprovedDepositError::ExecutorInitFailed(err)),
367        };
368        executor
369            .execute_approve_deposit(&deposit_request, signed_message)
370            .await
371            .inspect(|()| {
372                inner
373                    .metrics
374                    .sui_tx_submissions_total
375                    .with_label_values(&["approve_deposit", "success"])
376                    .inc();
377                info!("Successfully submitted deposit approval");
378            })
379            .inspect_err(|e| {
380                error!("Failed to submit deposit approval: {e}");
381                inner
382                    .metrics
383                    .sui_tx_submissions_total
384                    .with_label_values(&["approve_deposit", "failure"])
385                    .inc();
386            })
387            .map_err(UnapprovedDepositError::ApproveDepositFailed)?;
388        Ok(())
389    }

391    /// Submit `confirm_deposit` for a deposit that has already been
392    /// approved on-chain and whose time-delay window has elapsed. The
393    /// caller (`process_approved_deposit_requests`) checks the delay before
394    /// scheduling the task.
395    async fn process_approved_deposit(
396        inner: Arc<Hashi>,
397        deposit_request: DepositRequest,
398    ) -> Result<(), ApprovedDepositError> {
399        info!("Confirming approved deposit request");
400
401        let mut executor = match SuiTxExecutor::from_hashi(inner.clone()) {
402            Ok(executor) => executor,
403            Err(err) => return Err(ApprovedDepositError::ExecutorInitFailed(err)),
404        };
405        executor
406            .execute_confirm_deposit(deposit_request.id)
407            .await
408            .inspect(|()| {
409                inner
410                    .metrics
411                    .sui_tx_submissions_total
412                    .with_label_values(&["confirm_deposit", "success"])
413                    .inc();
414                inner.metrics.deposits_confirmed_total.inc();
415                info!("Successfully submitted deposit confirmation");
416            })
417            .inspect_err(|e| {
418                error!("Failed to submit deposit confirmation: {e}");
419                inner
420                    .metrics
421                    .sui_tx_submissions_total
422                    .with_label_values(&["confirm_deposit", "failure"])
423                    .inc();
424            })
425            .map_err(ApprovedDepositError::ConfirmDepositFailed)?;
426        Ok(())
427    }

429    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
430    async fn request_deposit_confirmation_signature(
431        inner: &Arc<Hashi>,
432        proto_request: SignDepositConfirmationRequest,
433        member: &CommitteeMember,
434    ) -> Option<MemberSignature> {
435        let validator_address = member.validator_address();
436        trace!("Requesting deposit confirmation signature");
437
438        let mut rpc_client = inner
439            .onchain_state()
440            .bridge_service_client(&validator_address)
441            .or_else(|| {
442                error!(
443                    "Cannot find client for validator address: {:?}",
444                    validator_address
445                );
446                None
447            })?;
448
449        let response = rpc_client
450            .sign_deposit_confirmation(proto_request.clone())
451            .await
452            .inspect_err(|e| {
453                error!(
454                    "Failed to get deposit confirmation signature from {}: {e}",
455                    validator_address
456                );
457            })
458            .ok()?;
459
460        trace!(
461            "Retrieved deposit confirmation signature from {}",
462            validator_address
463        );
464
465        response
466            .into_inner()
467            .member_signature
468            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
469            .and_then(parse_member_signature)
470            .inspect_err(|e| {
471                error!(
472                    "Failed to parse member signature from response from {}: {e}",
473                    validator_address
474                );
475            })
476            .ok()
477    }

480fn deposit_request_to_proto(req: &DepositRequest) -> SignDepositConfirmationRequest {
481    SignDepositConfirmationRequest {
482        id: req.id.as_bytes().to_vec().into(),
483        txid: req.utxo.id.txid.as_bytes().to_vec().into(),
484        vout: req.utxo.id.vout,
485        amount: req.utxo.amount,
486        derivation_path: req
487            .utxo
488            .derivation_path
489            .map(|p| p.as_bytes().to_vec().into()),
490        timestamp_ms: req.created_timestamp_ms,
491        requester_address: req.sender.as_bytes().to_vec().into(),
492        sui_tx_digest: req.sui_tx_digest.as_bytes().to_vec().into(),
493    }
