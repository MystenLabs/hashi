30    pub(super) fn process_deposits_on_bitcoin_block(&mut self, block_height: u32) {
31        debug!("New Bitcoin block {block_height}: processing deposit requests");
32        self.reload_pending_unapproved_deposit_requests();
33        self.process_unapproved_deposit_requests();
34    }
35
36    fn reload_pending_unapproved_deposit_requests(&mut self) {
37        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
38        deposit_requests.sort_by_key(|r| r.timestamp_ms);
39        let deposit_ids: HashSet<Address> =
40            deposit_requests.iter().map(|request| request.id).collect();
41        self.inflight_deposits
42            .retain(|deposit_id| deposit_ids.contains(deposit_id));
43        self.never_retry_deposit_ids
44            .retain(|deposit_id| deposit_ids.contains(deposit_id));
45        self.inner
46            .metrics
47            .never_retry_deposit_ids
48            .set(self.never_retry_deposit_ids.len() as i64);
49
50        let current_epoch = self.inner.onchain_state().epoch();
51        self.pending_unapproved_deposit_requests = deposit_requests
52            .into_iter()
53            .filter(|request| !self.never_retry_deposit_ids.contains(&request.id))
54            .filter(|request| {
55                request
56                    .approval_cert
57                    .as_ref()
58                    .is_none_or(|cert| cert.epoch != current_epoch)
59            })
60            .collect();
61        debug!(
62            pending_unapproved_deposits = self.pending_unapproved_deposit_requests.len(),
63            never_retry_deposits = self.never_retry_deposit_ids.len(),
64            "Reloaded pending unapproved deposit worklist"
65        );
66    }

68    fn process_unapproved_deposit_requests(&mut self) {
69        if self.check_halt_deposit_processing() {
70            return;
71        }
72
73        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
74        for deposit_request in self.pending_unapproved_deposit_requests.clone() {
75            if self.unapproved_deposit_tasks.len() >= max_concurrent {
76                break;
77            }
78            let deposit_id = deposit_request.id;
79            if self.inflight_deposits.contains(&deposit_id) {
80                continue;
81            }
82
83            let inner = self.inner.clone();
84
85            self.inflight_deposits.insert(deposit_id);
86            let task = async move {
87                let task = Self::process_unapproved_deposit(inner, deposit_request);
88                let result = match tokio::time::timeout(LEADER_TASK_TIMEOUT, task).await {
89                    Ok(result) => result,
90                    Err(_) => Err(UnapprovedDepositError::TimedOut(LEADER_TASK_TIMEOUT)),
91                };
92
93                (deposit_id, result)
94            };
95            self.unapproved_deposit_tasks.spawn(task);
96        }
97    }

99    pub(super) fn process_approved_deposit_requests(&mut self) {
100        if self.check_halt_deposit_processing() {
101            return;
102        }
103
104        let max_concurrent = self.inner.config.max_concurrent_leader_job_tasks();
105        let now_ms = self.inner.onchain_state().latest_checkpoint_timestamp_ms();
106        let delay_ms = self.inner.onchain_state().bitcoin_deposit_time_delay_ms();
107        let current_epoch = self.inner.onchain_state().epoch();
108
109        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
110        deposit_requests.sort_by_key(|r| r.timestamp_ms);
111        let approved_deposit_requests: Vec<_> = deposit_requests
112            .into_iter()
113            .filter(|request| !self.never_retry_deposit_ids.contains(&request.id))
114            .filter(|request| {
115                request
116                    .approval_cert
117                    .as_ref()
118                    .is_some_and(|cert| cert.epoch == current_epoch)
119            })
120            .collect();
121
122        let approved_deposit_ids: Vec<Address> = approved_deposit_requests
123            .iter()
124            .map(|request| request.id)
125            .collect();
126        self.approved_deposit_retry_tracker
127            .prune(&approved_deposit_ids);
128        self.inner
129            .metrics
130            .leader_items_in_backoff
131            .with_label_values(&["approved_deposit_confirmation"])
132            .set(self.approved_deposit_retry_tracker.in_backoff_count(now_ms) as i64);
133
134        for deposit_request in approved_deposit_requests {
135            let deposit_id = deposit_request.id;
136            if self.inflight_deposits.contains(&deposit_id) {
137                continue;
138            }
139            if self
140                .approved_deposit_retry_tracker
141                .should_skip(&deposit_id, now_ms)
142            {
143                continue;
144            }
145
146            let Some(approved_ms) = deposit_request.approval_timestamp_ms else {
147                warn!(
148                    deposit_id = %deposit_id,
149                    "Skipping deposit confirmation: approval timestamp is missing",
150                );
151                continue;
152            };
153            if approved_ms.saturating_add(delay_ms) > now_ms {
154                trace!(
155                    deposit_id = %deposit_id,
156                    approved_ms,
157                    delay_ms,
158                    now_ms,
159                    "Skipping deposit confirmation: time-delay has not elapsed",
160                );
161                continue;
162            }
163
164            if self.approved_deposit_tasks.len() >= max_concurrent {
165                break;
166            }
167
168            let inner = self.inner.clone();
169            self.inflight_deposits.insert(deposit_id);
170            self.approved_deposit_tasks.spawn(async move {
171                let task = Self::process_approved_deposit(inner, deposit_request);
172                let result = match tokio::time::timeout(LEADER_TASK_TIMEOUT, task).await {
173                    Ok(result) => result,
174                    Err(_) => Err(ApprovedDepositError::TimedOut(LEADER_TASK_TIMEOUT)),
175                };
176
177                (deposit_id, result)
178            });
179        }
180    }

182    pub(super) fn handle_completed_unapproved_deposit_task(
183        &mut self,
184        result: Result<(Address, Result<(), UnapprovedDepositError>), tokio::task::JoinError>,
185    ) {
186        match result {
187            Ok((deposit_id, result)) => {
188                self.inflight_deposits.remove(&deposit_id);
189                self.pending_unapproved_deposit_requests
190                    .retain(|request| request.id != deposit_id);
191                match result {
192                    Ok(()) => {
193                        info!(deposit_id = %deposit_id, "Deposit processed successfully");
194                    }
195                    Err(err) => match err.kind() {
196                        UnapprovedDepositErrorKind::RetryOnNextBlock => {
197                            warn!(deposit_id = %deposit_id, "Deferring deposit retry: {err:#}");
198                        }
199                        UnapprovedDepositErrorKind::NeverRetry => {
200                            self.never_retry_deposit_ids.insert(deposit_id);
201                            self.inner
202                                .metrics
203                                .never_retry_deposit_ids
204                                .set(self.never_retry_deposit_ids.len() as i64);
205                            warn!(deposit_id = %deposit_id, "Marking deposit as never retry: {err:#}");
206                        }
207                    },
208                }
209                self.process_unapproved_deposit_requests();
210            }
211            Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
212            Err(err) => error!("deposit task failed to join: {err}"),
213        }
214    }

216    pub(super) fn handle_completed_approved_deposit_task(
217        &mut self,
218        result: Result<(Address, Result<(), ApprovedDepositError>), tokio::task::JoinError>,
219    ) {
220        match result {
221            Ok((deposit_id, result)) => {
222                self.inflight_deposits.remove(&deposit_id);
223                match result {
224                    Ok(()) => {
225                        self.approved_deposit_retry_tracker.clear(&deposit_id);
226                        info!(deposit_id = %deposit_id, "Deposit processed successfully");
227                    }
228                    Err(err) => {
229                        let kind = err.kind();
230                        self.inner
231                            .metrics
232                            .leader_retries_total
233                            .with_label_values(&[
234                                "approved_deposit_confirmation",
235                                &format!("{kind:?}"),
236                            ])
237                            .inc();
238                        self.approved_deposit_retry_tracker.record_failure(
239                            kind,
240                            deposit_id,
241                            self.inner.onchain_state().latest_checkpoint_timestamp_ms(),
242                        );
243                    }
244                }
245            }
246            Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
247            Err(err) => error!("deposit task failed to join: {err}"),
248        }
249    }

251    fn check_halt_deposit_processing(&mut self) -> bool {
252        if !(self.inner.onchain_state().state().hashi().config.paused() || self.is_reconfiguring())
253        {
254            return false;
255        }
256
257        self.unapproved_deposit_tasks.abort_all();
258        self.approved_deposit_tasks.abort_all();
259        self.pending_unapproved_deposit_requests.clear();
260        self.inflight_deposits.clear();
261        true
262    }

264    async fn process_unapproved_deposit(
265        inner: Arc<Hashi>,
266        deposit_request: DepositRequest,
267    ) -> Result<(), UnapprovedDepositError> {
268        info!("Approving deposit request");
269
270        // Validate deposit_request before asking for signatures
271        inner
272            .validate_deposit_request(&deposit_request)
273            .await
274            .inspect_err(|err| debug!("Deposit validation failed: {err}"))?;
275
276        info!("Deposit request validated successfully");
277
278        let proto_request = deposit_request_to_proto(&deposit_request);
279        let members = inner
280            .onchain_state()
281            .current_committee_members()
282            .expect("No current committee members");
283
284        let committee = inner
285            .onchain_state()
286            .current_committee()
287            .expect("No current committee");
288
289        let required_weight = certificate_threshold(committee.total_weight());
290
291        // Fan out signature requests to all members in parallel.
292        let mut sig_tasks = JoinSet::new();
293        for member in members {
294            let inner = inner.clone();
295            let proto_request = proto_request.clone();
296            sig_tasks.spawn(async move {
297                Self::request_deposit_confirmation_signature(&inner, proto_request, &member).await
298            });
299        }
300
301        // Collect signatures, stopping once we reach quorum.
302        let confirmation_message = DepositConfirmationMessage {
303            request_id: deposit_request.id,
304            utxo: deposit_request.utxo.clone(),
305        };
306        let mut aggregator = BlsSignatureAggregator::new(&committee, confirmation_message);
307        while let Some(result) = sig_tasks.join_next().await {
308            let Ok(Some(sig)) = result else { continue };
309            if let Err(e) = aggregator.add_signature(sig) {
310                error!("Failed to add deposit signature: {e}");
311            }
312            if aggregator.weight() >= required_weight {
313                break;
314            }
315        }
316
317        if aggregator.weight() < required_weight {
318            return Err(UnapprovedDepositError::FailedQuorum {
319                weight: aggregator.weight(),
320                required_weight,
321            });
322        }
323
324        let signed_message = match aggregator.finish() {
325            Ok(signed_message) => signed_message,
326            Err(err) => return Err(UnapprovedDepositError::CertificateBuildFailed(err.into())),
327        };
328        let mut executor = match SuiTxExecutor::from_hashi(inner.clone()) {
329            Ok(executor) => executor,
330            Err(err) => return Err(UnapprovedDepositError::ExecutorInitFailed(err)),
331        };
332        executor
333            .execute_approve_deposit(&deposit_request, signed_message)
334            .await
335            .inspect(|()| {
336                inner
337                    .metrics
338                    .sui_tx_submissions_total
339                    .with_label_values(&["approve_deposit", "success"])
340                    .inc();
341                info!("Successfully submitted deposit approval");
342            })
343            .inspect_err(|e| {
344                error!("Failed to submit deposit approval: {e}");
345                inner
346                    .metrics
347                    .sui_tx_submissions_total
348                    .with_label_values(&["approve_deposit", "failure"])
349                    .inc();
350            })
351            .map_err(UnapprovedDepositError::ApproveDepositFailed)?;
352        Ok(())
353    }

355    /// Submit `confirm_deposit` for a deposit that has already been
356    /// approved on-chain and whose time-delay window has elapsed. The
357    /// caller (`process_approved_deposit_requests`) checks the delay before
358    /// scheduling the task.
359    async fn process_approved_deposit(
360        inner: Arc<Hashi>,
361        deposit_request: DepositRequest,
362    ) -> Result<(), ApprovedDepositError> {
363        info!("Confirming approved deposit request");
364
365        let mut executor = match SuiTxExecutor::from_hashi(inner.clone()) {
366            Ok(executor) => executor,
367            Err(err) => return Err(ApprovedDepositError::ExecutorInitFailed(err)),
368        };
369        executor
370            .execute_confirm_deposit(deposit_request.id)
371            .await
372            .inspect(|()| {
373                inner
374                    .metrics
375                    .sui_tx_submissions_total
376                    .with_label_values(&["confirm_deposit", "success"])
377                    .inc();
378                inner.metrics.deposits_confirmed_total.inc();
379                info!("Successfully submitted deposit confirmation");
380            })
381            .inspect_err(|e| {
382                error!("Failed to submit deposit confirmation: {e}");
383                inner
384                    .metrics
385                    .sui_tx_submissions_total
386                    .with_label_values(&["confirm_deposit", "failure"])
387                    .inc();
388            })
389            .map_err(ApprovedDepositError::ConfirmDepositFailed)?;
390        Ok(())
391    }

393    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
394    async fn request_deposit_confirmation_signature(
395        inner: &Arc<Hashi>,
396        proto_request: SignDepositConfirmationRequest,
397        member: &CommitteeMember,
398    ) -> Option<MemberSignature> {
399        let validator_address = member.validator_address();
400        trace!("Requesting deposit confirmation signature");
401
402        let mut rpc_client = inner
403            .onchain_state()
404            .bridge_service_client(&validator_address)
405            .or_else(|| {
406                error!(
407                    "Cannot find client for validator address: {:?}",
408                    validator_address
409                );
410                None
411            })?;
412
413        let response = rpc_client
414            .sign_deposit_confirmation(proto_request.clone())
415            .await
416            .inspect_err(|e| {
417                error!(
418                    "Failed to get deposit confirmation signature from {}: {e}",
419                    validator_address
420                );
421            })
422            .ok()?;
423
424        trace!(
425            "Retrieved deposit confirmation signature from {}",
426            validator_address
427        );
428
429        response
430            .into_inner()
431            .member_signature
432            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
433            .and_then(parse_member_signature)
434            .inspect_err(|e| {
435                error!(
436                    "Failed to parse member signature from response from {}: {e}",
437                    validator_address
438                );
439            })
440            .ok()
441    }

444fn deposit_request_to_proto(req: &DepositRequest) -> SignDepositConfirmationRequest {
445    SignDepositConfirmationRequest {
446        id: req.id.as_bytes().to_vec().into(),
447        txid: req.utxo.id.txid.as_bytes().to_vec().into(),
448        vout: req.utxo.id.vout,
449        amount: req.utxo.amount,
450        derivation_path: req
451            .utxo
452            .derivation_path
453            .map(|p| p.as_bytes().to_vec().into()),
454        timestamp_ms: req.timestamp_ms,
455        requester_address: req.sender.as_bytes().to_vec().into(),
456        sui_tx_digest: req.sui_tx_digest.as_bytes().to_vec().into(),
457    }
