41    pub(super) fn process_unapproved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
42        debug!("Entering process_unapproved_withdrawal_requests");
43        if self.is_reconfiguring() {
44            debug!("Reconfig in progress, skipping withdrawal approval processing");
45            return;
46        }
47
48        if self.withdrawal_approval_task.is_some() {
49            debug!("Withdrawal approval task already in-flight, skipping");
50            return;
51        }
52
53        let mut unapproved: Vec<_> = self
54            .inner
55            .onchain_state()
56            .withdrawal_requests()
57            .into_iter()
58            .filter(|r| r.status.is_requested())
59            .collect();
60        unapproved.sort_by_key(|r| r.created_timestamp_ms);
61
62        let unapproved_ids: Vec<Address> = unapproved.iter().map(|r| r.id).collect();
63        self.withdrawal_approval_retry_tracker
64            .prune(&unapproved_ids);
65
66        let mut to_process: Vec<_> = unapproved
67            .into_iter()
68            .filter(|r| {
69                !self
70                    .withdrawal_approval_retry_tracker
71                    .should_skip(&r.id, checkpoint_timestamp_ms)
72            })
73            .collect();
74
75        if to_process.len() > WITHDRAWAL_APPROVAL_PTB_BATCH_SIZE {
76            info!(
77                available = to_process.len(),
78                batch_size = WITHDRAWAL_APPROVAL_PTB_BATCH_SIZE,
79                "Limiting withdrawal approval batch to PTB batch size",
80            );
81            to_process.truncate(WITHDRAWAL_APPROVAL_PTB_BATCH_SIZE);
82        }
83
84        self.inner
85            .metrics
86            .leader_items_in_backoff
87            .with_label_values(&["withdrawal_approval"])
88            .set(
89                self.withdrawal_approval_retry_tracker
90                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
91            );
92
93        if to_process.is_empty() {
94            return;
95        }
96
97        let inner = self.inner.clone();
98        let retry_tracker = self.withdrawal_approval_retry_tracker.clone();
99
100        self.withdrawal_approval_task =
101            Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
102                Self::process_unapproved_withdrawal_requests_task(
103                    inner,
104                    retry_tracker,
105                    to_process,
106                    checkpoint_timestamp_ms,
107                )
108                .await
109            })));
110    }

112    #[tracing::instrument(level = "info", skip_all, fields(batch_size = to_process.len()))]
113    async fn process_unapproved_withdrawal_requests_task(
114        inner: Arc<Hashi>,
115        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
116        to_process: Vec<WithdrawalRequest>,
117        checkpoint_timestamp_ms: u64,
118    ) -> anyhow::Result<()> {
119        let max_concurrent = inner.config.max_concurrent_leader_job_tasks();
120
121        let this_validator_address = inner
122            .config
123            .validator_address()
124            .expect("No configured validator address");
125
126        let members = inner
127            .onchain_state()
128            .current_committee_members()
129            .expect("No current committee members");
130
131        let committee = inner
132            .onchain_state()
133            .current_committee()
134            .expect("No current committee");
135
136        let mut tasks = JoinSet::new();
137        let mut certified: Vec<(Address, CommitteeSignature)> = Vec::new();
138
139        for request in to_process {
140            if tasks.len() >= max_concurrent {
141                // Wait for one to finish before spawning more.
142                if let Some(result) = tasks.join_next().await {
143                    match &result {
144                        Err(err) if err.is_panic() => {
145                            error!("Withdrawal approval task panicked: {err}")
146                        }
147                        Err(err) => error!("Withdrawal approval task failed to join: {err}"),
148                        Ok(_) => {}
149                    }
150                    if let Ok((_request_id, Ok(Some(cert)))) = result {
151                        certified.push(cert);
152                    }
153                }
154            }
155
156            let inner = inner.clone();
157            let retry_tracker = retry_tracker.clone();
158            let members = members.clone();
159            let committee = committee.clone();
160            tasks.spawn(async move {
161                let request_id = request.id;
162                let task_result = tokio::time::timeout(
163                    LEADER_TASK_TIMEOUT,
164                    Self::process_unapproved_withdrawal_request(
165                        inner.clone(),
166                        retry_tracker.clone(),
167                        request,
168                        checkpoint_timestamp_ms,
169                        this_validator_address,
170                        &members,
171                        &committee,
172                    ),
173                )
174                .await;
175
176                let (result, failure_kind) = match task_result {
177                    Ok(result) => (result, None),
178                    Err(_) => {
179                        let kind = WithdrawalApprovalErrorKind::TimedOut;
180                        inner
181                            .metrics
182                            .leader_retries_total
183                            .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
184                            .inc();
185                        retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
186                        (Ok(None), Some(kind))
187                    }
188                };
189
190                if result.is_err() && failure_kind.is_none() {
191                    let kind = WithdrawalApprovalErrorKind::TaskFailed;
192                    inner
193                        .metrics
194                        .leader_retries_total
195                        .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
196                        .inc();
197                    retry_tracker.record_failure(kind, request_id, checkpoint_timestamp_ms);
198                }
199                if let Err(err) = &result {
200                    error!(request_id = %request_id, "Withdrawal approval failed: {err:#}");
201                }
202
203                (request_id, result)
204            });
205        }
206
207        while let Some(result) = tasks.join_next().await {
208            match &result {
209                Err(err) if err.is_panic() => error!("Withdrawal approval task panicked: {err}"),
210                Err(err) => error!("Withdrawal approval task failed to join: {err}"),
211                Ok(_) => {}
212            }
213            if let Ok((_request_id, Ok(Some(cert)))) = result {
214                certified.push(cert);
215            }
216        }
217
218        if certified.is_empty() {
219            return Ok(());
220        }
221
222        Self::submit_approve_withdrawal_requests_with_retry(&inner, certified).await;
223        Ok(())
224    }

226    #[tracing::instrument(level = "info", skip_all, fields(request_id = %request.id))]
227    async fn process_unapproved_withdrawal_request(
228        inner: Arc<Hashi>,
229        retry_tracker: RetryTracker<WithdrawalApprovalErrorKind>,
230        request: WithdrawalRequest,
231        checkpoint_timestamp_ms: u64,
232        this_validator_address: Address,
233        members: &[CommitteeMember],
234        committee: &hashi_types::committee::Committee,
235    ) -> anyhow::Result<Option<(Address, CommitteeSignature)>> {
236        let approval = WithdrawalRequestApproval {
237            request_id: request.id,
238        };
239
240        // Validate, screen, and sign locally first
241        let local_sig = match inner
242            .validate_and_sign_withdrawal_request_approval(&approval)
243            .await
244        {
245            Ok(sig) => {
246                retry_tracker.clear(&request.id);
247                parse_member_signature(sig).unwrap()
248            }
249            Err(e) => {
250                let kind = e.kind();
251                inner
252                    .metrics
253                    .leader_retries_total
254                    .with_label_values(&["withdrawal_approval", &format!("{kind:?}")])
255                    .inc();
256                retry_tracker.record_failure(kind, request.id, checkpoint_timestamp_ms);
257                return Ok(None);
258            }
259        };
260
261        let proto_request = approval.to_proto();
262        let required_weight = certificate_threshold(committee.total_weight());
263
264        let mut aggregator = BlsSignatureAggregator::new(committee, approval);
265        if let Err(e) = aggregator.add_signature(local_sig) {
266            error!("Failed to add local approval signature: {e}");
267        }
268
269        // Fan out signature requests to remote members in parallel.
270        let mut sig_tasks = JoinSet::new();
271        for member in members {
272            if member.validator_address() == this_validator_address {
273                continue;
274            }
275            let inner = inner.clone();
276            let proto_request = proto_request.clone();
277            let member = member.clone();
278            sig_tasks.spawn(async move {
279                Self::request_withdrawal_approval_signature(&inner, proto_request, &member).await
280            });
281        }
282
283        // Collect signatures, stopping once we reach quorum.
284        while let Some(result) = sig_tasks.join_next().await {
285            let Ok(Some(sig)) = result else { continue };
286            if let Err(e) = aggregator.add_signature(sig) {
287                error!("Failed to add approval signature: {e}");
288            }
289            if aggregator.weight() >= required_weight {
290                break;
291            }
292        }
293
294        let weight = aggregator.weight();
295        if weight < required_weight {
296            inner
297                .metrics
298                .leader_retries_total
299                .with_label_values(&["withdrawal_approval", "FailedQuorum"])
300                .inc();
301            retry_tracker.record_failure(
302                WithdrawalApprovalErrorKind::FailedQuorum,
303                request.id,
304                checkpoint_timestamp_ms,
305            );
306            error!("Insufficient approval signatures: weight {weight} < {required_weight}");
307            return Ok(None);
308        }
309
310        match aggregator.finish() {
311            Ok(signed) => Ok(Some((request.id, signed.committee_signature().clone()))),
312            Err(e) => {
313                error!("Failed to build approval certificate: {e}");
314                Ok(None)
315            }
316        }
317    }

319    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
320    async fn request_withdrawal_approval_signature(
321        inner: &Arc<Hashi>,
322        proto_request: SignWithdrawalRequestApprovalRequest,
323        member: &CommitteeMember,
324    ) -> Option<MemberSignature> {
325        let validator_address = member.validator_address();
326        trace!("Requesting withdrawal request approval signature");
327
328        let response = Self::call_peer_with_retry(
329            inner,
330            validator_address,
331            "withdrawal request approval signature",
332            move |mut client| {
333                let request = proto_request.clone();
334                async move { client.sign_withdrawal_request_approval(request).await }
335            },
336        )
337        .await?;
338
339        trace!(
340            "Retrieved withdrawal request approval signature from {}",
341            validator_address
342        );
343
344        response
345            .into_inner()
346            .member_signature
347            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
348            .and_then(parse_member_signature)
349            .inspect_err(|e| {
350                error!(
351                    "Failed to parse member signature from withdrawal request approval response from {}: {e}",
352                    validator_address
353                );
354            })
355            .ok()
356    }

358    async fn submit_approve_withdrawal_requests_with_retry(
359        inner: &Arc<Hashi>,
360        mut certified: Vec<(Address, CommitteeSignature)>,
361    ) {
362        loop {
363            let approvals: Vec<(Address, &CommitteeSignature)> =
364                certified.iter().map(|(id, cert)| (*id, cert)).collect();
365
366            let result = Self::submit_approve_withdrawal_requests(inner, &approvals)
367                .await
368                .inspect(|()| {
369                    inner
370                        .metrics
371                        .sui_tx_submissions_total
372                        .with_label_values(&["approve_withdrawal", "success"])
373                        .inc();
374                })
375                .inspect_err(|_| {
376                    inner
377                        .metrics
378                        .sui_tx_submissions_total
379                        .with_label_values(&["approve_withdrawal", "failure"])
380                        .inc();
381                });
382            let Err(e) = result else { return };
383
384            let err_msg = format!("{e}");
385            error!("approve_request PTB failed: {err_msg}");
386
387            // Try to identify which request caused the failure by checking
388            // which ones no longer exist in the queue (canceled).
389            let before_len = certified.len();
390            certified.retain(|(id, _)| inner.onchain_state().withdrawal_request(id).is_some());
391
392            if certified.len() == before_len {
393                error!("Could not identify failed request, aborting retry");
394                return;
395            }
396            if certified.is_empty() {
397                return;
398            }
399
400            info!(
401                "Retrying approve_request with {} remaining requests",
402                certified.len()
403            );
404        }
405    }

407    async fn submit_approve_withdrawal_requests(
408        inner: &Arc<Hashi>,
409        approvals: &[(Address, &CommitteeSignature)],
410    ) -> anyhow::Result<()> {
411        info!(
412            "Submitting approve_request PTB for {} requests",
413            approvals.len()
414        );
415
416        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
417        executor
418            .execute_approve_withdrawal_requests(approvals)
419            .await
420    }

426    pub(super) fn process_approved_withdrawal_requests(&mut self, checkpoint_timestamp_ms: u64) {
427        debug!("Entering process_approved_withdrawal_requests");
428        if self.is_reconfiguring() {
429            debug!("Reconfig in progress, skipping withdrawal commitment processing");
430            return;
431        }
432
433        if self.withdrawal_commitment_task.is_some() {
434            debug!("Withdrawal commitment task already in-flight, skipping");
435            return;
436        }
437
438        // Pairs with the spawn-side `max_concurrent = 1` cap: don't
439        // double-commit before the prior signing task has spawned.
440        if self.inner.onchain_state().has_unsigned_withdrawal_txn() {
441            debug!("Unsigned withdrawal txn already on-chain, skipping commitment");
442            return;
443        }
444
445        let mut approved: Vec<_> = self
446            .inner
447            .onchain_state()
448            .withdrawal_requests()
449            .into_iter()
450            .filter(|r| r.status.is_approved())
451            .collect();
452        approved.sort_by_key(|r| r.created_timestamp_ms);
453
454        // Prune stuck-warn entries so a re-stuck request warns again.
455        let pending_ids: HashSet<Address> = approved.iter().map(|r| r.id).collect();
456        self.stuck_withdrawal_warned
457            .retain(|id| pending_ids.contains(id));
458
459        if self
460            .withdrawal_commitment_retry_tracker
461            .should_skip(checkpoint_timestamp_ms)
462        {
463            self.inner
464                .metrics
465                .leader_items_in_backoff
466                .with_label_values(&["withdrawal_commitment"])
467                .set(
468                    self.withdrawal_commitment_retry_tracker
469                        .in_backoff_count(checkpoint_timestamp_ms) as i64,
470                );
471            return;
472        }
473
474        self.inner
475            .metrics
476            .leader_items_in_backoff
477            .with_label_values(&["withdrawal_commitment"])
478            .set(
479                self.withdrawal_commitment_retry_tracker
480                    .in_backoff_count(checkpoint_timestamp_ms) as i64,
481            );
482
483        if approved.is_empty() {
484            return;
485        }
486
487        // Skip oversize requests (would HOL-block forever) and take the
488        // longest prefix of the rest that fits current capacity. The
489        // dropped tail flips `at_capacity` so we don't sit on demand for
490        // the full batching window.
491        let (batch, at_capacity) = if let Some(limiter) = self.inner.local_limiter() {
492            let timestamp_secs = checkpoint_timestamp_ms / 1000;
493            let max_bucket = limiter.config().max_bucket_capacity;
494            let capacity = limiter.capacity_at(timestamp_secs);
495
496            let mut batch: Vec<WithdrawalRequest> = Vec::new();
497            let mut cumulative = 0u64;
498            let mut at_capacity = false;
499            for req in approved {
500                if req.btc_amount > max_bucket {
501                    if self.stuck_withdrawal_warned.insert(req.id) {
502                        warn!(
503                            request_id = %req.id,
504                            btc_amount = req.btc_amount,
505                            max_bucket_capacity = max_bucket,
506                            "Withdrawal exceeds limiter max bucket; skipping"
507                        );
508                        self.inner
509                            .metrics
510                            .guardian_limiter_stuck_oversize_skipped_total
511                            .inc();
512                    }
513                    continue;
514                }
515                let Some(next) = cumulative.checked_add(req.btc_amount) else {
516                    at_capacity = true;
517                    break;
518                };
519                if next > capacity {
520                    at_capacity = true;
521                    break;
522                }
523                cumulative = next;
524                batch.push(req);
525            }
526
527            if batch.is_empty() {
528                // All-oversize (already warned) or refill-bound head.
529                self.inner
530                    .metrics
531                    .guardian_limiter_batch_stuck_head_total
532                    .inc();
533                return;
534            }
535            if at_capacity {
536                self.inner
537                    .metrics
538                    .guardian_limiter_batch_truncated_total
539                    .inc();
540            }
541            (batch, at_capacity)
542        } else {
543            (approved, false)
544        };
545
546        let max_batch = self.inner.config.withdrawal_max_batch_size();
547        let delay_ms = self.inner.config.withdrawal_batching_delay_ms();
548
549        let batch_is_full = batch.len() >= max_batch;
550        let oldest_has_waited = batch
551            .first()
552            .is_some_and(|r| checkpoint_timestamp_ms >= r.created_timestamp_ms + delay_ms);
553
554        if !batch_is_full && !oldest_has_waited && !at_capacity {
555            debug!(
556                "Holding {} approved request(s): oldest is {}ms old, \
557                 waiting for {}ms delay or {} requests",
558                batch.len(),
559                checkpoint_timestamp_ms.saturating_sub(batch[0].created_timestamp_ms),
560                delay_ms,
561                max_batch,
562            );
563            return;
564        }
565
566        let inner = self.inner.clone();
567        let retry_tracker = self.withdrawal_commitment_retry_tracker.clone();
568
569        self.withdrawal_commitment_task =
570            Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
571                let task_result = tokio::time::timeout(
572                    LEADER_TASK_TIMEOUT,
573                    Self::process_approved_withdrawal_request_batch(
574                        inner.clone(),
575                        retry_tracker.clone(),
576                        batch,
577                        checkpoint_timestamp_ms,
578                    ),
579                )
580                .await;
581
582                match task_result {
583                    Ok(result) => result,
584                    Err(_) => {
585                        let kind = WithdrawalCommitmentErrorKind::TimedOut;
586                        inner
587                            .metrics
588                            .leader_retries_total
589                            .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
590                            .inc();
591                        Err(anyhow::anyhow!(
592                            "withdrawal commitment timed out after {LEADER_TASK_TIMEOUT:?}"
593                        ))
594                    }
595                }
596            })));
597    }

599    #[tracing::instrument(level = "info", skip_all, fields(batch_size = requests.len()))]
600    async fn process_approved_withdrawal_request_batch(
601        inner: Arc<Hashi>,
602        retry_tracker: GlobalRetryTracker<WithdrawalCommitmentErrorKind>,
603        requests: Vec<WithdrawalRequest>,
604        checkpoint_timestamp_ms: u64,
605    ) -> anyhow::Result<()> {
606        info!(
607            withdrawal_request_ids = ?requests.iter().map(|r| r.id).collect::<Vec<_>>(),
608            "Processing batch of {} approved withdrawal request(s)",
609            requests.len(),
610        );
611
612        // Build the withdrawal tx commitment for the batch.
613        let approval = match inner.build_withdrawal_tx_commitment(&requests).await {
614            Ok(approval) => {
615                retry_tracker.clear();
616                approval
617            }
618            Err(e) => {
619                let kind = e.kind();
620                inner
621                    .metrics
622                    .leader_retries_total
623                    .with_label_values(&["withdrawal_commitment", &format!("{kind:?}")])
624                    .inc();
625                retry_tracker.record_failure(kind, checkpoint_timestamp_ms);
626                return Ok(());
627            }
628        };
629
630        // Fan out to committee for BLS signatures over the commitment message
631        let members = inner
632            .onchain_state()
633            .current_committee_members()
634            .expect("No current committee members");
635
636        let committee = inner
637            .onchain_state()
638            .current_committee()
639            .expect("No current committee");
640
641        let required_weight = certificate_threshold(committee.total_weight());
642        let proto_request = approval.to_proto();
643
644        // Fan out signature requests to all members in parallel.
645        let mut sig_tasks = JoinSet::new();
646        for member in members {
647            let inner = inner.clone();
648            let proto_request = proto_request.clone();
649            sig_tasks.spawn(async move {
650                Self::request_withdrawal_tx_commitment_signature(&inner, proto_request, &member)
651                    .await
652            });
653        }
654
655        // Collect signatures, stopping once we reach quorum.
656        let mut aggregator = BlsSignatureAggregator::new(&committee, approval.clone());
657        while let Some(result) = sig_tasks.join_next().await {
658            let Ok(Some(sig)) = result else { continue };
659            if let Err(e) = aggregator.add_signature(sig) {
660                error!("Failed to add withdrawal commitment signature: {e}");
661            }
662            if aggregator.weight() >= required_weight {
663                break;
664            }
665        }
666
667        if aggregator.weight() < required_weight {
668            inner
669                .metrics
670                .leader_retries_total
671                .with_label_values(&["withdrawal_commitment", "FailedQuorum"])
672                .inc();
673            retry_tracker.record_failure(
674                WithdrawalCommitmentErrorKind::FailedQuorum,
675                checkpoint_timestamp_ms,
676            );
677            error!(
678                "Insufficient withdrawal commitment signatures: weight {} < {required_weight}",
679                aggregator.weight()
680            );
681            return Ok(());
682        }
683
684        let signed_approval = match aggregator.finish() {
685            Ok(signed_approval) => signed_approval,
686            Err(e) => {
687                error!("Failed to build withdrawal commitment certificate: {e}");
688                return Ok(());
689            }
690        };
691
692        // Proactively trigger a presig refill if this commit will allocate
693        // indices beyond the current pool.
694        {
695            let num_inputs = approval.selected_utxos.len() as u64;
696            let num_consumed = inner.onchain_state().state().hashi().num_consumed_presigs;
697            let needed_end = num_consumed + num_inputs;
698            if let Some(signing_manager) = inner.current_signing_manager() {
699                let available_end = signing_manager.available_presig_end_index();
700                if needed_end > available_end {
701                    info!(
702                        "Presig pool may be insufficient for this withdrawal: \
703                         need index {needed_end}, pool ends at {available_end}. \
704                         Triggering proactive refill.",
705                    );
706                    signing_manager.trigger_refill();
707                }
708            }
709        }
710
711        // Submit commit_withdrawal_tx to Sui
712        Self::submit_commit_withdrawal_tx(&inner, &approval, signed_approval.committee_signature())
713            .await
714            .inspect(|()| {
715                inner
716                    .metrics
717                    .sui_tx_submissions_total
718                    .with_label_values(&["commit_withdrawal", "success"])
719                    .inc();
720            })
721            .inspect_err(|e| {
722                inner
723                    .metrics
724                    .sui_tx_submissions_total
725                    .with_label_values(&["commit_withdrawal", "failure"])
726                    .inc();
727                error!("Failed to submit commit_withdrawal_tx: {e}");
728            })?;
729
730        Ok(())
731    }
732
733    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
734    async fn request_withdrawal_tx_commitment_signature(
735        inner: &Arc<Hashi>,
736        proto_request: SignWithdrawalTxConstructionRequest,
737        member: &CommitteeMember,
738    ) -> Option<MemberSignature> {
739        let validator_address = member.validator_address();
740        trace!("Requesting withdrawal tx commitment signature");
741
742        let response = Self::call_peer_with_retry(
743            inner,
744            validator_address,
745            "withdrawal tx commitment signature",
746            move |mut client| {
747                let request = proto_request.clone();
748                async move { client.sign_withdrawal_tx_construction(request).await }
749            },
750        )
751        .await?;
752
753        trace!(
754            "Retrieved withdrawal approval signature from {}",
755            validator_address
756        );
757
758        response
759            .into_inner()
760            .member_signature
761            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
762            .and_then(parse_member_signature)
763            .inspect_err(|e| {
764                error!(
765                    "Failed to parse member signature from withdrawal approval response from {}: {e}",
766                    validator_address
767                );
768            })
769            .ok()
770    }
771
772    async fn submit_commit_withdrawal_tx(
773        inner: &Arc<Hashi>,
774        approval: &WithdrawalTxCommitment,
775        cert: &CommitteeSignature,
776    ) -> anyhow::Result<()> {
777        info!(
778            "Submitting commit_withdrawal_tx for txid {:?}",
779            approval.txid
780        );
781
782        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
783        executor.execute_commit_withdrawal_tx(approval, cert).await
784    }
