31    pub(super) fn check_reconcile_guardian_committee(&mut self) {
32        if self.inner.guardian_client().is_none() {
33            return;
34        }
35        // Don't overwrite an existing handle — finished or not. The select!
36        // arm clears the slot and logs the result. Letting it run first
37        // avoids dropping a completed task's error on the floor.
38        if self.guardian_committee_reconcile_task.is_some() {
39            return;
40        }
41        // Only kick a reconcile when the hashi epoch advances (or on the
42        // first leader tick). A no-op reconcile still costs a GetGuardianInfo
43        // RPC, so don't run it every checkpoint.
44        let hashi_epoch = self.inner.onchain_state().epoch();
45        if self.last_guardian_reconcile_epoch == Some(hashi_epoch) {
46            return;
47        }
48        self.last_guardian_reconcile_epoch = Some(hashi_epoch);
49        let inner = self.inner.clone();
50        let handle = tokio::spawn(async move { Self::reconcile_guardian_committee(&inner).await });
51        self.guardian_committee_reconcile_task = Some(AbortOnDropHandle::new(handle));
52    }

58    /// Returns the per-input guardian Schnorr signatures (64 bytes each)
59    /// for inclusion in the on-chain `sign_withdrawal` PTB.
60    #[tracing::instrument(level = "info", skip_all, fields(withdrawal_txn_id = %txn.id, seq))]
61    pub(super) async fn finalize_withdrawal_through_guardian(
62        inner: &Arc<Hashi>,
63        txn: &WithdrawalTransaction,
64        members: &[CommitteeMember],
65        guardian: &crate::grpc::guardian_client::GuardianClient,
66        timestamp_secs: u64,
67        seq: u64,
68    ) -> anyhow::Result<Vec<Vec<u8>>> {
69        let signed_request =
70            Self::collect_guardian_withdrawal_signatures(inner, txn, members, timestamp_secs, seq)
71                .await?;
72        let proto_request = signed_standard_withdrawal_request_to_pb(&signed_request);
73
74        let rpc_start = std::time::Instant::now();
75        let rpc_result = guardian.standard_withdrawal(proto_request).await;
76        let rpc_elapsed = rpc_start.elapsed().as_secs_f64();
77
78        let response_pb = rpc_result.map_err(|status| {
79            let (rpc_outcome, retry_label) = if status.message().contains("seq mismatch") {
80                (
81                    crate::metrics::GUARDIAN_RPC_OUTCOME_SEQ_MISMATCH,
82                    "GuardianSeqMismatch",
83                )
84            } else if status.message().contains("Rate limit exceeded") {
85                warn!("Guardian rate-limited withdrawal, will retry later");
86                (
87                    crate::metrics::GUARDIAN_RPC_OUTCOME_RATE_LIMITED,
88                    "GuardianRateLimited",
89                )
90            } else {
91                error!("Guardian call failed: {}", status.message());
92                (
93                    crate::metrics::GUARDIAN_RPC_OUTCOME_UNAVAILABLE,
94                    "GuardianUnavailable",
95                )
96            };
97            Self::record_guardian_rpc_outcome(inner, rpc_outcome, rpc_elapsed);
98            inner
99                .metrics
100                .leader_retries_total
101                .with_label_values(&["withdrawal_signing", retry_label])
102                .inc();
103            anyhow::anyhow!("Guardian rejected withdrawal: {}", status.message())
104        })?;
105
106        let signed_response: GuardianSigned<StandardWithdrawalResponse> = response_pb
107            .try_into()
108            .inspect_err(|_| {
109                Self::record_guardian_rpc_outcome(
110                    inner,
111                    crate::metrics::GUARDIAN_RPC_OUTCOME_PARSE_ERROR,
112                    rpc_elapsed,
113                );
114            })
115            .map_err(|e| anyhow::anyhow!("Failed to parse guardian withdrawal response: {e}"))?;
116        // Authenticated by TLS; the per-input BTC witness signatures below only
117        // spend the 2-of-2 if they verify against the on-chain guardian BTC key
118        // (enforced by Bitcoin), so the response itself isn't signature-checked.
119        let response = signed_response.into_data_unchecked();
120
121        anyhow::ensure!(
122            response.enclave_signatures.len() == txn.inputs.len(),
123            "Guardian returned {} enclave_signatures but tx has {} inputs",
124            response.enclave_signatures.len(),
125            txn.inputs.len(),
126        );
127        let guardian_signatures: Vec<Vec<u8>> = response
128            .enclave_signatures
129            .iter()
130            .enumerate()
131            .map(|(i, sig)| {
132                let bytes = sig.to_vec();
133                anyhow::ensure!(
134                    bytes.len() == 64,
135                    "Guardian enclave_signatures[{i}] is {} bytes, expected 64",
136                    bytes.len(),
137                );
138                Ok(bytes)
139            })
140            .collect::<anyhow::Result<_>>()?;
141
142        Self::record_guardian_rpc_outcome(
143            inner,
144            crate::metrics::GUARDIAN_RPC_OUTCOME_OK,
145            rpc_elapsed,
146        );
147        info!(seq, "Guardian approved withdrawal");
148        Ok(guardian_signatures)
149    }

151    fn record_guardian_rpc_outcome(inner: &Arc<Hashi>, outcome: &str, elapsed_secs: f64) {
152        inner.metrics.record_guardian_rpc(
153            crate::metrics::GUARDIAN_RPC_METHOD_STANDARD_WITHDRAWAL,
154            outcome,
155            elapsed_secs,
156        );

159    async fn collect_guardian_withdrawal_signatures(
160        inner: &Arc<Hashi>,
161        txn: &WithdrawalTransaction,
162        members: &[CommitteeMember],
163        timestamp_secs: u64,
164        seq: u64,
165    ) -> anyhow::Result<SignedMessage<StandardWithdrawalRequest>> {
166        let guardian_request =
167            crate::withdrawals::build_guardian_withdrawal_request(inner, txn, timestamp_secs, seq)?;
168
169        let committee = inner
170            .onchain_state()
171            .current_committee()
172            .expect("No current committee");
173        let required_weight = certificate_threshold(committee.total_weight());
174
175        let proto_request = SignGuardianWithdrawalRequestRequest {
176            withdrawal_txn_id: txn.id.as_bytes().to_vec().into(),
177            timestamp_secs,
178            seq,
179        };
180
181        let mut sig_tasks = JoinSet::new();
182        for member in members {
183            let inner = inner.clone();
184            let proto_request = proto_request.clone();
185            let member = member.clone();
186            sig_tasks.spawn(async move {
187                Self::request_guardian_withdrawal_signature(&inner, proto_request, &member).await
188            });
189        }
190
191        let mut aggregator = BlsSignatureAggregator::new(&committee, guardian_request);
192        while let Some(result) = sig_tasks.join_next().await {
193            let Ok(Some(sig)) = result else { continue };
194            if let Err(e) = aggregator.add_signature(sig) {
195                error!(
196                    withdrawal_txn_id = %txn.id,
197                    "Failed to add guardian withdrawal signature: {e}"
198                );
199            }
200            if aggregator.weight() >= required_weight {
201                break;
202            }
203        }
204
205        let weight = aggregator.weight();
206        if weight < required_weight {
207            anyhow::bail!(
208                "Insufficient guardian withdrawal signatures: weight {weight} < {required_weight}"
209            );
210        }
211
212        Ok(aggregator.finish()?)
213    }

215    #[tracing::instrument(level = "debug", skip_all, fields(validator = %member.validator_address()))]
216    async fn request_guardian_withdrawal_signature(
217        inner: &Arc<Hashi>,
218        proto_request: SignGuardianWithdrawalRequestRequest,
219        member: &CommitteeMember,
220    ) -> Option<MemberSignature> {
221        let validator_address = member.validator_address();
222        trace!("Requesting guardian withdrawal signature");
223
224        let response = Self::call_peer_with_retry(
225            inner,
226            validator_address,
227            "guardian withdrawal signature",
228            move |mut client| {
229                let request = proto_request.clone();
230                async move { client.sign_guardian_withdrawal_request(request).await }
231            },
232        )
233        .await?;
234
235        response
236            .into_inner()
237            .member_signature
238            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
239            .and_then(parse_member_signature)
240            .inspect_err(|e| {
241                error!(
242                    "Failed to parse guardian withdrawal member signature from {validator_address}: {e}"
243                );
244            })
245            .ok()
246    }

252    /// Replay stored guardian handoffs until the guardian matches the chain.
253    async fn reconcile_guardian_committee(inner: &Arc<Hashi>) -> anyhow::Result<()> {
254        let Some(guardian) = inner.guardian_client() else {
255            return Ok(());
256        };
257
258        // Seed `guardian_epoch` once from `GetGuardianInfo`, then build the
259        // full stored handoff chain and submit it to the guardian in one RPC.
260        let info = inner.fetch_guardian_info_data().await?;
261        let Some(mut guardian_epoch) = info.current_committee_epoch else {
262            // ProvisionerInit hasn't run yet; the bootstrap CLI seeds it.
263            return Ok(());
264        };
265        inner
266            .metrics
267            .guardian_current_committee_epoch
268            .set(guardian_epoch as i64);
269
270        let hashi_epoch = inner.onchain_state().epoch();
271        let initial_guardian_epoch = guardian_epoch;
272        let mut transitions = Vec::new();
273        let mut final_to_epoch = guardian_epoch;
274        loop {
275            if guardian_epoch > hashi_epoch {
276                // The guardian only advances via certs that hashi signs, so
277                // it should never run ahead of the hashi chain. If we see
278                // this, something is wrong (e.g., a stale onchain read).
279                warn!(
280                    guardian_epoch,
281                    hashi_epoch, "guardian is ahead of hashi — unexpected"
282                );
283                return Ok(());
284            }
285            if guardian_epoch == hashi_epoch {
286                break;
287            }
288
289            let from_epoch = guardian_epoch;
290            info!(
291                from_epoch,
292                hashi_epoch, "Queueing stored guardian committee handoff"
293            );
294            let signed = inner
295                .onchain_state()
296                .committee_handoff(from_epoch)
297                .ok_or_else(|| {
298                    anyhow::anyhow!(
299                        "missing on-chain committee handoff for epoch {from_epoch}; cannot advance guardian"
300                    )
301                })?;
302            let to_epoch = signed.message().new_committee.epoch;
303            if to_epoch <= from_epoch {
304                anyhow::bail!("stored guardian handoff did not advance: {from_epoch}->{to_epoch}");
305            }
306            transitions.push(signed_committee_transition_to_pb(&signed));
307            final_to_epoch = to_epoch;
308            guardian_epoch = to_epoch;
309        }
310
311        if transitions.is_empty() {
312            return Ok(());
313        }
314
315        let resp = guardian
316            .update_committee_chain(UpdateCommitteeChainRequest { transitions })
317            .await
318            .map_err(|status| {
319                anyhow::anyhow!("UpdateCommitteeChain failed: {}", status.message())
320            })?;
321        let new_guardian_epoch = resp.current_committee_epoch.ok_or_else(|| {
322            anyhow::anyhow!("UpdateCommitteeChain response missing current_committee_epoch")
323        })?;
324        inner
325            .metrics
326            .guardian_current_committee_epoch
327            .set(new_guardian_epoch as i64);
328        info!(
329            from_epoch = initial_guardian_epoch,
330            to_epoch = new_guardian_epoch,
331            "Advanced guardian committee"
332        );
333
334        if new_guardian_epoch < final_to_epoch {
335            anyhow::bail!(
336                "guardian failed to advance to {final_to_epoch}: ended at {new_guardian_epoch}"
337            );
338        }
339
340        Ok(())
341    }

343    pub(crate) async fn collect_committee_transition_signatures(
344        inner: &Arc<Hashi>,
345        from_epoch: u64,
346    ) -> anyhow::Result<SignedMessage<CommitteeTransitionRequest>> {
347        let (to_epoch, from_committee, new_committee) = {
348            let onchain = inner.onchain_state();
349            let state = onchain.state();
350            let committees_map = state.hashi().committees.committees();
351            let from = committees_map
352                .get(&from_epoch)
353                .cloned()
354                .ok_or_else(|| anyhow::anyhow!("no on-chain committee for epoch {from_epoch}"))?;
355            // Hashi committee epochs are sparse: each reconfig only adds a
356            // new entry when Sui's epoch advances past hashi's AND the MPC
357            // reconfig completes, so the next entry is generally not
358            // `from_epoch + 1`. Both leader and followers derive the same
359            // `to_epoch` from on-chain state, so they sign the same transition.
360            let (to_epoch, to) = committees_map
361                .range((from_epoch + 1)..)
362                .next()
363                .map(|(&k, c)| (k, c.clone()))
364                .ok_or_else(|| anyhow::anyhow!("no on-chain committee epoch after {from_epoch}"))?;
365            (to_epoch, from, to)
366        };
367
368        let transition = CommitteeTransitionRequest {
369            new_committee: hashi_types::move_types::Committee::from(&new_committee),
370        };
371        let required_weight = certificate_threshold(from_committee.total_weight());
372
373        let proto_request = SignCommitteeTransitionRequest { from_epoch };
374        let mut sig_tasks = JoinSet::new();
375        for member in from_committee.members() {
376            let inner = inner.clone();
377            let member = member.clone();
378            sig_tasks.spawn(async move {
379                Self::request_committee_transition_signature(&inner, proto_request, &member).await
380            });
381        }
382
383        let mut aggregator = BlsSignatureAggregator::new(&from_committee, transition);
384        while let Some(result) = sig_tasks.join_next().await {
385            let Ok(Some(sig)) = result else { continue };
386            if let Err(e) = aggregator.add_signature(sig) {
387                error!(
388                    from_epoch,
389                    "Failed to add committee transition signature: {e}"
390                );
391            }
392            if aggregator.weight() >= required_weight {
393                break;
394            }
395        }
396        let weight = aggregator.weight();
397        if weight < required_weight {
398            anyhow::bail!(
399                "insufficient committee transition signatures for {from_epoch}->{to_epoch}: weight {weight} < {required_weight}"
400            );
401        }
402        Ok(aggregator.finish()?)
403    }

405    #[tracing::instrument(
406        level = "debug",
407        skip_all,
408        fields(validator = %member.validator_address())
409    )]
410    async fn request_committee_transition_signature(
411        inner: &Arc<Hashi>,
412        proto_request: SignCommitteeTransitionRequest,
413        member: &CommitteeMember,
414    ) -> Option<MemberSignature> {
415        let validator_address = member.validator_address();
416        let mut rpc_client = inner
417            .onchain_state()
418            .bridge_service_client(&validator_address)
419            .or_else(|| {
420                error!(
421                    "Cannot find client for validator address: {:?}",
422                    validator_address
423                );
424                None
425            })?;
426        let response = rpc_client
427            .sign_committee_transition(proto_request)
428            .await
429            .inspect_err(|e| {
430                error!(
431                    "Failed to get committee transition signature from {validator_address}: {e}"
432                );
433            })
434            .ok()?;
435        response
436            .into_inner()
437            .member_signature
438            .ok_or_else(|| anyhow::anyhow!("No member_signature in response"))
439            .and_then(parse_member_signature)
440            .inspect_err(|e| {
441                error!(
442                    "Failed to parse committee transition signature from {validator_address}: {e}"
443                );
444            })
445            .ok()
446    }
