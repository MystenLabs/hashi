4use std::collections::HashMap;
5use std::collections::HashSet;
6use std::fmt::Debug;
7use std::sync::Arc;
8use std::sync::Mutex;
9use sui_sdk_types::Address;
10use tracing::error;
11
12pub(crate) trait RetryPolicy {
13    fn retry_base_delay_ms(self) -> u64;
14    fn max_delay_ms(self) -> u64;
15    fn max_retries(self) -> u32;
16
17    fn retry_delay_ms(self, attempt: u32) -> u64
18    where
19        Self: Copy,
20    {
21        let exponent = attempt.saturating_sub(1).min(63);
22        let multiplier = 1u64.checked_shl(exponent).unwrap_or(u64::MAX);
23        self.retry_base_delay_ms()
24            .saturating_mul(multiplier)
25            .min(self.max_delay_ms())
26    }
27}

29#[derive(Clone)]
30pub(super) struct RetryTracker<K> {
31    state: Arc<Mutex<HashMap<Address, RetryState<K>>>>,
32}
33
34#[derive(Clone, Copy, Debug)]
35struct RetryState<K> {
36    attempt: u32,
37    next_retry_at_ms: u64,
38    last_error_kind: K,
39}
40
41impl<K> RetryTracker<K>
42where
43    K: RetryPolicy + Copy + Eq + Debug,
44{
45    pub(super) fn new() -> Self {
46        Self {
47            state: Arc::new(Mutex::new(HashMap::new())),
48        }
49    }
50
51    pub(super) fn prune(&self, active_ids: &[Address]) {
52        let active: HashSet<Address> = active_ids.iter().copied().collect();
53        self.state
54            .lock()
55            .unwrap()
56            .retain(|request_id, _| active.contains(request_id));
57    }
58
59    pub(super) fn should_skip(&self, request_id: &Address, checkpoint_timestamp_ms: u64) -> bool {
60        let states = self.state.lock().unwrap();
61        match states.get(request_id) {
62            Some(state) => {
63                state.attempt >= state.last_error_kind.max_retries()
64                    || state.next_retry_at_ms > checkpoint_timestamp_ms
65            }
66            None => false,
67        }
68    }
69
70    pub(super) fn clear(&self, request_id: &Address) {
71        self.state.lock().unwrap().remove(request_id);
72    }
73
74    pub(super) fn in_backoff_count(&self, checkpoint_timestamp_ms: u64) -> usize {
75        let states = self.state.lock().unwrap();
76        states
77            .values()
78            .filter(|s| {
79                s.attempt < s.last_error_kind.max_retries()
80                    && s.next_retry_at_ms > checkpoint_timestamp_ms
81            })
82            .count()
83    }
84
85    pub(super) fn record_failure(
86        &self,
87        error_kind: K,
88        request_id: Address,
89        checkpoint_timestamp_ms: u64,
90    ) {
91        let mut states = self.state.lock().unwrap();
92        let previous = states.get(&request_id).copied();
93
94        let attempt = match previous {
95            Some(state) if state.last_error_kind == error_kind => state.attempt.saturating_add(1),
96            _ => 1,
97        };
98        let next_retry_at_ms = if attempt >= error_kind.max_retries() {
99            checkpoint_timestamp_ms
100        } else {
101            let delay_ms = error_kind.retry_delay_ms(attempt);
102            checkpoint_timestamp_ms.saturating_add(delay_ms)
103        };
104        let state = RetryState {
105            attempt,
106            next_retry_at_ms,
107            last_error_kind: error_kind,
108        };
109        states.insert(request_id, state);
110
111        if state.attempt >= error_kind.max_retries() {
112            error!(
113                "Request {:?} failed validation ({error_kind:?}). Reached max retries ({}), no further retries",
114                request_id, state.attempt,
115            );
116        } else {
117            let retry_delay_ms = state
118                .next_retry_at_ms
119                .saturating_sub(checkpoint_timestamp_ms);
120            error!(
121                "Request {:?} failed validation ({error_kind:?}). Next retry in {} ms (attempt {})",
122                request_id, retry_delay_ms, state.attempt,
123            );
124        }
125    }
126}
