186    /// Store an encryption private key, keyed by its public key, plus the
187    /// `epoch -> pubkey` side-index entry. Both writes commit in one atomic
188    /// `fjall` batch.
189    ///
190    /// No-op if a key was already prepared for this epoch (idempotent for
191    /// restart safety).
192    pub fn store_encryption_key(
193        &self,
194        epoch: u64,
195        encryption_key: &EncryptionPrivateKey,
196    ) -> Result<()> {
197        let epoch_key = epoch.to_be_bytes();
198        if self.encryption_epoch_index.contains_key(epoch_key)? {
199            return Ok(());
200        }
201        let pubkey = EncryptionPublicKey::from_private_key(encryption_key)
202            .as_element()
203            .to_byte_array()
204            .to_vec();
205        let value = bcs::to_bytes(encryption_key).unwrap();
206        let mut batch = self.db.batch();
207        batch.insert(&self.encryption_keys, pubkey.clone(), value);
208        batch.insert(&self.encryption_epoch_index, epoch_key, pubkey);
209        batch.commit()
210    }

486    /// Prune all MPC keyspaces.
487    pub(crate) fn prune_messages_below(
488        &self,
489        cutoff_epoch: u64,
490        pruning_references: &PruningReferences,
491    ) -> Result<()> {
492        let retention_cutoff = cutoff_epoch.saturating_sub(RETENTION_EXTRA_EPOCHS);
493        // A key is retained if and only if its public key is referenced by a live committee or pending registration,
494        // or it was created within the retention buffer.
495        // The primary and its side-index rows are evicted together atomically.
496        prune_pubkey_keyspace(
497            &self.db,
498            &self.encryption_keys,
499            &self.encryption_epoch_index,
500            &pruning_references.encryption_keys,
501            retention_cutoff,
502        )?;
503        prune_pubkey_keyspace(
504            &self.db,
505            &self.signing_keys,
506            &self.signing_epoch_index,
507            &pruning_references.signing_keys,
508            retention_cutoff,
509        )?;
510        let is_referenced_epoch =
511            |epoch: u64, _value: &[u8]| pruning_references.committee_epochs.contains(&epoch);
512        prune_keyspace_with(&self.dealer_messages, retention_cutoff, is_referenced_epoch)?;
513        prune_keyspace_with(
514            &self.rotation_messages,
515            retention_cutoff,
516            is_referenced_epoch,
517        )?;
518        prune_keyspace(&self.nonce_messages, cutoff_epoch)?;
519        prune_keyspace(&self.avid_round_states, cutoff_epoch)?;
520        Ok(())
521    }

561/// List all `(Address, T)` pairs from a keyspace where keys match the given prefix.
562/// Keys are expected to end with a 32-byte address suffix.
563fn list_messages_by_prefix<T: DeserializeOwned>(
564    keyspace: &Keyspace,
565    prefix: &[u8],
566) -> Result<Vec<(Address, T)>> {
567    let addr_len = 32;
568    let mut results = Vec::new();
569    for guard in keyspace.prefix(prefix) {
570        let (key, value) = guard.into_inner()?;
571        let addr_start = key.len().checked_sub(addr_len).ok_or_else(|| {
572            fjall::Error::Io(std::io::Error::new(
573                std::io::ErrorKind::InvalidData,
574                "key too short for address",
575            ))
576        })?;
577        let address_bytes: [u8; 32] = key[addr_start..].try_into().map_err(|_| {
578            fjall::Error::Io(std::io::Error::new(
579                std::io::ErrorKind::InvalidData,
580                "invalid key length",
581            ))
582        })?;
583        let address = Address::new(address_bytes);
584        let message: T = bcs::from_bytes(&value).map_err(|_| {
585            fjall::Error::Io(std::io::Error::new(
586                std::io::ErrorKind::InvalidData,
587                "invalid message data",
588            ))
589        })?;
590        results.push((address, message));
591    }
592    Ok(results)
593}

595/// Delete entries from `keyspace` whose leading big-endian u64 epoch is `< cutoff_epoch`.
596fn prune_keyspace(keyspace: &Keyspace, cutoff_epoch: u64) -> Result<()> {
597    let keys_to_delete: Vec<_> = keyspace
598        .iter()
599        .filter_map(|guard| {
600            let key = guard.key().ok()?;
601            let epoch_bytes: [u8; 8] = key.as_ref().get(..8)?.try_into().ok()?;
602            let epoch = u64::from_be_bytes(epoch_bytes);
603            (epoch < cutoff_epoch).then(|| key.to_vec())
604        })
605        .collect();
606    for key in keys_to_delete {
607        keyspace.remove(key)?;
608    }
609    Ok(())
610}

612/// Delete entries from `keyspace` whose leading big-endian u64 epoch is
613/// `< cutoff_epoch`, unless `is_referenced(epoch, value)` returns `true`.
614fn prune_keyspace_with<F>(keyspace: &Keyspace, cutoff_epoch: u64, is_referenced: F) -> Result<()>
615where
616    F: Fn(u64, &[u8]) -> bool,
617{
618    let keys_to_delete: Vec<_> = keyspace
619        .iter()
620        .filter_map(|guard| {
621            let (key, value) = guard.into_inner().ok()?;
622            let epoch_bytes: [u8; 8] = key.as_ref().get(..8)?.try_into().ok()?;
623            let epoch = u64::from_be_bytes(epoch_bytes);
624            if epoch >= cutoff_epoch {
625                return None;
626            }
627            (!is_referenced(epoch, &value)).then(|| key.to_vec())
628        })
629        .collect();
630    for key in keys_to_delete {
631        keyspace.remove(key)?;
632    }
633    Ok(())
634}
635
636fn latest_epoch(side_index: &Keyspace) -> Result<Option<u64>> {
637    let mut latest: Option<u64> = None;
638    for guard in side_index.iter() {
639        let key = guard.key()?;
640        if let Ok(bytes) = <[u8; 8]>::try_from(key.as_ref()) {
641            let epoch = u64::from_be_bytes(bytes);
642            latest = Some(latest.map_or(epoch, |l: u64| l.max(epoch)));
643        }
644    }
645    Ok(latest)
646}

687/// Committee-walk GC for a pubkey-keyed key keyspace and its epoch side-index.
688fn prune_pubkey_keyspace(
689    db: &fjall::Database,
690    keyspace: &Keyspace,
691    side_index: &Keyspace,
692    referenced: &std::collections::HashSet<Vec<u8>>,
693    retention_cutoff: u64,
694) -> Result<()> {
695    let mut epochs_of: std::collections::HashMap<Vec<u8>, Vec<u64>> =
696        std::collections::HashMap::new();
697    for guard in side_index.iter() {
698        let (key, pubkey) = guard.into_inner()?;
699        let Ok(epoch_bytes) = <[u8; 8]>::try_from(key.as_ref()) else {
700            continue;
701        };
702        epochs_of
703            .entry(pubkey.to_vec())
704            .or_default()
705            .push(u64::from_be_bytes(epoch_bytes));
706    }
707    let to_delete: Vec<Vec<u8>> = keyspace
708        .iter()
709        .filter_map(|guard| {
710            let key = guard.key().ok()?;
711            let pubkey = key.as_ref();
712            let recent = epochs_of
713                .get(pubkey)
714                .and_then(|epochs| epochs.iter().max())
715                .is_some_and(|epoch| *epoch >= retention_cutoff);
716            (!referenced.contains(pubkey) && !recent).then(|| key.to_vec())
717        })
718        .collect();
719    for pubkey in &to_delete {
720        let mut batch = db.batch();
721        batch.remove(keyspace, pubkey.as_slice());
722        for epoch in epochs_of.get(pubkey.as_slice()).into_iter().flatten() {
723            batch.remove(side_index, epoch.to_be_bytes());
724        }
725        batch.commit()?;
726    }
727    Ok(())
728}

1966        db.store_signing_key(1, &pinned_signing).unwrap();
1967
1968        // Each epoch prepares a fresh key (as in production) referenced only at
1969        // its own epoch, then prunes — mirroring per-epoch reconfig + GC.
1970        for epoch in 2..=epochs {
1971            let encryption = EncryptionPrivateKey::new(&mut rand::thread_rng());
1972            let signing = Bls12381PrivateKey::generate(&mut rand::thread_rng());
1973            db.store_encryption_key(epoch, &encryption).unwrap();
1974            db.store_signing_key(epoch, &signing).unwrap();
1975
1976            let mut referenced = PruningReferences::default();
1977            referenced.add_member_pubkeys(
1978                &EncryptionPublicKey::from_private_key(&pinned_encryption),
1979                &pinned_signing.public_key(),
1980            );
1981            referenced.add_member_pubkeys(
1982                &EncryptionPublicKey::from_private_key(&encryption),
1983                &signing.public_key(),
1984            );
1985            db.prune_messages_below(epoch, &referenced).unwrap();
1986        }
1987
1988        // Retained = referenced (the one pinned key) ∪ the recent window
1989        // (RETENTION_EXTRA_EPOCHS epochs below the final cutoff) — that is
1990        // O(RETENTION_EXTRA_EPOCHS + references), independent of `epochs`.
1991        let bound = RETENTION_EXTRA_EPOCHS as usize + 3;
1992        let encryption_count = db.encryption_keys.iter().count();
1993        let signing_count = db.signing_keys.iter().count();
1994        assert!(
1995            encryption_count <= bound,
1996            "encryption key storage must stay bounded by references + recent window, \
1997             got {encryption_count} after {epochs} epochs"
1998        );
1999        assert!(
2000            signing_count <= bound,
2001            "signing key storage must stay bounded, got {signing_count} after {epochs} epochs"
2002        );

2004        // epoch count — fails if GC does not run.

2011        // across many evictions.
