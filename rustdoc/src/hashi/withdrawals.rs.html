52/// BTC that leaves the pool when this txn broadcasts — the amount that
53/// consumes the guardian's limit. Equivalent to
54/// `sum(withdrawal_outputs) + miner_fee`; we use `inputs - change` to
55/// avoid relying on a separate fee field.
56pub(crate) fn withdrawal_limiter_consumption_amount(txn: &WithdrawalTransaction) -> u64 {
57    let inputs: u64 = txn.inputs.iter().map(|u| u.amount).sum();
58    let change: u64 = txn.change_output.as_ref().map_or(0, |c| c.amount);
59    inputs.saturating_sub(change)
60}

62/// Conservative runtime-object budget for withdrawal commit transactions.
63///
64/// Sui's hard object-runtime cache limit is 1000 objects. Empirical
65/// measurements put withdrawal commit transactions at roughly
66/// `selected_utxos + 3 * requests + fixed_overhead` runtime objects. Target
67/// 922 to leave 7.8% headroom below the hard 1000 cap.
68const WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET: usize = 922;
69const WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS: usize = 12;
70const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST: usize = 3;
71const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT: usize = 1;
72
73fn safe_withdrawal_commit_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
74    let request_objects =
75        request_count.saturating_mul(WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST);
76    let runtime_input_budget = WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET
77        .saturating_sub(WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS)
78        .saturating_sub(request_objects);
79    let input_budget = runtime_input_budget / WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT;
80    configured_max_inputs.min(input_budget)
81}
82
83/// Confirm is no longer input-bound: spent UTXOs emit events only (0
84/// objects per input). The confirm transaction's object count is
85/// `requests + ~43` fixed overhead — well within the 1000-object limit
86/// for any practical request count. The commit path is the binding
87/// constraint.
88fn safe_withdrawal_flow_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
89    let request_input_budget =
90        request_count.saturating_mul(CoinSelectionParams::DEFAULT_INPUT_BUDGET);
91    configured_max_inputs
92        .min(request_input_budget)
93        .min(safe_withdrawal_commit_max_inputs(
94            request_count,
95            configured_max_inputs,
96        ))
97}

173    #[tracing::instrument(level = "debug", skip_all, fields(bitcoin_txid = %approval.txid))]
174    pub async fn validate_withdrawal_tx_commitment(
175        &self,
176        approval: &WithdrawalTxCommitment,
177    ) -> anyhow::Result<()> {
178        anyhow::ensure!(!approval.request_ids.is_empty(), "No request IDs");
179        anyhow::ensure!(!approval.selected_utxos.is_empty(), "No selected UTXOs");
180        anyhow::ensure!(!approval.outputs.is_empty(), "No outputs");
181
182        // Check for duplicate request IDs
183        let unique_request_ids: std::collections::BTreeSet<_> =
184            approval.request_ids.iter().collect();
185        anyhow::ensure!(
186            unique_request_ids.len() == approval.request_ids.len(),
187            "Duplicate request IDs"
188        );
189
190        // Check for duplicate UTXO IDs
191        let unique_utxo_ids: std::collections::BTreeSet<_> =
192            approval.selected_utxos.iter().collect();
193        anyhow::ensure!(
194            unique_utxo_ids.len() == approval.selected_utxos.len(),
195            "Duplicate UTXO IDs"
196        );
197
198        // 1. Verify each request_id exists and is approved
199        let requests: Vec<WithdrawalRequest> = approval
200            .request_ids
201            .iter()
202            .map(|id| {
203                let request = self
204                    .onchain_state()
205                    .withdrawal_request(id)
206                    .ok_or_else(|| anyhow!("Withdrawal request {id} not found in queue"))?;
207                anyhow::ensure!(
208                    request.status.is_approved(),
209                    "Withdrawal request {id} has not been approved"
210                );
211                Ok(request)
212            })
213            .collect::<anyhow::Result<_>>()?;
214
215        // 2. Verify each selected UTXO exists, is not locked, and collect
216        //    full UTXO data. We look up via utxo_records so we can
217        //    distinguish "missing" from "locked by another withdrawal" and
218        //    also inspect the `produced_by` chain for ancestor depth.
219        let (utxo_records, withdrawal_txns) = {
220            let state = self.onchain_state().state();
221            (
222                state.hashi().utxo_pool.utxo_records().clone(),
223                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
224            )
225        };
226
227        let selected_records: Vec<&UtxoRecord> = approval
228            .selected_utxos
229            .iter()
230            .map(|id| {
231                let record = utxo_records
232                    .get(id)
233                    .ok_or_else(|| anyhow!("UTXO {id:?} not found in the pool"))?;
234                anyhow::ensure!(
235                    record.locked_by.is_none(),
236                    "UTXO {id:?} is locked by pending withdrawal {:?}",
237                    record.locked_by.unwrap()
238                );
239                Ok(record)
240            })
241            .collect::<anyhow::Result<_>>()?;
242
243        // 2b. Verify that no selected UTXO has an unconfirmed ancestor
244        //     chain deeper than Bitcoin Core's relay limit. The limit
245        //     (DEFAULT_ANCESTOR_LIMIT = 25) counts the candidate tx
246        //     itself, so the existing chain must leave room for the
247        //     transaction we are about to construct.
248        for record in &selected_records {
249            let depth = unconfirmed_ancestor_depth(record, &withdrawal_txns, &utxo_records);
250            anyhow::ensure!(
251                depth < MAX_ANCESTOR_DEPTH,
252                "UTXO {:?} has an unconfirmed ancestor chain of depth {} \
253                 which, together with the new transaction, would exceed \
254                 Bitcoin Core's ancestor limit of {}",
255                record.utxo.id,
256                depth,
257                MAX_ANCESTOR_DEPTH,
258            );
259        }
260
261        let selected_utxos: Vec<Utxo> = selected_records.iter().map(|r| r.utxo.clone()).collect();
262
263        // 3. Verify output count: one per request, plus at most one change output
264        let request_count = requests.len();
265        let output_count = approval.outputs.len();
266        anyhow::ensure!(
267            output_count == request_count || output_count == request_count + 1,
268            "Expected {} or {} outputs, got {}",
269            request_count,
270            request_count + 1,
271            output_count
272        );
273
274        // 4. Compute miner fee and verify the per-user fee split
275        let input_total: u64 = selected_utxos.iter().map(|u| u.amount).sum();
276        let output_total: u64 = approval.outputs.iter().map(|o| o.amount).sum();
277        anyhow::ensure!(
278            input_total >= output_total,
279            "Inputs ({input_total}) < outputs ({output_total})"
280        );
281        let fee = input_total - output_total;
282
283        let per_user_miner_fee = fee / request_count as u64;
284
285        // Verify per-user miner fee does not exceed worst-case budget
286        let max_network_fee = self.onchain_state().worst_case_network_fee();
287        anyhow::ensure!(
288            per_user_miner_fee <= max_network_fee,
289            "Per-user miner fee {} sats exceeds worst-case budget {} sats",
290            per_user_miner_fee,
291            max_network_fee
292        );
293
294        // Verify each positional withdrawal output matches the expected amount and address.
295        // request.btc_amount is the full withdrawal amount.
296        for (i, request) in requests.iter().enumerate() {
297            let output = &approval.outputs[i];
298            let expected_amount = request.btc_amount - per_user_miner_fee;
299            anyhow::ensure!(
300                expected_amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
301                "Withdrawal output {} sats is below dust threshold {} sats",
302                expected_amount,
303                utxo_pool::TR_DUST_RELAY_MIN_VALUE
304            );
305            anyhow::ensure!(
306                output.amount == expected_amount,
307                "Output {i} amount {} does not match expected {} for request {:?}",
308                output.amount,
309                expected_amount,
310                request.id
311            );
312            anyhow::ensure!(
313                output.bitcoin_address == request.bitcoin_address,
314                "Output {i} address does not match request {:?}",
315                request.id
316            );
317        }
318
319        // 5. Verify change output (if present) goes to hashi root pubkey
320        if output_count == request_count + 1 {
321            let change_output = &approval.outputs[request_count];
322            let expected_address =
323                hashi_bitcoin::witness_program_from_address(&self.get_deposit_address(None)?)?;
324            anyhow::ensure!(
325                change_output.bitcoin_address == expected_address,
326                "Change output does not go to hashi root pubkey"
327            );
328            anyhow::ensure!(
329                change_output.amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
330                "Change output {} sats is below dust threshold {} sats",
331                change_output.amount,
332                utxo_pool::TR_DUST_RELAY_MIN_VALUE
333            );
334        }
335
336        // 6. Validate fee is reasonable.
337        //
338        // TODO: When spending unconfirmed change UTXOs the effective fee
339        // rate that matters for mining is the *package* fee rate across
340        // the entire ancestor chain (CPFP). This check currently
341        // evaluates the transaction in isolation, which may over- or
342        // under-estimate the true cost to get the package mined. A
343        // future revision should compute the aggregate ancestor weight
344        // and fees and validate the package fee rate instead.
345        {
346            // Estimate transaction weight.
347            let num_inputs = selected_utxos.len() as u64;
348            let input_weight = hashi_bitcoin::SCRIPT_PATH_2OF2_TXIN_WEIGHT * num_inputs;
349            let output_weight: u64 = approval
350                .outputs
351                .iter()
352                .map(|o| hashi_bitcoin::output_weight_for_witness_program(&o.bitcoin_address))
353                .collect::<anyhow::Result<Vec<_>>>()?
354                .iter()
355                .sum();
356            let tx_weight =
357                Weight::from_wu(hashi_bitcoin::TX_FIXED_WEIGHT_WU + input_weight + output_weight);
358
359            let min_fee_rate = FeeRate::from_sat_per_vb_unchecked(1);
360
361            // Fee must be at least the minimum relay fee (1 sat/vB).
362            let min_fee = min_fee_rate
363                .fee_wu(tx_weight)
364                .map(|a| a.to_sat())
365                .unwrap_or(0);
366            anyhow::ensure!(
367                fee >= min_fee,
368                "Fee {fee} sats is below minimum relay fee {min_fee} sats"
369            );
370
371            // Fee ceiling: clamp the fee rate from our Bitcoin node to
372            // a floor of 1 sat/vB, then cap at the tolerance multiplier.
373            let kyoto_fee_rate = self
374                .btc_monitor()
375                .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
376                .await?;
377            let clamped_fee_rate = std::cmp::max(kyoto_fee_rate, min_fee_rate);
378            let estimated_fee = clamped_fee_rate
379                .fee_wu(tx_weight)
380                .map(|a| a.to_sat())
381                .unwrap_or(0);
382            let max_fee = estimated_fee.saturating_mul(FEE_RATE_TOLERANCE_MULTIPLIER);
383            anyhow::ensure!(
384                fee <= max_fee,
385                "Fee {fee} sats exceeds maximum allowed {max_fee} sats \
386                 ({FEE_RATE_TOLERANCE_MULTIPLIER}x the clamped estimate of \
387                 {estimated_fee} sats at {clamped_fee_rate})"
388            );
389        }
390
391        // 7. Rebuild unsigned tx and verify txid matches.
392        let tx = self.build_unsigned_withdrawal_tx(&selected_utxos, &approval.outputs)?;
393        let expected_txid = BitcoinTxid::from(tx.compute_txid());
394        anyhow::ensure!(
395            approval.txid == expected_txid,
396            "Txid mismatch: approval has {:?}, rebuilt tx has {:?}",
397            approval.txid,
398            expected_txid
399        );
400
401        Ok(())
402    }

466    #[tracing::instrument(level = "info", skip_all, fields(withdrawal_id = %message.withdrawal_id))]
467    pub fn validate_and_sign_withdrawal_tx_signing(
468        &self,
469        message: &WithdrawalTxSigning,
470    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
471        let txn = self
472            .onchain_state()
473            .withdrawal_txn(&message.withdrawal_id)
474            .ok_or_else(|| {
475                anyhow!(
476                    "WithdrawalTransaction {} not found on-chain",
477                    message.withdrawal_id
478                )
479            })?;
480
481        anyhow::ensure!(
482            txn.signatures.is_none(),
483            "WithdrawalTransaction {} is already signed",
484            message.withdrawal_id
485        );
486
487        anyhow::ensure!(
488            message.request_ids == txn.request_ids,
489            "Request IDs mismatch for WithdrawalTransaction {}",
490            message.withdrawal_id
491        );
492
493        anyhow::ensure!(
494            message.signatures.len() == txn.inputs.len(),
495            "MPC signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
496            message.signatures.len(),
497            txn.inputs.len(),
498            message.withdrawal_id
499        );
500        anyhow::ensure!(
501            message.guardian_signatures.len() == txn.inputs.len(),
502            "Guardian signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
503            message.guardian_signatures.len(),
504            txn.inputs.len(),
505            message.withdrawal_id
506        );
507
508        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
509        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
510        let guardian_btc_pubkey = self.guardian_btc_pubkey().copied().ok_or_else(|| {
511            anyhow!("Guardian BTC pubkey not yet pinned; cannot validate withdrawal")
512        })?;
513        let guardian_schnorr_pk =
514            SchnorrPublicKey::from_byte_array(&guardian_btc_pubkey.serialize())
515                .map_err(|e| anyhow!("Failed to convert guardian BTC pubkey: {e}"))?;
516
517        for (i, ((mpc_sig_bytes, guardian_sig_bytes), sighash)) in message
518            .signatures
519            .iter()
520            .zip(message.guardian_signatures.iter())
521            .zip(signing_messages.iter())
522            .enumerate()
523        {
524            // MPC: verify against the derived hashi child key.
525            let mpc_arr: &[u8; 64] = mpc_sig_bytes.as_slice().try_into().map_err(|_| {
526                anyhow!(
527                    "MPC signature {i} is not 64 bytes for WithdrawalTransaction {}",
528                    message.withdrawal_id
529                )
530            })?;
531            let mpc_sig = SchnorrSignature::from_byte_array(mpc_arr)
532                .map_err(|e| anyhow!("Invalid MPC Schnorr signature at input {i}: {e}"))?;
533            let input_pubkey = self.deposit_pubkey(txn.inputs[i].derivation_path.as_ref())?;
534            let mpc_schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
535                .map_err(|e| anyhow!("Failed to convert mpc pubkey for input {i}: {e}"))?;
536            mpc_schnorr_pk
537                .verify(sighash, &mpc_sig)
538                .map_err(|e| anyhow!("MPC signature verification failed for input {i}: {e}"))?;
539
540            // Guardian: verify against the on-chain enclave BTC pubkey.
541            // Same sighash — both sigs commit to the same multi_a leaf.
542            let guardian_arr: &[u8; 64] =
543                guardian_sig_bytes.as_slice().try_into().map_err(|_| {
544                    anyhow!(
545                        "Guardian signature {i} is not 64 bytes for WithdrawalTransaction {}",
546                        message.withdrawal_id
547                    )
548                })?;
549            let guardian_sig = SchnorrSignature::from_byte_array(guardian_arr)
550                .map_err(|e| anyhow!("Invalid guardian Schnorr signature at input {i}: {e}"))?;
551            guardian_schnorr_pk
552                .verify(sighash, &guardian_sig)
553                .map_err(|e| {
554                    anyhow!("Guardian signature verification failed for input {i}: {e}")
555                })?;
556        }
557
558        self.sign_message_proto(message)
559    }

572    pub(crate) fn sign_message_proto_at_epoch<T: serde::Serialize>(
573        &self,
574        message: &T,
575        epoch: u64,
576    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
577        let validator_address = self
578            .config
579            .validator_address()
580            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
581        let committee = self
582            .onchain_state()
583            .state()
584            .hashi()
585            .committees
586            .committees()
587            .get(&epoch)
588            .cloned()
589            .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?;
590        let private_key =
591            self.find_signing_key_for_committee(&committee, validator_address, epoch)?;
592        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();
593        let signature_bytes = private_key
594            .sign(epoch, validator_address, message)
595            .signature()
596            .as_bytes()
597            .to_vec()
598            .into();
599
600        Ok(hashi_types::proto::MemberSignature {
601            epoch: Some(epoch),
602            address: Some(validator_address.to_string()),
603            public_key: Some(public_key_bytes),
604            signature: Some(signature_bytes),
605        })
606    }

610    /// Rebuild the transition from on-chain state and sign with the historical key.
611    #[tracing::instrument(level = "info", skip_all, fields(from_epoch))]
612    pub fn validate_and_sign_committee_transition(
613        &self,
614        from_epoch: u64,
615    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
616        let validator_address = self
617            .config
618            .validator_address()
619            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
620
621        let onchain = self.onchain_state();
622        let state = onchain.state();
623        let committees_map = state.hashi().committees.committees();
624        let from_committee = committees_map
625            .get(&from_epoch)
626            .ok_or_else(|| anyhow!("no on-chain committee for epoch {from_epoch}"))?;
627        if !from_committee
628            .members()
629            .iter()
630            .any(|m| m.validator_address() == validator_address)
631        {
632            anyhow::bail!("not a member of the committee at epoch {from_epoch}");
633        }
634
635        // Hashi committee epochs are sparse: the next entry after `from_epoch`
636        // is generally not `from_epoch + 1`. Both leader and followers derive
637        // the same `to_epoch` from on-chain state, so they sign the same
638        // transition.
639        let new_committee = committees_map
640            .range((from_epoch + 1)..)
641            .next()
642            .map(|(_, c)| c)
643            .ok_or_else(|| anyhow!("no on-chain committee epoch after {from_epoch}"))?;
644
645        let transition = hashi_types::guardian::CommitteeTransitionRequest {
646            new_committee: hashi_types::move_types::Committee::from(new_committee),
647        };
648
649        self.sign_message_proto_at_epoch(&transition, from_epoch)
650    }

702    /// Produce MPC Schnorr signatures for an unsigned withdrawal transaction.
703    #[tracing::instrument(
704        level = "debug",
705        skip_all,
706        fields(withdrawal_txn_id = %txn.id, input_count = txn.inputs.len()),
707    )]
708    async fn mpc_sign_withdrawal_tx(
709        &self,
710        txn: &WithdrawalTransaction,
711        unsigned_tx: &bitcoin::Transaction,
712        expected_limiter_seq: Option<u64>,
713        timestamp_secs: Option<u64>,
714        sink: tokio::sync::mpsc::Sender<
715            Result<hashi_types::proto::SignWithdrawalTransactionPartial, tonic::Status>,
716        >,
717    ) -> anyhow::Result<()> {
718        let onchain_state = self.onchain_state().clone();
719        let epoch = onchain_state.epoch();
720        if txn.epoch != epoch {
721            anyhow::bail!(
722                "Stale presig assignment: pending withdrawal {} has epoch {}, current is {}. \
723                 Either the leader hasn't called allocate_presigs_for_withdrawal_txn yet, \
724                 or this node's on-chain state is behind.",
725                txn.id,
726                txn.epoch,
727                epoch,
728            );
729        }
730        // Validate-only: the watcher advances the limiter on
731        // `WithdrawalSignedEvent`, never from this path.
732        match (self.local_limiter(), expected_limiter_seq) {
733            (Some(limiter), Some(expected_seq)) => {
734                let amount_sats = withdrawal_limiter_consumption_amount(txn);
735                // Leader's live checkpoint time (drift-bounded) so a throttled
736                // batch becomes signable as it refills; pre-upgrade leaders omit it.
737                let timestamp_secs = match timestamp_secs {
738                    Some(ts) => {
739                        self.bound_leader_timestamp(ts)?;
740                        ts
741                    }
742                    None => txn.timestamp_ms / 1000,
743                };
744                let result = limiter.validate_consume(expected_seq, timestamp_secs, amount_sats);
745                self.metrics.record_limiter_validate(
746                    &result,
747                    crate::metrics::GUARDIAN_LIMITER_CALLSITE_MPC_SIGNING,
748                );
749                result.map_err(|e| anyhow!("Limiter rejected withdrawal {}: {e}", txn.id))?;
750            }
751            (None, None) => {}
752            (Some(_), None) => anyhow::bail!(
753                "Local limiter is configured but request for withdrawal {} lacks expected_limiter_seq",
754                txn.id
755            ),
756            (None, Some(_)) => anyhow::bail!(
757                "Request for withdrawal {} carries expected_limiter_seq but local limiter is not configured",
758                txn.id
759            ),
760        }
761        let p2p_channel =
762            RpcP2PChannel::new(onchain_state, epoch, crate::metrics::MPC_LABEL_SIGNING);
763        let signing_manager = self.signing_manager_for(epoch).ok_or_else(|| {
764            anyhow::anyhow!(
765                "SigningManager not available for epoch {epoch}; \
766                 reconciliation may be catching up"
767            )
768        })?;
769        let beacon = S::from_bytes_mod_order(&txn.randomness);
770        let signing_messages = self.withdrawal_signing_messages(unsigned_tx, &txn.inputs)?;
771        let concurrency = self.config.withdrawal_signing_concurrency();
772        let signing_manager_ref = &signing_manager;
773        let p2p_channel_ref = &p2p_channel;
774        let beacon_ref = &beacon;
775        let metrics_ref = &*self.metrics;
776        let txn_id = txn.id;
777        let presig_start = txn.presig_start_index;
778        let inputs = &txn.inputs;
779        let sink_ref = &sink;
780        // Run the per-input MPC signs concurrently (bounded by `concurrency`).
781        futures::stream::iter(signing_messages.into_iter().enumerate())
782            .map(|(input_index, message)| async move {
783                let request_id = withdrawal_input_signing_request_id(&txn_id, input_index as u32);
784                // Change UTXOs (`derivation_path = None`) ride the `[0; 32]`
785                // path everywhere else (leaf script, `deposit_pubkey`). MPC
786                // must too — passing `None` signs for master `G`, not the
787                // `derive(G, [0; 32])` child the 2-of-2 leaf binds.
788                let derivation_address = inputs
789                    .get(input_index)
790                    .map(|input| {
791                        crate::deposits::path_bytes_or_zero(input.derivation_path.as_ref())
792                    })
793                    .expect("input_index iterated from signing_messages.len() == txn.inputs.len()");
794                let sign_start = std::time::Instant::now();
795                let global_presig_index = presig_start + input_index as u64;
796                let sign_result = signing_manager_ref
797                    .sign(
798                        p2p_channel_ref,
799                        request_id,
800                        &message,
801                        global_presig_index,
802                        beacon_ref,
803                        Some(&derivation_address),
804                        WITHDRAWAL_SIGNING_TIMEOUT,
805                        metrics_ref,
806                    )
807                    .await;
808                let sign_duration = sign_start.elapsed().as_secs_f64();
809                match &sign_result {
810                    Ok(_) => {
811                        metrics_ref
812                            .mpc_sign_duration_seconds
813                            .with_label_values(&["success"])
814                            .observe(sign_duration);
815                        metrics_ref
816                            .presig_pool_remaining
817                            .set(signing_manager_ref.presignatures_remaining() as i64);
818                    }
819                    Err(e) => {
820                        metrics_ref
821                            .mpc_sign_duration_seconds
822                            .with_label_values(&["failure"])
823                            .observe(sign_duration);
824                        let reason = match e {
825                            crate::mpc::types::SigningError::Timeout { .. } => "timeout",
826                            crate::mpc::types::SigningError::PoolExhausted => "pool_exhausted",
827                            crate::mpc::types::SigningError::TooManyInvalidSignatures {
828                                ..
829                            } => "too_many_invalid",
830                            crate::mpc::types::SigningError::CryptoError(_) => "crypto_error",
831                            _ => "other",
832                        };
833                        metrics_ref
834                            .mpc_sign_failures_total
835                            .with_label_values(&[reason])
836                            .inc();
837                    }
838                }
839                let partial = sign_result
840                    .map(|sig| hashi_types::proto::SignWithdrawalTransactionPartial {
841                        input_index: input_index as u32,
842                        signature: sig.to_byte_array().to_vec().into(),
843                    })
844                    .map_err(|e| {
845                        tonic::Status::internal(format!(
846                            "Failed to sign withdrawal transaction input {input_index}: {e}"
847                        ))
848                    });
849                let _ = sink_ref.send(partial).await;
850            })
851            .buffer_unordered(concurrency)
852            .for_each(|()| std::future::ready(()))
853            .await;
854        Ok(())
855    }

858        &self,
859        unsigned_tx: &bitcoin::Transaction,
860        inputs: &[Utxo],
861    ) -> anyhow::Result<Vec<[u8; 32]>> {
862        let spend_inputs = inputs
863            .iter()
864            .map(|input| {
865                let address = self.get_deposit_address(input.derivation_path.as_ref())?;
866                let (_, _, leaf_hash) =
867                    self.deposit_spend_artifacts(input.derivation_path.as_ref())?;

869                    TxOut {
870                        value: Amount::from_sat(input.amount),
871                        script_pubkey: address.script_pubkey(),
872                    },
873                    leaf_hash,
874                ))
875            })
876            .collect::<anyhow::Result<Vec<_>>>()?;
877        let prevouts = spend_inputs
878            .iter()
879            .map(|(txout, _)| txout.clone())
880            .collect::<Vec<_>>();
881        let leaf_hashes = spend_inputs
882            .iter()
883            .map(|(_, leaf_hash)| *leaf_hash)
884            .collect::<Vec<TapLeafHash>>();
885
886        Ok(hashi_bitcoin::taproot_script_spend_sighashes(
887            unsigned_tx,
888            &prevouts,
889            &leaf_hashes,
890        ))
891    }

924    /// Build a withdrawal commitment for a batch of approved requests: select
925    /// UTXOs using the batching-aware coin selection algorithm, build the
926    /// unsigned BTC tx, and return a `WithdrawalTxCommitment` covering the
927    /// selected requests.
928    #[tracing::instrument(level = "debug", skip_all, fields(request_count = requests.len()))]
929    pub async fn build_withdrawal_tx_commitment(
930        &self,
931        requests: &[WithdrawalRequest],
932    ) -> Result<WithdrawalTxCommitment, WithdrawalCommitmentError> {
933        // Fetch current fee rate from the Bitcoin node, clamped to a high
934        // fee rate threshold to avoid overpaying during fee spikes.
935        let kyoto_fee_rate = self
936            .btc_monitor()
937            .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
938            .await
939            .map_err(|e| WithdrawalCommitmentError::FeeEstimateFailed(anyhow!(e)))?;
940        let min_fee_rate = CoinSelectionParams::DEFAULT_MIN_FEE_RATE;
941        let max_fee_rate = CoinSelectionParams::DEFAULT_HIGH_FEE_RATE_THRESHOLD;
942        let fee_rate = kyoto_fee_rate.clamp(min_fee_rate, max_fee_rate);
943
944        let change_address = self
945            .get_deposit_address(None)
946            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
947
948        let configured_max_inputs = CoinSelectionParams::DEFAULT_MAX_INPUTS;
949        let configured_long_term_fee_rate = CoinSelectionParams::DEFAULT_LONG_TERM_FEE_RATE;
950        let configured_max_requests = self.config.withdrawal_max_batch_size().min(requests.len());
951
952        // Snapshot both maps under a single read-lock so they are always
953        // mutually consistent (e.g., a WithdrawalConfirmedEvent cannot update
954        // one map but not the other between the two reads).
955        let (withdrawal_txns, utxo_records) = {
956            let state = self.onchain_state().state();
957            (
958                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
959                state.hashi().utxo_pool.utxo_records().clone(),
960            )
961        };
962
963        // Query Bitcoin in parallel for the confirmation count of every
964        // pending withdrawal so we can accurately fill AncestorTx::confirmations
965        // instead of always hardcoding 0.
966        let tx_confirmations = fetch_withdrawal_tx_confirmations(self, &withdrawal_txns).await;
967
968        // Map available (unlocked) UTXOs to UtxoCandidates.
969        let candidates: Vec<UtxoCandidate> = utxo_records
970            .values()
971            .filter(|r| r.locked_by.is_none())
972            .map(|r| {
973                let status =
974                    build_utxo_status(self, r, &withdrawal_txns, &tx_confirmations, &utxo_records);
975                UtxoCandidate {
976                    id: r.utxo.id,
977                    amount: r.utxo.amount,
978                    spend_path: SpendPath::TaprootScriptPath2of2,
979                    status,
980                }
981            })
982            .collect();
983
984        // Map on-chain WithdrawalRequests to the coin-selector view.
985        // btc_amount is the full withdrawal amount.
986        let mapped_requests: Vec<utxo_pool::WithdrawalRequest> = requests
987            .iter()
988            .map(|r| utxo_pool::WithdrawalRequest {
989                id: r.id,
990                recipient: r.bitcoin_address.clone(),
991                amount: r.btc_amount,
992                timestamp_ms: r.timestamp_ms,
993            })
994            .collect();
995
996        let mut last_selection_error = None;
997        let mut result = None;
998        for request_count in (1..=configured_max_requests).rev() {
999            let max_inputs = safe_withdrawal_flow_max_inputs(request_count, configured_max_inputs);
1000            if max_inputs == 0 {
1001                continue;
1002            }
1003
1004            let params = CoinSelectionParams {
1005                max_inputs,
1006                long_term_fee_rate: configured_long_term_fee_rate,
1007                max_fee_per_request: self.onchain_state().worst_case_network_fee(),
1008                max_withdrawal_requests: request_count,
1009                max_mempool_chain_depth: self.config.max_mempool_chain_depth(),
1010                ..CoinSelectionParams::new(change_address.clone())
1011            };
1012
1013            match utxo_pool::select_coins(&candidates, &mapped_requests, &params, fee_rate) {
1014                Ok(selection) => {
1015                    if request_count < configured_max_requests {
1016                        tracing::info!(
1017                            selected_requests = selection.selected_requests.len(),
1018                            selected_inputs = selection.inputs.len(),
1019                            configured_max_requests,
1020                            configured_max_inputs,
1021                            max_inputs,
1022                            "Reduced withdrawal batch to stay within Sui commit limits",
1023                        );
1024                    }
1025                    result = Some(selection);
1026                    break;
1027                }
1028                Err(e) => last_selection_error = Some(e),
1029            }
1030        }
1031
1032        let result = result.ok_or_else(|| {
1033            WithdrawalCommitmentError::UtxoSelectionFailed(anyhow!(
1034                last_selection_error
1035                    .map(|e| e.to_string())
1036                    .unwrap_or_else(|| "no withdrawal request count fits Sui commit limits".into())
1037            ))
1038        })?;
1039
1040        // Build outputs: one per selected request (net amount already deducted),
1041        // plus an optional change output.
1042        let mut outputs: Vec<OutputUtxo> = result
1043            .withdrawal_outputs
1044            .iter()
1045            .map(|o| OutputUtxo {
1046                amount: o.amount,
1047                bitcoin_address: o.recipient.clone(),
1048            })
1049            .collect();
1050
1051        if let Some(change_amount) = result.change {
1052            outputs.push(OutputUtxo {
1053                amount: change_amount,
1054                bitcoin_address: hashi_bitcoin::witness_program_from_address(&change_address)
1055                    .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?,
1056            });
1057        }
1058
1059        let selected_utxos: Vec<UtxoId> = result.inputs.iter().map(|u| u.id).collect();
1060        let request_ids: Vec<Address> = result.selected_requests.iter().map(|r| r.id).collect();
1061
1062        // Resolve UtxoCandidates back to full Utxo objects for tx building.
1063        let selected_input_utxos: Vec<Utxo> = result
1064            .inputs
1065            .iter()
1066            .map(|c| self.onchain_state().active_utxo(&c.id))
1067            .collect::<Option<Vec<_>>>()
1068            .ok_or_else(|| {
1069                WithdrawalCommitmentError::BtcTxBuildFailed(anyhow!(
1070                    "a selected UTXO disappeared from the pool between selection and tx build"
1071                ))
1072            })?;
1073
1074        let tx = self
1075            .build_unsigned_withdrawal_tx(&selected_input_utxos, &outputs)
1076            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
1077        let txid = BitcoinTxid::from(tx.compute_txid());
1078
1079        Ok(WithdrawalTxCommitment {
1080            request_ids,
1081            selected_utxos,
1082            outputs,
1083            txid,
1084        })
1085    }

1087    /// Run AML/Sanctions checks for a withdrawal request.
1088    /// If no screener client is configured, checks are skipped.
1089    #[tracing::instrument(level = "debug", skip_all, fields(request_id = %request.id))]
1090    pub(crate) async fn screen_withdrawal(
1091        &self,
1092        request: &WithdrawalRequest,
1093    ) -> Result<(), WithdrawalApprovalError> {
1094        let Some(screener) = self.screener_client() else {
1095            tracing::debug!("AML checks skipped: no screener configured");
1096            return Ok(());
1097        };
1098
1099        // Source: Sui tx digest (base58 string)
1100        let source_tx_hash = request.sui_tx_digest.to_string();
1101
1102        // Destination: Bitcoin address (raw witness bytes -> bech32 string)
1103        let destination_address = hashi_bitcoin::address_string_from_witness_program(
1104            &request.bitcoin_address,
1105            self.config.bitcoin_network(),
1106        )
1107        .map_err(WithdrawalApprovalError::NeverRetry)?;
1108
1109        let approved = screener
1110            .approve_withdrawal(
1111                &source_tx_hash,
1112                &destination_address,
1113                self.config.sui_chain_id(),
1114                self.config.bitcoin_chain_id(),
1115            )
1116            .await
1117            .map_err(|e| WithdrawalApprovalError::AmlServiceError(anyhow!(e)))?;
1118
1119        if !approved {
1120            return Err(WithdrawalApprovalError::NeverRetry(anyhow!(
1121                "AML checks failed for withdrawal request {:?} to {}",
1122                request.id,
1123                destination_address,
1124            )));
1125        }
1126
1127        Ok(())
1128    }

1273/// Query Bitcoin in parallel for the confirmation count of every pending
1274/// withdrawal transaction. Returns a map from withdrawal ID to confirmation
1275/// count. Withdrawals that are in the mempool, not found, or whose RPC call
1276/// fails are mapped to 0 (treated as unconfirmed).
1277async fn fetch_withdrawal_tx_confirmations(
1278    hashi: &Hashi,
1279    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1280) -> HashMap<Address, u32> {
1281    let futures: Vec<_> = withdrawal_txns
1282        .iter()
1283        .map(|(id, txn)| async {
1284            let btc_txid = txn.txid.into();
1285            let confs = match hashi.btc_monitor().get_transaction_status(btc_txid).await {
1286                Ok(TxStatus::Confirmed { confirmations }) => confirmations,
1287                // Mempool, not found, or RPC error — treat as unconfirmed.
1288                _ => 0,
1289            };
1290            (*id, confs)
1291        })
1292        .collect();
1293    futures::future::join_all(futures)
1294        .await
1295        .into_iter()
1296        .collect()
1297}

1299/// Build the [`UtxoStatus`] for a UTXO record using a pre-fetched snapshot.
1300///
1301/// For confirmed UTXOs (`produced_by = None`) this is simply
1302/// [`UtxoStatus::Confirmed`]. For unconfirmed change outputs
1303/// (`produced_by = Some(withdrawal_id)`) we walk the full ancestor chain
1304/// so that CPFP weight and mempool depth are accurately computed even for
1305/// multi-level chains. If the producing withdrawal has already been removed
1306/// from `withdrawal_txns` (confirmed and cleared), we promote the UTXO
1307/// to `Confirmed` — it is safe to spend immediately.
1308fn build_utxo_status(
1309    hashi: &Hashi,
1310    record: &UtxoRecord,
1311    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1312    tx_confirmations: &HashMap<Address, u32>,
1313    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1314) -> UtxoStatus {
1315    let Some(producing_id) = record.produced_by else {
1316        return UtxoStatus::Confirmed;
1317    };
1318
1319    let chain = build_ancestor_chain(
1320        hashi,
1321        producing_id,
1322        withdrawal_txns,
1323        tx_confirmations,
1324        utxo_records,
1325    );
1326
1327    if chain.is_empty() {
1328        // The producing withdrawal was confirmed and removed from
1329        // withdrawal_txns. The UTXO is safe to spend.
1330        UtxoStatus::Confirmed
1331    } else {
1332        UtxoStatus::Pending { chain }
1333    }
1334}

1336/// Maximum ancestor depth permitted by Bitcoin Core's relay policy
1337/// (`DEFAULT_ANCESTOR_LIMIT = 25`). Bitcoin Core counts the candidate
1338/// transaction itself in the ancestor set, so a UTXO whose existing
1339/// unconfirmed ancestor depth is already `MAX_ANCESTOR_DEPTH - 1` is
1340/// the deepest we can safely spend.
1341pub const MAX_ANCESTOR_DEPTH: usize = 25;
1342
1343/// Count the number of unconfirmed ancestors for a UTXO record by walking
1344/// the `produced_by` chain. Every ancestor that still appears in
1345/// `withdrawal_txns` is conservatively treated as unconfirmed (we skip
1346/// querying Bitcoin for actual confirmation counts). This is used during
1347/// commitment validation to reject UTXOs whose ancestor chain would exceed
1348/// Bitcoin Core's relay limit.
1349///
1350/// The walk is a BFS over the ancestor DAG. Each item on the queue is a
1351/// `(producing_withdrawal_id, depth)` pair. We track the maximum depth
1352/// seen across all branches.
1353fn unconfirmed_ancestor_depth(
1354    record: &UtxoRecord,
1355    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1356    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1357) -> usize {
1358    let Some(producing_id) = record.produced_by else {
1359        return 0;
1360    };
1361
1362    let mut max_depth: usize = 0;
1363    let mut queue = std::collections::VecDeque::new();
1364    queue.push_back((producing_id, 1usize));
1365
1366    while let Some((wid, depth)) = queue.pop_front() {
1367        if depth > MAX_ANCESTOR_DEPTH {
1368            return depth;
1369        }
1370
1371        let Some(txn) = withdrawal_txns.get(&wid) else {
1372            // The producing withdrawal has been confirmed and removed;
1373            // it does not contribute to the unconfirmed chain.
1374            continue;
1375        };
1376
1377        max_depth = std::cmp::max(max_depth, depth);
1378
1379        // Enqueue any inputs that are themselves unconfirmed change
1380        // outputs of an earlier withdrawal.
1381        for input_utxo in &txn.inputs {
1382            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1383                && let Some(parent_id) = input_record.produced_by
1384            {
1385                queue.push_back((parent_id, depth + 1));
1386            }
1387        }
1388    }
1389
1390    max_depth
1391}

1393/// Build the ancestor chain for a UTXO produced by `producing_id`. Each
1394/// unconfirmed ancestor that still appears in `withdrawal_txns` gets
1395/// one [`AncestorTx`] entry with its confirmation count, weight, and fee.
1396/// The walk is a BFS over the ancestor DAG, capped at
1397/// [`MAX_ANCESTOR_DEPTH`] levels.
1398fn build_ancestor_chain(
1399    hashi: &Hashi,
1400    producing_id: Address,
1401    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1402    tx_confirmations: &HashMap<Address, u32>,
1403    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1404) -> Vec<AncestorTx> {
1405    let mut chain = Vec::new();
1406    let mut queue = std::collections::VecDeque::new();
1407    queue.push_back((producing_id, 0usize));
1408
1409    while let Some((wid, depth)) = queue.pop_front() {
1410        if depth >= MAX_ANCESTOR_DEPTH {
1411            continue;
1412        }
1413
1414        let Some(txn) = withdrawal_txns.get(&wid) else {
1415            continue;
1416        };
1417
1418        let Ok(tx) = hashi.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs()) else {
1419            continue;
1420        };
1421
1422        let confirmations = tx_confirmations.get(&wid).copied().unwrap_or(0);
1423        let input_total: u64 = txn.inputs.iter().map(|u| u.amount).sum();
1424        let output_total: u64 = txn.all_outputs().iter().map(|o| o.amount).sum();
1425
1426        chain.push(AncestorTx {
1427            confirmations,
1428            tx_weight: tx.weight(),
1429            tx_fee: input_total.saturating_sub(output_total),
1430        });
1431
1432        for input_utxo in &txn.inputs {
1433            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1434                && let Some(parent_id) = input_record.produced_by
1435            {
1436                queue.push_back((parent_id, depth + 1));
1437            }
1438        }
1439    }
1440
1441    chain
1442}

1444/// Deterministic from on-chain state and the leader-supplied
1445/// `(timestamp_secs, seq)`, so every validator reconstructs the same request.
1446pub fn build_guardian_withdrawal_request(
1447    hashi: &Hashi,
1448    txn: &WithdrawalTransaction,
1449    timestamp_secs: u64,
1450    seq: u64,
1451) -> anyhow::Result<hashi_types::guardian::StandardWithdrawalRequest> {
1452    use hashi_types::bitcoin::InputUTXO;
1453    use hashi_types::bitcoin::OutputUTXOWire;
1454    use hashi_types::bitcoin::TxUTXOs;
1455
1456    let network = hashi.config.bitcoin_network();
1457
1458    let inputs: Vec<_> = txn.inputs.iter().map(InputUTXO::from).collect();
1459
1460    // First N outputs are external payouts; any trailing output is internal change.
1461    let all_outputs = txn.all_outputs();
1462    let num_requests = txn.request_ids.len();
1463    let outputs = all_outputs
1464        .iter()
1465        .enumerate()
1466        .map(|(i, output)| {
1467            if i < num_requests {
1468                let script_pubkey =
1469                    hashi_bitcoin::script_pubkey_from_witness_program(&output.bitcoin_address)?;
1470                let address =
1471                    hashi_bitcoin::BitcoinAddress::from_script(&script_pubkey, network)
1472                        .map_err(|e| anyhow!("Cannot derive address from output script: {e}"))?;
1473                Ok(OutputUTXOWire::external(
1474                    address.into_unchecked(),
1475                    Amount::from_sat(output.amount),
1476                ))
1477            } else {
1478                Ok(OutputUTXOWire::internal(
1479                    sui_sdk_types::Address::ZERO,
1480                    Amount::from_sat(output.amount),
1481                ))
1482            }
1483        })
1484        .collect::<anyhow::Result<Vec<_>>>()?;
1485
1486    let utxos = TxUTXOs::new(inputs, outputs, network)
1487        .map_err(|e| anyhow!("Failed to build guardian TxUTXOs: {e}"))?;
1488
1489    // The on-chain `WithdrawalTransaction` UID doubles as the guardian-side `wid`.
1490    let wid: hashi_types::guardian::WithdrawalID = txn.id;
1491
1492    Ok(hashi_types::guardian::StandardWithdrawalRequest::new(
1493        wid,
1494        utxos,
1495        timestamp_secs,
1496        seq,
1497    ))
1498}
