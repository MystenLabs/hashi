52fn select_withdrawal_signing_indices(
53    signing: &hashi_types::move_types::SigningBatch,
54    requested_input_indices: &[u64],
55) -> anyhow::Result<Vec<usize>> {
56    if requested_input_indices.is_empty() {
57        return Ok(signing
58            .unsigned_indices()
59            .into_iter()
60            .map(|i| i as usize)
61            .collect());
62    }
63
64    let mut seen = BTreeSet::new();
65    let mut selected = Vec::with_capacity(requested_input_indices.len());
66    for &input_index in requested_input_indices {
67        if !seen.insert(input_index) {
68            anyhow::bail!("duplicate input index {input_index} in withdrawal signing request");
69        }
70        let i = usize::try_from(input_index)
71            .map_err(|_| anyhow!("input index {input_index} out of range"))?;
72        if i >= signing.num_inputs() {
73            anyhow::bail!(
74                "input index {input_index} out of range for withdrawal with {} inputs",
75                signing.num_inputs()
76            );
77        }
78        if signing.pending_index(i).is_none() {
79            anyhow::bail!("input index {input_index} is already signed");
80        }
81        selected.push(i);
82    }
83    Ok(selected)
84}

86/// BTC that leaves the pool when this txn broadcasts — the amount that
87/// consumes the guardian's limit. Equivalent to
88/// `sum(withdrawal_outputs) + miner_fee`; we use `inputs - change` to
89/// avoid relying on a separate fee field.
90pub(crate) fn withdrawal_limiter_consumption_amount(txn: &WithdrawalTransaction) -> u64 {
91    let inputs: u64 = txn.inputs.iter().map(|u| u.amount).sum();
92    let change: u64 = txn.change_outputs.iter().map(|c| c.amount).sum();
93    inputs.saturating_sub(change)
94}

96/// Conservative runtime-object budget for withdrawal commit transactions.
97///
98/// Sui's hard object-runtime cache limit is 1000 objects. Empirical
99/// measurements put withdrawal commit transactions at roughly
100/// `selected_utxos + 3 * requests + fixed_overhead` runtime objects. Target
101/// 922 to leave 7.8% headroom below the hard 1000 cap.
102const WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET: usize = 922;
103const WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS: usize = 12;
104const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST: usize = 3;
105const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT: usize = 1;
106
107fn safe_withdrawal_commit_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
108    let request_objects =
109        request_count.saturating_mul(WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST);
110    let runtime_input_budget = WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET
111        .saturating_sub(WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS)
112        .saturating_sub(request_objects);
113    let input_budget = runtime_input_budget / WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT;
114    configured_max_inputs.min(input_budget)
115}
116
117/// Confirm is no longer input-bound: spent UTXOs emit events only (0
118/// objects per input). The confirm transaction's object count is
119/// `requests + ~43` fixed overhead — well within the 1000-object limit
120/// for any practical request count. The commit path is the binding
121/// constraint.
122fn safe_withdrawal_flow_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
123    let request_input_budget =
124        request_count.saturating_mul(CoinSelectionParams::DEFAULT_INPUT_BUDGET);
125    configured_max_inputs
126        .min(request_input_budget)
127        .min(safe_withdrawal_commit_max_inputs(
128            request_count,
129            configured_max_inputs,
130        ))
131}

218    #[tracing::instrument(level = "debug", skip_all, fields(bitcoin_txid = %approval.txid))]
219    pub async fn validate_withdrawal_tx_commitment(
220        &self,
221        approval: &WithdrawalTxCommitment,
222    ) -> anyhow::Result<()> {
223        anyhow::ensure!(!approval.request_ids.is_empty(), "No request IDs");
224        anyhow::ensure!(!approval.selected_utxos.is_empty(), "No selected UTXOs");
225        anyhow::ensure!(!approval.outputs.is_empty(), "No outputs");
226
227        // Check for duplicate request IDs
228        let unique_request_ids: std::collections::BTreeSet<_> =
229            approval.request_ids.iter().collect();
230        anyhow::ensure!(
231            unique_request_ids.len() == approval.request_ids.len(),
232            "Duplicate request IDs"
233        );
234
235        // Check for duplicate UTXO IDs
236        let unique_utxo_ids: std::collections::BTreeSet<_> =
237            approval.selected_utxos.iter().collect();
238        anyhow::ensure!(
239            unique_utxo_ids.len() == approval.selected_utxos.len(),
240            "Duplicate UTXO IDs"
241        );
242
243        // 1. Verify each request_id exists and is approved
244        let requests: Vec<WithdrawalRequest> = approval
245            .request_ids
246            .iter()
247            .map(|id| {
248                let request = self
249                    .onchain_state()
250                    .withdrawal_request(id)
251                    .ok_or_else(|| anyhow!("Withdrawal request {id} not found in queue"))?;
252                anyhow::ensure!(
253                    request.status.is_approved(),
254                    "Withdrawal request {id} has not been approved"
255                );
256                Ok(request)
257            })
258            .collect::<anyhow::Result<_>>()?;
259
260        // 2. Verify each selected UTXO exists, is not locked, and collect
261        //    full UTXO data. We look up via utxo_records so we can
262        //    distinguish "missing" from "locked by another withdrawal" and
263        //    also inspect the `produced_by` chain for ancestor depth.
264        let (utxo_records, withdrawal_txns) = {
265            let state = self.onchain_state().state();
266            (
267                state.hashi().utxo_pool.utxo_records().clone(),
268                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
269            )
270        };
271
272        let selected_records: Vec<&UtxoRecord> = approval
273            .selected_utxos
274            .iter()
275            .map(|id| {
276                let record = utxo_records
277                    .get(id)
278                    .ok_or_else(|| anyhow!("UTXO {id:?} not found in the pool"))?;
279                anyhow::ensure!(
280                    record.locked_by.is_none(),
281                    "UTXO {id:?} is locked by pending withdrawal {:?}",
282                    record.locked_by.unwrap()
283                );
284                Ok(record)
285            })
286            .collect::<anyhow::Result<_>>()?;
287
288        // 2b. Verify that no selected UTXO has an unconfirmed ancestor
289        //     chain deeper than Bitcoin Core's relay limit. The limit
290        //     (DEFAULT_ANCESTOR_LIMIT = 25) counts the candidate tx
291        //     itself, so the existing chain must leave room for the
292        //     transaction we are about to construct.
293        for record in &selected_records {
294            let depth = unconfirmed_ancestor_depth(record, &withdrawal_txns, &utxo_records);
295            anyhow::ensure!(
296                depth < MAX_ANCESTOR_DEPTH,
297                "UTXO {:?} has an unconfirmed ancestor chain of depth {} \
298                 which, together with the new transaction, would exceed \
299                 Bitcoin Core's ancestor limit of {}",
300                record.utxo.id,
301                depth,
302                MAX_ANCESTOR_DEPTH,
303            );
304        }
305
306        let selected_utxos: Vec<Utxo> = selected_records.iter().map(|r| r.utxo.clone()).collect();
307
308        // 3. Verify output count: one per request, followed by zero or more
309        //    trailing change outputs.
310        let request_count = requests.len();
311        let output_count = approval.outputs.len();
312        anyhow::ensure!(
313            output_count >= request_count,
314            "Expected at least {} outputs (one per request), got {}",
315            request_count,
316            output_count
317        );
318
319        // 4. Compute miner fee and verify the per-user fee split
320        let input_total: u64 = selected_utxos.iter().map(|u| u.amount).sum();
321        let output_total: u64 = approval.outputs.iter().map(|o| o.amount).sum();
322        anyhow::ensure!(
323            input_total >= output_total,
324            "Inputs ({input_total}) < outputs ({output_total})"
325        );
326        let fee = input_total - output_total;
327
328        let per_user_miner_fee = fee / request_count as u64;
329
330        // Verify per-user miner fee does not exceed worst-case budget
331        let max_network_fee = self.onchain_state().worst_case_network_fee();
332        anyhow::ensure!(
333            per_user_miner_fee <= max_network_fee,
334            "Per-user miner fee {} sats exceeds worst-case budget {} sats",
335            per_user_miner_fee,
336            max_network_fee
337        );
338
339        // Verify each positional withdrawal output matches the expected amount and address.
340        // request.btc_amount is the full withdrawal amount.
341        for (i, request) in requests.iter().enumerate() {
342            let output = &approval.outputs[i];
343            let expected_amount = request.btc_amount - per_user_miner_fee;
344            anyhow::ensure!(
345                expected_amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
346                "Withdrawal output {} sats is below dust threshold {} sats",
347                expected_amount,
348                utxo_pool::TR_DUST_RELAY_MIN_VALUE
349            );
350            anyhow::ensure!(
351                output.amount == expected_amount,
352                "Output {i} amount {} does not match expected {} for request {:?}",
353                output.amount,
354                expected_amount,
355                request.id
356            );
357            anyhow::ensure!(
358                output.bitcoin_address == request.bitcoin_address,
359                "Output {i} address does not match request {:?}",
360                request.id
361            );
362        }
363
364        // 5. Verify every change output (the trailing outputs after the
365        //    per-request ones) goes to the hashi root pubkey and is above dust.
366        if output_count > request_count {
367            let expected_address =
368                hashi_bitcoin::witness_program_from_address(&self.get_deposit_address(None)?)?;
369            for (j, change_output) in approval.outputs[request_count..].iter().enumerate() {
370                anyhow::ensure!(
371                    change_output.bitcoin_address == expected_address,
372                    "Change output {j} does not go to hashi root pubkey"
373                );
374                anyhow::ensure!(
375                    change_output.amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
376                    "Change output {j} ({} sats) is below dust threshold {} sats",
377                    change_output.amount,
378                    utxo_pool::TR_DUST_RELAY_MIN_VALUE
379                );
380            }
381        }
382
383        // 6. Validate fee is reasonable.
384        //
385        // TODO: When spending unconfirmed change UTXOs the effective fee
386        // rate that matters for mining is the *package* fee rate across
387        // the entire ancestor chain (CPFP). This check currently
388        // evaluates the transaction in isolation, which may over- or
389        // under-estimate the true cost to get the package mined. A
390        // future revision should compute the aggregate ancestor weight
391        // and fees and validate the package fee rate instead.
392        {
393            // Estimate transaction weight.
394            let num_inputs = selected_utxos.len() as u64;
395            let input_weight = hashi_bitcoin::SCRIPT_PATH_2OF2_TXIN_WEIGHT * num_inputs;
396            let output_weight: u64 = approval
397                .outputs
398                .iter()
399                .map(|o| hashi_bitcoin::output_weight_for_witness_program(&o.bitcoin_address))
400                .collect::<anyhow::Result<Vec<_>>>()?
401                .iter()
402                .sum();
403            let tx_weight =
404                Weight::from_wu(hashi_bitcoin::TX_FIXED_WEIGHT_WU + input_weight + output_weight);
405
406            let min_fee_rate = FeeRate::from_sat_per_vb_unchecked(1);
407
408            // Fee must be at least the minimum relay fee (1 sat/vB).
409            let min_fee = min_fee_rate
410                .fee_wu(tx_weight)
411                .map(|a| a.to_sat())
412                .unwrap_or(0);
413            anyhow::ensure!(
414                fee >= min_fee,
415                "Fee {fee} sats is below minimum relay fee {min_fee} sats"
416            );
417
418            // Fee ceiling: clamp the fee rate from our Bitcoin node to
419            // a floor of 1 sat/vB, then cap at the tolerance multiplier.
420            let kyoto_fee_rate = self
421                .btc_monitor()
422                .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
423                .await?;
424            let clamped_fee_rate = std::cmp::max(kyoto_fee_rate, min_fee_rate);
425            let estimated_fee = clamped_fee_rate
426                .fee_wu(tx_weight)
427                .map(|a| a.to_sat())
428                .unwrap_or(0);
429            let max_fee = estimated_fee.saturating_mul(FEE_RATE_TOLERANCE_MULTIPLIER);
430            anyhow::ensure!(
431                fee <= max_fee,
432                "Fee {fee} sats exceeds maximum allowed {max_fee} sats \
433                 ({FEE_RATE_TOLERANCE_MULTIPLIER}x the clamped estimate of \
434                 {estimated_fee} sats at {clamped_fee_rate})"
435            );
436        }
437
438        // 7. Rebuild unsigned tx and verify txid matches.
439        let tx = self.build_unsigned_withdrawal_tx(&selected_utxos, &approval.outputs)?;
440        let expected_txid = BitcoinTxid::from(tx.compute_txid());
441        anyhow::ensure!(
442            approval.txid == expected_txid,
443            "Txid mismatch: approval has {:?}, rebuilt tx has {:?}",
444            approval.txid,
445            expected_txid
446        );
447
448        Ok(())
449    }

513    #[tracing::instrument(level = "info", skip_all, fields(withdrawal_id = %message.withdrawal_id))]
514    pub fn validate_and_sign_withdrawal_tx_signing(
515        &self,
516        message: &WithdrawalTxSigning,
517        expected_limiter_seq: Option<u64>,
518        timestamp_secs: Option<u64>,
519    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
520        let txn = self
521            .onchain_state()
522            .withdrawal_txn(&message.withdrawal_id)
523            .ok_or_else(|| {
524                anyhow!(
525                    "WithdrawalTransaction {} not found on-chain",
526                    message.withdrawal_id
527                )
528            })?;
529
530        anyhow::ensure!(
531            !txn.is_fully_signed(),
532            "WithdrawalTransaction {} is already finalized",
533            message.withdrawal_id
534        );
535
536        anyhow::ensure!(
537            message.request_ids == txn.request_ids,
538            "Request IDs mismatch for WithdrawalTransaction {}",
539            message.withdrawal_id
540        );
541
542        anyhow::ensure!(
543            message.signatures.len() == txn.inputs.len(),
544            "MPC signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
545            message.signatures.len(),
546            txn.inputs.len(),
547            message.withdrawal_id
548        );
549        anyhow::ensure!(
550            message.guardian_signatures.len() == txn.inputs.len(),
551            "Guardian signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
552            message.guardian_signatures.len(),
553            txn.inputs.len(),
554            message.withdrawal_id
555        );
556
557        // Single committee-side rate-limit gate: signing is driven unconditionally,
558        // so the committee independently re-validates the limit once here, before
559        // certifying the finalize — refusing to sign blocks an over-limit broadcast
560        // even if the guardian co-signed. Validate-only: the watcher advances the
561        // limiter on `WithdrawalSignedEvent`, never from this path.
562        //
563        // TODO(guardian-seq-durability): this gate is necessarily *after* the
564        // guardian co-signed (the cert binds the guardian sigs), so a rejection
565        // here — which only happens when this mirror disagrees with the guardian
566        // (the defense-in-depth case) — leaves the guardian's seq consumed with no
567        // on-chain finalize, nudging that gap wider. It self-heals via the
568        // reconcile loop and fires only when catching a guardian/mirror divergence,
569        // so likely fine as-is; if it ever proves to matter, one option worth
570        // exploring would be a separate committee limiter check *before* the
571        // guardian co-signs (extra round-trip, but no consumed-but-not-finalized gap).
572        match (self.local_limiter(), expected_limiter_seq) {
573            (Some(limiter), Some(expected_seq)) => {
574                let amount_sats = withdrawal_limiter_consumption_amount(&txn);
575                // Leader's live checkpoint time (drift-bounded); pre-upgrade leaders omit it.
576                let timestamp_secs = match timestamp_secs {
577                    Some(ts) => {
578                        self.bound_leader_timestamp(ts)?;
579                        ts
580                    }
581                    None => txn.timestamp_ms / 1000,
582                };
583                let result = limiter.validate_consume(expected_seq, timestamp_secs, amount_sats);
584                self.metrics.record_limiter_validate(
585                    &result,
586                    crate::metrics::GUARDIAN_LIMITER_CALLSITE_FINALIZE_CERT,
587                );
588                result.map_err(|e| {
589                    anyhow!("Limiter rejected withdrawal {}: {e}", message.withdrawal_id)
590                })?;
591            }
592            (None, None) => {}
593            (Some(_), None) => anyhow::bail!(
594                "Local limiter is configured but finalize request for withdrawal {} lacks expected_limiter_seq",
595                message.withdrawal_id
596            ),
597            (None, Some(_)) => anyhow::bail!(
598                "Finalize request for withdrawal {} carries expected_limiter_seq but local limiter is not configured",
599                message.withdrawal_id
600            ),
601        }
602
603        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
604        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
605        let guardian_btc_pubkey = self.guardian_btc_pubkey().copied().ok_or_else(|| {
606            anyhow!("Guardian BTC pubkey not yet pinned; cannot validate withdrawal")
607        })?;
608        let guardian_schnorr_pk =
609            SchnorrPublicKey::from_byte_array(&guardian_btc_pubkey.serialize())
610                .map_err(|e| anyhow!("Failed to convert guardian BTC pubkey: {e}"))?;
611
612        for (i, ((mpc_sig_bytes, guardian_sig_bytes), sighash)) in message
613            .signatures
614            .iter()
615            .zip(message.guardian_signatures.iter())
616            .zip(signing_messages.iter())
617            .enumerate()
618        {
619            // MPC: verify against the derived hashi child key.
620            let mpc_arr: &[u8; 64] = mpc_sig_bytes.as_slice().try_into().map_err(|_| {
621                anyhow!(
622                    "MPC signature {i} is not 64 bytes for WithdrawalTransaction {}",
623                    message.withdrawal_id
624                )
625            })?;
626            let mpc_sig = SchnorrSignature::from_byte_array(mpc_arr)
627                .map_err(|e| anyhow!("Invalid MPC Schnorr signature at input {i}: {e}"))?;
628            let input_pubkey = self.deposit_pubkey(txn.inputs[i].derivation_path.as_ref())?;
629            let mpc_schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
630                .map_err(|e| anyhow!("Failed to convert mpc pubkey for input {i}: {e}"))?;
631            mpc_schnorr_pk
632                .verify(sighash, &mpc_sig)
633                .map_err(|e| anyhow!("MPC signature verification failed for input {i}: {e}"))?;
634
635            // Guardian: verify against the on-chain enclave BTC pubkey.
636            // Same sighash — both sigs commit to the same multi_a leaf.
637            let guardian_arr: &[u8; 64] =
638                guardian_sig_bytes.as_slice().try_into().map_err(|_| {
639                    anyhow!(
640                        "Guardian signature {i} is not 64 bytes for WithdrawalTransaction {}",
641                        message.withdrawal_id
642                    )
643                })?;
644            let guardian_sig = SchnorrSignature::from_byte_array(guardian_arr)
645                .map_err(|e| anyhow!("Invalid guardian Schnorr signature at input {i}: {e}"))?;
646            guardian_schnorr_pk
647                .verify(sighash, &guardian_sig)
648                .map_err(|e| {
649                    anyhow!("Guardian signature verification failed for input {i}: {e}")
650                })?;
651        }
652
653        self.sign_message_proto(message)
654    }

656    /// Validate and BLS-sign one incremental chunk of per-input MPC signatures
657    /// (`MpcInputSignaturesMessage`). Each `(index, signature)` is verified
658    /// against that input's sighash before the member signs the chunk cert, so a
659    /// leader cannot obtain a cert over signatures the committee hasn't checked.
660    #[tracing::instrument(
661        level = "info",
662        skip_all,
663        fields(withdrawal_id = %message.withdrawal_id, chunk_size = message.indices.len()),
664    )]
665    pub fn validate_and_sign_mpc_input_signatures(
666        &self,
667        message: &MpcInputSignaturesMessage,
668    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
669        let txn = self
670            .onchain_state()
671            .withdrawal_txn(&message.withdrawal_id)
672            .ok_or_else(|| {
673                anyhow!(
674                    "WithdrawalTransaction {} not found on-chain",
675                    message.withdrawal_id
676                )
677            })?;
678
679        anyhow::ensure!(
680            !txn.is_fully_signed(),
681            "WithdrawalTransaction {} is already finalized",
682            message.withdrawal_id
683        );
684        anyhow::ensure!(
685            message.indices.len() == message.signatures.len(),
686            "Chunk indices ({}) and signatures ({}) length mismatch for WithdrawalTransaction {}",
687            message.indices.len(),
688            message.signatures.len(),
689            message.withdrawal_id
690        );
691
692        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
693        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
694
695        for (chunk_pos, (&input_index, mpc_sig_bytes)) in message
696            .indices
697            .iter()
698            .zip(message.signatures.iter())
699            .enumerate()
700        {
701            let i = input_index as usize;
702            anyhow::ensure!(
703                i < txn.inputs.len(),
704                "Chunk input index {i} out of range ({}) for WithdrawalTransaction {}",
705                txn.inputs.len(),
706                message.withdrawal_id
707            );
708            let sighash = &signing_messages[i];
709            let mpc_arr: &[u8; 64] = mpc_sig_bytes.as_slice().try_into().map_err(|_| {
710                anyhow!("MPC signature at chunk position {chunk_pos} (input {i}) is not 64 bytes")
711            })?;
712            let mpc_sig = SchnorrSignature::from_byte_array(mpc_arr)
713                .map_err(|e| anyhow!("Invalid MPC Schnorr signature at input {i}: {e}"))?;
714            let input_pubkey = self.deposit_pubkey(txn.inputs[i].derivation_path.as_ref())?;
715            let mpc_schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
716                .map_err(|e| anyhow!("Failed to convert mpc pubkey for input {i}: {e}"))?;
717            mpc_schnorr_pk
718                .verify(sighash, &mpc_sig)
719                .map_err(|e| anyhow!("MPC signature verification failed for input {i}: {e}"))?;
720        }
721
722        self.sign_message_proto(message)
723    }

774    /// Rebuild the transition from on-chain state and sign with the historical key.
775    #[tracing::instrument(level = "info", skip_all, fields(from_epoch))]
776    pub fn validate_and_sign_committee_transition(
777        &self,
778        from_epoch: u64,
779    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
780        if let Some(signature) = self.get_committee_handoff_signature(from_epoch) {
781            return Ok(signature);
782        }
783
784        let validator_address = self
785            .config
786            .validator_address()
787            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
788
789        let onchain = self.onchain_state();
790        let state = onchain.state();
791        let committees_map = state.hashi().committees.committees();
792        let from_committee = committees_map
793            .get(&from_epoch)
794            .ok_or_else(|| anyhow!("no on-chain committee for epoch {from_epoch}"))?;
795        if !from_committee
796            .members()
797            .iter()
798            .any(|m| m.validator_address() == validator_address)
799        {
800            anyhow::bail!("not a member of the committee at epoch {from_epoch}");
801        }
802
803        // Hashi committee epochs are sparse: the next entry after `from_epoch`
804        // is generally not `from_epoch + 1`. Both leader and followers derive
805        // the same `to_epoch` from on-chain state, so they sign the same
806        // transition.
807        let new_committee = committees_map
808            .range((from_epoch + 1)..)
809            .next()
810            .map(|(_, c)| c)
811            .ok_or_else(|| anyhow!("no on-chain committee epoch after {from_epoch}"))?;
812
813        let transition = hashi_types::guardian::CommitteeTransitionRequest {
814            new_committee: hashi_types::move_types::Committee::from(new_committee),
815        };
816
817        let signature = self.sign_message_proto_at_epoch(&transition, from_epoch)?;
818        self.store_committee_handoff_signature(from_epoch, signature.clone());
819        Ok(signature)
820    }

865    /// Produce MPC Schnorr signatures for an unsigned withdrawal transaction.
866    #[tracing::instrument(
867        level = "debug",
868        skip_all,
869        fields(withdrawal_txn_id = %txn.id, input_count = txn.inputs.len()),
870    )]
871    async fn mpc_sign_withdrawal_tx(
872        &self,
873        txn: &WithdrawalTransaction,
874        unsigned_tx: &bitcoin::Transaction,
875        requested_input_indices: &[u64],
876        sink: tokio::sync::mpsc::Sender<
877            Result<hashi_types::proto::SignWithdrawalTransactionPartial, tonic::Status>,
878        >,
879    ) -> anyhow::Result<()> {
880        let onchain_state = self.onchain_state().clone();
881        let epoch = onchain_state.epoch();
882        if txn.signing.epoch != epoch {
883            anyhow::bail!(
884                "Stale presig assignment: pending withdrawal {} has signing epoch {}, current is {}. \
885                 Either the leader hasn't called reallocate_presigs yet, \
886                 or this node's on-chain state is behind.",
887                txn.id,
888                txn.signing.epoch,
889                epoch,
890            );
891        }
892        let p2p_channel =
893            RpcP2PChannel::new(onchain_state, epoch, crate::metrics::MPC_LABEL_SIGNING);
894        let signing_manager = self.signing_manager_for(epoch).ok_or_else(|| {
895            anyhow::anyhow!(
896                "SigningManager not available for epoch {epoch}; \
897                 reconciliation may be catching up"
898            )
899        })?;
900        let beacon = S::from_bytes_mod_order(&txn.randomness);
901        let signing_messages = self.withdrawal_signing_messages(unsigned_tx, &txn.inputs)?;
902        let signing_manager_ref = &signing_manager;
903        let p2p_channel_ref = &p2p_channel;
904        let beacon_ref = &beacon;
905        let metrics_ref = &*self.metrics;
906        let txn_id = txn.id;
907        // Per-input presig index is read off the on-chain signing batch slot, so
908        // out-of-order / resume works and the index is always the current-epoch
909        // one assigned by `commit`/`reallocate`. Already-signed inputs are skipped.
910        let signing = &txn.signing;
911        let inputs = &txn.inputs;
912        let sink_ref = &sink;
913        let selected_input_indices =
914            select_withdrawal_signing_indices(signing, requested_input_indices)?;
915        let mut requests = Vec::with_capacity(selected_input_indices.len());
916        let mut index_by_id: HashMap<Address, usize> =
917            HashMap::with_capacity(selected_input_indices.len());
918        for input_index in selected_input_indices {
919            let message = signing_messages
920                .get(input_index)
921                .expect("validated input_index is in range for signing_messages");
922            let global_presig_index = signing
923                .pending_index(input_index)
924                .expect("validated input_index is pending");
925            let signing_id = withdrawal_input_signing_id(&txn_id, input_index as u32);
926            // Change UTXOs (`derivation_path = None`) ride the `[0; 32]` path
927            // everywhere else (leaf script, `deposit_pubkey`). MPC must too —
928            // passing `None` signs for master `G`, not the `derive(G, [0; 32])`
929            // child the 2-of-2 leaf binds.
930            let derivation_address = inputs
931                .get(input_index)
932                .map(|input| {
933                    crate::deposits::normalized_derivation_path(input.derivation_path.as_ref())
934                        .into_inner()
935                })
936                .expect("validated input_index is in range for txn.inputs");
937            index_by_id.insert(signing_id, input_index);
938            requests.push(crate::mpc::SignInput {
939                signing_id,
940                message: message.to_vec(),
941                global_presig_index,
942                derivation_address: Some(derivation_address),
943            });
944        }
945        let (result_tx, mut result_rx) = tokio::sync::mpsc::unbounded_channel();
946        let batch_start = std::time::Instant::now();
947        let collect = signing_manager_ref.sign(
948            p2p_channel_ref,
949            requests,
950            beacon_ref,
951            WITHDRAWAL_SIGNING_TIMEOUT,
952            metrics_ref,
953            result_tx,
954        );
955        let forward = async {
956            while let Some((signing_id, sign_result)) = result_rx.recv().await {
957                let input_index = index_by_id[&signing_id];
958                let sign_duration = batch_start.elapsed().as_secs_f64();
959                match &sign_result {
960                    Ok(_) => {
961                        metrics_ref
962                            .mpc_sign_duration_seconds
963                            .with_label_values(&["success"])
964                            .observe(sign_duration);
965                        metrics_ref
966                            .presig_pool_remaining
967                            .set(signing_manager_ref.presignatures_remaining() as i64);
968                    }
969                    Err(e) => {
970                        metrics_ref
971                            .mpc_sign_duration_seconds
972                            .with_label_values(&["failure"])
973                            .observe(sign_duration);
974                        let reason = match e {
975                            crate::mpc::types::SigningError::Timeout { .. } => "timeout",
976                            crate::mpc::types::SigningError::PoolExhausted => "pool_exhausted",
977                            crate::mpc::types::SigningError::TooManyInvalidSignatures {
978                                ..
979                            } => "too_many_invalid",
980                            crate::mpc::types::SigningError::CryptoError(_) => "crypto_error",
981                            _ => "other",
982                        };
983                        metrics_ref
984                            .mpc_sign_failures_total
985                            .with_label_values(&[reason])
986                            .inc();
987                    }
988                }
989                let partial = sign_result
990                    .map(|sig| hashi_types::proto::SignWithdrawalTransactionPartial {
991                        input_index: input_index as u32,
992                        signature: sig.to_byte_array().to_vec().into(),
993                    })
994                    .map_err(|e| {
995                        tonic::Status::internal(format!(
996                            "Failed to sign withdrawal transaction input {input_index}: {e}"
997                        ))
998                    });
999                let _ = sink_ref.send(partial).await;
1000            }
1001        };
1002        tokio::join!(collect, forward);
1003        Ok(())
1004    }

1073    /// Build a withdrawal commitment for a batch of approved requests: select
1074    /// UTXOs using the batching-aware coin selection algorithm, build the
1075    /// unsigned BTC tx, and return a `WithdrawalTxCommitment` covering the
1076    /// selected requests.
1077    #[tracing::instrument(level = "debug", skip_all, fields(request_count = requests.len()))]
1078    pub async fn build_withdrawal_tx_commitment(
1079        &self,
1080        requests: &[WithdrawalRequest],
1081    ) -> Result<WithdrawalTxCommitment, WithdrawalCommitmentError> {
1082        // Fetch current fee rate from the Bitcoin node, clamped to a high
1083        // fee rate threshold to avoid overpaying during fee spikes.
1084        let kyoto_fee_rate = self
1085            .btc_monitor()
1086            .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
1087            .await
1088            .map_err(|e| WithdrawalCommitmentError::FeeEstimateFailed(anyhow!(e)))?;
1089        let min_fee_rate = CoinSelectionParams::DEFAULT_MIN_FEE_RATE;
1090        let max_fee_rate = CoinSelectionParams::DEFAULT_HIGH_FEE_RATE_THRESHOLD;
1091        let fee_rate = kyoto_fee_rate.clamp(min_fee_rate, max_fee_rate);
1092
1093        let change_address = self
1094            .get_deposit_address(None)
1095            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
1096
1097        let configured_max_inputs = CoinSelectionParams::DEFAULT_MAX_INPUTS;
1098        let configured_long_term_fee_rate = CoinSelectionParams::DEFAULT_LONG_TERM_FEE_RATE;
1099        let configured_max_requests = self.config.withdrawal_max_batch_size().min(requests.len());
1100
1101        // Snapshot both maps under a single read-lock so they are always
1102        // mutually consistent (e.g., a WithdrawalConfirmedEvent cannot update
1103        // one map but not the other between the two reads).
1104        let (withdrawal_txns, utxo_records) = {
1105            let state = self.onchain_state().state();
1106            (
1107                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
1108                state.hashi().utxo_pool.utxo_records().clone(),
1109            )
1110        };
1111
1112        // Query Bitcoin in parallel for the confirmation count of every
1113        // pending withdrawal so we can accurately fill AncestorTx::confirmations
1114        // instead of always hardcoding 0.
1115        let tx_confirmations = fetch_withdrawal_tx_confirmations(self, &withdrawal_txns).await;
1116
1117        // Map available (unlocked) UTXOs to UtxoCandidates.
1118        let candidates: Vec<UtxoCandidate> = utxo_records
1119            .values()
1120            .filter(|r| r.locked_by.is_none())
1121            .map(|r| {
1122                let status =
1123                    build_utxo_status(self, r, &withdrawal_txns, &tx_confirmations, &utxo_records);
1124                UtxoCandidate {
1125                    id: r.utxo.id,
1126                    amount: r.utxo.amount,
1127                    spend_path: SpendPath::TaprootScriptPath2of2,
1128                    status,
1129                }
1130            })
1131            .collect();
1132
1133        // Map on-chain WithdrawalRequests to the coin-selector view.
1134        // btc_amount is the full withdrawal amount.
1135        let mapped_requests: Vec<utxo_pool::WithdrawalRequest> = requests
1136            .iter()
1137            .map(|r| utxo_pool::WithdrawalRequest {
1138                id: r.id,
1139                recipient: r.bitcoin_address.clone(),
1140                amount: r.btc_amount,
1141                timestamp_ms: r.timestamp_ms,
1142            })
1143            .collect();
1144
1145        let mut last_selection_error = None;
1146        let mut result = None;
1147        for request_count in (1..=configured_max_requests).rev() {
1148            let max_inputs = safe_withdrawal_flow_max_inputs(request_count, configured_max_inputs);
1149            if max_inputs == 0 {
1150                continue;
1151            }
1152
1153            let params = CoinSelectionParams {
1154                max_inputs,
1155                long_term_fee_rate: configured_long_term_fee_rate,
1156                max_fee_per_request: self.onchain_state().worst_case_network_fee(),
1157                max_withdrawal_requests: request_count,
1158                max_mempool_chain_depth: self.config.max_mempool_chain_depth(),
1159                ..CoinSelectionParams::new(change_address.clone())
1160            };
1161
1162            match utxo_pool::select_coins(&candidates, &mapped_requests, &params, fee_rate) {
1163                Ok(selection) => {
1164                    if request_count < configured_max_requests {
1165                        tracing::info!(
1166                            selected_requests = selection.selected_requests.len(),
1167                            selected_inputs = selection.inputs.len(),
1168                            configured_max_requests,
1169                            configured_max_inputs,
1170                            max_inputs,
1171                            "Reduced withdrawal batch to stay within Sui commit limits",
1172                        );
1173                    }
1174                    result = Some(selection);
1175                    break;
1176                }
1177                Err(e) => last_selection_error = Some(e),
1178            }
1179        }
1180
1181        let result = result.ok_or_else(|| {
1182            WithdrawalCommitmentError::UtxoSelectionFailed(anyhow!(
1183                last_selection_error
1184                    .map(|e| e.to_string())
1185                    .unwrap_or_else(|| "no withdrawal request count fits Sui commit limits".into())
1186            ))
1187        })?;
1188
1189        // Build outputs: one per selected request (net amount already deducted),
1190        // plus an optional change output.
1191        let mut outputs: Vec<OutputUtxo> = result
1192            .withdrawal_outputs
1193            .iter()
1194            .map(|o| OutputUtxo {
1195                amount: o.amount,
1196                bitcoin_address: o.recipient.clone(),
1197            })
1198            .collect();
1199
1200        if let Some(change_amount) = result.change {
1201            outputs.push(OutputUtxo {
1202                amount: change_amount,
1203                bitcoin_address: hashi_bitcoin::witness_program_from_address(&change_address)
1204                    .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?,
1205            });
1206        }
1207
1208        let selected_utxos: Vec<UtxoId> = result.inputs.iter().map(|u| u.id).collect();
1209        let request_ids: Vec<Address> = result.selected_requests.iter().map(|r| r.id).collect();
1210
1211        // Resolve UtxoCandidates back to full Utxo objects for tx building.
1212        let selected_input_utxos: Vec<Utxo> = result
1213            .inputs
1214            .iter()
1215            .map(|c| self.onchain_state().active_utxo(&c.id))
1216            .collect::<Option<Vec<_>>>()
1217            .ok_or_else(|| {
1218                WithdrawalCommitmentError::BtcTxBuildFailed(anyhow!(
1219                    "a selected UTXO disappeared from the pool between selection and tx build"
1220                ))
1221            })?;
1222
1223        let tx = self
1224            .build_unsigned_withdrawal_tx(&selected_input_utxos, &outputs)
1225            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
1226        let txid = BitcoinTxid::from(tx.compute_txid());
1227
1228        Ok(WithdrawalTxCommitment {
1229            request_ids,
1230            selected_utxos,
1231            outputs,
1232            txid,
1233        })
1234    }
1235
1236    /// Run AML/Sanctions checks for a withdrawal request.
1237    /// If no screener client is configured, checks are skipped.
1238    #[tracing::instrument(level = "debug", skip_all, fields(request_id = %request.id))]
1239    pub(crate) async fn screen_withdrawal(
1240        &self,
1241        request: &WithdrawalRequest,
1242    ) -> Result<(), WithdrawalApprovalError> {
1243        let Some(screener) = self.screener_client() else {
1244            tracing::debug!("AML checks skipped: no screener configured");
1245            return Ok(());
1246        };
1247
1248        // Source: Sui tx digest (base58 string)
1249        let source_tx_hash = request.sui_tx_digest.to_string();
1250
1251        // Destination: Bitcoin address (raw witness bytes -> bech32 string)
1252        let destination_address = hashi_bitcoin::address_string_from_witness_program(
1253            &request.bitcoin_address,
1254            self.config.bitcoin_network(),
1255        )
1256        .map_err(WithdrawalApprovalError::NeverRetry)?;
1257
1258        let approved = screener
1259            .approve_withdrawal(
1260                &source_tx_hash,
1261                &destination_address,
1262                self.config.sui_chain_id(),
1263                self.config.bitcoin_chain_id(),
1264            )
1265            .await
1266            .map_err(|e| WithdrawalApprovalError::AmlServiceError(anyhow!(e)))?;
1267
1268        if !approved {
1269            return Err(WithdrawalApprovalError::NeverRetry(anyhow!(
1270                "AML checks failed for withdrawal request {:?} to {}",
1271                request.id,
1272                destination_address,
1273            )));
1274        }
1275
1276        Ok(())
1277    }
1278}

1448/// Build the [`UtxoStatus`] for a UTXO record using a pre-fetched snapshot.
1449///
1450/// For confirmed UTXOs (`produced_by = None`) this is simply
1451/// [`UtxoStatus::Confirmed`]. For unconfirmed change outputs
1452/// (`produced_by = Some(withdrawal_id)`) we walk the full ancestor chain
1453/// so that CPFP weight and mempool depth are accurately computed even for
1454/// multi-level chains. If the producing withdrawal has already been removed
1455/// from `withdrawal_txns` (confirmed and cleared), we promote the UTXO
1456/// to `Confirmed` — it is safe to spend immediately.
1457fn build_utxo_status(
1458    hashi: &Hashi,
1459    record: &UtxoRecord,
1460    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1461    tx_confirmations: &HashMap<Address, u32>,
1462    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1463) -> UtxoStatus {
1464    let Some(producing_id) = record.produced_by else {
1465        return UtxoStatus::Confirmed;
1466    };
1467
1468    let chain = build_ancestor_chain(
1469        hashi,
1470        producing_id,
1471        withdrawal_txns,
1472        tx_confirmations,
1473        utxo_records,
1474    );
1475
1476    if chain.is_empty() {
1477        // The producing withdrawal was confirmed and removed from
1478        // withdrawal_txns. The UTXO is safe to spend.
1479        UtxoStatus::Confirmed
1480    } else {
1481        UtxoStatus::Pending { chain }
1482    }
1483}

1485/// Maximum ancestor depth permitted by Bitcoin Core's relay policy
1486/// (`DEFAULT_ANCESTOR_LIMIT = 25`). Bitcoin Core counts the candidate
1487/// transaction itself in the ancestor set, so a UTXO whose existing
1488/// unconfirmed ancestor depth is already `MAX_ANCESTOR_DEPTH - 1` is
1489/// the deepest we can safely spend.
1490pub const MAX_ANCESTOR_DEPTH: usize = 25;
1491
1492/// Count the number of unconfirmed ancestors for a UTXO record by walking
1493/// the `produced_by` chain. Every ancestor that still appears in
1494/// `withdrawal_txns` is conservatively treated as unconfirmed (we skip
1495/// querying Bitcoin for actual confirmation counts). This is used during
1496/// commitment validation to reject UTXOs whose ancestor chain would exceed
1497/// Bitcoin Core's relay limit.
1498///
1499/// The walk is a BFS over the ancestor DAG. Each item on the queue is a
1500/// `(producing_withdrawal_id, depth)` pair. We track the maximum depth
1501/// seen across all branches.
1502fn unconfirmed_ancestor_depth(
1503    record: &UtxoRecord,
1504    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1505    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1506) -> usize {
1507    let Some(producing_id) = record.produced_by else {
1508        return 0;
1509    };
1510
1511    let mut max_depth: usize = 0;
1512    let mut queue = std::collections::VecDeque::new();
1513    queue.push_back((producing_id, 1usize));
1514
1515    while let Some((wid, depth)) = queue.pop_front() {
1516        if depth > MAX_ANCESTOR_DEPTH {
1517            return depth;
1518        }
1519
1520        let Some(txn) = withdrawal_txns.get(&wid) else {
1521            // The producing withdrawal has been confirmed and removed;
1522            // it does not contribute to the unconfirmed chain.
1523            continue;
1524        };
1525
1526        max_depth = std::cmp::max(max_depth, depth);
1527
1528        // Enqueue any inputs that are themselves unconfirmed change
1529        // outputs of an earlier withdrawal.
1530        for input_utxo in &txn.inputs {
1531            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1532                && let Some(parent_id) = input_record.produced_by
1533            {
1534                queue.push_back((parent_id, depth + 1));
1535            }
1536        }
1537    }
1538
1539    max_depth
1540}

1542/// Build the ancestor chain for a UTXO produced by `producing_id`. Each
1543/// unconfirmed ancestor that still appears in `withdrawal_txns` gets
1544/// one [`AncestorTx`] entry with its confirmation count, weight, and fee.
1545/// The walk is a BFS over the ancestor DAG, capped at
1546/// [`MAX_ANCESTOR_DEPTH`] levels.
1547fn build_ancestor_chain(
1548    hashi: &Hashi,
1549    producing_id: Address,
1550    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1551    tx_confirmations: &HashMap<Address, u32>,
1552    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1553) -> Vec<AncestorTx> {
1554    let mut chain = Vec::new();
1555    let mut queue = std::collections::VecDeque::new();
1556    queue.push_back((producing_id, 0usize));
1557
1558    while let Some((wid, depth)) = queue.pop_front() {
1559        if depth >= MAX_ANCESTOR_DEPTH {
1560            continue;
1561        }
1562
1563        let Some(txn) = withdrawal_txns.get(&wid) else {
1564            continue;
1565        };
1566
1567        let Ok(tx) = hashi.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs()) else {
1568            continue;
1569        };
1570
1571        let confirmations = tx_confirmations.get(&wid).copied().unwrap_or(0);
1572        let input_total: u64 = txn.inputs.iter().map(|u| u.amount).sum();
1573        let output_total: u64 = txn.all_outputs().iter().map(|o| o.amount).sum();
1574
1575        chain.push(AncestorTx {
1576            confirmations,
1577            tx_weight: tx.weight(),
1578            tx_fee: input_total.saturating_sub(output_total),
1579        });
1580
1581        for input_utxo in &txn.inputs {
1582            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1583                && let Some(parent_id) = input_record.produced_by
1584            {
1585                queue.push_back((parent_id, depth + 1));
1586            }
1587        }
1588    }
1589
1590    chain
1591}

1593/// Deterministic from on-chain state and the leader-supplied
1594/// `(timestamp_secs, seq)`, so every validator reconstructs the same request.
1595pub fn build_guardian_withdrawal_request(
1596    hashi: &Hashi,
1597    txn: &WithdrawalTransaction,
1598    timestamp_secs: u64,
1599    seq: u64,
1600) -> anyhow::Result<hashi_types::guardian::StandardWithdrawalRequest> {
1601    use hashi_types::bitcoin::InputUTXO;
1602    use hashi_types::bitcoin::OutputUTXOWire;
1603    use hashi_types::bitcoin::TxUTXOs;
1604
1605    let network = hashi.config.bitcoin_network();
1606
1607    let inputs: Vec<_> = txn.inputs.iter().map(InputUTXO::from).collect();
1608
1609    // First N outputs are external payouts; any trailing output is internal change.
1610    let all_outputs = txn.all_outputs();
1611    let num_requests = txn.request_ids.len();
1612    let outputs = all_outputs
1613        .iter()
1614        .enumerate()
1615        .map(|(i, output)| {
1616            if i < num_requests {
1617                let script_pubkey =
1618                    hashi_bitcoin::script_pubkey_from_witness_program(&output.bitcoin_address)?;
1619                let address =
1620                    hashi_bitcoin::BitcoinAddress::from_script(&script_pubkey, network)
1621                        .map_err(|e| anyhow!("Cannot derive address from output script: {e}"))?;
1622                Ok(OutputUTXOWire::external(
1623                    address.into_unchecked(),
1624                    Amount::from_sat(output.amount),
1625                ))
1626            } else {
1627                Ok(OutputUTXOWire::internal(
1628                    sui_sdk_types::Address::ZERO,
1629                    Amount::from_sat(output.amount),
1630                ))
1631            }
1632        })
1633        .collect::<anyhow::Result<Vec<_>>>()?;
1634
1635    let utxos = TxUTXOs::new(inputs, outputs, network)
1636        .map_err(|e| anyhow!("Failed to build guardian TxUTXOs: {e}"))?;
1637
1638    // The on-chain `WithdrawalTransaction` UID doubles as the guardian-side `wid`.
1639    let wid: hashi_types::guardian::WithdrawalID = txn.id;
1640
1641    Ok(hashi_types::guardian::StandardWithdrawalRequest::new(
1642        wid,
1643        utxos,
1644        timestamp_secs,
1645        seq,
1646    ))
1647}
