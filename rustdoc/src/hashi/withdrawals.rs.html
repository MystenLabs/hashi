52fn select_withdrawal_signing_indices(
53    signing: &hashi_types::move_types::SigningBatch,
54    requested_input_indices: &[u64],
55) -> anyhow::Result<Vec<usize>> {
56    if requested_input_indices.is_empty() {
57        return Ok(signing
58            .unsigned_indices()
59            .into_iter()
60            .map(|i| i as usize)
61            .collect());
62    }
63
64    let mut seen = BTreeSet::new();
65    let mut selected = Vec::with_capacity(requested_input_indices.len());
66    for &input_index in requested_input_indices {
67        if !seen.insert(input_index) {
68            anyhow::bail!("duplicate input index {input_index} in withdrawal signing request");
69        }
70        let i = usize::try_from(input_index)
71            .map_err(|_| anyhow!("input index {input_index} out of range"))?;
72        if i >= signing.num_inputs() {
73            anyhow::bail!(
74                "input index {input_index} out of range for withdrawal with {} inputs",
75                signing.num_inputs()
76            );
77        }
78        if signing.pending_index(i).is_none() {
79            anyhow::bail!("input index {input_index} is already signed");
80        }
81        selected.push(i);
82    }
83    Ok(selected)
84}

86/// BTC that leaves the pool when this txn broadcasts — the amount that
87/// consumes the guardian's limit. Equivalent to
88/// `sum(withdrawal_outputs) + miner_fee`; we use `inputs - change` to
89/// avoid relying on a separate fee field.
90pub(crate) fn withdrawal_limiter_consumption_amount(txn: &WithdrawalTransaction) -> u64 {
91    let inputs: u64 = txn.inputs.iter().map(|u| u.amount).sum();
92    let change: u64 = txn.change_outputs.iter().map(|c| c.amount).sum();
93    inputs.saturating_sub(change)
94}

96/// Conservative runtime-object budget for withdrawal commit transactions.
97///
98/// Sui's hard object-runtime cache limit is 1000 objects. Empirical
99/// measurements put withdrawal commit transactions at roughly
100/// `selected_utxos + 3 * requests + fixed_overhead` runtime objects. Target
101/// 922 to leave 7.8% headroom below the hard 1000 cap.
102const WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET: usize = 922;
103const WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS: usize = 12;
104const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST: usize = 3;
105const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT: usize = 1;
106
107fn safe_withdrawal_commit_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
108    let request_objects =
109        request_count.saturating_mul(WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST);
110    let runtime_input_budget = WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET
111        .saturating_sub(WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS)
112        .saturating_sub(request_objects);
113    let input_budget = runtime_input_budget / WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT;
114    configured_max_inputs.min(input_budget)
115}
116
117/// Confirm is no longer input-bound: spent UTXOs emit events only (0
118/// objects per input). The confirm transaction's object count is
119/// `requests + ~43` fixed overhead — well within the 1000-object limit
120/// for any practical request count. The commit path is the binding
121/// constraint.
122fn safe_withdrawal_flow_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
123    let request_input_budget =
124        request_count.saturating_mul(CoinSelectionParams::DEFAULT_INPUT_BUDGET);
125    configured_max_inputs
126        .min(request_input_budget)
127        .min(safe_withdrawal_commit_max_inputs(
128            request_count,
129            configured_max_inputs,
130        ))
131}

239    #[tracing::instrument(level = "debug", skip_all, fields(bitcoin_txid = %approval.txid))]
240    pub async fn validate_withdrawal_tx_commitment(
241        &self,
242        approval: &WithdrawalTxCommitment,
243    ) -> anyhow::Result<()> {
244        anyhow::ensure!(!approval.request_ids.is_empty(), "No request IDs");
245        anyhow::ensure!(!approval.selected_utxos.is_empty(), "No selected UTXOs");
246        anyhow::ensure!(!approval.outputs.is_empty(), "No outputs");
247
248        // Check for duplicate request IDs
249        let unique_request_ids: std::collections::BTreeSet<_> =
250            approval.request_ids.iter().collect();
251        anyhow::ensure!(
252            unique_request_ids.len() == approval.request_ids.len(),
253            "Duplicate request IDs"
254        );
255
256        // Check for duplicate UTXO IDs
257        let unique_utxo_ids: std::collections::BTreeSet<_> =
258            approval.selected_utxos.iter().collect();
259        anyhow::ensure!(
260            unique_utxo_ids.len() == approval.selected_utxos.len(),
261            "Duplicate UTXO IDs"
262        );
263
264        // 1. Verify each request_id exists and is approved
265        let requests: Vec<WithdrawalRequest> = approval
266            .request_ids
267            .iter()
268            .map(|id| {
269                let request = self
270                    .onchain_state()
271                    .withdrawal_request(id)
272                    .ok_or_else(|| anyhow!("Withdrawal request {id} not found in queue"))?;
273                anyhow::ensure!(
274                    request.status.is_approved(),
275                    "Withdrawal request {id} has not been approved"
276                );
277                Ok(request)
278            })
279            .collect::<anyhow::Result<_>>()?;
280
281        // 2. Verify each selected UTXO exists, is not locked, and collect
282        //    full UTXO data. We look up via utxo_records so we can
283        //    distinguish "missing" from "locked by another withdrawal" and
284        //    also inspect the `produced_by` chain for ancestor depth.
285        let (utxo_records, withdrawal_txns) = {
286            let state = self.onchain_state().state();
287            (
288                state.hashi().utxo_pool.utxo_records().clone(),
289                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
290            )
291        };
292
293        let selected_records: Vec<&UtxoRecord> = approval
294            .selected_utxos
295            .iter()
296            .map(|id| {
297                let record = utxo_records
298                    .get(id)
299                    .ok_or_else(|| anyhow!("UTXO {id:?} not found in the pool"))?;
300                anyhow::ensure!(
301                    record.spent_by.is_none(),
302                    "UTXO {id:?} is locked by pending withdrawal {:?}",
303                    record.spent_by.unwrap()
304                );
305                Ok(record)
306            })
307            .collect::<anyhow::Result<_>>()?;
308
309        // 2b. Verify that no selected UTXO has an unconfirmed ancestor
310        //     chain deeper than Bitcoin Core's relay limit. The limit
311        //     (DEFAULT_ANCESTOR_LIMIT = 25) counts the candidate tx
312        //     itself, so the existing chain must leave room for the
313        //     transaction we are about to construct.
314        for record in &selected_records {
315            let depth = unconfirmed_ancestor_depth(record, &withdrawal_txns, &utxo_records);
316            anyhow::ensure!(
317                depth < MAX_ANCESTOR_DEPTH,
318                "UTXO {:?} has an unconfirmed ancestor chain of depth {} \
319                 which, together with the new transaction, would exceed \
320                 Bitcoin Core's ancestor limit of {}",
321                record.utxo.id,
322                depth,
323                MAX_ANCESTOR_DEPTH,
324            );
325        }
326
327        let selected_utxos: Vec<Utxo> = selected_records.iter().map(|r| r.utxo.clone()).collect();
328
329        // 3. Verify output count: one per request, followed by zero or more
330        //    trailing change outputs.
331        let request_count = requests.len();
332        let output_count = approval.outputs.len();
333        anyhow::ensure!(
334            output_count >= request_count,
335            "Expected at least {} outputs (one per request), got {}",
336            request_count,
337            output_count
338        );
339
340        // 4. Compute miner fee and verify the per-user fee split
341        let input_total: u64 = selected_utxos.iter().map(|u| u.amount).sum();
342        let output_total: u64 = approval.outputs.iter().map(|o| o.amount).sum();
343        anyhow::ensure!(
344            input_total >= output_total,
345            "Inputs ({input_total}) < outputs ({output_total})"
346        );
347        let fee = input_total - output_total;
348
349        let per_user_miner_fee = fee / request_count as u64;
350
351        // Verify per-user miner fee does not exceed worst-case budget
352        let max_network_fee = self.onchain_state().worst_case_network_fee();
353        anyhow::ensure!(
354            per_user_miner_fee <= max_network_fee,
355            "Per-user miner fee {} sats exceeds worst-case budget {} sats",
356            per_user_miner_fee,
357            max_network_fee
358        );
359
360        // Verify each positional withdrawal output matches the expected amount and address.
361        // request.btc_amount is the full withdrawal amount.
362        for (i, request) in requests.iter().enumerate() {
363            let output = &approval.outputs[i];
364            let expected_amount = request.btc_amount - per_user_miner_fee;
365            anyhow::ensure!(
366                expected_amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
367                "Withdrawal output {} sats is below dust threshold {} sats",
368                expected_amount,
369                utxo_pool::TR_DUST_RELAY_MIN_VALUE
370            );
371            anyhow::ensure!(
372                output.amount == expected_amount,
373                "Output {i} amount {} does not match expected {} for request {:?}",
374                output.amount,
375                expected_amount,
376                request.id
377            );
378            anyhow::ensure!(
379                output.bitcoin_address == request.bitcoin_address,
380                "Output {i} address does not match request {:?}",
381                request.id
382            );
383        }
384
385        // 5. Verify every change output (the trailing outputs after the
386        //    per-request ones) goes to the hashi root pubkey and is above dust.
387        if output_count > request_count {
388            let expected_address =
389                hashi_bitcoin::witness_program_from_address(&self.get_deposit_address(None)?)?;
390            for (j, change_output) in approval.outputs[request_count..].iter().enumerate() {
391                anyhow::ensure!(
392                    change_output.bitcoin_address == expected_address,
393                    "Change output {j} does not go to hashi root pubkey"
394                );
395                anyhow::ensure!(
396                    change_output.amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
397                    "Change output {j} ({} sats) is below dust threshold {} sats",
398                    change_output.amount,
399                    utxo_pool::TR_DUST_RELAY_MIN_VALUE
400                );
401            }
402        }
403
404        // 6. Validate fee is reasonable.
405        //
406        // TODO: When spending unconfirmed change UTXOs the effective fee
407        // rate that matters for mining is the *package* fee rate across
408        // the entire ancestor chain (CPFP). This check currently
409        // evaluates the transaction in isolation, which may over- or
410        // under-estimate the true cost to get the package mined. A
411        // future revision should compute the aggregate ancestor weight
412        // and fees and validate the package fee rate instead.
413        {
414            // Estimate transaction weight.
415            let num_inputs = selected_utxos.len() as u64;
416            let input_weight = hashi_bitcoin::SCRIPT_PATH_2OF2_TXIN_WEIGHT * num_inputs;
417            let output_weight: u64 = approval
418                .outputs
419                .iter()
420                .map(|o| hashi_bitcoin::output_weight_for_witness_program(&o.bitcoin_address))
421                .collect::<anyhow::Result<Vec<_>>>()?
422                .iter()
423                .sum();
424            let tx_weight =
425                Weight::from_wu(hashi_bitcoin::TX_FIXED_WEIGHT_WU + input_weight + output_weight);
426
427            let min_fee_rate = FeeRate::from_sat_per_vb_unchecked(1);
428
429            // Fee must be at least the minimum relay fee (1 sat/vB).
430            let min_fee = min_fee_rate
431                .fee_wu(tx_weight)
432                .map(|a| a.to_sat())
433                .unwrap_or(0);
434            anyhow::ensure!(
435                fee >= min_fee,
436                "Fee {fee} sats is below minimum relay fee {min_fee} sats"
437            );
438
439            // Fee ceiling: clamp the fee rate from our Bitcoin node to
440            // a floor of 1 sat/vB, then cap at the tolerance multiplier.
441            let kyoto_fee_rate = self
442                .btc_monitor()
443                .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
444                .await?;
445            let clamped_fee_rate = std::cmp::max(kyoto_fee_rate, min_fee_rate);
446            let estimated_fee = clamped_fee_rate
447                .fee_wu(tx_weight)
448                .map(|a| a.to_sat())
449                .unwrap_or(0);
450            let max_fee = estimated_fee.saturating_mul(FEE_RATE_TOLERANCE_MULTIPLIER);
451            anyhow::ensure!(
452                fee <= max_fee,
453                "Fee {fee} sats exceeds maximum allowed {max_fee} sats \
454                 ({FEE_RATE_TOLERANCE_MULTIPLIER}x the clamped estimate of \
455                 {estimated_fee} sats at {clamped_fee_rate})"
456            );
457        }
458
459        // 7. Rebuild unsigned tx and verify txid matches.
460        let tx = self.build_unsigned_withdrawal_tx(&selected_utxos, &approval.outputs)?;
461        let expected_txid = BitcoinTxid::from(tx.compute_txid());
462        anyhow::ensure!(
463            approval.txid == expected_txid,
464            "Txid mismatch: approval has {:?}, rebuilt tx has {:?}",
465            approval.txid,
466            expected_txid
467        );
468
469        Ok(())
470    }

479    pub async fn sign_withdrawal_confirmation(
480        &self,
481        withdrawal_txn_id: &Address,
482    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
483        let txn = self
484            .onchain_state()
485            .withdrawal_txn(withdrawal_txn_id)
486            .ok_or_else(|| {
487                anyhow!("WithdrawalTransaction {withdrawal_txn_id} not found on-chain")
488            })?;
489
490        // Confirmation is the terminal transition: it marks the withdrawal's
491        // inputs spent and finalizes requests whose BTC was already burned at
492        // commit. A malicious leader can aggregate confirmation signatures
493        // directly, so each validator must independently refuse to sign unless
494        // the withdrawal is genuinely finalizable — the leader's own checks are
495        // no defense here.
496
497        // (1) The on-chain transaction must be fully signed (every input has an
498        // MPC signature and the guardian signatures are attached), i.e. a
499        // broadcastable Bitcoin transaction actually exists.
500        anyhow::ensure!(
501            txn.is_fully_signed(),
502            "Refusing to sign confirmation for withdrawal {withdrawal_txn_id}: \
503             on-chain transaction is not fully signed"
504        );
505
506        // (2) That Bitcoin transaction must itself be confirmed to the
507        // configured threshold. This observational check is what a malicious
508        // leader cannot forge.
509        let confirmation_threshold = self.onchain_state().bitcoin_confirmation_threshold();
510        let txid: bitcoin::Txid = txn.txid.into();
511        match self.btc_monitor().get_transaction_status(txid).await? {
512            TxStatus::Confirmed { confirmations } if confirmations >= confirmation_threshold => {}
513            status => anyhow::bail!(
514                "Refusing to sign confirmation for withdrawal {withdrawal_txn_id}: \
515                 Bitcoin transaction {txid} not confirmed to threshold \
516                 {confirmation_threshold} (status: {status:?})"
517            ),
518        }
519
520        let confirmation = WithdrawalConfirmation {
521            withdrawal_id: txn.id,
522        };
523
524        self.sign_message_proto(&confirmation)
525    }

576                anyhow!(
577                    "WithdrawalTransaction {} not found on-chain",
578                    message.withdrawal_id
579                )
580            })?;
581
582        anyhow::ensure!(
583            !txn.is_fully_signed(),
584            "WithdrawalTransaction {} is already finalized",
585            message.withdrawal_id
586        );
587
588        anyhow::ensure!(
589            message.request_ids == txn.request_ids,
590            "Request IDs mismatch for WithdrawalTransaction {}",
591            message.withdrawal_id
592        );
593
594        anyhow::ensure!(
595            message.signatures.len() == txn.inputs.len(),
596            "MPC signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
597            message.signatures.len(),
598            txn.inputs.len(),
599            message.withdrawal_id
600        );
601        anyhow::ensure!(
602            message.guardian_signatures.len() == txn.inputs.len(),
603            "Guardian signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
604            message.guardian_signatures.len(),
605            txn.inputs.len(),
606            message.withdrawal_id
607        );
608
609        // Single committee-side rate-limit gate: signing is driven unconditionally,
610        // so the committee independently re-validates the limit once here, before
611        // certifying the finalize — refusing to sign blocks an over-limit broadcast
612        // even if the guardian co-signed. Validate-only: the watcher advances the
613        // limiter on `WithdrawalSigned`, never from this path.
614        //
615        // TODO(guardian-seq-durability): this gate is necessarily *after* the
616        // guardian co-signed (the cert binds the guardian sigs), so a rejection
617        // here — which only happens when this mirror disagrees with the guardian
618        // (the defense-in-depth case) — leaves the guardian's seq consumed with no
619        // on-chain finalize, nudging that gap wider. It self-heals via the
620        // reconcile loop and fires only when catching a guardian/mirror divergence,
621        // so likely fine as-is; if it ever proves to matter, one option worth
622        // exploring would be a separate committee limiter check *before* the
623        // guardian co-signs (extra round-trip, but no consumed-but-not-finalized gap).
624        match (self.local_limiter(), expected_limiter_seq) {
625            (Some(limiter), Some(expected_seq)) => {
626                let amount_sats = withdrawal_limiter_consumption_amount(&txn);
627                // Leader's live checkpoint time (drift-bounded); pre-upgrade leaders omit it.
628                let timestamp_secs = match timestamp_secs {
629                    Some(ts) => {
630                        self.bound_leader_timestamp(ts)?;
631                        ts
632                    }
633                    None => txn.created_timestamp_ms / 1000,
634                };
635                let result = limiter.validate_consume(expected_seq, timestamp_secs, amount_sats);
636                self.metrics.record_limiter_validate(
637                    &result,
638                    crate::metrics::GUARDIAN_LIMITER_CALLSITE_FINALIZE_CERT,
639                );
640                result.map_err(|e| {
641                    anyhow!("Limiter rejected withdrawal {}: {e}", message.withdrawal_id)
642                })?;
643            }
644            (None, None) => {}
645            (Some(_), None) => anyhow::bail!(
646                "Local limiter is configured but finalize request for withdrawal {} lacks expected_limiter_seq",
647                message.withdrawal_id
648            ),
649            (None, Some(_)) => anyhow::bail!(
650                "Finalize request for withdrawal {} carries expected_limiter_seq but local limiter is not configured",
651                message.withdrawal_id
652            ),
653        }
654
655        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
656        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
657        let guardian_btc_pubkey = self.guardian_btc_pubkey().copied().ok_or_else(|| {
658            anyhow!("Guardian BTC pubkey not yet pinned; cannot validate withdrawal")
659        })?;
660        let guardian_schnorr_pk =
661            SchnorrPublicKey::from_byte_array(&guardian_btc_pubkey.serialize())
662                .map_err(|e| anyhow!("Failed to convert guardian BTC pubkey: {e}"))?;
663
664        for (i, ((mpc_sig_bytes, guardian_sig_bytes), sighash)) in message
665            .signatures
666            .iter()
667            .zip(message.guardian_signatures.iter())
668            .zip(signing_messages.iter())
669            .enumerate()
670        {
671            // MPC: verify against the derived hashi child key.
672            let mpc_arr: &[u8; 64] = mpc_sig_bytes.as_slice().try_into().map_err(|_| {
673                anyhow!(
674                    "MPC signature {i} is not 64 bytes for WithdrawalTransaction {}",
675                    message.withdrawal_id
676                )
677            })?;
678            let mpc_sig = SchnorrSignature::from_byte_array(mpc_arr)
679                .map_err(|e| anyhow!("Invalid MPC Schnorr signature at input {i}: {e}"))?;
680            let input_pubkey = self.deposit_pubkey(txn.inputs[i].derivation_path.as_ref())?;
681            let mpc_schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
682                .map_err(|e| anyhow!("Failed to convert mpc pubkey for input {i}: {e}"))?;
683            mpc_schnorr_pk
684                .verify(sighash, &mpc_sig)
685                .map_err(|e| anyhow!("MPC signature verification failed for input {i}: {e}"))?;
686
687            // Guardian: verify against the on-chain enclave BTC pubkey.
688            // Same sighash — both sigs commit to the same multi_a leaf.
689            let guardian_arr: &[u8; 64] =
690                guardian_sig_bytes.as_slice().try_into().map_err(|_| {
691                    anyhow!(
692                        "Guardian signature {i} is not 64 bytes for WithdrawalTransaction {}",
693                        message.withdrawal_id
694                    )
695                })?;
696            let guardian_sig = SchnorrSignature::from_byte_array(guardian_arr)
697                .map_err(|e| anyhow!("Invalid guardian Schnorr signature at input {i}: {e}"))?;
698            guardian_schnorr_pk
699                .verify(sighash, &guardian_sig)
700                .map_err(|e| {
701                    anyhow!("Guardian signature verification failed for input {i}: {e}")
702                })?;
703        }
704
705        self.sign_message_proto(message)
706    }

708    /// Validate and BLS-sign one incremental chunk of per-input MPC signatures
709    /// (`MpcInputSignaturesMessage`). Each `(index, signature)` is verified
710    /// against that input's sighash before the member signs the chunk cert, so a
711    /// leader cannot obtain a cert over signatures the committee hasn't checked.
712    #[tracing::instrument(
713        level = "info",
714        skip_all,
715        fields(withdrawal_id = %message.withdrawal_id, chunk_size = message.indices.len()),
716    )]
717    pub fn validate_and_sign_mpc_input_signatures(
718        &self,
719        message: &MpcInputSignaturesMessage,
720    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
721        let txn = self
722            .onchain_state()
723            .withdrawal_txn(&message.withdrawal_id)
724            .ok_or_else(|| {
725                anyhow!(
726                    "WithdrawalTransaction {} not found on-chain",
727                    message.withdrawal_id
728                )
729            })?;
730
731        anyhow::ensure!(
732            !txn.is_fully_signed(),
733            "WithdrawalTransaction {} is already finalized",
734            message.withdrawal_id
735        );
736        anyhow::ensure!(
737            message.indices.len() == message.signatures.len(),
738            "Chunk indices ({}) and signatures ({}) length mismatch for WithdrawalTransaction {}",
739            message.indices.len(),
740            message.signatures.len(),
741            message.withdrawal_id
742        );
743
744        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
745        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
746
747        for (chunk_pos, (&input_index, mpc_sig_bytes)) in message
748            .indices
749            .iter()
750            .zip(message.signatures.iter())
751            .enumerate()
752        {
753            let i = input_index as usize;
754            anyhow::ensure!(
755                i < txn.inputs.len(),
756                "Chunk input index {i} out of range ({}) for WithdrawalTransaction {}",
757                txn.inputs.len(),
758                message.withdrawal_id
759            );
760            let sighash = &signing_messages[i];
761            let mpc_arr: &[u8; 64] = mpc_sig_bytes.as_slice().try_into().map_err(|_| {
762                anyhow!("MPC signature at chunk position {chunk_pos} (input {i}) is not 64 bytes")
763            })?;
764            let mpc_sig = SchnorrSignature::from_byte_array(mpc_arr)
765                .map_err(|e| anyhow!("Invalid MPC Schnorr signature at input {i}: {e}"))?;
766            let input_pubkey = self.deposit_pubkey(txn.inputs[i].derivation_path.as_ref())?;
767            let mpc_schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
768                .map_err(|e| anyhow!("Failed to convert mpc pubkey for input {i}: {e}"))?;
769            mpc_schnorr_pk
770                .verify(sighash, &mpc_sig)
771                .map_err(|e| anyhow!("MPC signature verification failed for input {i}: {e}"))?;
772        }
773
774        self.sign_message_proto(message)
775    }

787    /// Sign at a specific historical `epoch` using that epoch's DB key.
788    pub(crate) fn sign_message_proto_at_epoch<T: hashi_types::intent::IntentMessage>(
789        &self,
790        message: &T,
791        epoch: u64,
792    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
793        let validator_address = self
794            .config
795            .validator_address()
796            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
797        let committee = self
798            .onchain_state()
799            .state()
800            .hashi()
801            .committees
802            .committees()
803            .get(&epoch)
804            .cloned()
805            .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?;
806        let private_key =
807            self.find_signing_key_for_committee(&committee, validator_address, epoch)?;
808        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();

810            .sign(epoch, validator_address, message)
811            .signature()
812            .as_bytes()
813            .to_vec()
814            .into();
815
816        Ok(hashi_types::proto::MemberSignature {
817            epoch: Some(epoch),
818            address: Some(validator_address.to_string()),
819            public_key: Some(public_key_bytes),
820            signature: Some(signature_bytes),
821        })

826    /// Rebuild the transition from on-chain state and sign with the historical key.
827    #[tracing::instrument(level = "info", skip_all, fields(from_epoch))]
828    pub fn validate_and_sign_committee_transition(
829        &self,
830        from_epoch: u64,
831    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
832        if let Some(signature) = self.get_committee_handoff_signature(from_epoch) {
833            return Ok(signature);
834        }
835
836        let validator_address = self
837            .config
838            .validator_address()
839            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
840
841        let onchain = self.onchain_state();
842        let state = onchain.state();
843        let committees_map = state.hashi().committees.committees();
844        let from_committee = committees_map
845            .get(&from_epoch)
846            .ok_or_else(|| anyhow!("no on-chain committee for epoch {from_epoch}"))?;
847        if !from_committee
848            .members()
849            .iter()
850            .any(|m| m.validator_address() == validator_address)
851        {
852            anyhow::bail!("not a member of the committee at epoch {from_epoch}");
853        }
854
855        // Hashi committee epochs are sparse: the next entry after `from_epoch`
856        // is generally not `from_epoch + 1`. Both leader and followers derive
857        // the same `to_epoch` from on-chain state, so they sign the same
858        // transition.
859        let new_committee = committees_map
860            .range((from_epoch + 1)..)
861            .next()
862            .map(|(_, c)| c)
863            .ok_or_else(|| anyhow!("no on-chain committee epoch after {from_epoch}"))?;
864
865        let transition = hashi_types::guardian::CommitteeTransitionRequest {
866            new_committee: hashi_types::move_types::Committee::from(new_committee),
867        };
868
869        let signature = self.sign_message_proto_at_epoch(&transition, from_epoch)?;
870        self.store_committee_handoff_signature(from_epoch, signature.clone());
871        Ok(signature)
872    }

917    /// Produce MPC Schnorr signatures for an unsigned withdrawal transaction.
918    #[tracing::instrument(
919        level = "debug",
920        skip_all,
921        fields(withdrawal_txn_id = %txn.id, input_count = txn.inputs.len()),
922    )]
923    async fn mpc_sign_withdrawal_tx(
924        &self,
925        txn: &WithdrawalTransaction,
926        unsigned_tx: &bitcoin::Transaction,
927        requested_input_indices: &[u64],
928        sink: tokio::sync::mpsc::Sender<
929            Result<hashi_types::proto::SignWithdrawalTransactionPartial, tonic::Status>,
930        >,
931    ) -> anyhow::Result<()> {
932        let onchain_state = self.onchain_state().clone();
933        let epoch = onchain_state.epoch();
934        if txn.signing.epoch != epoch {
935            anyhow::bail!(
936                "Stale presig assignment: pending withdrawal {} has signing epoch {}, current is {}. \
937                 Either the leader hasn't called reallocate_presigs yet, \
938                 or this node's on-chain state is behind.",
939                txn.id,
940                txn.signing.epoch,
941                epoch,
942            );
943        }
944        let p2p_channel =
945            RpcP2PChannel::new(onchain_state, epoch, crate::metrics::MPC_LABEL_SIGNING);
946        let signing_manager = self.signing_manager_for(epoch).ok_or_else(|| {
947            anyhow::anyhow!(
948                "SigningManager not available for epoch {epoch}; \
949                 reconciliation may be catching up"
950            )
951        })?;
952        let beacon = S::from_bytes_mod_order(&txn.randomness);
953        let signing_messages = self.withdrawal_signing_messages(unsigned_tx, &txn.inputs)?;
954        let signing_manager_ref = &signing_manager;
955        let p2p_channel_ref = &p2p_channel;
956        let beacon_ref = &beacon;
957        let metrics_ref = &*self.metrics;
958        let txn_id = txn.id;
959        // Per-input presig index is read off the on-chain signing batch slot, so
960        // out-of-order / resume works and the index is always the current-epoch
961        // one assigned by `commit`/`reallocate`. Already-signed inputs are skipped.
962        let signing = &txn.signing;
963        let inputs = &txn.inputs;
964        let sink_ref = &sink;
965        let selected_input_indices =
966            select_withdrawal_signing_indices(signing, requested_input_indices)?;
967        let mut requests = Vec::with_capacity(selected_input_indices.len());
968        let mut index_by_id: HashMap<Address, usize> =
969            HashMap::with_capacity(selected_input_indices.len());
970        for input_index in selected_input_indices {
971            let message = signing_messages
972                .get(input_index)
973                .expect("validated input_index is in range for signing_messages");
974            let global_presig_index = signing
975                .pending_index(input_index)
976                .expect("validated input_index is pending");
977            let signing_id = withdrawal_input_signing_id(&txn_id, input_index as u32);
978            // Change UTXOs (`derivation_path = None`) ride the `[0; 32]` path
979            // everywhere else (leaf script, `deposit_pubkey`). MPC must too —
980            // passing `None` signs for master `G`, not the `derive(G, [0; 32])`
981            // child the 2-of-2 leaf binds.
982            let derivation_address = inputs
983                .get(input_index)
984                .map(|input| {
985                    crate::deposits::normalized_derivation_path(input.derivation_path.as_ref())
986                        .into_inner()
987                })
988                .expect("validated input_index is in range for txn.inputs");
989            index_by_id.insert(signing_id, input_index);
990            requests.push(crate::mpc::SignInput {
991                signing_id,
992                message: message.to_vec(),
993                global_presig_index,
994                derivation_address: Some(derivation_address),
995            });
996        }
997        let (result_tx, mut result_rx) = tokio::sync::mpsc::unbounded_channel();
998        let batch_start = std::time::Instant::now();
999        let collect = signing_manager_ref.sign(
1000            p2p_channel_ref,
1001            requests,
1002            beacon_ref,
1003            WITHDRAWAL_SIGNING_TIMEOUT,
1004            metrics_ref,
1005            result_tx,
1006        );
1007        let forward = async {
1008            while let Some((signing_id, sign_result)) = result_rx.recv().await {
1009                let input_index = index_by_id[&signing_id];
1010                let sign_duration = batch_start.elapsed().as_secs_f64();
1011                match &sign_result {
1012                    Ok(_) => {
1013                        metrics_ref
1014                            .mpc_sign_duration_seconds
1015                            .with_label_values(&["success"])
1016                            .observe(sign_duration);
1017                        metrics_ref
1018                            .presig_pool_remaining
1019                            .set(signing_manager_ref.presignatures_remaining() as i64);
1020                    }
1021                    Err(e) => {
1022                        metrics_ref
1023                            .mpc_sign_duration_seconds
1024                            .with_label_values(&["failure"])
1025                            .observe(sign_duration);
1026                        let reason = match e {
1027                            crate::mpc::types::SigningError::Timeout { .. } => "timeout",
1028                            crate::mpc::types::SigningError::PoolExhausted => "pool_exhausted",
1029                            crate::mpc::types::SigningError::TooManyInvalidSignatures {
1030                                ..
1031                            } => "too_many_invalid",
1032                            crate::mpc::types::SigningError::CryptoError(_) => "crypto_error",
1033                            _ => "other",
1034                        };
1035                        metrics_ref
1036                            .mpc_sign_failures_total
1037                            .with_label_values(&[reason])
1038                            .inc();
1039                    }
1040                }
1041                let partial = sign_result
1042                    .map(|sig| hashi_types::proto::SignWithdrawalTransactionPartial {
1043                        input_index: input_index as u32,
1044                        signature: sig.to_byte_array().to_vec().into(),
1045                    })
1046                    .map_err(|e| {
1047                        tonic::Status::internal(format!(
1048                            "Failed to sign withdrawal transaction input {input_index}: {e}"
1049                        ))
1050                    });
1051                let _ = sink_ref.send(partial).await;
1052            }
1053        };
1054        tokio::join!(collect, forward);
1055        Ok(())
1056    }

1125    /// Build a withdrawal commitment for a batch of approved requests: select
1126    /// UTXOs using the batching-aware coin selection algorithm, build the
1127    /// unsigned BTC tx, and return a `WithdrawalTxCommitment` covering the
1128    /// selected requests.
1129    #[tracing::instrument(level = "debug", skip_all, fields(request_count = requests.len()))]
1130    pub async fn build_withdrawal_tx_commitment(
1131        &self,
1132        requests: &[WithdrawalRequest],
1133    ) -> Result<WithdrawalTxCommitment, WithdrawalCommitmentError> {
1134        // Fetch current fee rate from the Bitcoin node, clamped to a high
1135        // fee rate threshold to avoid overpaying during fee spikes.
1136        let kyoto_fee_rate = self
1137            .btc_monitor()
1138            .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
1139            .await
1140            .map_err(|e| WithdrawalCommitmentError::FeeEstimateFailed(anyhow!(e)))?;
1141        let min_fee_rate = CoinSelectionParams::DEFAULT_MIN_FEE_RATE;
1142        let max_fee_rate = CoinSelectionParams::DEFAULT_HIGH_FEE_RATE_THRESHOLD;
1143        let fee_rate = kyoto_fee_rate.clamp(min_fee_rate, max_fee_rate);
1144
1145        let change_address = self
1146            .get_deposit_address(None)
1147            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
1148
1149        let configured_max_inputs = CoinSelectionParams::DEFAULT_MAX_INPUTS;
1150        let configured_long_term_fee_rate = CoinSelectionParams::DEFAULT_LONG_TERM_FEE_RATE;
1151        let configured_max_requests = self.config.withdrawal_max_batch_size().min(requests.len());
1152
1153        // Snapshot both maps under a single read-lock so they are always
1154        // mutually consistent (e.g., a WithdrawalConfirmed cannot update
1155        // one map but not the other between the two reads).
1156        let (withdrawal_txns, utxo_records) = {
1157            let state = self.onchain_state().state();
1158            (
1159                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
1160                state.hashi().utxo_pool.utxo_records().clone(),
1161            )
1162        };
1163
1164        // Query Bitcoin in parallel for the confirmation count of every
1165        // pending withdrawal so we can accurately fill AncestorTx::confirmations
1166        // instead of always hardcoding 0.
1167        let tx_confirmations = fetch_withdrawal_tx_confirmations(self, &withdrawal_txns).await;
1168
1169        // Map available (unlocked) UTXOs to UtxoCandidates.
1170        let candidates: Vec<UtxoCandidate> = utxo_records
1171            .values()
1172            .filter(|r| r.spent_by.is_none())
1173            .map(|r| {
1174                let status =
1175                    build_utxo_status(self, r, &withdrawal_txns, &tx_confirmations, &utxo_records);
1176                UtxoCandidate {
1177                    id: r.utxo.id,
1178                    amount: r.utxo.amount,
1179                    spend_path: SpendPath::TaprootScriptPath2of2,
1180                    status,
1181                }
1182            })
1183            .collect();
1184
1185        // Map on-chain WithdrawalRequests to the coin-selector view.
1186        // btc_amount is the full withdrawal amount.
1187        let mapped_requests: Vec<utxo_pool::WithdrawalRequest> = requests
1188            .iter()
1189            .map(|r| utxo_pool::WithdrawalRequest {
1190                id: r.id,
1191                recipient: r.bitcoin_address.clone(),
1192                amount: r.btc_amount,
1193                timestamp_ms: r.created_timestamp_ms,
1194            })
1195            .collect();
1196
1197        let mut last_selection_error = None;
1198        let mut result = None;
1199        for request_count in (1..=configured_max_requests).rev() {
1200            let max_inputs = safe_withdrawal_flow_max_inputs(request_count, configured_max_inputs);
1201            if max_inputs == 0 {
1202                continue;
1203            }
1204
1205            let params = CoinSelectionParams {
1206                max_inputs,
1207                long_term_fee_rate: configured_long_term_fee_rate,
1208                max_fee_per_request: self.onchain_state().worst_case_network_fee(),
1209                max_withdrawal_requests: request_count,
1210                max_mempool_chain_depth: self.config.max_mempool_chain_depth(),
1211                ..CoinSelectionParams::new(change_address.clone())
1212            };
1213
1214            match utxo_pool::select_coins(&candidates, &mapped_requests, &params, fee_rate) {
1215                Ok(selection) => {
1216                    if request_count < configured_max_requests {
1217                        tracing::info!(
1218                            selected_requests = selection.selected_requests.len(),
1219                            selected_inputs = selection.inputs.len(),
1220                            configured_max_requests,
1221                            configured_max_inputs,
1222                            max_inputs,
1223                            "Reduced withdrawal batch to stay within Sui commit limits",
1224                        );
1225                    }
1226                    result = Some(selection);
1227                    break;
1228                }
1229                Err(e) => last_selection_error = Some(e),
1230            }
1231        }
1232
1233        let result = result.ok_or_else(|| {
1234            WithdrawalCommitmentError::UtxoSelectionFailed(anyhow!(
1235                last_selection_error
1236                    .map(|e| e.to_string())
1237                    .unwrap_or_else(|| "no withdrawal request count fits Sui commit limits".into())
1238            ))
1239        })?;
1240
1241        // Build outputs: one per selected request (net amount already deducted),
1242        // plus an optional change output.
1243        let mut outputs: Vec<OutputUtxo> = result
1244            .withdrawal_outputs
1245            .iter()
1246            .map(|o| OutputUtxo {
1247                amount: o.amount,
1248                bitcoin_address: o.recipient.clone(),
1249            })
1250            .collect();
1251
1252        if let Some(change_amount) = result.change {
1253            outputs.push(OutputUtxo {
1254                amount: change_amount,
1255                bitcoin_address: hashi_bitcoin::witness_program_from_address(&change_address)
1256                    .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?,
1257            });
1258        }
1259
1260        let selected_utxos: Vec<UtxoId> = result.inputs.iter().map(|u| u.id).collect();
1261        let request_ids: Vec<Address> = result.selected_requests.iter().map(|r| r.id).collect();
1262
1263        // Resolve UtxoCandidates back to full Utxo objects for tx building.
1264        let selected_input_utxos: Vec<Utxo> = result
1265            .inputs
1266            .iter()
1267            .map(|c| self.onchain_state().active_utxo(&c.id))
1268            .collect::<Option<Vec<_>>>()
1269            .ok_or_else(|| {
1270                WithdrawalCommitmentError::BtcTxBuildFailed(anyhow!(
1271                    "a selected UTXO disappeared from the pool between selection and tx build"
1272                ))
1273            })?;
1274
1275        let tx = self
1276            .build_unsigned_withdrawal_tx(&selected_input_utxos, &outputs)
1277            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
1278        let txid = BitcoinTxid::from(tx.compute_txid());
1279
1280        Ok(WithdrawalTxCommitment {
1281            request_ids,
1282            selected_utxos,
1283            outputs,
1284            txid,
1285        })
1286    }
1287
1288    /// Run AML/Sanctions checks for a withdrawal request.
1289    /// If no screener client is configured, checks are skipped.
1290    #[tracing::instrument(level = "debug", skip_all, fields(request_id = %request.id))]
1291    pub(crate) async fn screen_withdrawal(
1292        &self,
1293        request: &WithdrawalRequest,
1294    ) -> Result<(), WithdrawalApprovalError> {
1295        let Some(screener) = self.screener_client() else {
1296            tracing::debug!("AML checks skipped: no screener configured");
1297            return Ok(());
1298        };
1299
1300        // Source: Sui tx digest (base58 string)
1301        let source_tx_hash = request.sui_tx_digest.to_string();
1302
1303        // Destination: Bitcoin address (raw witness bytes -> bech32 string)
1304        let destination_address = hashi_bitcoin::address_string_from_witness_program(
1305            &request.bitcoin_address,
1306            self.config.bitcoin_network(),
1307        )
1308        .map_err(WithdrawalApprovalError::NeverRetry)?;
1309
1310        let approved = screener
1311            .approve_withdrawal(
1312                &source_tx_hash,
1313                &destination_address,
1314                self.config.sui_chain_id(),
1315                self.config.bitcoin_chain_id(),
1316            )
1317            .await
1318            .map_err(|e| WithdrawalApprovalError::AmlServiceError(anyhow!(e)))?;
1319
1320        if !approved {
1321            return Err(WithdrawalApprovalError::NeverRetry(anyhow!(
1322                "AML checks failed for withdrawal request {:?} to {}",
1323                request.id,
1324                destination_address,
1325            )));
1326        }
1327
1328        Ok(())
1329    }
1330}

1474/// Query Bitcoin in parallel for the confirmation count of every pending
1475/// withdrawal transaction. Returns a map from withdrawal ID to confirmation
1476/// count. Withdrawals that are in the mempool, not found, or whose RPC call
1477/// fails are mapped to 0 (treated as unconfirmed).
1478async fn fetch_withdrawal_tx_confirmations(
1479    hashi: &Hashi,
1480    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1481) -> HashMap<Address, u32> {
1482    let futures: Vec<_> = withdrawal_txns
1483        .iter()
1484        .map(|(id, txn)| async {
1485            let btc_txid = txn.txid.into();
1486            let confs = match hashi.btc_monitor().get_transaction_status(btc_txid).await {
1487                Ok(TxStatus::Confirmed { confirmations }) => confirmations,
1488                // Mempool, not found, or RPC error — treat as unconfirmed.
1489                _ => 0,
1490            };
1491            (*id, confs)
1492        })
1493        .collect();
1494    futures::future::join_all(futures)
1495        .await
1496        .into_iter()
1497        .collect()
1498}

1500/// Build the [`UtxoStatus`] for a UTXO record using a pre-fetched snapshot.
1501///
1502/// For confirmed UTXOs (`produced_by = None`) this is simply
1503/// [`UtxoStatus::Confirmed`]. For unconfirmed change outputs
1504/// (`produced_by = Some(withdrawal_id)`) we walk the full ancestor chain
1505/// so that CPFP weight and mempool depth are accurately computed even for
1506/// multi-level chains. If the producing withdrawal has already been removed
1507/// from `withdrawal_txns` (confirmed and cleared), we promote the UTXO
1508/// to `Confirmed` — it is safe to spend immediately.
1509fn build_utxo_status(
1510    hashi: &Hashi,
1511    record: &UtxoRecord,
1512    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1513    tx_confirmations: &HashMap<Address, u32>,
1514    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1515) -> UtxoStatus {
1516    let Some(producing_id) = record.produced_by else {
1517        return UtxoStatus::Confirmed;
1518    };
1519
1520    let chain = build_ancestor_chain(
1521        hashi,
1522        producing_id,
1523        withdrawal_txns,
1524        tx_confirmations,
1525        utxo_records,
1526    );
1527
1528    if chain.is_empty() {
1529        // The producing withdrawal was confirmed and removed from
1530        // withdrawal_txns. The UTXO is safe to spend.
1531        UtxoStatus::Confirmed
1532    } else {
1533        UtxoStatus::Pending { chain }
1534    }
1535}

1537/// Maximum ancestor depth permitted by Bitcoin Core's relay policy
1538/// (`DEFAULT_ANCESTOR_LIMIT = 25`). Bitcoin Core counts the candidate
1539/// transaction itself in the ancestor set, so a UTXO whose existing
1540/// unconfirmed ancestor depth is already `MAX_ANCESTOR_DEPTH - 1` is
1541/// the deepest we can safely spend.
1542pub const MAX_ANCESTOR_DEPTH: usize = 25;
1543
1544/// Count the number of unconfirmed ancestors for a UTXO record by walking
1545/// the `produced_by` chain. Every ancestor that still appears in
1546/// `withdrawal_txns` is conservatively treated as unconfirmed (we skip
1547/// querying Bitcoin for actual confirmation counts). This is used during
1548/// commitment validation to reject UTXOs whose ancestor chain would exceed
1549/// Bitcoin Core's relay limit.
1550///
1551/// The walk is a BFS over the ancestor DAG. Each item on the queue is a
1552/// `(producing_withdrawal_id, depth)` pair. We track the maximum depth
1553/// seen across all branches.
1554fn unconfirmed_ancestor_depth(
1555    record: &UtxoRecord,
1556    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1557    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1558) -> usize {
1559    let Some(producing_id) = record.produced_by else {
1560        return 0;
1561    };
1562
1563    let mut max_depth: usize = 0;
1564    let mut queue = std::collections::VecDeque::new();
1565    queue.push_back((producing_id, 1usize));
1566
1567    while let Some((wid, depth)) = queue.pop_front() {
1568        if depth > MAX_ANCESTOR_DEPTH {
1569            return depth;
1570        }
1571
1572        let Some(txn) = withdrawal_txns.get(&wid) else {
1573            // The producing withdrawal has been confirmed and removed;
1574            // it does not contribute to the unconfirmed chain.
1575            continue;
1576        };
1577
1578        max_depth = std::cmp::max(max_depth, depth);
1579
1580        // Enqueue any inputs that are themselves unconfirmed change
1581        // outputs of an earlier withdrawal.
1582        for input_utxo in &txn.inputs {
1583            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1584                && let Some(parent_id) = input_record.produced_by
1585            {
1586                queue.push_back((parent_id, depth + 1));
1587            }
1588        }
1589    }
1590
1591    max_depth
1592}

1594/// Build the ancestor chain for a UTXO produced by `producing_id`. Each
1595/// unconfirmed ancestor that still appears in `withdrawal_txns` gets
1596/// one [`AncestorTx`] entry with its confirmation count, weight, and fee.
1597/// The walk is a BFS over the ancestor DAG, capped at
1598/// [`MAX_ANCESTOR_DEPTH`] levels.
1599fn build_ancestor_chain(
1600    hashi: &Hashi,
1601    producing_id: Address,
1602    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1603    tx_confirmations: &HashMap<Address, u32>,
1604    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1605) -> Vec<AncestorTx> {
1606    let mut chain = Vec::new();
1607    let mut queue = std::collections::VecDeque::new();
1608    queue.push_back((producing_id, 0usize));
1609
1610    while let Some((wid, depth)) = queue.pop_front() {
1611        if depth >= MAX_ANCESTOR_DEPTH {
1612            continue;
1613        }
1614
1615        let Some(txn) = withdrawal_txns.get(&wid) else {
1616            continue;
1617        };
1618
1619        let Ok(tx) = hashi.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs()) else {
1620            continue;
1621        };
1622
1623        let confirmations = tx_confirmations.get(&wid).copied().unwrap_or(0);
1624        let input_total: u64 = txn.inputs.iter().map(|u| u.amount).sum();
1625        let output_total: u64 = txn.all_outputs().iter().map(|o| o.amount).sum();
1626
1627        chain.push(AncestorTx {
1628            confirmations,
1629            tx_weight: tx.weight(),
1630            tx_fee: input_total.saturating_sub(output_total),
1631        });
1632
1633        for input_utxo in &txn.inputs {
1634            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1635                && let Some(parent_id) = input_record.produced_by
1636            {
1637                queue.push_back((parent_id, depth + 1));
1638            }
1639        }
1640    }
1641
1642    chain
1643}

1645/// Deterministic from on-chain state and the leader-supplied
1646/// `(timestamp_secs, seq)`, so every validator reconstructs the same request.
1647pub fn build_guardian_withdrawal_request(
1648    hashi: &Hashi,
1649    txn: &WithdrawalTransaction,
1650    timestamp_secs: u64,
1651    seq: u64,
1652) -> anyhow::Result<hashi_types::guardian::StandardWithdrawalRequest> {
1653    use hashi_types::bitcoin::InputUTXO;
1654    use hashi_types::bitcoin::OutputUTXOWire;
1655    use hashi_types::bitcoin::TxUTXOs;
1656
1657    let network = hashi.config.bitcoin_network();
1658
1659    let inputs: Vec<_> = txn.inputs.iter().map(InputUTXO::from).collect();
1660
1661    // First N outputs are external payouts; any trailing output is internal change.
1662    let all_outputs = txn.all_outputs();
1663    let num_requests = txn.request_ids.len();
1664    let outputs = all_outputs
1665        .iter()
1666        .enumerate()
1667        .map(|(i, output)| {
1668            if i < num_requests {
1669                let script_pubkey =
1670                    hashi_bitcoin::script_pubkey_from_witness_program(&output.bitcoin_address)?;
1671                let address =
1672                    hashi_bitcoin::BitcoinAddress::from_script(&script_pubkey, network)
1673                        .map_err(|e| anyhow!("Cannot derive address from output script: {e}"))?;
1674                Ok(OutputUTXOWire::external(
1675                    address.into_unchecked(),
1676                    Amount::from_sat(output.amount),
1677                ))
1678            } else {
1679                Ok(OutputUTXOWire::internal(
1680                    sui_sdk_types::Address::ZERO,
1681                    Amount::from_sat(output.amount),
1682                ))
1683            }
1684        })
1685        .collect::<anyhow::Result<Vec<_>>>()?;
1686
1687    let utxos = TxUTXOs::new(inputs, outputs, network)
1688        .map_err(|e| anyhow!("Failed to build guardian TxUTXOs: {e}"))?;
1689
1690    // The on-chain `WithdrawalTransaction` UID doubles as the guardian-side `wid`.
1691    let wid: hashi_types::guardian::WithdrawalID = txn.id;
1692
1693    Ok(hashi_types::guardian::StandardWithdrawalRequest::new(
1694        wid,
1695        utxos,
1696        timestamp_secs,
1697        seq,
1698    ))
1699}
