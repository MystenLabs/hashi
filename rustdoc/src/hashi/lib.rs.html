154    /// Record a successful guardian finalize; monotonic in `seq`.
155    pub(crate) fn record_guardian_finalized(&self, seq: u64, wid: sui_sdk_types::Address) {
156        let mut last = self.guardian_last_finalized.write().unwrap();
157        match *last {
158            Some((prev_seq, _)) if seq < prev_seq => {}
159            _ => *last = Some((seq, wid)),
160        }

179    pub fn set_mpc_manager(&self, manager: mpc::MpcManager) {
180        self.metrics
181            .mpc_manager_epoch
182            .set(manager.mpc_config.epoch as i64);
183        match self.mpc_manager.get() {
184            Some(lock) => {
185                // RwLock::write only fails if poisoned (a thread panicked while holding the lock).
186                // Poisoning indicates a bug, so we propagate the panic rather than recover.
187                *lock.write().unwrap() = manager;
188            }
189            None => {
190                // First-time initialization (e.g. new committee member joining mid-rotation).
191                let _ = self.mpc_manager.set(Arc::new(RwLock::new(manager)));
192            }
193        }
194    }

275    pub fn guardian_client(&self) -> Option<&grpc::guardian_client::GuardianClient> {
276        if self.guardian_client.get().is_none() {
277            // Pre-launch boot: no guardian endpoint existed at startup.
278            // `guardian_url` lands on-chain with the launch tx
279            // (finish_publish); resolve the client on first use afterwards.
280            let endpoint = self.onchain_state_opt().and_then(|onchain| {
281                let state = onchain.state();
282                state.hashi().config.guardian_url().map(|s| s.to_string())
283            });
284            if let Some(endpoint) = endpoint {
285                match grpc::guardian_client::GuardianClient::new(&endpoint) {
286                    Ok(guardian) => {
287                        let guardian = guardian.with_metrics(self.metrics.clone());
288                        tracing::info!(
289                            "Guardian client configured from on-chain config for {}",
290                            guardian.endpoint()
291                        );
292                        self.metrics.guardian_enabled.set(1);
293                        let _ = self.guardian_client.set(Some(guardian));
294                    }
295                    Err(e) => {
296                        tracing::warn!("Failed to configure guardian client for {endpoint}: {e}")
297                    }
298                }
299            }
300        }
301        self.guardian_client.get().and_then(|opt| opt.as_ref())
302    }

328    pub fn prepare_encryption_key(&self, epoch: u64) -> anyhow::Result<EncryptionPublicKey> {
329        if let Some(existing) = self.db.get_encryption_key(epoch)? {
330            return Ok(EncryptionPublicKey::from_private_key(&existing));
331        }
332        let private_key = EncryptionPrivateKey::new(&mut rand::thread_rng());
333        let public_key = EncryptionPublicKey::from_private_key(&private_key);
334        self.db
335            .store_encryption_key(epoch, &private_key)
336            .map_err(|e| anyhow!("failed to store encryption key for epoch {epoch}: {e}"))?;
337        Ok(public_key)
338    }

386        tracing::info!(

390        );

394    fn find_encryption_key_for_committee(
395        &self,
396        committee: &hashi_types::committee::Committee,
397        validator_address: sui_sdk_types::Address,
398        epoch: u64,
399    ) -> anyhow::Result<EncryptionPrivateKey> {
400        let member = committee
401            .members()
402            .iter()
403            .find(|m| m.validator_address() == validator_address)
404            .ok_or_else(|| anyhow!("validator not in committee for epoch {epoch}"))?;
405        let pub_key = member.encryption_public_key();
406        self.db
407            .find_encryption_key_matching(pub_key)
408            .map_err(|e| anyhow!("DB error looking up encryption key for epoch {epoch}: {e}"))?
409            .ok_or_else(|| {
410                anyhow!(
411                    "no DB encryption key matches committee record for epoch {epoch}; \
412                     restore the node DB to rejoin this epoch — otherwise replacement \
413                     keys are registered automatically and the node rejoins at the next \
414                     reconfig"
415                )
416            })
417    }

494    pub(crate) async fn next_reconfig_epoch(&self) -> anyhow::Result<u64> {
495        use sui_rpc::proto::sui::rpc::v2::GetServiceInfoRequest;
496        let mut client = self.onchain_state().client();
497        let service_info = client
498            .ledger_client()
499            .get_service_info(GetServiceInfoRequest::default())
500            .await?
501            .into_inner();
502        let sui_epoch = service_info.epoch();
503        let hashi_epoch = self.onchain_state().epoch();
504        let is_genesis = hashi_epoch == 0 && self.onchain_state().current_committee().is_none();
505        Ok(if is_genesis || hashi_epoch < sui_epoch {
506            sui_epoch
507        } else {
508            sui_epoch + 1
509        })
510    }

512    fn resolve_previous_encryption_key(
513        &self,
514        committee_set: &onchain::types::CommitteeSet,
515        target_epoch: u64,
516        validator_address: sui_sdk_types::Address,
517    ) -> anyhow::Result<Option<EncryptionPrivateKey>> {
518        let previous_committee_info = committee_set.previous_committee_for_target(target_epoch);
519        if previous_committee_info.is_none() && target_epoch > 0 {
520            let sui_epoch = self
521                .onchain_state_opt()
522                .map(|s| s.latest_checkpoint_epoch());
523            tracing::info!(
524                target_epoch,
525                committee_set_epoch = committee_set.epoch(),
526                pending_epoch_change = ?committee_set.pending_epoch_change(),
527                sui_epoch = ?sui_epoch,
528                "create_mpc_manager: target_epoch>0 with no previous committee recorded; \
529                 previous_encryption_key=None (genesis bootstrap onto chain at sui_epoch>0)"
530            );
531        }
532        previous_committee_info
533            .map(|(prev_ep, prev_committee)| {
534                self.find_encryption_key_for_committee(prev_committee, validator_address, prev_ep)
535                    .map(Some)
536                    .or_else(|e| {
537                        if !prev_committee
538                            .members()
539                            .iter()
540                            .any(|m| m.validator_address() == validator_address)
541                        {
542                            Ok(None)
543                        } else if self
544                            .committee_encryption_key_lost(prev_committee, validator_address)
545                        {
546                            tracing::warn!(
547                                previous_epoch = prev_ep,
548                                "previous-epoch encryption key lost; continuing without it \
549                                 (party-only: cannot deal previous shares)"
550                            );
551                            Ok(None)
552                        } else {
553                            Err(e)
554                        }
555                    })
556            })
557            .transpose()
558            .map(|opt| opt.flatten())
559    }

561    pub fn create_mpc_manager(
562        &self,
563        epoch: u64,
564        protocol_type: mpc::types::ProtocolType,
565    ) -> anyhow::Result<mpc::MpcManager> {
566        let state = self.onchain_state().state();
567        let hashi = state.hashi();
568        let committee_set = &hashi.committees;
569        let session_id = mpc::SessionId::new(self.config.sui_chain_id(), epoch, &protocol_type);
570        let validator_address = self.config.validator_address()?;
571        let encryption_key = self.find_encryption_key_for_committee(
572            committee_set
573                .committees()
574                .get(&epoch)
575                .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?,
576            validator_address,
577            epoch,
578        )?;
579        let previous_encryption_key =
580            self.resolve_previous_encryption_key(committee_set, epoch, validator_address)?;
581        let signing_key = self.find_signing_key_for_committee(
582            committee_set
583                .committees()
584                .get(&epoch)
585                .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?,
586            validator_address,
587            epoch,
588        )?;
589        let store = Box::new(storage::EpochPublicMessagesStore::new(
590            self.db.clone(),
591            epoch,
592        ));
593        let address = self.config.validator_address()?;
594        let chain_id = self.config.sui_chain_id();
595        let batch_size_per_weight =
596            if let Some(override_val) = self.config.test_batch_size_per_weight {
597                assert_test_only_config(
598                    chain_id,
599                    self.config.bitcoin_chain_id(),
600                    "test_batch_size_per_weight",
601                );
602                override_val
603            } else {
604                BATCH_SIZE_PER_WEIGHT
605            };
606        if self.config.test_corrupt_shares_for.is_some() {
607            assert_test_only_config(
608                chain_id,
609                self.config.bitcoin_chain_id(),
610                "test_corrupt_shares_for",
611            );
612        }
613        let presignature_derivation_activation_epoch =
614            if let Some(epoch) = self.config.test_presignature_derivation_activation_epoch {
615                assert_test_only_config(
616                    chain_id,
617                    self.config.bitcoin_chain_id(),
618                    "test_presignature_derivation_activation_epoch",
619                );
620                epoch
621            } else {
622                constants::presignature_derivation_activation_epoch(chain_id)
623            };
624        Ok(mpc::MpcManager::new(
625            address,
626            committee_set,
627            epoch,
628            session_id,
629            encryption_key,
630            previous_encryption_key,
631            signing_key,
632            store,
633            chain_id,
634            self.config.test_weight_divisor,

761    pub async fn start(self: Arc<Self>) -> anyhow::Result<Service> {
762        let screener = if let Some(endpoint) = self.config.screener_endpoint() {
763            match grpc::screener_client::ScreenerClient::new(endpoint) {
764                Ok(client) => {
765                    tracing::info!("Screener client configured for {}", client.endpoint());
766                    Some(client)
767                }
768                Err(e) => {
769                    tracing::warn!(
770                        "Failed to configure screener client for {}: {}",
771                        endpoint,
772                        e
773                    );
774                    None
775                }
776            }
777        } else {
778            tracing::warn!("No screener endpoint configured; AML screening will be skipped");
779            None
780        };
781
782        self.metrics
783            .screener_enabled
784            .set(if screener.is_some() { 1 } else { 0 });
785
786        self.screener_client
787            .set(screener)
788            .map_err(|_| anyhow!("Screener client already initialized"))?;
789
790        // Verify Sui RPC is on the expected chain before loading any state.
791        self.verify_sui_chain_id().await?;
792
793        // Initialize on-chain state first so we can read guardian config from it.
794        let onchain_service = self.initialize_onchain_state().await?;
795
796        // Guardian endpoint: prefer the on-chain value (published by the
797        // launch tx, `finish_publish`), falling back to local config. On a
798        // pre-launch boot neither may exist yet — nodes still need to start
799        // to register their keys — so leave the client unresolved;
800        // `guardian_client()` resolves it lazily from on-chain config once
801        // the launch lands.
802        let guardian_endpoint = {
803            let state = self.onchain_state().state();
804            state.hashi().config.guardian_url().map(|s| s.to_string())
805        }
806        .or_else(|| self.config.guardian_endpoint().map(|s| s.to_string()));
807
808        match guardian_endpoint {
809            Some(guardian_endpoint) => {
810                let guardian = grpc::guardian_client::GuardianClient::new(&guardian_endpoint)
811                    .map_err(|e| {
812                        anyhow!("Failed to configure guardian client for {guardian_endpoint}: {e}")
813                    })?
814                    .with_metrics(self.metrics.clone());
815                tracing::info!("Guardian client configured for {}", guardian.endpoint());
816
817                self.metrics.guardian_enabled.set(1);
818                self.guardian_client
819                    .set(Some(guardian))
820                    .map_err(|_| anyhow!("Guardian client already initialized"))?;
821            }
822            None => tracing::warn!(
823                "Guardian endpoint not available yet (pre-launch: `guardian_url` lands \
824                 on-chain with finish_publish and no local `guardian_endpoint` is set); \
825                 will configure the guardian client once it appears on-chain"
826            ),
827        }
828
829        // Verify the local bitcoin_chain_id matches the on-chain value.
830        self.verify_bitcoin_chain_id()?;
831
832        // Sweep any SUI in the configured account to AB to enable parallelization of txns
833        sui_tx_executor::sweep_to_address_balance(&mut self.onchain_state().client(), &self.config)
834            .await?;
835
836        let next_epoch_keys = match self.next_reconfig_epoch().await {
837            Ok(next_epoch) => self
838                .prepare_next_epoch_keys(next_epoch)
839                .inspect_err(|e| {
840                    tracing::warn!(
841                        "Failed to prepare encryption/signing keys for epoch {next_epoch}: {e}; \
842                         will retry before next start_reconfig"
843                    )
844                })
845                .ok(),
846            Err(e) => {
847                tracing::warn!("Failed to compute next reconfig epoch: {e}");
848                None
849            }
850        };
851
852        // Register validator (if not already registered) and update any stale metadata.
853        match sui_tx_executor::SuiTxExecutor::from_config(&self.config, self.onchain_state())?
854            .execute_register_or_update_validator(
855                &self.config,
856                None,
857                next_epoch_keys.as_ref().map(|k| &k.encryption_public_key),
858                next_epoch_keys.as_ref().map(|k| &k.signing_private_key),
859            )
860            .await
861        {
862            Ok(true) => tracing::info!("Validator registered/updated on-chain"),
863            Ok(false) => tracing::debug!("Validator metadata is already up-to-date"),
864            Err(e) => tracing::warn!("Failed to register/update validator metadata: {e}"),
865        }
866
867        if self.is_in_current_committee() {
868            tracing::info!("Node is in the current committee; MPC service will recover state");
869        } else if self.onchain_state().epoch() == 0
870            && self.onchain_state().current_committee().is_none()
871        {
872            tracing::info!("No initial committee yet; MPC service will handle genesis bootstrap");
873        } else {
874            tracing::info!(
875                "Node is not in the current committee; skipping initial DKG manager creation"
876            );
877        }
878
879        let (backup_service, backup_handle) = backup::BackupService::new(self.clone());
880        let (mpc_service, mpc_handle) = mpc::MpcService::new(self.clone(), backup_handle);
881        self.mpc_handle
882            .set(mpc_handle)
883            .expect("MpcHandle already set");
884
885        let btc_monitor_service = self.initialize_btc_monitor().map_err(|e| {
886            tracing::error!("Failed to initialize BtcMonitor: {e}");
887            e
888        })?;
889
890        // Start services
891        let (_http_addr, http_service) = grpc::HttpService::new(self.clone()).start().await;
892        let leader_service = leader::LeaderService::new(self.clone()).start();
893        let backup_service = backup_service.start();
894        let mpc_service = mpc_service.start();
895        let guardian_bootstrap_service = self.clone().start_guardian_bootstrap();
896        let sui_balance_service = self.clone().start_sui_balance_metric();
897
898        let service = Service::new()
899            .merge(onchain_service)
900            .merge(btc_monitor_service)
901            .merge(http_service)
902            .merge(leader_service)
903            .merge(backup_service)
904            .merge(mpc_service)
905            .merge(guardian_bootstrap_service)
906            .merge(sui_balance_service);
907
908        Ok(service)
909    }

911    async fn try_seed_guardian_state(&self) -> bool {
912        self.metrics.guardian_bootstrap_attempts_total.inc();
913        let Ok(info) = self.fetch_guardian_info_data().await else {
914            return false;
915        };
916        if !self.verify_and_pin_guardian_btc_pubkey(info.enclave_btc_pubkey) {
917            return false;
918        }
919        let (Some(state), Some(config)) = (info.limiter_state, info.limiter_config) else {
920            self.metrics.record_guardian_bootstrap_outcome(
921                metrics::GUARDIAN_BOOTSTRAP_OUTCOME_NO_LIMITER_YET,
922            );
923            tracing::debug!("guardian bootstrap: guardian has no limiter yet");
924            return false;
925        };
926        let limiter = Arc::new(guardian_limiter::LocalLimiter::new(config, state));
927        if self.local_limiter.set(limiter.clone()).is_ok() {
928            tracing::info!(
929                ?state,
930                ?config,
931                "Local guardian limiter seeded from GetGuardianInfo",
932            );
933            self.metrics.record_limiter_state(&state, &config);
934            self.metrics.guardian_limiter_initialized.set(1);
935            // Hand the same Arc to OnchainState so the watcher can advance
936            // it inline when WithdrawalSigned fires.
937            self.onchain_state().set_local_limiter(limiter);
938        }
939        self.metrics
940            .record_guardian_bootstrap_outcome(metrics::GUARDIAN_BOOTSTRAP_OUTCOME_SUCCESS);
941        true
942    }

1052    /// Snap the local limiter to the guardian once `tracker` confirms the
1053    /// drift has persisted past ordinary in-flight lag.
1054    async fn reconcile_guardian_limiter(
1055        &self,
1056        tracker: &mut guardian_limiter::LimiterStallTracker,
1057    ) {
1058        let Some((limiter, state)) = self.guardian_limiter_and_state().await else {
1059            return;
1060        };
1061        let local_seq = limiter.next_seq();
1062        if tracker.observe(local_seq, state.next_seq) {
1063            tracing::warn!(
1064                local_seq,
1065                guardian_seq = state.next_seq,
1066                "Local guardian limiter stalled away from the guardian; reconciled to authoritative state",
1067            );
1068            self.apply_limiter_reconcile(&limiter, state);
1069        } else if limiter.reconcile_token_drift(state) {
1070            // Equal seq, drifted bucket: the mirror debits at sign-time, the
1071            // guardian at finalize-time — invisible to the seq-only tracker above.
1072            self.record_limiter_reconcile(&limiter, state);
1073            tracing::debug!(
1074                seq = state.next_seq,
1075                "Local guardian limiter token-drifted from the guardian; reconciled",
1076            );
1077        }
1078    }

1080    /// Reconcile on a watcher rescrape. A rescrape can't tell a dropped signed event
1081    /// (real drift) from an in-flight withdrawal (consumed by the guardian, event
1082    /// pending), so forward-snapping `next_seq` would double-count the in-flight ones
1083    /// (the snap counts them, then their `WithdrawalSigned` counts them again).
1084    /// So only re-align the bucket at a matching seq; seq drift is left to the stall tick.
1085    async fn reconcile_guardian_limiter_on_rescrape(&self) {
1086        let Some((limiter, state)) = self.guardian_limiter_and_state().await else {
1087            return;
1088        };
1089        if limiter.reconcile_token_drift(state) {
1090            self.record_limiter_reconcile(&limiter, state);
1091            tracing::debug!(
1092                seq = state.next_seq,
1093                "Local guardian limiter bucket reconciled after watcher rescrape",
1094            );
1095        }
1096    }

1098    /// Poll the operator gas wallet's total SUI balance (owned coins plus
1099    /// address balance — `sui client gas` misses the latter) into
1100    /// `hashi_sui_balance`, so operators can alert on drain before
1101    /// transactions start failing gas selection.
1102    fn start_sui_balance_metric(self: Arc<Self>) -> Service {
1103        const BALANCE_POLL_INTERVAL: std::time::Duration = std::time::Duration::from_secs(60);
1104        Service::new().spawn_aborting(async move {
1105            // -1 = never sampled; keeps a `0 <= balance < threshold` alert
1106            // from firing on nodes where this poller is disabled or has not
1107            // yet succeeded (the gauge would otherwise read a false 0).
1108            self.metrics.sui_balance.set(-1);
1109            let owner = match self.config.operator_private_key() {
1110                Ok(key) => key.verifying_key().derive_address(),
1111                Err(e) => {
1112                    tracing::info!("SUI balance metric disabled; operator key unavailable: {e}");
1113                    return Ok(());
1114                }
1115            };
1116            let request = sui_rpc::proto::sui::rpc::v2::GetBalanceRequest::default()
1117                .with_owner(owner)
1118                .with_coin_type(sui_sdk_types::StructTag::sui());
1119            let mut client = self.onchain_state().client();
1120            let mut interval = tokio::time::interval(BALANCE_POLL_INTERVAL);
1121            interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
1122            loop {
1123                interval.tick().await;
1124                match client.state_client().get_balance(request.clone()).await {
1125                    Ok(response) => {
1126                        if let Some(balance) = response.into_inner().balance.and_then(|b| b.balance)
1127                        {
1128                            self.metrics
1129                                .sui_balance
1130                                .set(balance.min(i64::MAX as u64) as i64);
1131                        }
1132                    }
1133                    Err(e) => tracing::debug!("failed to fetch operator SUI balance: {e}"),
1134                }
1135            }
1136        })

1139    fn start_guardian_bootstrap(self: Arc<Self>) -> Service {
1140        use backon::Retryable;
1141        // Cadence for the limiter reconciliation safety net below.
1142        const RECONCILE_INTERVAL: std::time::Duration = std::time::Duration::from_secs(15);
1143        Service::new().spawn_aborting(async move {
1144            // The guardian may be set up after this node boots: on a
1145            // pre-launch boot the on-chain guardian_url only lands with
1146            // finish_publish, and an external guardian can be provisioned
1147            // later still. `guardian_client()` re-resolves from on-chain
1148            // config on every call, so wait for it rather than giving up —
1149            // returning here would permanently skip limiter seeding and the
1150            // reconcile loop for this process lifetime.
1151            while self.guardian_client().is_none() {
1152                tokio::time::sleep(std::time::Duration::from_secs(10)).await;
1153            }
1154            let policy = backon::ExponentialBuilder::default()
1155                .with_min_delay(std::time::Duration::from_secs(1))
1156                .with_max_delay(std::time::Duration::from_secs(30))
1157                .without_max_times();
1158            let _ = (|| async {
1159                if self.try_seed_guardian_state().await {
1160                    Ok::<(), ()>(())
1161                } else {
1162                    Err(())
1163                }
1164            })
1165            .retry(policy)
1166            .await;
1167            tracing::info!("Guardian bootstrap complete");
1168
1169            // Safety net for the event path: the periodic tick catches slow seq drift
1170            // (stall-gated) + bucket drift; a rescrape notify re-aligns the bucket.
1171            let reconcile_notify = self.onchain_state().limiter_reconcile_notify();
1172            // Pinned + re-armed so a notify can't be lost to `select!` cancellation.
1173            let rescraped = reconcile_notify.notified();
1174            tokio::pin!(rescraped);
1175            let mut interval = tokio::time::interval(RECONCILE_INTERVAL);
1176            interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
1177            let mut tracker = guardian_limiter::LimiterStallTracker::default();
1178            loop {
1179                tokio::select! {
1180                    _ = interval.tick() => {
1181                        self.reconcile_guardian_limiter(&mut tracker).await;
1182                    }
1183                    _ = &mut rescraped => {
1184                        // Re-arm first so a rescrape during the reconcile isn't lost.
1185                        rescraped.set(reconcile_notify.notified());
1186                        self.reconcile_guardian_limiter_on_rescrape().await;
1187                    }
1188                }
1189            }
1190        })
1191    }

1204/// Wrap a bitcoind response-decode failure with a version hint. A response
1205/// that is valid JSON but does not match the v29 schema is almost always a
1206/// pre-v29 bitcoind (`warnings` was a string, not an array, until Core v28),
1207/// so name the real problem instead of surfacing a bare serde error.
1208fn bitcoind_rpc_error_with_version_hint(e: corepc_client::client_sync::Error) -> anyhow::Error {
1209    match &e {
1210        corepc_client::client_sync::Error::JsonRpc(jsonrpc::Error::Json(_))
1211        | corepc_client::client_sync::Error::Json(_) => anyhow!(
1212            "unexpected getblockchaininfo response from bitcoind: {e}. Hashi requires \
1213             Bitcoin Core v29 or newer; check the node's version with `bitcoin-cli getnetworkinfo`"
1214        ),
1215        _ => anyhow!(e),
1216    }
1217}

1248/// Cross-check the live guardian's BTC pubkey against the on-chain pin.
1249/// `expected = None` skips the check (the pin may be bound later via
1250/// governance). Extra fatal case: when the on-chain key is `Some` but the live
1251/// guardian doesn't return one — the deploy must have come from a guardian that
1252/// did.
1253fn verify_btc_pub_key_matches(
1254    live: Option<&hashi_types::bitcoin::BitcoinPubkey>,
1255    expected: Option<&[u8]>,
1256    metrics: &metrics::Metrics,
1257) -> bool {
1258    let Some(expected) = expected else {
1259        return true;
1260    };
1261    let Some(live) = live else {
1262        metrics.record_guardian_bootstrap_outcome(
1263            metrics::GUARDIAN_BOOTSTRAP_OUTCOME_BTC_KEY_MISSING_FROM_INFO,
1264        );
1265        tracing::error!(
1266            on_chain = %hex::encode(expected),
1267            "FATAL: on-chain guardian_btc_public_key is set but guardian \
1268             /info did not return enclave_btc_pubkey; refusing to seed \
1269             or reconcile local limiter",
1270        );
1271        return false;
1272    };
1273    if live.serialize().as_slice() == expected {
1274        return true;
1275    }
1276    metrics.record_guardian_bootstrap_outcome(metrics::GUARDIAN_BOOTSTRAP_OUTCOME_BTC_KEY_MISMATCH);
1277    tracing::error!(
1278        on_chain = %hex::encode(expected),
1279        from_info = %hex::encode(live.serialize()),
1280        "FATAL: guardian /info enclave_btc_pubkey does not match on-chain \
1281         guardian_btc_public_key; refusing to seed or reconcile local limiter",
1282    );
1283    false
1284}
