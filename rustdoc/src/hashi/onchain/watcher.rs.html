34/// Reconnect if the checkpoint stream goes silent this long. A half-open h2
35/// stream yields neither an item nor an error, so an unbounded read hangs — the
36/// SDK's keepalive only trips on a fully-dead connection, not a live one whose
37/// server silently stopped sending checkpoints.
38const CHECKPOINT_STREAM_STALL_TIMEOUT: std::time::Duration = std::time::Duration::from_secs(120);
39
40/// How often to unconditionally rescrape the full on-chain state. The event
41/// stream can drift from chain state (e.g. writes that emit no event, such as
42/// `cleanup_spent_utxos`); a periodic rescrape bounds how long any such drift
43/// can persist.
44const PERIODIC_RESCRAPE_INTERVAL: std::time::Duration = std::time::Duration::from_secs(10 * 60);
45
46#[tracing::instrument(name = "watcher", skip_all)]
47pub async fn watcher(sui_rpc_url: String, state: OnchainState, metrics: Option<Arc<Metrics>>) {
48    let subscription_read_mask = FieldMask::from_paths([
49        Checkpoint::path_builder().sequence_number(),
50        Checkpoint::path_builder().summary().timestamp(),
51        Checkpoint::path_builder().summary().epoch(),
52        Checkpoint::path_builder()
53            .transactions()
54            .events()
55            .events()
56            .contents()
57            .finish(),
58        Checkpoint::path_builder().transactions().digest(),
59        Checkpoint::path_builder()
60            .transactions()
61            .effects()
62            .status()
63            .finish(),
64    ]);
65
66    let mut rescrape_state = false;
67    // Boot scrape just populated the snapshot; start the poll clocks from here.
68    let mut last_config_refresh = std::time::Instant::now();
69    let mut last_rescrape = std::time::Instant::now();
70
71    loop {
72        // Reconnect with a fresh client each iteration. Re-subscribing on the
73        // same channel can reuse a wedged h2 connection — the one whose stream
74        // just stalled — and silently hang again; a new client forces a clean
75        // connection. Reconnects are rare, so the extra setup cost is fine.
76        let mut client = match crate::sui_rpc_client::new_sui_rpc_client(&sui_rpc_url) {
77            Ok(client) => client,
78            Err(e) => {
79                tracing::warn!("error creating Sui RPC client: {e}");
80                rescrape_state = true;
81                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
82                continue;
83            }
84        };
85        // Same decode limit as the bootstrap scrape — this client does the
86        // reconnect rescrape, which reads the full (large) on-chain state.
87        if let Some(limit) = state.grpc_max_decoding_message_size() {
88            client = client.with_max_decoding_message_size(limit);
89        }
90
91        let mut subscription = match client
92            .subscription_client()
93            .subscribe_checkpoints(
94                SubscribeCheckpointsRequest::default()
95                    .with_read_mask(subscription_read_mask.clone()),
96            )
97            .await
98        {
99            Ok(subscription) => subscription,
100            Err(e) => {
101                tracing::warn!("error trying to subscribe to checkpoints: {e}");
102                rescrape_state = true;
103                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
104                continue;
105            }
106        }
107        .into_inner();
108
109        // Rescrape the chain state in the event our subscription broke
110        if rescrape_state {
111            if let Err(e) = rescrape_hashi_state(&client, &state, metrics.as_ref()).await {
112                tracing::warn!("error trying to rescrape hashi's state: {e}");
113                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
114                continue;
115            }
116            last_config_refresh = std::time::Instant::now();
117            last_rescrape = std::time::Instant::now();
118        }
119
120        while let Ok(Some(item)) =
121            tokio::time::timeout(CHECKPOINT_STREAM_STALL_TIMEOUT, subscription.next()).await
122        {
123            let checkpoint = match item {
124                Ok(checkpoint) => checkpoint,
125                Err(e) => {
126                    tracing::warn!("error in checkpoint stream: {e}");
127                    break;
128                }
129            };
130
131            let ckpt = checkpoint.cursor();
132            tracing::trace!("received checkpoint {ckpt}");
133            let timestamp_ms = checkpoint
134                .checkpoint()
135                .summary()
136                .timestamp
137                .and_then(|t| proto_to_timestamp_ms(t).ok())
138                .unwrap_or(0);
139            let epoch = checkpoint.checkpoint().summary().epoch();
140            let previous_epoch = state.latest_checkpoint_epoch();
141            if epoch != previous_epoch {
142                tracing::debug!("Sui epoch changed from {previous_epoch} to {epoch}");
143                state.notify(Notification::SuiEpochChanged(epoch));
144            }
145            let mut events = Vec::new();
146            {
147                let state = state.state();
148
149                for txn in checkpoint.checkpoint().transactions() {
150                    // Skip txns that were not successful
151                    if !txn.effects().status().success() {
152                        continue;
153                    }
154
155                    for event in txn.events().events() {
156                        match HashiEvent::try_parse(&state.package_ids, event.contents()) {
157                            Ok(Some(event)) => {
158                                tracing::debug!("found event {:?}", event);
159                                events.push(event);
160                            }
161                            Ok(None) => {}
162                            Err(e) => tracing::error!("unable to parse event: {e}"),
163                        }
164                    }
165                }
166            }
167
168            handle_events(&mut client, &state, timestamp_ms, &events).await;
169
170            if last_config_refresh.elapsed() >= state.config_poll_interval()
171                && launch_pending(&state)
172            {
173                refresh_config_if_changed(&client, &state).await;
174                last_config_refresh = std::time::Instant::now();
175            }
176
177            // Finally update the latest checkpoint info
178            state.update_latest_checkpoint_info(CheckpointInfo {
179                height: ckpt,
180                timestamp_ms,
181                epoch,
182            });
183
184            if let Some(metrics) = &metrics {
185                metrics.update_onchain_state(&state);
186            }
187
188            // Unconditional periodic rescrape. Runs on the watcher task, so
189            // it can't race event application; any checkpoints the scrape
190            // already included are replayed by the stream afterward, which
191            // re-applies them as no-ops. On failure, just wait for the next
192            // interval — the reconnect rescrape still guards correctness.
193            if last_rescrape.elapsed() >= PERIODIC_RESCRAPE_INTERVAL {
194                if let Err(e) = rescrape_hashi_state(&client, &state, metrics.as_ref()).await {
195                    tracing::warn!("periodic rescrape of hashi's state failed: {e}");
196                } else {
197                    last_config_refresh = std::time::Instant::now();
198                }
199                last_rescrape = std::time::Instant::now();
200            }
201        }
202
203        // The stream stalled, errored, or closed. Loop to rebuild the client,
204        // re-subscribe, and rescrape to recover any gap we missed.
205        tracing::warn!("checkpoint stream ended; reconnecting Sui client and rescraping");
206        rescrape_state = true;
207    }
208}

210/// Scrape the full on-chain state and install it as the new snapshot. Used
211/// both after a reconnect (to recover any gap the stream missed) and on the
212/// periodic rescrape interval (to heal drift from eventless writes).
213async fn rescrape_hashi_state(
214    client: &Client,
215    state: &OnchainState,
216    metrics: Option<&Arc<Metrics>>,
217) -> anyhow::Result<()> {
218    let (checkpoint_info, hashi) = super::scrape_hashi(
219        client.clone(),
220        state.hashi_id(),
221        state.package_id_original(),
222    )
223    .await?;
224    state.install_scraped_state(checkpoint_info, hashi)?;
225    if let Some(metrics) = metrics {
226        metrics.update_onchain_state(state);
227    }
228    // Rescrape gapped the event-driven limiter; trigger an eager reconcile.
229    state.request_limiter_reconcile();
230    Ok(())
231}

233/// The poll only exists to catch `finish_publish` (guardian url + BTC key,
234/// written with no event); every other config write emits `ProposalExecuted`.
235/// Once both fields are in the snapshot the launch has landed: stop polling.
236fn launch_pending(state: &OnchainState) -> bool {
237    let state = state.state();
238    let config = &state.hashi().config;
239    config.guardian_url().is_none() || config.guardian_btc_public_key().is_none()
240}

242/// Runs on the watcher task, so it can't race the event-driven config
243/// refresh; bounded so a wedged connection can't hang the checkpoint loop.
244async fn refresh_config_if_changed(client: &Client, state: &OnchainState) {
245    let scrape = super::scrape_hashi_config(client.clone(), state.hashi_id());
246    let config = match tokio::time::timeout(Duration::from_secs(10), scrape).await {
247        Ok(Ok(config)) => config,
248        Ok(Err(e)) => {
249            tracing::warn!("error polling on-chain config: {e}");
250            return;
251        }
252        Err(_) => {
253            tracing::warn!("timed out polling on-chain config");
254            return;
255        }
256    };
257    let mut state = state.state_mut();
258    if state.hashi.config != config {
259        state.hashi.config = config;
260        tracing::info!("on-chain config changed; snapshot refreshed by periodic poll");
261    }
262}

264async fn handle_events(
265    client: &mut Client,
266    state: &OnchainState,
267    checkpoint_timestamp_ms: u64,
268    events: &[HashiEvent],
269) {
270    if events.is_empty() {
271        return;
272    }
273
274    let mut validator_updates = BTreeSet::new();
275
276    for event in events {
277        match event {
278            HashiEvent::ValidatorRegistered(validator_registered) => {
279                validator_updates.insert(validator_registered.validator);
280            }
281            HashiEvent::ValidatorUpdated(validator_updated) => {
282                validator_updates.insert(validator_updated.validator);
283            }
284            HashiEvent::VoteCast(_) => {}
285            HashiEvent::VoteRemoved(_) => {}
286            HashiEvent::ProposalCreated(proposal_created_event) => {
287                let proposal = Proposal {
288                    id: proposal_created_event.proposal_id,
289                    timestamp_ms: proposal_created_event.timestamp_ms,
290                    proposal_type: parse_proposal_type_from_type_tag(
291                        &proposal_created_event.proposal_type,
292                    ),
293                };
294                state
295                    .state_mut()
296                    .hashi
297                    .proposals
298                    .active
299                    .insert(proposal.id, proposal);
300            }
301            HashiEvent::ProposalDeleted(proposal_deleted_event) => {
302                state
303                    .state_mut()
304                    .hashi
305                    .proposals
306                    .active
307                    .remove(&proposal_deleted_event.proposal_id);
308            }
309            HashiEvent::ProposalExecuted(proposal_executed_event) => {
310                // Move the entry from `active` to `executed` to mirror
311                // the on-chain dual-bag layout. Scope the write lock so
312                // it's released before the async config refetch below.
313                {
314                    let mut state_lock = state.state_mut();
315                    let proposals = &mut state_lock.hashi.proposals;
316                    if let Some(proposal) = proposals
317                        .active
318                        .remove(&proposal_executed_event.proposal_id)
319                    {
320                        proposals
321                            .executed
322                            .insert(proposal_executed_event.proposal_id, proposal);
323                    }
324                }
325
326                // When an UpdateConfig or EmergencyPause proposal executes,
327                // the Hashi object's config field changes on-chain. Re-fetch
328                // the config from the Hashi object to keep the in-memory
329                // state current. (The event now carries the typed `data`,
330                // so this could be applied directly in the future.)
331                if matches!(
332                    parse_proposal_type_from_type_tag(&proposal_executed_event.proposal_type),
333                    ProposalType::UpdateConfig
334                        | ProposalType::EmergencyPause
335                        | ProposalType::UpdateGuardian
336                ) {
337                    match super::scrape_hashi_config(client.clone(), state.hashi_id()).await {
338                        Ok(config) => {
339                            state.state_mut().hashi.config = config;
340                            tracing::info!(
341                                "on-chain config refreshed after config-changing proposal"
342                            );
343                        }
344                        Err(e) => {
345                            tracing::error!(
346                                "failed to refresh config after config-changing proposal: {e}"
347                            );
348                        }
349                    }
350                }
351
352                if matches!(
353                    parse_proposal_type_from_type_tag(&proposal_executed_event.proposal_type),
354                    ProposalType::AbortReconfig
355                ) {
356                    match bcs::from_bytes::<AbortReconfig>(&proposal_executed_event.data_bcs) {
357                        Ok(abort_reconfig) => {
358                            let mut state = state.state_mut();
359                            state
360                                .hashi
361                                .committees
362                                .committees_mut()
363                                .remove(&abort_reconfig.epoch);
364                            state.hashi.committees.set_pending_epoch_change(None);
365                        }
366                        Err(e) => {
367                            tracing::error!("unable to decode AbortReconfig proposal data: {e}");
368                        }
369                    }
370                }
371            }
372            HashiEvent::QuorumReached(_) => {}
373            HashiEvent::PackageUpgraded(package_upgraded_event) => {
374                state.add_package_version(
375                    package_upgraded_event.version,
376                    package_upgraded_event.package,
377                );
378            }
379            HashiEvent::Minted(Minted { coin_type, .. })
380            | HashiEvent::Burned(Burned { coin_type, .. }) => {
381                refresh_treasury_cap_supply(client, state, coin_type).await;
382            }
383            HashiEvent::DepositRequested(deposit_requested_event) => {
384                tracing::info!(deposit_request_id = %deposit_requested_event.request_id, "Deposit request detected");
385                let deposit_request = DepositRequest {
386                    id: deposit_requested_event.request_id,
387                    sender: deposit_requested_event.requester_address,
388                    created_timestamp_ms: deposit_requested_event.timestamp_ms,
389                    sui_tx_digest: deposit_requested_event.sui_tx_digest,
390                    utxo: super::types::Utxo {
391                        id: deposit_requested_event.utxo_id,
392                        amount: deposit_requested_event.amount,
393                        derivation_path: deposit_requested_event.derivation_path,
394                    },
395                    approval_cert: None,
396                    approved_timestamp_ms: None,
397                    confirmed_timestamp_ms: None,
398                };
399                state
400                    .state_mut()
401                    .hashi
402                    .deposit_queue
403                    .requests
404                    .insert(deposit_request.id, deposit_request);
405            }
406            HashiEvent::DepositApproved(deposit_approved_event) => {
407                tracing::info!(deposit_request_id = %deposit_approved_event.request_id, "Deposit approved");
408                let mut state = state.state_mut();
409                // Stamp the in-memory request so the leader's next pass
410                // knows it's already approved and only needs to call
411                // `confirm_deposit` once the time-delay elapses. The
412                // event carries the same Sui-clock timestamp the
413                // on-chain request stores, so the local view matches
414                // the on-chain value exactly.
415                if let Some(request) = state
416                    .hashi
417                    .deposit_queue
418                    .requests
419                    .get_mut(&deposit_approved_event.request_id)
420                {
421                    request.approval_cert = Some(deposit_approved_event.cert.clone());
422                    request.approved_timestamp_ms =
423                        Some(deposit_approved_event.approved_timestamp_ms);
424                }
425            }
426            HashiEvent::DepositConfirmed(deposit_confirmed_event) => {
427                tracing::info!(deposit_request_id = %deposit_confirmed_event.request_id, "Deposit confirmed");
428                let mut state = state.state_mut();
429
430                let utxo = deposit_confirmed_event.utxo.clone();
431
432                state
433                    .hashi
434                    .deposit_queue
435                    .requests
436                    .remove(&deposit_confirmed_event.request_id);
437                state.hashi.utxo_pool.utxo_records.insert(
438                    utxo.id,
439                    super::types::UtxoRecord {
440                        utxo,
441                        produced_by: None,
442                        spent_by: None,
443                        spent_epoch: None,
444                    },
445                );
446            }
447            HashiEvent::ExpiredDepositDeleted(expired_deposit_deleted_event) => {
448                tracing::info!(deposit_request_id = %expired_deposit_deleted_event.request_id, "Expired deposit deleted");
449                state
450                    .state_mut()
451                    .hashi
452                    .deposit_queue
453                    .requests
454                    .remove(&expired_deposit_deleted_event.request_id);
455            }
456            HashiEvent::WithdrawalRequested(withdrawal_requested_event) => {
457                tracing::info!(withdrawal_request_id = %withdrawal_requested_event.request_id, "Withdrawal request detected");
458                let withdrawal_request = WithdrawalRequest {
459                    id: withdrawal_requested_event.request_id,
460                    sender: withdrawal_requested_event.requester_address,
461                    btc_amount: withdrawal_requested_event.btc_amount,
462                    bitcoin_address: withdrawal_requested_event.bitcoin_address.clone(),
463                    created_timestamp_ms: withdrawal_requested_event.timestamp_ms,
464                    status: super::types::WithdrawalStatus::Requested,
465                    approval_cert: None,
466                    approved_timestamp_ms: None,
467                    withdrawal_txn_id: None,
468                    sui_tx_digest: withdrawal_requested_event.sui_tx_digest,
469                    btc: withdrawal_requested_event.btc_amount,
470                };
471                state
472                    .state_mut()
473                    .hashi
474                    .withdrawal_queue
475                    .requests
476                    .insert(withdrawal_request.id, withdrawal_request);
477            }
478            HashiEvent::WithdrawalApproved(event) => {
479                tracing::info!(withdrawal_request_id = %event.request_id, "Withdrawal approved");
480                if let Some(request) = state
481                    .state_mut()
482                    .hashi
483                    .withdrawal_queue
484                    .requests
485                    .get_mut(&event.request_id)
486                {
487                    request.status = super::types::WithdrawalStatus::Approved;
488                }
489            }
490            HashiEvent::WithdrawalPickedForProcessing(event) => {
491                tracing::info!(withdrawal_txn_id = %event.withdrawal_txn_id, "Withdrawal picked for processing");
492                // Remove requests from the queue
493                {
494                    let mut state = state.state_mut();
495                    for request_id in &event.request_ids {
496                        state.hashi.withdrawal_queue.requests.remove(request_id);
497                    }
498                }
499
500                // Fetch the full withdrawal transaction from chain
501                match super::fetch_withdrawal_txn(client, event.withdrawal_txn_id).await {
502                    Ok(txn) => {
503                        state
504                            .state_mut()
505                            .hashi
506                            .withdrawal_queue
507                            .withdrawal_txns
508                            .insert(txn.id, txn);
509                    }
510                    Err(e) => {
511                        tracing::error!(
512                            withdrawal_txn_id = %event.withdrawal_txn_id,
513                            "Failed to fetch withdrawal transaction: {e}",
514                        );
515                    }
516                }
517
518                // Lock each input UTXO in the pool and insert the pending
519                // change UTXOs (if any) so they are immediately selectable.
520                {
521                    let mut state = state.state_mut();
522                    for input in &event.inputs {
523                        if let Some(record) = state.hashi.utxo_pool.utxo_records.get_mut(&input.id)
524                        {
525                            record.spent_by = Some(event.withdrawal_txn_id);
526                        }
527                    }
528                    // Change outputs are the trailing outputs: change output `j`
529                    // sits at vout `withdrawal_outputs.len() + j`.
530                    let base_vout = event.withdrawal_outputs.len() as u32;
531                    for (j, change_output) in event.change_outputs.iter().enumerate() {
532                        let change_utxo_id = super::types::UtxoId {
533                            txid: event.txid,
534                            vout: base_vout + j as u32,
535                        };
536                        let change_utxo = super::types::Utxo {
537                            id: change_utxo_id,
538                            amount: change_output.amount,
539                            derivation_path: None,
540                        };
541                        state.hashi.utxo_pool.utxo_records.insert(
542                            change_utxo_id,
543                            super::types::UtxoRecord {
544                                utxo: change_utxo,
545                                produced_by: Some(event.withdrawal_txn_id),
546                                spent_by: None,
547                                spent_epoch: None,
548                            },
549                        );
550                    }
551                }
552            }
553            HashiEvent::WithdrawalSigned(event) => {
554                tracing::info!(withdrawal_txn_id = %event.withdrawal_txn_id, "Withdrawal signatures stored on-chain");
555                // Watcher is the sole mutator of the local limiter
556                // post-bootstrap; advance it inline with the mirror update.
557                // Advance uses the event's checkpoint timestamp (~sign-time)
558                // rather than `txn.created_timestamp_ms` (creation time) to stay in
559                // lockstep with the guardian's `last_updated_at`.
560                // Gate on `signatures.is_none()` for idempotency across checkpoint redelivery and bootstrap replay.
561                let (limiter_inputs, pick_to_sign_ms) = {
562                    let mut state = state.state_mut();
563                    state
564                        .withdrawal_signed_at_ms
565                        .insert(event.withdrawal_txn_id, checkpoint_timestamp_ms);
566                    match state
567                        .hashi
568                        .withdrawal_queue
569                        .withdrawal_txns
570                        .get_mut(&event.withdrawal_txn_id)
571                    {
572                        Some(txn) if !txn.is_fully_signed() => {
573                            // Reflect the finalized 2-of-2 witness in-memory: mark
574                            // every input Signed and attach guardian sigs — together
575                            // those make `is_fully_signed()` derive true.
576                            txn.signing.signatures = event
577                                .signatures
578                                .iter()
579                                .cloned()
580                                .map(MpcSig::Signed)
581                                .collect();
582                            txn.guardian_signatures = Some(event.guardian_signatures.clone());
583                            let amount_sats = withdrawal_limiter_consumption_amount(txn);
584                            let timestamp_secs = checkpoint_timestamp_ms / 1000;
585                            let pick_to_sign =
586                                checkpoint_timestamp_ms.saturating_sub(txn.created_timestamp_ms);
587                            (Some((amount_sats, timestamp_secs)), Some(pick_to_sign))
588                        }
589                        _ => (None, None),
590                    }
591                };
592                if limiter_inputs.is_none() {
593                    tracing::debug!(
594                        withdrawal_txn_id = %event.withdrawal_txn_id,
595                        "Skipping limiter apply: WithdrawalSigned for already-signed txn"
596                    );
597                    if let Some(metrics) = state.metrics() {
598                        metrics.guardian_limiter_anchor_events_skipped_total.inc();
599                    }
600                }
601                if let Some(d) = pick_to_sign_ms
602                    && let Some(metrics) = state.metrics()
603                {
604                    metrics
605                        .withdrawal_duration_seconds
606                        .with_label_values(&["pick_to_sign"])
607                        .observe(Duration::from_millis(d).as_secs_f64());
608                }
609                if let Some((amount_sats, timestamp_secs)) = limiter_inputs {
610                    if let Some(limiter) = state.local_limiter() {
611                        let seq = limiter.next_seq();
612                        let result = limiter.apply_consume(seq, timestamp_secs, amount_sats);
613                        if let Some(metrics) = state.metrics() {
614                            metrics.record_limiter_apply(&result);
615                        }
616                        match &result {
617                            Ok(()) => {
618                                if let Some(metrics) = state.metrics() {
619                                    metrics.guardian_limiter_anchor_events_total.inc();
620                                    metrics.record_limiter_state(
621                                        &limiter.snapshot(),
622                                        limiter.config(),
623                                    );
624                                }
625                                tracing::info!(
626                                    seq,
627                                    amount_sats,
628                                    timestamp_secs,
629                                    withdrawal_txn_id = %event.withdrawal_txn_id,
630                                    "Local limiter advanced from on-chain WithdrawalSigned",
631                                );
632                            }
633                            Err(e) => {
634                                if let Some(metrics) = state.metrics() {
635                                    metrics.guardian_limiter_drifted.set(1);
636                                }
637                                tracing::error!(
638                                    ?e,
639                                    seq,
640                                    withdrawal_txn_id = %event.withdrawal_txn_id,
641                                    "Local limiter apply_consume failed; node is now drifted from guardian"
642                                );
643                            }
644                        }
645                    } else if let Some(metrics) = state.metrics() {
646                        metrics
647                            .guardian_limiter_apply_total
648                            .with_label_values(&[
649                                crate::metrics::GUARDIAN_LIMITER_OUTCOME_NO_LIMITER,
650                            ])
651                            .inc();
652                    }
653                }
654            }
655            HashiEvent::WithdrawalInputsSigned(event) => {
656                tracing::info!(
657                    withdrawal_txn_id = %event.withdrawal_txn_id,
658                    signed_count = event.signed_count,
659                    num_inputs = event.num_inputs,
660                    "Withdrawal input signatures committed on-chain",
661                );
662                // The committed per-input signatures live on the object; refresh
663                // the local mirror so the next leader tick resumes from the
664                // up-to-date unsigned set.
665                match super::fetch_withdrawal_txn(client, event.withdrawal_txn_id).await {
666                    Ok(txn) => {
667                        state
668                            .state_mut()
669                            .hashi
670                            .withdrawal_queue
671                            .withdrawal_txns
672                            .insert(txn.id, txn);
673                    }
674                    Err(e) => {
675                        tracing::error!(
676                            withdrawal_txn_id = %event.withdrawal_txn_id,
677                            "Failed to refresh withdrawal signing batch: {e}",
678                        );
679                    }
680                }
681            }
682            HashiEvent::WithdrawalPresigsReassigned(event) => {
683                tracing::info!(
684                    withdrawal_txn_id = %event.withdrawal_txn_id,
685                    epoch = event.epoch,
686                    presig_start_index = event.presig_start_index,
687                    "Withdrawal presigs reassigned on-chain",
688                );
689                // Per-input presig indices changed in the batch; refresh from chain.
690                match super::fetch_withdrawal_txn(client, event.withdrawal_txn_id).await {
691                    Ok(txn) => {
692                        state
693                            .state_mut()
694                            .hashi
695                            .withdrawal_queue
696                            .withdrawal_txns
697                            .insert(txn.id, txn);
698                    }
699                    Err(e) => {
700                        tracing::error!(
701                            withdrawal_txn_id = %event.withdrawal_txn_id,
702                            "Failed to refresh withdrawal transaction: {e}",
703                        );
704                    }
705                }
706            }
707            HashiEvent::WithdrawalConfirmed(event) => {
708                tracing::info!(withdrawal_txn_id = %event.withdrawal_txn_id, "Withdrawal confirmed on-chain");
709                let (sign_to_confirm_ms, total_ms) = {
710                    let mut state = state.state_mut();
711
712                    // Promote the change UTXOs from pending to confirmed by
713                    // clearing `produced_by`. The UTXOs were already inserted at
714                    // commit time; input UTXOs are marked spent via UtxoSpent
715                    // and their records then stay in the mirror until a full
716                    // rescrape (reconnect or restart) — nothing prunes them.
717                    // The cleanup GC deliberately decides from a fresh chain
718                    // snapshot rather than these records, so the staleness is
719                    // cosmetic (spent_by stays set, excluding them from coin
720                    // selection).
721                    for change_utxo_id in &event.change_utxo_ids {
722                        if let Some(record) =
723                            state.hashi.utxo_pool.utxo_records.get_mut(change_utxo_id)
724                        {
725                            record.produced_by = None;
726                        }
727                    }
728
729                    let signed_at = state
730                        .withdrawal_signed_at_ms
731                        .remove(&event.withdrawal_txn_id);
732                    let pick_at = state
733                        .hashi
734                        .withdrawal_queue
735                        .withdrawal_txns
736                        .remove(&event.withdrawal_txn_id)
737                        .map(|txn| txn.created_timestamp_ms);
738                    (
739                        signed_at.map(|s| checkpoint_timestamp_ms.saturating_sub(s)),
740                        pick_at.map(|p| checkpoint_timestamp_ms.saturating_sub(p)),
741                    )
742                };
743                if let Some(metrics) = state.metrics() {
744                    if let Some(d) = sign_to_confirm_ms {
745                        metrics
746                            .withdrawal_duration_seconds
747                            .with_label_values(&["sign_to_confirm"])
748                            .observe(Duration::from_millis(d).as_secs_f64());
749                    }
750                    if let Some(d) = total_ms {
751                        metrics
752                            .withdrawal_duration_seconds
753                            .with_label_values(&["total"])
754                            .observe(Duration::from_millis(d).as_secs_f64());
755                    }
756                }
757            }
758            HashiEvent::UtxoSpent(utxo_spent_event) => {
759                let mut state = state.state_mut();
760                // Mark the local record as spent so the orphan scanner can discover it.
761                if let Some(record) = state
762                    .hashi
763                    .utxo_pool
764                    .utxo_records
765                    .get_mut(&utxo_spent_event.utxo_id)
766                {
767                    record.spent_epoch = Some(utxo_spent_event.spent_epoch);
768                }
769                state
770                    .hashi
771                    .utxo_pool
772                    .spent_utxos
773                    .insert(utxo_spent_event.utxo_id, utxo_spent_event.spent_epoch);
774            }
775            HashiEvent::ReconfigStarted(start_reconfig_event) => {
776                let epoch = start_reconfig_event.epoch;
777                // Fetch new committee
778                let committees_id = state.state().hashi().committees.committees_id();
779                //TODO maybe include info in the event
780                match super::scrape_committee(client.clone(), committees_id, epoch).await {
781                    Ok(committee) => {
782                        {
783                            let mut state = state.state_mut();
784                            state
785                                .hashi
786                                .committees
787                                .committees_mut()
788                                .insert(epoch, committee);
789                            state.hashi.committees.set_pending_epoch_change(Some(epoch));
790                        }
791                        state.notify(Notification::StartReconfig(epoch));
792                    }
793                    Err(e) => tracing::error!(
794                        epoch,
795                        "failed to scrape committee on ReconfigStarted; skipping (rescrape recovers): {e}"
796                    ),
797                }
798            }
799            HashiEvent::ReconfigEnded(end_reconfig_event) => {
800                let committees_id = state.state().hashi().committees.committees_id();
801                let scraped_committees =
802                    super::scrape_committees(client.clone(), committees_id).await;
803                let mut state = state.state_mut();
804                match scraped_committees {
805                    Ok((committees, committee_handoffs)) => {
806                        state
807                            .hashi
808                            .committees
809                            .set_committees(committees)
810                            .set_committee_handoffs(committee_handoffs);
811                    }
812                    Err(e) => tracing::error!(
813                        from_epoch = end_reconfig_event.from_epoch,
814                        "failed to scrape committee handoffs after ReconfigEnded: {e}"
815                    ),
816                }
817                state
818                    .hashi
819                    .committees
820                    .set_epoch(end_reconfig_event.epoch)
821                    .set_pending_epoch_change(None)
822                    .set_mpc_public_key(end_reconfig_event.mpc_public_key.clone());
823            }
824        }
825    }
826
827    let members_id = state.state().hashi().committees.members_id();
828    for validator in validator_updates {
829        match scrape_member_info(client.clone(), members_id, validator).await {
830            Ok(info) => {
831                state.state_mut().hashi.committees.update_validator(info);
832                state.notify(Notification::ValidatorInfoUpdated(validator));
833            }
834            Err(e) => tracing::error!("unable to query validator {validator}'s info: {e}"),
835        }
836    }
837}

839async fn refresh_treasury_cap_supply(
840    client: &mut Client,
841    state: &OnchainState,
842    coin_type: &TypeTag,
843) {
844    let treasury_cap_id = state
845        .state()
846        .hashi
847        .treasury
848        .treasury_caps
849        .get(coin_type)
850        .map(|tc| tc.id);
851
852    let Some(id) = treasury_cap_id else {
853        return;
854    };
855
856    match super::fetch_treasury_cap(client, id).await {
857        Ok(treasury_cap) => {
858            state
859                .state_mut()
860                .hashi
861                .treasury
862                .treasury_caps
863                .insert(coin_type.clone(), treasury_cap);
864        }
865        Err(e) => {
866            tracing::error!("failed to fetch treasury cap for {coin_type}: {e}");
867        }
868    }
869}

871/// Parse the proposal type from the TypeTag extracted from the event's phantom type parameter.
872fn parse_proposal_type_from_type_tag(type_tag: &TypeTag) -> ProposalType {
873    let TypeTag::Struct(struct_tag) = type_tag else {
874        return ProposalType::Unknown(format!("{:?}", type_tag));
875    };
876
877    match (struct_tag.module().as_str(), struct_tag.name().as_str()) {
878        ("update_config", "UpdateConfig") => ProposalType::UpdateConfig,
879        ("enable_version", "EnableVersion") => ProposalType::EnableVersion,
880        ("disable_version", "DisableVersion") => ProposalType::DisableVersion,
881        ("upgrade", "Upgrade") => ProposalType::Upgrade,
882        ("emergency_pause", "EmergencyPause") => ProposalType::EmergencyPause,
883        ("abort_reconfig", "AbortReconfig") => ProposalType::AbortReconfig,
884        ("update_guardian", "UpdateGuardian") => ProposalType::UpdateGuardian,
885        _ => ProposalType::Unknown(format!("{}::{}", struct_tag.module(), struct_tag.name())),
886    }
887}
