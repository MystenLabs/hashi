33    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
34    pub async fn validate_deposit_request(
35        &self,
36        deposit_request: &DepositRequest,
37    ) -> Result<(), UnapprovedDepositError> {
38        self.validate_deposit_request_on_sui(deposit_request)?;
39        self.validate_deposit_request_on_bitcoin(deposit_request)
40            .await?;
41        self.screen_deposit(deposit_request).await?;
42        Ok(())

45    /// Run AML/Sanctions checks for the deposit request.
46    /// If no screener client is configured, checks are skipped.
47    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
48    async fn screen_deposit(
49        &self,
50        deposit_request: &DepositRequest,
51    ) -> Result<(), UnapprovedDepositError> {
52        let Some(screener) = self.screener_client() else {
53            tracing::debug!("AML checks skipped: no screener configured");
54            return Ok(());
55        };
56
57        // bitcoin
58        let source_tx_hash = deposit_request.utxo.id.txid.to_string();
59        let bitcoin_chain_id = self.config.bitcoin_chain_id().to_string();
60
61        // sui
62        let destination_address = deposit_request.id.to_string();
63        let sui_chain_id = self.config.sui_chain_id().to_string();
64
65        let approved = screener
66            .approve_deposit(
67                &source_tx_hash,
68                &destination_address,
69                &bitcoin_chain_id,
70                &sui_chain_id,
71            )
72            .await
73            .map_err(|e| UnapprovedDepositError::AmlServiceError(anyhow!(e)))?;
74
75        if !approved {
76            return Err(UnapprovedDepositError::AmlRejected(anyhow!(
77                "AML checks failed for source tx {source_tx_hash}, destination {destination_address}, bitcoin chain {bitcoin_chain_id}, sui chain {sui_chain_id}"
78            )));
79        }
80
81        Ok(())
82    }

84    /// Validate that the deposit request exists on Sui
85    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
86    fn validate_deposit_request_on_sui(
87        &self,
88        deposit_request: &DepositRequest,
89    ) -> Result<(), UnapprovedDepositError> {
90        let state = self.onchain_state().state();
91        let deposit_queue = &state.hashi().deposit_queue;
92        match deposit_queue.requests().get(&deposit_request.id) {
93            None => {
94                return Err(UnapprovedDepositError::InvalidOnchainRequest(anyhow!(
95                    "Deposit request not found on Sui"
96                )));
97            }
98            Some(onchain_request) => {
99                // Approval state lives on the on-chain request but is not
100                // carried by `DepositRequest` constructed from the proto, so
101                // compare the immutable core fields only.
102                let core_matches = onchain_request.id == deposit_request.id
103                    && onchain_request.sender == deposit_request.sender
104                    && onchain_request.created_timestamp_ms == deposit_request.created_timestamp_ms
105                    && onchain_request.sui_tx_digest == deposit_request.sui_tx_digest
106                    && onchain_request.utxo == deposit_request.utxo;
107                if !core_matches {
108                    return Err(UnapprovedDepositError::InvalidOnchainRequest(anyhow!(
109                        "Deposit request fields do not match on-chain state"
110                    )));
111                }
112
113                // Refuse to sign a re-approval if the on-chain request is
114                // already approved by the current committee. The on-chain
115                // `approve_deposit` would reject it anyway, so we don't
116                // want to waste a signature exchange or a transaction.
117                let current_epoch = self.onchain_state().epoch();
118                if let Some(cert) = &onchain_request.approval_cert
119                    && cert.epoch == current_epoch
120                {
121                    return Err(UnapprovedDepositError::AlreadyApprovedThisEpoch);
122                }
123            }
124        }
125
126        let utxo_pool = &state.hashi().utxo_pool;
127        if utxo_pool
128            .utxo_records()
129            .contains_key(&deposit_request.utxo.id)
130            || utxo_pool
131                .spent_utxos()
132                .contains_key(&deposit_request.utxo.id)
133        {
134            return Err(UnapprovedDepositError::DuplicateOrSpentOnSui(anyhow!(
135                "UTXO {:?} is already active or spent",
136                deposit_request.utxo.id
137            )));
138        }
139
140        Ok(())
141    }

143    /// Validate that there is a txout on Bitcoin that matches the deposit request
144    #[tracing::instrument(
145        level = "debug",
146        skip_all,
147        fields(
148            deposit_id = %deposit_request.id,
149            bitcoin_txid = %deposit_request.utxo.id.txid,
150            vout = deposit_request.utxo.id.vout,
151        ),
152    )]
153    async fn validate_deposit_request_on_bitcoin(
154        &self,
155        deposit_request: &DepositRequest,
156    ) -> Result<(), UnapprovedDepositError> {
157        let outpoint = bitcoin::OutPoint {
158            txid: deposit_request.utxo.id.txid.into(),
159            vout: deposit_request.utxo.id.vout,
160        };
161        // Read the threshold live from on-chain state so governance
162        // updates take effect for new deposits without a restart.
163        let confirmation_threshold = self.onchain_state().bitcoin_confirmation_threshold();
164        let txout = self
165            .btc_monitor()
166            .confirm_deposit(outpoint, confirmation_threshold)
167            .await
168            .map_err(|e| match e {
169                DepositConfirmError::UtxoSpent { .. } => {
170                    self.metrics.deposits_rejected_utxo_spent.inc();
171                    UnapprovedDepositError::BitcoinUtxoSpent(anyhow!(e))
172                }
173                DepositConfirmError::Other(err) => {
174                    UnapprovedDepositError::BitcoinConfirmFailed(err)
175                }
176            })?;
177        if txout.value.to_sat() != deposit_request.utxo.amount {
178            return Err(UnapprovedDepositError::DepositDataMismatch(anyhow!(
179                "Bitcoin deposit amount mismatch: got {}, onchain is {}",
180                deposit_request.utxo.amount,
181                txout.value.to_sat(),
182            )));
183        }
184
185        self.validate_deposit_request_derivation_path(&txout.script_pubkey, deposit_request)
186            .await?;
187        Ok(())
188    }

190    async fn validate_deposit_request_derivation_path(
191        &self,
192        script_pubkey: &ScriptBuf,
193        deposit_request: &DepositRequest,
194    ) -> Result<(), UnapprovedDepositError> {
195        let deposit_address = hashi_bitcoin::BitcoinAddress::from_script(
196            script_pubkey,
197            self.config.bitcoin_network(),
198        )
199        .map_err(|e| {
200            UnapprovedDepositError::DepositDataMismatch(anyhow!(
201                "Failed to extract address from script_pubkey: {e}"
202            ))
203        })?;
204        let expected_address = self
205            .get_deposit_address(deposit_request.utxo.derivation_path.as_ref())
206            .map_err(UnapprovedDepositError::DepositDataMismatch)?;
207
208        if deposit_address != expected_address {
209            return Err(UnapprovedDepositError::DepositDataMismatch(anyhow!(
210                "Expected address {expected_address}, got address {deposit_address}",
211            )));
212        }
213
214        Ok(())
215    }

261    /// Hashi committee's child pubkey at `derivation_path`. `None` maps
262    /// to a zero-byte path (change outputs).
263    pub(crate) fn deposit_pubkey(
264        &self,
265        derivation_path: Option<&sui_sdk_types::Address>,
266    ) -> anyhow::Result<XOnlyPublicKey> {
267        // Prefer the local signing manager (available after DKG preparation).
268        // Fall back to the on-chain key, which is guaranteed present once the
269        // initial committee has formed and `end_reconfig` has been processed.
270        let verifying_key = self
271            .signing_verifying_key()
272            .map(Ok)
273            .unwrap_or_else(|| self.onchain_state().onchain_verifying_key_g())
274            .context("MPC public key not available yet")?;
275        let derivation_path = normalized_derivation_path(derivation_path).into_inner();
276        let derived = fastcrypto_tbls::threshold_schnorr::key_derivation::derive_verifying_key(
277            &verifying_key,
278            &derivation_path,
279        )
280        .context("derive child verifying key")?;
281        XOnlyPublicKey::from_slice(&derived.to_byte_array()).context("valid 32-byte x-only key")
282    }

290    fn sign_deposit_confirmation(
291        &self,
292        deposit_request: &DepositRequest,
293    ) -> anyhow::Result<MemberSignature> {
294        let epoch = self.onchain_state().epoch();
295        let validator_address = self
296            .config
297            .validator_address()
298            .map_err(|e| anyhow!("No validator address configured: {}", e))?;
299        let committee = self
300            .onchain_state()
301            .state()
302            .hashi()
303            .committees
304            .committees()
305            .get(&epoch)
306            .cloned()
307            .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?;
308        let private_key =
309            self.find_signing_key_for_committee(&committee, validator_address, epoch)?;
310        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();
311
312        let message = DepositConfirmationMessage {
313            request_id: deposit_request.id,
314            utxo: deposit_request.utxo.clone(),
315        };
316
317        let signature_bytes = private_key
318            .sign(epoch, validator_address, &message)
319            .signature()
320            .as_bytes()
321            .to_vec()
322            .into();
323
324        Ok(MemberSignature {
325            epoch: Some(epoch),
326            address: Some(validator_address.to_string()),
327            public_key: Some(public_key_bytes),
328            signature: Some(signature_bytes),
329        })
330    }

333/// `tr(NUMS, {multi_a(2, guardian_btc_pubkey, h), and_v(v:older(delay), pk(h))})`
334/// where `h = derive(mpc_pubkey, path)`: an immediate guardian+MPC 2-of-2 leaf,
335/// plus an MPC-only recovery leaf spendable after
336/// `HASHI_MPC_RECOVERY_DELAY_SECONDS`.
337/// `path = None` (change address) maps to a zero-byte path so deposit
338/// and withdrawal sides agree on the leaf key without a special case.
339///
340/// `mpc_key` is the raw MPC verifying key (`G`). The derivation is taken
341/// directly against this point — using only the x-only projection would
342/// silently force the parent to even-y and produce a different child key
343/// for ~half of all MPC vks, breaking the 2-of-2 leaf script.
344pub fn derive_deposit_address(
345    mpc_key: &ProjectivePoint,
346    guardian_btc_pubkey: &XOnlyPublicKey,
347    derivation_path: Option<&sui_sdk_types::Address>,
348    btc_network: bitcoin::Network,
349) -> anyhow::Result<hashi_bitcoin::BitcoinAddress> {
350    Ok(hashi_bitcoin::taproot_address(
351        guardian_btc_pubkey,
352        mpc_key,
353        &normalized_derivation_path(derivation_path),
354        btc_network,
355    ))
356}

358pub(crate) fn normalized_derivation_path(
359    derivation_path: Option<&sui_sdk_types::Address>,
360) -> sui_sdk_types::Address {
361    derivation_path
362        .copied()
363        .unwrap_or(sui_sdk_types::Address::ZERO)

372#[derive(Debug, Error)]
373pub enum UnapprovedDepositError {
374    #[error("Failed to confirm Bitcoin deposit: {0}")]
375    BitcoinConfirmFailed(#[source] anyhow::Error),
376
377    #[error("Screener service error: {0}")]
378    AmlServiceError(#[source] anyhow::Error),
379
380    #[error("Invalid on-chain deposit request: {0}")]
381    InvalidOnchainRequest(#[source] anyhow::Error),
382
383    #[error("UTXO is already active or spent on Sui: {0}")]
384    DuplicateOrSpentOnSui(#[source] anyhow::Error),
385
386    #[error("Deposit UTXO has already been spent on Bitcoin: {0}")]
387    BitcoinUtxoSpent(#[source] anyhow::Error),
388
389    #[error("Deposit data mismatch: {0}")]
390    DepositDataMismatch(#[source] anyhow::Error),
391
392    #[error("AML checks rejected deposit: {0}")]
393    AmlRejected(#[source] anyhow::Error),
394
395    #[error("Failed quorum: weight {weight} < {required_weight}")]
396    FailedQuorum { weight: u64, required_weight: u64 },
397
398    #[error("Failed to build deposit certificate: {0}")]
399    CertificateBuildFailed(#[source] anyhow::Error),
400
401    #[error("Failed to create Sui transaction executor: {0}")]
402    ExecutorInitFailed(#[source] anyhow::Error),
403
404    #[error("Failed to approve deposit on Sui: {0}")]
405    ApproveDepositFailed(#[source] anyhow::Error),
406
407    #[error("Failed to sign deposit confirmation: {0}")]
408    SignDepositFailed(#[source] anyhow::Error),
409
410    #[error("Deposit has already been approved by the current committee")]
411    AlreadyApprovedThisEpoch,
412
413    #[error("Deposit processing timed out after {0:?}")]
414    TimedOut(std::time::Duration),
415}
416
417impl UnapprovedDepositError {
418    pub(crate) fn kind(&self) -> UnapprovedDepositErrorKind {
419        match self {
420            Self::BitcoinConfirmFailed(_)
421            | Self::AmlServiceError(_)
422            | Self::FailedQuorum { .. }
423            | Self::CertificateBuildFailed(_)
424            | Self::ExecutorInitFailed(_)
425            | Self::ApproveDepositFailed(_)
426            | Self::SignDepositFailed(_)
427            | Self::AlreadyApprovedThisEpoch
428            | Self::TimedOut(_) => UnapprovedDepositErrorKind::RetryOnNextBlock,
429            Self::InvalidOnchainRequest(_)
430            | Self::DuplicateOrSpentOnSui(_)
431            | Self::BitcoinUtxoSpent(_)
432            | Self::DepositDataMismatch(_)
433            | Self::AmlRejected(_) => UnapprovedDepositErrorKind::NeverRetry,
434        }
435    }
436}

445#[derive(Debug, Error)]
446pub enum ApprovedDepositError {
447    #[error("Failed to create Sui transaction executor: {0}")]
448    ExecutorInitFailed(#[source] anyhow::Error),
449
450    #[error("Failed to confirm deposit on Sui: {0}")]
451    ConfirmDepositFailed(#[source] anyhow::Error),
452
453    #[error("Deposit processing timed out after {0:?}")]
454    TimedOut(std::time::Duration),
455}
456
457impl ApprovedDepositError {
458    pub(crate) fn kind(&self) -> ApprovedDepositErrorKind {
459        match self {
460            Self::ExecutorInitFailed(_) => ApprovedDepositErrorKind::ExecutorInitFailed,
461            Self::ConfirmDepositFailed(_) => ApprovedDepositErrorKind::ConfirmDepositFailed,
462            Self::TimedOut(_) => ApprovedDepositErrorKind::TimedOut,
463        }
464    }
465}

467// This backoff only applies after an eligible approved deposit confirmation
468// actually fails. Deposits whose time-delay has not elapsed are skipped before
469// entering retry tracking.
470impl RetryPolicy for ApprovedDepositErrorKind {
471    fn retry_base_delay_ms(self) -> u64 {
472        // All approved-deposit failures are transient in practice (epoch
473        // mismatch, transient RPC errors, slow network). A short base delay
474        // with exponential backoff handles the common case while throttling
475        // genuinely broken deposits.
476        3_000
477    }
478
479    fn max_delay_ms(self) -> u64 {
480        30 * 60 * 1000
481    }
482
483    fn max_retries(self) -> u32 {
484        u32::MAX
485    }
