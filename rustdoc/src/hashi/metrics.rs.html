944    pub fn update_onchain_state(&self, state: &crate::onchain::OnchainState) {
945        self.latest_checkpoint_height
946            .set(state.latest_checkpoint_height() as i64);
947        self.latest_checkpoint_timestamp_ms
948            .set(state.latest_checkpoint_timestamp_ms() as i64);
949        self.sui_epoch.set(state.latest_checkpoint_epoch() as i64);
950
951        let guard = state.state();
952        let hashi = guard.hashi();
953
954        self.epoch.set(hashi.committees.epoch() as i64);
955        self.reconfig_in_progress
956            .set(if hashi.committees.pending_epoch_change().is_some() {
957                1
958            } else {
959                0
960            });
961        self.paused.set(if hashi.config.paused() { 1 } else { 0 });
962        self.deposit_queue_size
963            .set(hashi.deposit_queue.requests().len() as i64);
964        let (requested, approved) = hashi
965            .withdrawal_queue
966            .requests()
967            .values()
968            .partition::<Vec<_>, _>(|r| r.status.is_requested());
969        // Three on-chain withdrawal-txn states, distinguished for operator
970        // visibility now that signing spans a multi-checkpoint window:
971        //   signed  = fully signed (2-of-2 witness assembled), broadcast-ready
972        //   signing = some inputs MPC-signed but not yet finalized (in progress)
973        //   pending = committed on-chain but no inputs signed yet
974        let mut signed = Vec::new();
975        let mut signing = Vec::new();
976        let mut pending = Vec::new();
977        for w in hashi.withdrawal_queue.withdrawal_txns().values() {
978            if w.is_fully_signed() {
979                signed.push(w);
980            } else if w.signing.signed_count() > 0 {
981                signing.push(w);
982            } else {
983                pending.push(w);
984            }
985        }
986        self.withdrawal_queue_size
987            .with_label_values(&["requested"])
988            .set(requested.len() as i64);
989        self.withdrawal_queue_value
990            .with_label_values(&["requested", "BTC"])
991            .set(requested.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
992        self.withdrawal_queue_size
993            .with_label_values(&["approved"])
994            .set(approved.len() as i64);
995        self.withdrawal_queue_value
996            .with_label_values(&["approved", "BTC"])
997            .set(approved.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
998        self.withdrawal_queue_size
999            .with_label_values(&["pending"])
1000            .set(pending.len() as i64);
1001        self.withdrawal_queue_value
1002            .with_label_values(&["pending", "BTC"])
1003            .set(
1004                pending
1005                    .iter()
1006                    .flat_map(|w| &w.withdrawal_outputs)
1007                    .map(|o| o.amount)
1008                    .sum::<u64>() as i64,
1009            );
1010        self.withdrawal_queue_size
1011            .with_label_values(&["signing"])
1012            .set(signing.len() as i64);
1013        self.withdrawal_queue_value
1014            .with_label_values(&["signing", "BTC"])
1015            .set(
1016                signing
1017                    .iter()
1018                    .flat_map(|w| &w.withdrawal_outputs)
1019                    .map(|o| o.amount)
1020                    .sum::<u64>() as i64,
1021            );
1022        self.withdrawal_queue_size
1023            .with_label_values(&["signed"])
1024            .set(signed.len() as i64);
1025        self.withdrawal_queue_value
1026            .with_label_values(&["signed", "BTC"])
1027            .set(
1028                signed
1029                    .iter()
1030                    .flat_map(|w| &w.withdrawal_outputs)
1031                    .map(|o| o.amount)
1032                    .sum::<u64>() as i64,
1033            );
1034        // Track three views of utxo_records:
1035        // - available:         all selectable UTXOs (locked_by = None), whether
1036        //                      confirmed or not; this is the coin-selection pool
1037        // - unconfirmed_change: subset of available whose producing withdrawal
1038        //                      has not yet confirmed on Bitcoin (produced_by =
1039        //                      Some); useful for gauging mempool chain depth
1040        // - locked:            committed to a pending withdrawal, awaiting
1041        //                      Bitcoin confirmation (locked_by = Some)
1042        let mut available_count = 0i64;
1043        let mut unconfirmed_change_count = 0i64;
1044        let mut locked_count = 0i64;
1045        let mut available_value = 0u64;
1046        let mut unconfirmed_change_value = 0u64;
1047        let mut locked_value = 0u64;
1048        for record in hashi.utxo_pool.utxo_records().values() {
1049            if record.locked_by.is_some() {
1050                locked_count += 1;
1051                locked_value += record.utxo.amount;
1052            } else {
1053                available_count += 1;
1054                available_value += record.utxo.amount;
1055                if record.produced_by.is_some() {
1056                    unconfirmed_change_count += 1;
1057                    unconfirmed_change_value += record.utxo.amount;
1058                }
1059            }
1060        }
1061        self.utxo_pool_size
1062            .with_label_values(&["available"])
1063            .set(available_count);
1064        self.utxo_pool_size
1065            .with_label_values(&["unconfirmed_change"])
1066            .set(unconfirmed_change_count);
1067        self.utxo_pool_size
1068            .with_label_values(&["locked"])
1069            .set(locked_count);
1070        self.utxo_pool_value
1071            .with_label_values(&["available"])
1072            .set(available_value as i64);
1073        self.utxo_pool_value
1074            .with_label_values(&["unconfirmed_change"])
1075            .set(unconfirmed_change_value as i64);
1076        self.utxo_pool_value
1077            .with_label_values(&["locked"])
1078            .set(locked_value as i64);
1079        {
1080            use crate::onchain::types::ProposalType;
1081            let mut counts = std::collections::HashMap::<&str, i64>::new();
1082            for proposal in hashi.proposals.active().values() {
1083                *counts.entry(proposal.proposal_type.as_str()).or_default() += 1;
1084            }
1085            for label in ProposalType::all_labels() {
1086                self.proposals
1087                    .with_label_values(&[label])
1088                    .set(*counts.get(label).unwrap_or(&0));
1089            }
1090        }
1091        self.num_consumed_presigs
1092            .set(hashi.num_consumed_presigs as i64);
1093        for (type_tag, cap) in &hashi.treasury.treasury_caps {
1094            if let sui_sdk_types::TypeTag::Struct(struct_tag) = type_tag {
1095                self.treasury_supply
1096                    .with_label_values(&[struct_tag.name().as_str()])
1097                    .set(cap.supply as i64);
1098            }
1099        }
1100
1101        self.package_version_enabled.reset();
1102        for version in &hashi.config.enabled_versions {
1103            let version_str = version.to_string();
1104            let package_id_str = guard
1105                .package_versions()
1106                .get(version)
1107                .map(|addr| addr.to_string())
1108                .unwrap_or_else(|| "unknown".to_string());
1109            self.package_version_enabled
1110                .with_label_values(&[&version_str, &package_id_str])
1111                .set(1);
1112        }
1113    }

1147fn limiter_outcome_label(
1148    result: &Result<(), crate::guardian_limiter::LocalLimiterError>,
1149) -> &'static str {
1150    use crate::guardian_limiter::LocalLimiterError;
1151    match result {
1152        Ok(()) => GUARDIAN_LIMITER_OUTCOME_SUCCESS,
1153        Err(LocalLimiterError::SeqMismatch { .. }) => GUARDIAN_LIMITER_OUTCOME_SEQ_MISMATCH,
1154        Err(LocalLimiterError::StaleTimestamp { .. }) => GUARDIAN_LIMITER_OUTCOME_STALE_TIMESTAMP,
1155        Err(LocalLimiterError::InsufficientCapacity { .. }) => {
1156            GUARDIAN_LIMITER_OUTCOME_INSUFFICIENT_CAPACITY
1157        }
1158    }
1159}

1169    package_id: &str,
1170    hashi_object_id: &str,
1171) -> Box<dyn prometheus::core::Collector> {
1172    let opts = prometheus::opts!("uptime", "uptime of the node service in seconds")
1173        .variable_label("version")
1174        .variable_label("sui_chain_id")
1175        .variable_label("bitcoin_chain_id")
1176        .variable_label("package_id")
1177        .variable_label("hashi_object_id");
1178
1179    let start_time = std::time::Instant::now();
1180    let uptime = move || start_time.elapsed().as_secs();
1181    let metric = prometheus_closure_metric::ClosureMetric::new(
1182        opts,
1183        prometheus_closure_metric::ValueType::Counter,
1184        uptime,
1185        &[
1186            version,
1187            sui_chain_id,
1188            bitcoin_chain_id,
1189            package_id,
1190            hashi_object_id,
1191        ],
1192    )
1193    .unwrap();

1195    Box::new(metric)

1200// Creates a new http server that has as a sole purpose to expose
1201// an endpoint that prometheus agent can use to poll for the metrics.
1202pub fn start_prometheus_server(
1203    addr: std::net::SocketAddr,
1204    registry: prometheus::Registry,
1205) -> sui_http::ServerHandle {
1206    let router = axum::Router::new()
1207        .route(METRICS_ROUTE, axum::routing::get(metrics))
1208        .with_state(registry);
1209
1210    sui_http::Builder::new().serve(addr, router).unwrap()
1211}

1213async fn metrics(
1214    axum::extract::State(registry): axum::extract::State<prometheus::Registry>,
1215) -> (http::StatusCode, String) {
1216    let metrics_families = registry.gather();
1217    match prometheus::TextEncoder.encode_to_string(&metrics_families) {
1218        Ok(metrics) => (http::StatusCode::OK, metrics),
1219        Err(error) => (
1220            http::StatusCode::INTERNAL_SERVER_ERROR,
1221            format!("unable to encode metrics: {error}"),
1222        ),
1223    }
1224}
