5//!
6//! This module handles the mechanics of building backup manifests, encrypting
7//! files into age-wrapped tar archives, and extracting them. CLI-specific
8//! orchestration (config loading, recipient resolution, user output) lives in
9//! [`crate::cli::commands::backup`].

33pub const BACKUP_MANIFEST_FILE_NAME: &str = "hashi-config-backup-manifest.toml";
34pub const DB_SNAPSHOT_TAR_PREFIX: &str = "hashi-db-snapshot";

36/// Open `path` for writing with mode `0o600`, failing if anything already
37/// exists there. The `AlreadyExists` case is mapped to a clear "refusing to
38/// overwrite" error so callers don't have to repeat the same pattern.
39fn create_file_strict(path: &Path) -> Result<File> {
40    OpenOptions::new()
41        .write(true)
42        .create_new(true)
43        .mode(0o600)
44        .open(path)
45        .map_err(|e| match e.kind() {
46            ErrorKind::AlreadyExists => {
47                anyhow::anyhow!("Refusing to overwrite existing file: {}", path.display())
48            }
49            _ => anyhow::Error::from(e).context(format!("Failed to create {}", path.display())),
50        })
51}
52
53/// Create directory `path` non-recursively, failing if anything already
54/// exists at that exact location. Caller is responsible for ensuring the
55/// parent exists.
56fn create_dir_strict(path: &Path) -> Result<()> {
57    fs::DirBuilder::new()
58        .recursive(false)
59        .create(path)
60        .map_err(|e| match e.kind() {
61            ErrorKind::AlreadyExists => {
62                anyhow::anyhow!(
63                    "Refusing to overwrite existing directory: {}",
64                    path.display()
65                )
66            }
67            _ => anyhow::Error::from(e)
68                .context(format!("Failed to create directory {}", path.display())),
69        })
70}

90pub fn build_backup_manifest(files: &[PathBuf], db_original_path: &Path) -> Result<BackupManifest> {
91    let db_archive_entries = backup_keyspace_archive_entries(Path::new(DB_SNAPSHOT_TAR_PREFIX));
92    // Reserve the db snapshot directory prefix and every keyspace archive
93    // basename so a user file with one of those names gets disambiguated
94    // instead of silently colliding with a backed-up database entry.
95    let mut archive_names = HashSet::new();
96    archive_names.insert(DB_SNAPSHOT_TAR_PREFIX.to_string());
97    for entry in &db_archive_entries {
98        let name = entry.file_name().and_then(|n| n.to_str()).ok_or_else(|| {
99            anyhow::anyhow!(
100                "Database backup entry does not have a valid file name: {}",
101                entry.display()
102            )
103        })?;
104        archive_names.insert(name.to_string());
105    }
106    let mut manifest_paths = Vec::new();
107
108    for file in files {
109        let base_name = file
110            .file_name()
111            .ok_or_else(|| {
112                anyhow::anyhow!("Backup input does not have a file name: {}", file.display())
113            })?
114            .to_string_lossy();
115
116        let archive_name = if archive_names.contains(base_name.as_ref()) {
117            let stem = Path::new(base_name.as_ref())
118                .file_stem()
119                .unwrap_or_default()
120                .to_string_lossy();
121            let ext = Path::new(base_name.as_ref())
122                .extension()
123                .map(|e| format!(".{}", e.to_string_lossy()))
124                .unwrap_or_default();
125
126            let mut suffix = 2u32;
127            loop {
128                let candidate = format!("{stem}-{suffix}{ext}");
129                if archive_names.insert(candidate.clone()) {
130                    info!(
131                        original = %file.display(),
132                        renamed = %candidate,
133                        "Archive name collision for {base_name}",
134                    );
135                    break PathBuf::from(candidate);
136                }
137                suffix += 1;
138            }
139        } else {
140            archive_names.insert(base_name.to_string());
141            PathBuf::from(base_name.as_ref())
142        };
143
144        manifest_paths.push(BackupManifestEntry {
145            archive_name,
146            original_path: file.clone(),
147        });
148    }
149
150    Ok(BackupManifest {
151        paths: manifest_paths,
152        db: DbManifestEntry {
153            original_path: db_original_path.to_path_buf(),
154            archive_entries: db_archive_entries,
155        },
156    })
157}

159pub fn encrypt_files_to_age_archive(
160    manifest: &BackupManifest,
161    db: &Database,
162    recipient: &dyn age::Recipient,
163    output_path: &Path,
164) -> Result<()> {
165    let output = create_file_strict(output_path)?;
166    let encryptor = Encryptor::with_recipients(std::iter::once(recipient))?;
167    let mut encrypted = encryptor.wrap_output(output)?;
168    {
169        let mut archive = tar::Builder::new(&mut encrypted);
170        append_backup_manifest(&mut archive, manifest)?;
171
172        for entry in &manifest.paths {
173            archive.append_path_with_name(&entry.original_path, &entry.archive_name)?;
174            info!(
175                original = %entry.original_path.display(),
176                archive_name = %entry.archive_name.display(),
177                "Added file to backup archive",
178            );
179        }
180
181        append_db_backup_to_tar(db, &mut archive, &manifest.db.archive_entries)?;
182        info!("Added database backup to backup archive");
183
184        archive.finish()?;
185    }
186    encrypted.finish()?;
187
188    Ok(())
189}

191pub fn encrypted_backup_file_name() -> PathBuf {
192    // ISO 8601 basic format in UTC, e.g. 20260409T230419Z. Compact, sorts
193    // lexicographically, and contains no characters that need escaping on any
194    // common filesystem.
195    let timestamp = jiff::Timestamp::now()
196        .to_zoned(jiff::tz::TimeZone::UTC)
197        .strftime("%Y%m%dT%H%M%SZ")
198        .to_string();
199    PathBuf::from(format!("hashi-config-backup-{timestamp}.tar.age"))
200}

202fn append_backup_manifest<W: std::io::Write>(
203    archive: &mut tar::Builder<W>,
204    manifest: &BackupManifest,
205) -> Result<()> {
206    let manifest_bytes = toml::to_string_pretty(manifest)?.into_bytes();
207    let mut header = tar::Header::new_gnu();
208    header.set_size(manifest_bytes.len() as u64);
209    header.set_mode(0o644);
210    header.set_cksum();
211    archive.append_data(
212        &mut header,
213        BACKUP_MANIFEST_FILE_NAME,
214        manifest_bytes.as_slice(),
215    )?;
216    Ok(())
217}

225fn backup_keyspace_archive_entries(tar_prefix: &Path) -> Vec<PathBuf> {
226    Database::backup_keyspace_names()
227        .into_iter()
228        .map(|name| tar_prefix.join(format!("{name}.bin")))
229        .collect()
230}

232fn backup_keyspace_name_from_file_name(file_name: &str) -> Option<&'static str> {
233    let name = file_name.strip_suffix(".bin")?;
234    Database::backup_keyspace_names()
235        .into_iter()
236        .find(|keyspace_name| *keyspace_name == name)
237}

239fn append_db_backup_to_tar<W: Write>(
240    db: &Database,
241    archive: &mut tar::Builder<W>,
242    archive_entries: &[PathBuf],
243) -> Result<()> {
244    let snapshot = db.snapshot();
245    let keyspaces = db.backup_keyspaces();
246
247    for archive_path in archive_entries {
248        validate_db_archive_path(archive_path)?;
249        let file_name = archive_path
250            .file_name()
251            .and_then(|name| name.to_str())
252            .ok_or_else(|| anyhow::anyhow!("Database entry has no valid file name"))?;
253        let name = backup_keyspace_name_from_file_name(file_name).ok_or_else(|| {
254            anyhow::anyhow!(
255                "Database entry must be a known keyspace .bin file: {}",
256                archive_path.display()
257            )
258        })?;
259        let source_ks = keyspaces
260            .iter()
261            .find_map(|(keyspace_name, keyspace)| (*keyspace_name == name).then_some(*keyspace))
262            .expect("backup_keyspace_name_from_file_name matched a backup keyspace");
263        let mut records = Vec::new();
264        for guard in snapshot.iter(source_ks) {
265            let (key, value) = guard.into_inner()?;
266            records.push(DbBackupRecord {
267                key: key.to_vec(),
268                value: value.to_vec(),
269            });
270        }
271        let bytes = bcs::to_bytes(&records)
272            .with_context(|| format!("failed to serialize backup keyspace {name}"))?;
273        let mut header = tar::Header::new_gnu();
274        header.set_entry_type(tar::EntryType::Regular);
275        header.set_size(bytes.len() as u64);
276        header.set_mode(0o600);
277        header.set_cksum();
278        archive
279            .append_data(&mut header, archive_path, bytes.as_slice())
280            .with_context(|| {
281                format!(
282                    "failed to append database backup entry {}",
283                    archive_path.display()
284                )
285            })?;
286    }
287
288    Ok(())
289}

291/// Determine the directory name to extract a backup tarball into.
292///
293/// Strips the `.tar.age` or `.age` suffix from the tarball's file name, so
294/// `hashi-config-backup-20260409T230419Z.tar.age` becomes
295/// `hashi-config-backup-20260409T230419Z`. An input without one of those
296/// suffixes is rejected rather than silently used verbatim, to avoid
297/// surprising extraction directory names when users point at the wrong file.
298pub fn extract_dir_name(backup_tarball: &Path) -> Result<PathBuf> {
299    let file_name = backup_tarball
300        .file_name()
301        .and_then(|name| name.to_str())
302        .ok_or_else(|| {
303            anyhow::anyhow!(
304                "Backup tarball path has no file name: {}",
305                backup_tarball.display()
306            )
307        })?;
308
309    let stem = file_name
310        .strip_suffix(".tar.age")
311        .or_else(|| file_name.strip_suffix(".age"))
312        .ok_or_else(|| {
313            anyhow::anyhow!(
314                "Backup tarball must have a .tar.age or .age suffix: {}",
315                backup_tarball.display()
316            )
317        })?;
318
319    // Require the stem to be exactly one plain directory component when
320    // joined under the user's output dir. Catches both the empty case (zero
321    // components) and traversal attempts like `../../tmp/pwn.tar.age` (a
322    // `ParentDir` component, not `Normal`).
323    let stem_path = Path::new(stem);
324    let mut components = stem_path.components();
325    let only = components.next();
326    let extra = components.next();
327    if extra.is_some() || !matches!(only, Some(Component::Normal(_))) {
328        anyhow::bail!(
329            "Backup tarball file name must be a single path component without separators or `..`: {}",
330            backup_tarball.display()
331        );
332    }
333
334    Ok(PathBuf::from(stem))
335}

337pub fn read_backup_manifest<R: Read>(
338    mut entry: tar::Entry<'_, R>,
339) -> Result<(BackupManifest, String)> {
340    let path = entry.path()?.into_owned();
341    let file_name = path.file_name().ok_or_else(|| {
342        anyhow::anyhow!(
343            "First tar entry does not have a file name: {}",
344            path.display()
345        )
346    })?;
347    if file_name != BACKUP_MANIFEST_FILE_NAME {
348        anyhow::bail!(
349            "Expected backup manifest {} as the first tar entry, found {}",
350            BACKUP_MANIFEST_FILE_NAME,
351            path.display()
352        );
353    }
354
355    let mut manifest_toml = String::new();
356    entry.read_to_string(&mut manifest_toml)?;
357    let manifest: BackupManifest = toml::from_str(&manifest_toml)?;
358
359    Ok((manifest, manifest_toml))
360}

362pub fn write_manifest_to_extract_dir(extract_dir: &Path, manifest_toml: &str) -> Result<()> {
363    let manifest_path = extract_dir.join(BACKUP_MANIFEST_FILE_NAME);
364    let mut file = create_file_strict(&manifest_path)?;
365    io::Write::write_all(&mut file, manifest_toml.as_bytes())
366        .with_context(|| format!("Failed to write manifest to {}", manifest_path.display()))?;
367    info!(path = %manifest_path.display(), "Restored manifest");
368    Ok(())
369}

371pub fn restore_backup_entries<R: Read>(
372    entries: tar::Entries<'_, R>,
373    output_dir: &Path,
374    manifest: &BackupManifest,
375) -> Result<()> {
376    let db_prefix = Path::new(DB_SNAPSHOT_TAR_PREFIX);
377    let expected_files: HashSet<PathBuf> = manifest
378        .paths
379        .iter()
380        .map(|entry| entry.archive_name.clone())
381        .collect();
382    let expected_db_entries: HashSet<PathBuf> =
383        manifest.db.archive_entries.iter().cloned().collect();
384    let mut restored_db_entries = HashSet::new();
385    let mut restored_count: usize = 0;
386    let db_path = output_dir.join(DB_SNAPSHOT_TAR_PREFIX);
387    let db = fjall::Database::builder(&db_path).open().map_err(|e| {
388        anyhow::Error::new(e).context(format!(
389            "failed to open destination database at {}",
390            db_path.display()
391        ))
392    })?;
393
394    for entry in entries {
395        let mut entry = entry?;
396        let archive_path = entry.path()?.into_owned();
397
398        if archive_path.starts_with(db_prefix) {
399            validate_db_archive_path(&archive_path)?;
400            if !expected_db_entries.contains(&archive_path) {
401                anyhow::bail!(
402                    "Backup archive contains unexpected database entry: {}",
403                    archive_path.display()
404                );
405            }
406            if !restored_db_entries.insert(archive_path.clone()) {
407                anyhow::bail!(
408                    "Backup archive contains duplicate database entry: {}",
409                    archive_path.display()
410                );
411            }
412            restore_db_entry(&mut entry, &archive_path, &db)?;
413        } else {
414            restore_config_entry(&mut entry, &archive_path, output_dir, &expected_files)?;
415            restored_count += 1;
416        }
417    }
418
419    if restored_count != expected_files.len() {
420        anyhow::bail!(
421            "Backup archive is missing file entries: expected {}, restored {}",
422            expected_files.len(),
423            restored_count
424        );
425    }
426
427    if restored_db_entries != expected_db_entries {
428        let mut missing_db_entries: Vec<_> = expected_db_entries
429            .difference(&restored_db_entries)
430            .cloned()
431            .collect();
432        missing_db_entries.sort();
433        anyhow::bail!(
434            "Backup archive is missing database entries: {}",
435            missing_db_entries
436                .into_iter()
437                .map(|path| path.display().to_string())
438                .collect::<Vec<_>>()
439                .join(", ")
440        );
441    }
442
443    db.persist(fjall::PersistMode::SyncAll)?;
444
445    Ok(())
446}

448fn validate_db_archive_path(archive_path: &Path) -> Result<()> {
449    let mut components = archive_path.components();
450    if !matches!(components.next(), Some(Component::Normal(prefix)) if prefix == DB_SNAPSHOT_TAR_PREFIX)
451    {
452        anyhow::bail!(
453            "Database entry must live under {}: {}",
454            DB_SNAPSHOT_TAR_PREFIX,
455            archive_path.display()
456        );
457    }
458
459    let file_name = match (components.next(), components.next()) {
460        (Some(Component::Normal(file_name)), None) => file_name,
461        _ => {
462            anyhow::bail!(
463                "Database entry must be a single keyspace file under {}: {}",
464                DB_SNAPSHOT_TAR_PREFIX,
465                archive_path.display()
466            );
467        }
468    };
469
470    file_name.to_str().ok_or_else(|| {
471        anyhow::anyhow!(
472            "Database entry file name is not valid UTF-8: {}",
473            archive_path.display()
474        )
475    })?;
476
477    Ok(())
478}

480fn restore_db_entry<R: Read>(
481    entry: &mut tar::Entry<'_, R>,
482    archive_path: &Path,
483    db: &fjall::Database,
484) -> Result<()> {
485    let entry_type = entry.header().entry_type();
486    if entry_type != tar::EntryType::Regular {
487        anyhow::bail!(
488            "Database entry {} has unexpected type {entry_type:?}; only regular files are supported",
489            archive_path.display(),
490        );
491    }
492
493    let file_name = archive_path
494        .file_name()
495        .and_then(|name| name.to_str())
496        .ok_or_else(|| anyhow::anyhow!("Database entry has no valid file name"))?;
497    let keyspace_name = backup_keyspace_name_from_file_name(file_name).ok_or_else(|| {
498        anyhow::anyhow!(
499            "Database entry must be a known keyspace .bin file: {}",
500            archive_path.display()
501        )
502    })?;
503    restore_db_backup_keyspace(db, keyspace_name, entry).with_context(|| {
504        format!(
505            "Failed to restore database entry {}",
506            archive_path.display()
507        )
508    })?;
509    Ok(())
510}

512fn restore_db_backup_keyspace<R: Read>(
513    db: &fjall::Database,
514    keyspace_name: &str,
515    mut reader: R,
516) -> Result<()> {
517    let dest_ks = db.keyspace(keyspace_name, KeyspaceCreateOptions::default)?;
518    let mut ingestion = dest_ks.start_ingestion()?;
519
520    let mut bytes = Vec::new();
521    reader
522        .read_to_end(&mut bytes)
523        .with_context(|| format!("failed to read backup keyspace {keyspace_name}"))?;
524    let records: Vec<DbBackupRecord> = bcs::from_bytes(&bytes)
525        .with_context(|| format!("failed to deserialize backup keyspace {keyspace_name}"))?;
526    for record in records {
527        ingestion.write(&record.key, &record.value)?;
528    }
529
530    ingestion.finish()?;
531    Ok(())
532}

534/// Extract a single config-file tar entry into `output_dir`. Config entries
535/// must be regular files at the tar root and must appear in the manifest;
536/// anything else is rejected so a tampered archive can't sneak unexpected
537/// files past the restore.
538fn restore_config_entry<R: Read>(
539    entry: &mut tar::Entry<'_, R>,
540    archive_path: &Path,
541    output_dir: &Path,
542    expected_files: &HashSet<PathBuf>,
543) -> Result<()> {
544    let archive_name = PathBuf::from(archive_path.file_name().ok_or_else(|| {
545        anyhow::anyhow!(
546            "Backup entry does not have a file name: {}",
547            archive_path.display()
548        )
549    })?);
550
551    let entry_type = entry.header().entry_type();
552    if entry_type != tar::EntryType::Regular {
553        anyhow::bail!(
554            "Backup entry {} has unexpected type {entry_type:?}; only regular files are supported",
555            archive_name.display(),
556        );
557    }
558
559    if archive_path != archive_name {
560        anyhow::bail!(
561            "Backup entry must be at the tar root: {}",
562            archive_path.display()
563        );
564    }
565
566    if !expected_files.contains(&archive_name) {
567        anyhow::bail!(
568            "Backup archive contains unexpected file: {}",
569            archive_name.display()
570        );
571    }
572
573    let output_path = output_dir.join(&archive_name);
574    let mut output_file = create_file_strict(&output_path)?;
575    io::copy(entry, &mut output_file).with_context(|| {
576        format!(
577            "Failed to write restored file contents to {}",
578            output_path.display()
579        )
580    })?;
581    info!(
582        archive_name = %archive_name.display(),
583        output = %output_path.display(),
584        "Restored file",
585    );
586    Ok(())
587}

589/// Copy each restored config file from the extract directory to its original
590/// path as recorded in the manifest. Refuses to overwrite anything.
591///
592/// `extract_dir` is where the tarball was unpacked; this function joins it
593/// with each entry's `archive_name` to find the source. The reconstructed
594/// database directory is restored separately by
595/// [`copy_db_snapshot_to_original_path`].
596pub fn copy_restored_files_to_original_paths(
597    extract_dir: &Path,
598    manifest: &BackupManifest,
599) -> Result<()> {
600    for file in manifest.paths.iter() {
601        let restored_path = extract_dir.join(&file.archive_name);
602
603        if let Some(parent) = file.original_path.parent() {
604            fs::create_dir_all(parent).with_context(|| {
605                format!("Failed to create parent directory {}", parent.display())
606            })?;
607        }
608
609        let mut source = File::open(&restored_path)
610            .with_context(|| format!("Failed to open {}", restored_path.display()))?;
611        // Custom AlreadyExists message: "original path" hints that the
612        // collision is with a path the manifest pointed at, not a freshly
613        // chosen output location.
614        let mut dest = OpenOptions::new()
615            .write(true)
616            .create_new(true)
617            .mode(0o600)
618            .open(&file.original_path)
619            .map_err(|e| match e.kind() {
620                ErrorKind::AlreadyExists => anyhow::anyhow!(
621                    "Refusing to overwrite existing original path: {}",
622                    file.original_path.display()
623                ),
624                _ => anyhow::Error::from(e)
625                    .context(format!("Failed to create {}", file.original_path.display())),
626            })?;
627        io::copy(&mut source, &mut dest).with_context(|| {
628            format!(
629                "Failed to copy {} to {}",
630                restored_path.display(),
631                file.original_path.display()
632            )
633        })?;
634        info!(
635            from = %restored_path.display(),
636            to = %file.original_path.display(),
637            "Copied restored file to original path",
638        );
639    }
640
641    Ok(())
642}

644/// Copy the extracted database snapshot directory to its original path.
645///
646/// The reconstructed fjall DB directory at
647/// `extract_dir / DB_SNAPSHOT_TAR_PREFIX` is recursively copied into a
648/// sibling staging directory of the manifest's DB original path and then
649/// renamed into place on success. The rename is atomic on a single
650/// filesystem, so a crash mid-copy leaves the staging directory behind
651/// without ever creating a half-populated DB at `dest`. The staging
652/// directory is created with `tempdir_in(dest.parent())`, so it lives on
653/// the same filesystem as `dest`; that same-filesystem placement is
654/// required because `rename` does not work across mount points.
655///
656/// Fails if `dest` already exists.
657pub fn copy_db_snapshot_to_original_path(
658    extract_dir: &Path,
659    manifest: &BackupManifest,
660) -> Result<()> {
661    let source = extract_dir.join(DB_SNAPSHOT_TAR_PREFIX);
662    let dest = &manifest.db.original_path;
663
664    // Pre-flight: refuse early if the destination already exists. This is
665    // checked again implicitly by the final rename, but failing here gives a
666    // clear error before doing all the copy work.
667    if dest
668        .try_exists()
669        .with_context(|| format!("Failed to stat database destination {}", dest.display()))?
670    {
671        anyhow::bail!(
672            "Refusing to overwrite existing database directory: {}",
673            dest.display()
674        );
675    }
676
677    let parent = dest.parent().ok_or_else(|| {
678        anyhow::anyhow!(
679            "Database destination has no parent directory: {}",
680            dest.display()
681        )
682    })?;
683    fs::create_dir_all(parent)
684        .with_context(|| format!("Failed to create parent directory {}", parent.display()))?;
685
686    // Stage into a sibling temp directory so the entire copy is invisible
687    // until the final rename. Using a sibling guarantees we're on the same
688    // filesystem, which is required for `rename` to be atomic.
689    let staging = tempfile::Builder::new()
690        .prefix(".hashi-db-restore-")
691        .tempdir_in(parent)
692        .with_context(|| {
693            format!(
694                "Failed to create db staging directory in {}",
695                parent.display()
696            )
697        })?;
698
699    copy_dir_recursive_strict(&source, staging.path()).with_context(|| {
700        format!(
701            "Failed to copy database snapshot from {} to staging directory {}",
702            source.display(),
703            staging.path().display()
704        )
705    })?;
706
707    // Convert the TempDir into an owned path before rename so we can do
708    // explicit best-effort cleanup ourselves if the rename fails.
709    let staging_path = staging.keep();
710    // `staging_path` was created as a sibling of `dest`, so this rename stays
711    // on one filesystem rather than crossing mounts; `fs::rename` does not
712    // work across mount points.
713    fs::rename(&staging_path, dest).map_err(|e| {
714        // Best-effort cleanup of the staging directory if the rename failed,
715        // since `into_path` consumed the auto-deleting guard.
716        let _ = fs::remove_dir_all(&staging_path);
717        match e.kind() {
718            ErrorKind::AlreadyExists => anyhow::anyhow!(
719                "Refusing to overwrite existing database directory: {}",
720                dest.display()
721            ),
722            _ => anyhow::Error::from(e).context(format!(
723                "Failed to move staged database into place at {}",
724                dest.display()
725            )),
726        }
727    })?;
728
729    info!(
730        from = %source.display(),
731        to = %dest.display(),
732        "Copied database snapshot to original path",
733    );
734
735    Ok(())
736}
737
738/// Recursively copy a directory tree from `src` to `dest` without ever
739/// overwriting existing files.
740///
741/// - The root `dest` is assumed to have already been created by the caller.
742/// - Subdirectories below the root use `create_dir` (non-recursive), so any
743///   collision surfaces as an error rather than merging into existing state.
744/// - Files are opened with `create_new(true)` and mode `0o600`, matching the
745///   protection `copy_restored_files_to_original_paths` applies to config
746///   files. fjall on-disk files are not sensitive individually, but we keep
747///   the mode consistent so the restored DB never advertises looser
748///   permissions than the backup did.
749/// - Symlinks are rejected outright.
750fn copy_dir_recursive_strict(src: &Path, dest: &Path) -> Result<()> {
751    for entry in walkdir::WalkDir::new(src).min_depth(1) {
752        let entry = entry?;
753
754        if entry.file_type().is_symlink() {
755            anyhow::bail!(
756                "Refusing to restore symlink from database snapshot: {}",
757                entry.path().display()
758            );
759        }
760
761        let relative = entry.path().strip_prefix(src).expect("walkdir under src");
762        let target = dest.join(relative);
763
764        if entry.file_type().is_dir() {
765            create_dir_strict(&target)?;
766        } else {
767            let mut source_file = File::open(entry.path())
768                .with_context(|| format!("Failed to open {}", entry.path().display()))?;
769            let mut dest_file = create_file_strict(&target)?;
770            io::copy(&mut source_file, &mut dest_file).with_context(|| {
771                format!(
772                    "Failed to copy {} to {}",
773                    entry.path().display(),
774                    target.display()
775                )
776            })?;
777        }
778    }
779    Ok(())
780}

797    fn append_regular_file<W: std::io::Write>(
798        archive: &mut tar::Builder<W>,
799        path: &str,
800        contents: &[u8],
801    ) {
802        let mut header = tar::Header::new_gnu();
803        header.set_entry_type(tar::EntryType::Regular);
804        header.set_mode(0o600);
805        header.set_size(contents.len() as u64);
806        header.set_cksum();
807        archive.append_data(&mut header, path, contents).unwrap();
808    }

815            for (path, contents) in db_files {
816                append_regular_file(&mut archive, path, contents);
817            }
818            archive.finish().unwrap();
819        }
820        tar_bytes
821    }
