503/// Transaction options passed to commands
504pub struct TxOptions {
505    /// Gas budget - None means estimate via dry-run
506    pub gas_budget: Option<u64>,
507    pub skip_confirm: bool,
508    /// If true, simulate the transaction without executing
509    pub dry_run: bool,
510    /// If true, build and print the unsigned transaction as base64 instead of
511    /// signing/executing it.
512    pub serialize_unsigned: bool,
513    /// Explicit sender address (e.g. a multisig). `None` => derive from the
514    /// configured keypair.
515    pub sender: Option<sui_sdk_types::Address>,
516    /// Pin a specific gas coin object id (`None` => fullnode gas selection).
517    pub gas_object: Option<sui_sdk_types::Address>,
518    /// Gas price override in MIST/unit (`None` => reference price).
519    pub gas_price: Option<u64>,
520}
521
522impl TxOptions {
523    /// The finalization mode implied by the flags. `--serialize-unsigned-transaction`
524    /// wins over `--dry-run` (they are also mutually exclusive at the clap layer).
525    pub fn mode(&self) -> crate::sui_tx_executor::TxMode {
526        use crate::sui_tx_executor::TxMode;
527        if self.serialize_unsigned {
528            TxMode::SerializeUnsigned
529        } else if self.dry_run {
530            TxMode::DryRun
531        } else {
532            TxMode::Execute
533        }
534    }
535
536    /// The manual gas overrides implied by the flags.
537    pub fn gas_overrides(&self) -> crate::sui_tx_executor::GasOverrides {
538        crate::sui_tx_executor::GasOverrides {
539            gas_object: self.gas_object,
540            gas_budget: self.gas_budget,
541            gas_price: self.gas_price,
542        }
543    }
544
545    /// Get gas budget, using the provided estimate if not explicitly set
546    pub fn gas_budget_or(&self, estimate: u64) -> u64 {
547        self.gas_budget.unwrap_or(estimate)
548    }
549
550    /// Get gas budget with a safety margin (1.2x the estimate)
551    pub fn gas_budget_or_with_margin(&self, estimate: u64) -> u64 {
552        self.gas_budget.unwrap_or_else(|| {
553            // Add 20% safety margin to estimates
554            estimate.saturating_mul(120).saturating_div(100)
555        })
556    }
557}

559#[cfg(test)]
560mod tx_options_tests {
561    use super::TxOptions;
562    use crate::sui_tx_executor::TxMode;
563
564    fn base() -> TxOptions {
565        TxOptions {
566            gas_budget: None,
567            skip_confirm: false,
568            dry_run: false,
569            serialize_unsigned: false,
570            sender: None,
571            gas_object: None,
572            gas_price: None,
573        }
574    }
575
576    #[test]
577    fn mode_defaults_to_execute() {
578        assert_eq!(base().mode(), TxMode::Execute);
579    }
580
581    #[test]
582    fn dry_run_maps_to_dry_run() {
583        let opts = TxOptions {
584            dry_run: true,
585            ..base()
586        };
587        assert_eq!(opts.mode(), TxMode::DryRun);
588    }
589
590    #[test]
591    fn serialize_unsigned_wins_over_dry_run() {
592        // The flags are mutually exclusive at the clap layer, but if both were
593        // set, serialize-unsigned must take precedence (we never execute).
594        let opts = TxOptions {
595            serialize_unsigned: true,
596            dry_run: true,
597            ..base()
598        };
599        assert_eq!(opts.mode(), TxMode::SerializeUnsigned);
600    }
601
602    #[test]
603    fn gas_overrides_pass_through() {
604        let gas = sui_sdk_types::Address::from_static("0x2");
605        let opts = TxOptions {
606            gas_object: Some(gas),
607            gas_budget: Some(123),
608            gas_price: Some(7),
609            ..base()
610        };
611        let overrides = opts.gas_overrides();
612        assert_eq!(overrides.gas_object, Some(gas));
613        assert_eq!(overrides.gas_budget, Some(123));
614        assert_eq!(overrides.gas_price, Some(7));
615    }

1080/// Parse metadata arguments from "key=value" format into a Vec of tuples
1081fn parse_metadata(args: Vec<String>) -> Vec<(String, String)> {
1082    args.into_iter()
1083        .filter_map(|s| {
1084            let mut parts = s.splitn(2, '=');
1085            match (parts.next(), parts.next()) {
1086                (Some(key), Some(value)) => Some((key.to_string(), value.to_string())),
1087                _ => {
1088                    print_warning(&format!(
1089                        "Ignoring invalid metadata format: '{}' (expected key=value)",
1090                        s
1091                    ));
1092                    None
1093                }
1094            }
1095        })
1096        .collect()
1097}

1154/// Render the result of a finalized transaction and return the execution
1155/// response when one was produced (execute mode only). The serialized unsigned

1157pub fn print_tx_outcome(
1158    outcome: crate::sui_tx_executor::TxOutcome,
1159) -> Option<Box<sui_rpc::proto::sui::rpc::v2::ExecuteTransactionResponse>> {
1160    use crate::sui_tx_executor::TxOutcome;

1165        }
1166        TxOutcome::Simulated {
1167            sender,
1168            gas_budget,
1169            gas_price,
1170        } => {

1172            print_detail(&format!(
1173                "  {} {}",
1174                "Sender:".dimmed(),
1175                sender.to_hex().cyan()
1176            ));
1177            print_detail(&format!(
1178                "  {} {} MIST",

1180                gas_budget.to_string().cyan()
1181            ));
1182            print_detail(&format!(
1183                "  {} {} MIST/unit",
1184                "Gas Price:".dimmed(),
1185                gas_price.to_string().cyan()
1186            ));
1187            let max_cost_sui = (gas_budget as f64) / 1_000_000_000.0;
1188            print_detail(&format!(
1189                "  {} {:.6} SUI",
1190                "Max Cost:".dimmed(),

1196            ));

1201            print_detail(&format!(
1202                "\n{} Transaction submitted: {}",
1203                "✓".green(),
1204                digest.to_string().cyan()
1205            ));

1207        }
1208    }

1224/// Run the `publish` command – build, publish, and initialise the Hashi package.
1225pub async fn run_publish(opts: PublishOpts) -> anyhow::Result<()> {
1226    crate::init_crypto_provider();
1227    init_tracing(opts.verbose);
1228
1229    // Load signer
1230    let signer = crate::keys::load_keypair_from_path(&opts.keypair)?;
1231    let sender = signer.verifying_key().derive_address();
1232    print_info(&format!("Sender address: {sender}"));
1233
1234    // Build
1235    print_info(&format!(
1236        "Building package at {} ...",
1237        opts.package_path.display()
1238    ));
1239    let params = crate::publish::BuildParams {
1240        sui_binary: &opts.sui_binary,
1241        package_path: &opts.package_path,
1242        client_config: opts.sui_client_config.as_deref(),
1243        environment: opts.environment.as_deref(),
1244    };
1245    let compiled = crate::publish::build_package(&params)?;
1246    print_success(&format!(
1247        "Package built ({} module(s))",
1248        compiled.modules.len()
1249    ));
1250
1251    if !opts.yes {
1252        print_info("This will publish the package (1 transaction).");
1253        print_info("Use --yes / -y to skip this prompt.");
1254        eprint!("Continue? [y/N] ");
1255        let mut answer = String::new();
1256        std::io::stdin().read_line(&mut answer)?;
1257        if !answer.trim().eq_ignore_ascii_case("y") {
1258            print_warning("Aborted.");
1259            return Ok(());
1260        }
1261    }
1262
1263    // Connect to RPC
1264    let mut client = crate::sui_rpc_client::new_sui_rpc_client(&opts.sui_rpc_url)?;
1265
1266    // Publish
1267    print_info("Publishing ...");
1268    let crate::publish::PublishOutput {
1269        ids,
1270        upgrade_cap_id,
1271    } = crate::publish::publish_package(&mut client, &signer, compiled).await?;
1272    print_success(&format!("package_id:      {}", ids.package_id));
1273    print_success(&format!("hashi_object_id: {}", ids.hashi_object_id));
1274    print_success(&format!("upgrade_cap_id:  {upgrade_cap_id}"));
1275    print_info(
1276        "The UpgradeCap stays in the publisher's wallet and the deploy is not yet \
1277         configured. Once all expected validators have registered, run `hashi launch` \
1278         with the chain id and guardian parameters to configure the deploy and unlock \
1279         genesis.",
1280    );
1281
1282    // Write ids to hashi_ids.json
1283    let json = serde_json::to_string_pretty(&ids)?;
1284    let out_path = "hashi_ids.json";
1285    std::fs::write(out_path, &json)?;
1286    print_success(&format!("Wrote {out_path}"));
1287
1288    Ok(())
1289}

1291/// Run the `launch` command – send `hashi::finish_publish` (the launch
1292/// switch): configure the deploy (chain id, guardian, overrides) and hand
1293/// the package `UpgradeCap` into on-chain custody, which unlocks genesis.
1294/// The initial committee forms from the validators fully registered at that
1295/// moment.
1296pub async fn run_launch(opts: LaunchOpts) -> anyhow::Result<()> {
1297    use sui_sdk_types::bcs::ToBcs;
1298
1299    crate::init_crypto_provider();
1300    init_tracing(opts.verbose);
1301
1302    // In serialize/status mode keep stdout clean (base64 / the LAUNCH_STATUS
1303    // line only); notes go to stderr.
1304    set_notes_to_stderr(opts.serialize_unsigned || opts.status);
1305
1306    // Resolve ids: explicit flags win, else the hashi_ids.json from publish.
1307    let ids: crate::config::HashiIds = match (&opts.package_id, &opts.hashi_object_id) {
1308        (Some(package_id), Some(hashi_object_id)) => crate::config::HashiIds {
1309            package_id: package_id.parse()?,
1310            hashi_object_id: hashi_object_id.parse()?,
1311        },
1312        (None, None) => {
1313            let raw = std::fs::read_to_string(&opts.hashi_ids).with_context(|| {
1314                format!(
1315                    "failed to read {} (or pass --package-id and --hashi-object-id)",
1316                    opts.hashi_ids.display()
1317                )
1318            })?;
1319            serde_json::from_str(&raw)?
1320        }
1321        _ => anyhow::bail!("--package-id and --hashi-object-id must be provided together"),
1322    };
1323
1324    print_info(&format!("Sui RPC: {}", opts.sui_rpc_url));
1325
1326    let mut client = crate::sui_rpc_client::new_sui_rpc_client(&opts.sui_rpc_url)?;
1327
1328    // Pre-flight: read the launch state and who would form the initial
1329    // committee. No hard failures yet — status mode reports every state.
1330    let (onchain, _watcher) =
1331        crate::onchain::OnchainState::new(&opts.sui_rpc_url, ids, None, None, None, None).await?;
1332    let (launched, roster): (bool, Vec<(sui_sdk_types::Address, bool)>) = {
1333        let state = onchain.state();
1334        (
1335            state.hashi().config.upgrade_cap.is_some(),
1336            state
1337                .hashi()
1338                .committees
1339                .members()
1340                .iter()
1341                .map(|(address, member)| {
1342                    (
1343                        *address,
1344                        member.next_epoch_encryption_public_key().is_some(),
1345                    )
1346                })
1347                .collect(),
1348        )
1349    };
1350
1351    // Stake-weight the roster: the genesis committee's security is the Sui
1352    // voting power it carries, not its head-count. Sui voting power sums to
1353    // 10_000 basis points across the active validator set.
1354    let voting_powers = fetch_voting_powers(&mut client).await?;
1355    let total_power: u64 = voting_powers.values().sum();
1356    let power_of = |address: &sui_sdk_types::Address| -> u64 {
1357        voting_powers.get(address).copied().unwrap_or(0)
1358    };
1359    let percent = |power: u64| -> f64 {
1360        if total_power == 0 {
1361            0.0
1362        } else {
1363            100.0 * power as f64 / total_power as f64
1364        }
1365    };
1366
1367    print_info(&format!("Registered validators ({}):", roster.len()));
1368    for (address, ready) in &roster {
1369        let status = if !voting_powers.contains_key(address) {
1370            "NOT AN ACTIVE SUI VALIDATOR"
1371        } else if *ready {
1372            "ready"
1373        } else {
1374            "MISSING NEXT-EPOCH KEYS (will be excluded from the committee)"
1375        };
1376        print_info(&format!(
1377            "  {address}  {:6.2}% stake  {status}",
1378            percent(power_of(address)),
1379        ));
1380    }
1381    let num_ready = roster.iter().filter(|(_, ready)| *ready).count();
1382    let ready_power: u64 = roster
1383        .iter()
1384        .filter(|(_, ready)| *ready)
1385        .map(|(address, _)| power_of(address))
1386        .sum();
1387    let registered_power: u64 = roster.iter().map(|(address, _)| power_of(address)).sum();
1388    print_info(&format!(
1389        "Ready to launch: {num_ready}/{} validators, {:.2}% of total Sui voting power \
1390         (registered: {:.2}%)",
1391        roster.len(),
1392        percent(ready_power),
1393        percent(registered_power),
1394    ));
1395
1396    if opts.status {
1397        // STABLE CONTRACT: the only stdout line in --status mode, parsed by
1398        // deployment automation (sui-operations deploy-hashi.yaml). Change
1399        // only in lockstep with its consumers.
1400        let bps = |power: u64| -> u64 { (power * 10_000).checked_div(total_power).unwrap_or(0) };
1401        println!(
1402            "LAUNCH_STATUS launched={launched} registered={} ready={num_ready} \
1403             ready_stake_bps={} registered_stake_bps={}",
1404            roster.len(),
1405            bps(ready_power),
1406            bps(registered_power),
1407        );
1408        return Ok(());
1409    }
1410
1411    anyhow::ensure!(
1412        !launched,
1413        "the UpgradeCap is already in on-chain custody — the launch (finish_publish) \
1414         has already happened"
1415    );
1416    anyhow::ensure!(
1417        !roster.is_empty(),
1418        "no validators are registered yet; the genesis committee would be empty"
1419    );
1420    anyhow::ensure!(
1421        num_ready > 0,
1422        "no validator has finished key registration; genesis would stall"
1423    );
1424    if num_ready < roster.len() {
1425        print_warning(&format!(
1426            "{} of {} registered validators ({:.2}% of total Sui voting power) have not \
1427             finished key registration and would be excluded from the genesis committee",
1428            roster.len() - num_ready,
1429            roster.len(),
1430            percent(registered_power - ready_power),
1431        ));
1432    }
1433
1434    // Guardian parameters are published on-chain by the launch tx (clap
1435    // requires them in every mode except --status).
1436    let bitcoin_chain_id = opts
1437        .bitcoin_chain_id
1438        .expect("required unless --status (enforced by clap)");
1439    let guardian_btc_public_key = opts
1440        .guardian_btc_public_key
1441        .expect("required unless --status (enforced by clap)");
1442    let guardian_url = opts
1443        .guardian_url
1444        .expect("required unless --status (enforced by clap)");
1445    let btc_public_key = hex::decode(
1446        guardian_btc_public_key
1447            .strip_prefix("0x")
1448            .unwrap_or(&guardian_btc_public_key),
1449    )
1450    .context("Invalid hex for --guardian-btc-public-key")?;
1451    anyhow::ensure!(
1452        btc_public_key.len() == 32,
1453        "--guardian-btc-public-key must be 32 bytes (x-only), got {} bytes",
1454        btc_public_key.len(),
1455    );
1456    let guardian = crate::publish::GuardianConfig {
1457        url: guardian_url,
1458        btc_public_key,
1459    };
1460    let bitcoin_overrides = crate::publish::BitcoinConfigOverrides {
1461        confirmation_threshold: opts.bitcoin_confirmation_threshold,
1462        deposit_time_delay_ms: opts.bitcoin_deposit_time_delay_ms,
1463    };
1464
1465    // Resolve the sender (the UpgradeCap owner): a local keypair, or an
1466    // explicit --sender for the serialize-unsigned (multisig) path.
1467    let signer = opts
1468        .keypair
1469        .as_deref()
1470        .map(crate::keys::load_keypair_from_path)
1471        .transpose()?;
1472    let sender: sui_sdk_types::Address = match (&signer, &opts.sender) {
1473        (Some(signer), None) => signer.verifying_key().derive_address(),
1474        (None, Some(sender)) => sender.parse()?,
1475        (Some(_), Some(_)) => anyhow::bail!("pass either --keypair or --sender, not both"),
1476        (None, None) => anyhow::bail!(
1477            "pass --keypair, or --sender together with --serialize-unsigned-transaction"
1478        ),
1479    };
1480    if signer.is_none() && !opts.serialize_unsigned {
1481        anyhow::bail!("--sender requires --serialize-unsigned-transaction (no key to sign with)");
1482    }
1483    print_info(&format!("Sender (UpgradeCap owner): {sender}"));
1484
1485    // Locate the cap.
1486    let upgrade_cap_id: sui_sdk_types::Address = match &opts.upgrade_cap {
1487        Some(id) => id.parse()?,
1488        None => {
1489            print_info("Locating UpgradeCap among the sender's owned objects ...");
1490            crate::publish::find_upgrade_cap(&mut client, sender, ids.package_id).await?
1491        }
1492    };
1493    print_info(&format!("UpgradeCap: {upgrade_cap_id}"));
1494
1495    if opts.serialize_unsigned {
1496        let tx = crate::publish::build_finish_publish_tx(
1497            &mut client,
1498            sender,
1499            &ids,
1500            upgrade_cap_id,
1501            &bitcoin_chain_id,
1502            &guardian,
1503            &bitcoin_overrides,
1504        )
1505        .await?;
1506        println!("{}", tx.to_bcs_base64()?);
1507        return Ok(());
1508    }
1509
1510    if !opts.yes {
1511        print_info(
1512            "This sends hashi::finish_publish: it configures the deploy (chain id, \
1513             guardian) and hands the UpgradeCap into on-chain custody, UNLOCKING \
1514             GENESIS — the initial committee forms from the validators that are fully \
1515             registered right now (1 transaction).",
1516        );
1517        print_info("Use --yes / -y to skip this prompt.");
1518        eprint!("Continue? [y/N] ");
1519        let mut answer = String::new();
1520        std::io::stdin().read_line(&mut answer)?;
1521        if !answer.trim().eq_ignore_ascii_case("y") {
1522            print_warning("Aborted.");
1523            return Ok(());
1524        }
1525    }
1526
1527    let signer = signer.expect("presence checked during sender resolution");
1528    print_info("Sending launch transaction (hashi::finish_publish) ...");
1529    crate::publish::finish_publish(
1530        &mut client,
1531        &signer,
1532        &ids,
1533        upgrade_cap_id,
1534        &bitcoin_chain_id,
1535        &guardian,
1536        &bitcoin_overrides,
1537    )
1538    .await?;
1539    print_success("finish_publish executed — genesis unlocked. Validators will now run DKG.");
1540    Ok(())
1541}

1543/// Active Sui validators' voting power (basis points of the 10_000 total),
1544/// keyed by validator address.
1545async fn fetch_voting_powers(
1546    client: &mut sui_rpc::Client,
1547) -> anyhow::Result<std::collections::HashMap<sui_sdk_types::Address, u64>> {
1548    use sui_rpc::field::FieldMaskUtil;
1549
1550    let mut request = sui_rpc::proto::sui::rpc::v2::GetEpochRequest::default();
1551    request.read_mask = Some(sui_rpc::field::FieldMask::from_paths([
1552        "system_state.validators.active_validators",
1553    ]));
1554    let response = client
1555        .ledger_client()
1556        .get_epoch(request)
1557        .await?
1558        .into_inner();
1559
1560    let validators = response
1561        .epoch
1562        .and_then(|epoch| epoch.system_state)
1563        .and_then(|system_state| system_state.validators)
1564        .map(|validator_set| validator_set.active_validators)
1565        .unwrap_or_default();
1566
1567    let mut powers = std::collections::HashMap::new();
1568    for validator in validators {
1569        if let (Some(address), Some(power)) = (validator.address, validator.voting_power) {
1570            powers.insert(address.parse::<sui_sdk_types::Address>()?, power);
1571        }
1572    }
1573    Ok(powers)
1574}

1576/// Run the `register` command – register a validator on-chain.
1577pub async fn run_register(opts: RegisterOpts) -> anyhow::Result<()> {
1578    use sui_sdk_types::bcs::ToBcs;
1579
1580    init_tracing(opts.verbose);
1581
1582    // In serialize mode keep stdout clean (base64 only); notes go to stderr.
1583    set_notes_to_stderr(opts.serialize_unsigned);
1584
1585    // Load the validator config
1586    let config = crate::config::Config::load(&opts.config)?;
1587
1588    // Resolve Sui RPC URL: CLI flag > config file
1589    let sui_rpc_url = opts
1590        .sui_rpc_url
1591        .or_else(|| config.sui_rpc.clone())
1592        .ok_or_else(|| {
1593            anyhow::anyhow!("Sui RPC URL not provided (use --sui-rpc-url or set in config file)")
1594        })?;
1595
1596    // Parse optional operator address
1597    let operator_address = opts
1598        .operator_address
1599        .map(|s| s.parse::<sui_sdk_types::Address>())
1600        .transpose()?;
1601
1602    let validator_address = config.validator_address()?;
1603    print_info(&format!("Validator address: {validator_address}"));
1604    print_info(&format!("Sui RPC: {sui_rpc_url}"));
1605
1606    if opts.serialize_unsigned {
1607        // Build the transaction and print as base64 without executing.
1608        // No private key is required for this path.
1609        let mut client = crate::sui_rpc_client::new_sui_rpc_client(&sui_rpc_url)?;
1610        let hashi_ids = config.hashi_ids();
1611
1612        print_info("Building registration transaction ...");
1613        let transaction = crate::sui_tx_executor::build_register_or_update_validator_tx(
1614            &mut client,
1615            &hashi_ids,
1616            &config,
1617            operator_address,
1618            None,
1619            None,
1620            None,
1621        )
1622        .await?;
1623
1624        match transaction {
1625            Some(tx) => {
1626                let tx_base64 = tx.to_bcs_base64()?;
1627                println!("{tx_base64}");
1628            }
1629            None => print_info("Validator metadata is already up-to-date; nothing to do."),
1630        }
1631        return Ok(());
1632    }
1633
1634    if !opts.yes {
1635        print_info("This will register the validator on-chain (1 transaction).");
1636        print_info("Use --yes / -y to skip this prompt.");
1637        eprint!("Continue? [y/N] ");
1638        let mut answer = String::new();
1639        std::io::stdin().read_line(&mut answer)?;
1640        if !answer.trim().eq_ignore_ascii_case("y") {
1641            print_warning("Aborted.");
1642            return Ok(());
1643        }
1644    }
1645
1646    let client = crate::sui_rpc_client::new_sui_rpc_client(&sui_rpc_url)?;
1647    let signer = config.operator_private_key()?;
1648    let hashi_ids = config.hashi_ids();
1649    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(client, signer, hashi_ids);
1650
1651    print_info("Registering validator ...");
1652    let updated = executor
1653        .execute_register_or_update_validator(&config, operator_address, None, None)
1654        .await?;
1655
1656    if updated {
1657        print_success("Validator registered/updated successfully");
1658    } else {
1659        print_info("Validator metadata is already up-to-date; nothing to do.");
1660    }
1661    Ok(())
1662}
