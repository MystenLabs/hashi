25/// Print metadata if present
26fn print_metadata(metadata: &[(String, String)]) {
27    if !metadata.is_empty() {
28        print_detail(&format!("  {}", "Metadata:".bold()));
29        for (key, value) in metadata {
30            print_detail(&format!("    {}: {}", key.dimmed(), value));
31        }
32    }
33}

35/// Finalize a transaction according to `tx_opts`: serialize it unsigned,
36/// dry-run it, or sign and submit it.
37///
38/// Returns `Some(response)` when a real transaction was executed, and `None`
39/// for dry-run, serialize-unsigned, or when execution is requested but no
40/// keypair is configured.
41async fn execute_or_simulate(
42    client: &mut HashiClient,
43    tx: sui_transaction_builder::TransactionBuilder,
44    tx_opts: &TxOptions,
45) -> Result<Option<ExecuteTransactionResponse>> {
46    use crate::sui_tx_executor::TxMode;
47
48    // Only the execute path needs a keypair; serialize/dry-run build with just
49    // the sender address.
50    if tx_opts.mode() == TxMode::Execute && !client.can_execute() {
51        print_warning(
52            "Transaction execution requires a keypair (--keypair). Use \
53             --serialize-unsigned-transaction to emit an unsigned transaction, or --dry-run.",
54        );
55        return Ok(None);
56    }
57
58    match tx_opts.mode() {
59        TxMode::SerializeUnsigned => print_info("Building unsigned transaction..."),
60        TxMode::DryRun => print_info("Simulating transaction (dry-run)..."),
61        TxMode::Execute => print_info("Executing transaction..."),
62    }
63
64    let outcome = client.finalize_tx(tx, tx_opts).await?;
65    Ok(crate::cli::print_tx_outcome(outcome).map(|response| *response))
66}

68/// Print the newly-created proposal's ID after a `create_*_proposal` call,
69/// when the response is available (real execute, not dry-run).
70fn print_created_proposal_id(response: Option<&ExecuteTransactionResponse>) {
71    let Some(response) = response else {
72        return;
73    };
74    match crate::cli::upgrade::extract_proposal_id_from_response(response) {
75        Ok(id) => println!("  {} {}", "Proposal ID:".bold(), id.to_hex().cyan()),
76        Err(e) => {
77            tracing::warn!("Could not extract proposal ID from response: {e}");
78        }
79    }
80}

82/// List all active proposals
83pub async fn list_proposals(
84    config: &CliConfig,
85    type_filter: Option<String>,
86    detailed: bool,
87) -> Result<()> {
88    let client = HashiClient::new(config).await?;
89
90    print_info("Fetching proposals...");
91
92    let proposals = client.fetch_proposals();
93
94    if proposals.is_empty() {
95        println!("\n{}", "No active proposals found.".dimmed());
96        return Ok(());
97    }
98
99    // Filter by type if specified
100    let proposals: Vec<_> = if let Some(ref filter) = type_filter {
101        let filter_lower = filter.to_lowercase();
102        proposals
103            .into_iter()
104            .filter(|p| {
105                display::format_proposal_type(&p.proposal_type)
106                    .to_lowercase()
107                    .contains(&filter_lower)
108            })
109            .collect()
110    } else {
111        proposals
112    };
113
114    if proposals.is_empty() {
115        println!(
116            "\n{}",
117            format!(
118                "No proposals found matching type filter: {}",
119                type_filter.unwrap_or_default()
120            )
121            .dimmed()
122        );
123        return Ok(());
124    }
125
126    println!("\n📋 Active Proposals:\n");
127
128    if detailed {
129        // List mode skips the per-proposal vote/quorum fetch to avoid N extra
130        // network calls; use `proposal view <id>` for full vote progress.
131        for proposal in &proposals {
132            print_proposal_detailed(proposal, None, None);
133            println!();
134        }
135    } else {
136        #[derive(Tabled)]
137        struct ProposalRow {
138            #[tabled(rename = "ID")]
139            id: String,
140            #[tabled(rename = "Type")]
141            proposal_type: String,
142            #[tabled(rename = "Created")]
143            timestamp: String,
144        }
145
146        let rows: Vec<ProposalRow> = proposals
147            .iter()
148            .map(|p| ProposalRow {
149                id: display::format_address(&p.id),
150                proposal_type: display::format_proposal_type(&p.proposal_type),
151                timestamp: display::format_timestamp(p.timestamp_ms),
152            })
153            .collect();
154
155        let table = Table::new(rows).to_string();
156        println!("{}", table);
157    }
158
159    println!(
160        "\n{} {} proposal(s) found",
161        "ℹ".blue(),
162        proposals.len().to_string().bold()
163    );
164
165    Ok(())
166}

190/// Vote on a proposal, optionally chaining an execute if this vote pushes
191/// the proposal over quorum.
192pub async fn vote(
193    config: &CliConfig,
194    proposal_id: &str,
195    execute: bool,
196    tx_opts: &TxOptions,
197) -> Result<()> {
198    let mut client = HashiClient::new(config).await?;
199
200    let proposal_addr = Address::from_hex(proposal_id)
201        .with_context(|| format!("Invalid proposal ID: {}", proposal_id))?;
202
203    print_info(&format!("Fetching proposal {}...", proposal_id));
204
205    let proposal = client
206        .fetch_proposal(&proposal_addr)
207        .ok_or_else(|| anyhow::anyhow!("Proposal not found: {}", proposal_id))?;
208
209    let proposal_type_str = display::format_proposal_type(&proposal.proposal_type);
210
211    print_detail(&format!("\n{}", "Proposal Details:".bold()));
212    print_detail(&format!("  Type: {}", proposal_type_str.cyan()));
213
214    prompt_continue("vote on this proposal", tx_opts).await?;
215
216    print_info("Building vote transaction...");
217
218    // Infer the type tag from the on-chain proposal type
219    let type_arg = get_proposal_type_arg(client.hashi_ids().package_id, &proposal.proposal_type)?;
220    let tx = client.build_vote_transaction(proposal_addr, type_arg)?;
221
222    print_info(&format!(
223        "Transaction: proposal::vote<{}> on {}",
224        proposal_type_str, proposal_id
225    ));
226
227    let vote_response = execute_or_simulate(&mut client, tx, tx_opts).await?;
228
229    if !execute {
230        return Ok(());
231    }
232
233    // `--execute` was requested. Only meaningful after a real execute (not a
234    // dry-run / missing-keypair).
235    if vote_response.is_none() {
236        return Ok(());
237    }
238
239    // Upgrade proposals require the dedicated upgrade flow — the generic
240    // `<module>::execute` path can't construct an UpgradeTicket.
241    use crate::onchain::types::ProposalType;
242    if matches!(proposal.proposal_type, ProposalType::Upgrade) {
243        print_warning(
244            "--execute is not supported for Upgrade proposals; run the \
245             dedicated upgrade flow once quorum is reached.",
246        );
247        return Ok(());
248    }
249
250    // Re-fetch live vote state to see whether the vote we just submitted
251    // pushed us over quorum. `HashiClient`'s cached scrape is from CLI start,
252    // so this has to be the live `list_dynamic_fields` call.
253    let details = client
254        .fetch_proposal_details(proposal_addr)
255        .await
256        .context("failed to re-fetch proposal state after voting")?;
257    let committee = client
258        .fetch_current_committee()
259        .ok_or_else(|| anyhow::anyhow!("no committee available to compute quorum"))?;
260
261    let total_weight = committee.total_weight();
262    let voted_weight: u64 = details
263        .votes
264        .iter()
265        .map(|voter| {
266            committee
267                .members()
268                .iter()
269                .find(|m| m.validator_address() == *voter)
270                .map(|m| m.weight())
271                .unwrap_or(0)
272        })
273        .sum();
274    let threshold_weight = total_weight
275        .saturating_mul(details.quorum_threshold_bps)
276        .div_ceil(10_000);
277
278    if voted_weight < threshold_weight {
279        print_info(&format!(
280            "Quorum not reached yet ({voted_weight}/{threshold_weight} weight); \
281             skipping --execute."
282        ));
283        return Ok(());
284    }
285
286    print_info(&format!(
287        "Quorum reached ({voted_weight}/{threshold_weight} weight); executing..."
288    ));
289    let execute_tx =
290        client.build_execute_proposal_transaction(proposal_addr, &proposal.proposal_type)?;
291    print_info(&format!(
292        "Transaction: {}::execute on {}",
293        proposal.proposal_type.as_str(),
294        proposal_id
295    ));
296    execute_or_simulate(&mut client, execute_tx, tx_opts).await?;
297    Ok(())
298}

377/// Create an upgrade proposal.
378///
379/// Exactly one of `digest` or `package_path` must be `Some`. When
380/// `package_path` is provided, the CLI builds the package via `sui move build`
381/// and verifies that its `PACKAGE_VERSION` constant is exactly +1 of the
382/// currently published version (pre-flight check) before submitting the
383/// proposal. The `--digest` path skips that check and is retained only for
384/// callers with a pre-built package.
385pub async fn create_upgrade_proposal(
386    config: &CliConfig,
387    digest: Option<&str>,
388    package_path: Option<&std::path::Path>,
389    sui_binary: &std::path::Path,
390    sui_client_config: Option<&std::path::Path>,
391    metadata: Vec<(String, String)>,
392    tx_opts: &TxOptions,
393) -> Result<()> {
394    let mut client = HashiClient::new(config).await?;
395
396    let digest_bytes = match (digest, package_path) {
397        (Some(d), None) => {
398            print_warning(
399                "--digest skips pre-flight checks (PACKAGE_VERSION = current + 1). \
400                 Prefer --package-path.",
401            );
402            hex::decode(d.trim_start_matches("0x")).context("Invalid digest hex")?
403        }
404        (None, Some(path)) => {
405            let current_version = client.highest_package_version().context(
406                "could not determine current package version from on-chain state; \
407                 is the package deployed?",
408            )?;
409            let expected_version = current_version + 1;
410            print_info(&format!(
411                "Building upgrade package at {} (expecting PACKAGE_VERSION = {expected_version})",
412                path.display()
413            ));
414            let (_compiled, digest) = crate::cli::upgrade::build_upgrade_package(
415                sui_binary,
416                path,
417                sui_client_config,
418                expected_version,
419            )
420            .context("failed to build upgrade package")?;
421            digest
422        }
423        (None, None) => {
424            anyhow::bail!("must provide either --digest or --package-path");
425        }
426        (Some(_), Some(_)) => unreachable!("clap enforces mutual exclusion"),
427    };
428
429    print_detail(&format!("\n{}", "Creating Upgrade Proposal:".bold()));
430    print_detail(&format!("  Digest: 0x{}", hex::encode(&digest_bytes)));
431    print_metadata(&metadata);
432
433    prompt_continue("create this upgrade proposal", tx_opts).await?;
434
435    let tx = client.build_create_proposal_transaction(CreateProposalParams::Upgrade {
436        digest: digest_bytes,
437        metadata,
438    })?;
439
440    print_info("Transaction: upgrade::propose");
441    let response = execute_or_simulate(&mut client, tx, tx_opts).await?;
442    print_created_proposal_id(response.as_ref());
443    Ok(())
444}

477pub async fn create_update_mpc_config_proposal(
478    config: &CliConfig,
479    threshold_bps: Option<u64>,
480    max_faulty_bps: Option<u64>,
481    weight_reduction_allowed_delta: Option<u64>,
482    nonce_generation_protocol: Option<u64>,
483    metadata: Vec<(String, String)>,
484    tx_opts: &TxOptions,
485) -> Result<()> {
486    const MAX_BPS: u64 = 10_000;
487    if let Some(t) = threshold_bps {
488        anyhow::ensure!(
489            (1..=MAX_BPS).contains(&t),
490            "--threshold-bps must be in 1..={MAX_BPS}, got {t}"
491        );
492    }
493    if let Some(f) = max_faulty_bps {
494        anyhow::ensure!(
495            f <= MAX_BPS,
496            "--max-faulty-bps must be in 0..={MAX_BPS}, got {f}"
497        );
498    }
499    if let Some(d) = weight_reduction_allowed_delta {
500        anyhow::ensure!(
501            d <= MAX_BPS,
502            "--weight-reduction-allowed-delta must be in 0..={MAX_BPS}, got {d}"
503        );
504    }
505    if let Some(p) = nonce_generation_protocol {
506        anyhow::ensure!(
507            p <= 1,
508            "--nonce-generation-protocol must be 0 (vanilla) or 1 (avid), got {p}"
509        );
510    }
511
512    let count = [
513        threshold_bps,
514        max_faulty_bps,
515        weight_reduction_allowed_delta,
516        nonce_generation_protocol,
517    ]
518    .iter()
519    .filter(|v| v.is_some())
520    .count();
521    if count == 0 {
522        anyhow::bail!(
523            "must provide at least one of --threshold-bps, --max-faulty-bps, --weight-reduction-allowed-delta, --nonce-generation-protocol"
524        );
525    }
526
527    prompt_continue("create this MPC config update proposal", tx_opts).await?;
528
529    let mut client = HashiClient::new(config).await?;
530    let tx = client.build_create_proposal_transaction(CreateProposalParams::UpdateMpcConfig {
531        threshold_bps,
532        max_faulty_bps,
533        weight_reduction_allowed_delta,
534        nonce_generation_protocol,
535        metadata,
536    })?;
537
538    let response = execute_or_simulate(&mut client, tx, tx_opts).await?;
539    print_created_proposal_id(response.as_ref());
540    Ok(())
541}

543/// Parse a CLI config value string like "u64:1000" or "bool:true" into a ConfigValueParam.
544fn parse_config_value(s: &str) -> Result<hashi_types::move_types::ConfigValue> {
545    use hashi_types::move_types::ConfigValue;
546
547    let (type_prefix, raw) = s
548        .split_once(':')
549        .ok_or_else(|| anyhow::anyhow!("expected type:value format (e.g. u64:1000)"))?;
550
551    match type_prefix {
552        "u64" => Ok(ConfigValue::U64(raw.parse().context("invalid u64")?)),
553        "bool" => Ok(ConfigValue::Bool(raw.parse().context("invalid bool")?)),
554        "string" => Ok(ConfigValue::String(raw.to_string())),
555        "address" => Ok(ConfigValue::Address(
556            raw.parse().context("invalid address")?,
557        )),
558        other => anyhow::bail!(
559            "unknown type prefix '{}' (expected u64, bool, string, address)",
560            other
561        ),
562    }
563}

665    print_info("Transaction: update_guardian::propose");
666    let response = execute_or_simulate(&mut client, tx, tx_opts).await?;
667    print_created_proposal_id(response.as_ref());
668    Ok(())
669}

671/// Create an emergency pause (or, with `unpause == true`, unpause) proposal.
672///
673/// Pausing uses a deliberately low quorum (default 5%) so a small fraction of
674/// committee weight can halt the protocol quickly; unpausing requires the
675/// normal 2/3 supermajority. Both paths target `emergency_pause::propose`.
676pub async fn create_emergency_pause_proposal(
677    config: &CliConfig,
678    unpause: bool,
679    metadata: Vec<(String, String)>,
680    tx_opts: &TxOptions,
681) -> Result<()> {
682    let action = if unpause { "Unpause" } else { "Pause" };
683    let title = format!("Creating Emergency {action} Proposal:");
684    print_detail(&format!("\n{}", title.bold()));
685    print_detail(&format!("  Action: {action}"));
686    print_metadata(&metadata);
687
688    prompt_continue(
689        &format!("create this emergency {} proposal", action.to_lowercase()),
690        tx_opts,
691    )
692    .await?;
693
694    let mut client = HashiClient::new(config).await?;
695    let tx = client.build_create_proposal_transaction(CreateProposalParams::EmergencyPause {
696        pause: !unpause,
697        metadata,
698    })?;
699
700    print_info("Transaction: emergency_pause::propose");
701    let response = execute_or_simulate(&mut client, tx, tx_opts).await?;
702    print_created_proposal_id(response.as_ref());
703    Ok(())
704}

708fn print_proposal_detailed(
709    proposal: &Proposal,
710    details: Option<&crate::cli::client::ProposalDetails>,
711    committee: Option<&hashi_types::committee::Committee>,
712) {
713    println!("{}", "━".repeat(60).dimmed());
714    println!(
715        "  {} {}",
716        "ID:".bold(),
717        display::format_address_full(&proposal.id).cyan()
718    );
719    println!(
720        "  {} {}",
721        "Type:".bold(),
722        display::format_proposal_type(&proposal.proposal_type).green()
723    );
724    println!(
725        "  {} {}",
726        "Created:".bold(),
727        display::format_timestamp(proposal.timestamp_ms)
728    );
729
730    if let Some(details) = details {
731        println!(
732            "  {} {}",
733            "Creator:".bold(),
734            details.creator.to_hex().dimmed()
735        );
736
737        // Vote tally + quorum progress.
738        let total_weight = committee.map(|c| c.total_weight()).unwrap_or(0);
739        let voted_weight: u64 = details
740            .votes
741            .iter()
742            .map(|voter| {
743                committee
744                    .and_then(|c| c.members().iter().find(|m| m.validator_address() == *voter))
745                    .map(|m| m.weight())
746                    .unwrap_or(0)
747            })
748            .sum();
749        let threshold_weight = total_weight
750            .saturating_mul(details.quorum_threshold_bps)
751            .div_ceil(10_000);
752        let quorum_met = voted_weight >= threshold_weight && total_weight > 0;
753
754        let status = if quorum_met {
755            "QUORUM REACHED".green().bold()
756        } else {
757            format!(
758                "{}/{} weight ({} more needed)",
759                voted_weight,
760                threshold_weight,
761                threshold_weight.saturating_sub(voted_weight)
762            )
763            .yellow()
764        };
765        println!(
766            "  {} {} voter(s) — {} of total weight {} — {}",
767            "Votes:".bold(),
768            details.votes.len().to_string().cyan(),
769            voted_weight.to_string().cyan(),
770            total_weight.to_string().dimmed(),
771            status
772        );
773        println!(
774            "  {} {} bps ({:.2}%)",
775            "Threshold:".bold(),
776            details.quorum_threshold_bps,
777            details.quorum_threshold_bps as f64 / 100.0
778        );
779        if !details.votes.is_empty() {
780            println!("  {}", "Voters:".bold());
781            for voter in &details.votes {
782                println!("    - {}", voter.to_hex().dimmed());
783            }
784        }
785
786        if !details.metadata.contents.is_empty() {
787            println!("  {}", "Metadata:".bold());
788            for entry in &details.metadata.contents {
789                println!("    {}: {}", entry.key.dimmed(), entry.value);
790            }
791        }
792    }
793
794    println!("{}", "━".repeat(60).dimmed());
795}

797/// Pause for user acknowledgement before an actual execution. No-op when the
798/// user passed `-y/--yes`, or in dry-run / serialize-unsigned mode — those
799/// change no on-chain state, and serialize mode must keep stdout clean.
800async fn prompt_continue(action: &str, tx_opts: &TxOptions) -> Result<()> {
801    use crate::sui_tx_executor::TxMode;
802    use tokio::io::AsyncBufReadExt;
803    use tokio::io::BufReader;
804
805    if tx_opts.skip_confirm || tx_opts.mode() != TxMode::Execute {
806        return Ok(());
807    }
808
809    eprintln!(
810        "\n{}",
811        format!("Press enter to {action}, or Ctrl+C to cancel...").yellow()
812    );
813
814    let mut reader = BufReader::new(tokio::io::stdin());
815    let mut input = String::new();
816    reader.read_line(&mut input).await?;
817    Ok(())
818}
