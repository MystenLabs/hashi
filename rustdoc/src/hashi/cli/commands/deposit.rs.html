20pub async fn run(action: DepositCommands, config: &CliConfig, tx_opts: &TxOptions) -> Result<()> {
21    match action {
22        DepositCommands::GenerateAddress { recipient } => {
23            generate_address(config, &recipient).await
24        }
25        DepositCommands::Request {
26            txid,
27            vout,
28            amount,
29            recipient,
30        } => request(config, tx_opts, &txid, vout, amount, recipient.as_deref()).await,
31        DepositCommands::Status { request_id } => status(config, &request_id).await,
32        DepositCommands::List => list(config).await,
33    }
34}

36/// Parse raw on-chain MPC public key bytes and derive the deposit address.
37pub fn cli_derive_deposit_address(
38    mpc_pubkey_bytes: &[u8],
39    recipient: Option<&sui_sdk_types::Address>,
40    btc_network: bitcoin::Network,
41) -> Result<bitcoin::Address> {
42    use fastcrypto::groups::secp256k1::ProjectivePoint;
43    use fastcrypto::serde_helpers::ToFromByteArray;
44
45    let mpc_key = match mpc_pubkey_bytes.len() {
46        33 => <ProjectivePoint as ToFromByteArray<33>>::from_byte_array(
47            mpc_pubkey_bytes
48                .try_into()
49                .context("MPC key must be 33 bytes")?,
50        )
51        .context("Failed to parse MPC key as ProjectivePoint")?,
52        32 => {
53            if recipient.is_some() {
54                anyhow::bail!(
55                    "Key derivation requires the full 33-byte compressed MPC key, \
56                     but only 32-byte x-only key is available"
57                );
58            }
59            let xonly = XOnlyPublicKey::from_slice(mpc_pubkey_bytes)
60                .context("Failed to parse 32-byte MPC key")?;
61            return Ok(
62                hashi_types::guardian::bitcoin_utils::single_key_taproot_script_path_address(
63                    &xonly,
64                    btc_network,
65                ),
66            );
67        }
68        n => anyhow::bail!("Unexpected MPC public key length: {} bytes", n),
69    };
70
71    crate::deposits::derive_deposit_address(&mpc_key, recipient, btc_network)
72}

74async fn generate_address(config: &CliConfig, recipient: &str) -> Result<()> {
75    let client = HashiClient::new(config).await?;
76
77    let mpc_pubkey = client.fetch_mpc_public_key();
78    if mpc_pubkey.is_empty() {
79        anyhow::bail!("MPC public key not available on-chain. Has the committee completed DKG?");
80    }
81
82    let is_change = recipient.is_empty();
83    let recipient_addr = if is_change {
84        None
85    } else {
86        Some(
87            recipient
88                .parse::<sui_sdk_types::Address>()
89                .context("Invalid recipient Sui address")?,
90        )
91    };
92
93    let btc_network = crate::btc_monitor::config::parse_btc_network(
94        config.bitcoin.as_ref().and_then(|b| b.network.as_deref()),
95    )?;
96
97    let address = cli_derive_deposit_address(&mpc_pubkey, recipient_addr.as_ref(), btc_network)?;
98
99    let title = if is_change {
100        "Deposit Address (Change Address)"
101    } else {
102        "Deposit Address"
103    };
104
105    println!("\n{}", title.bold());
106    println!("{}", "━".repeat(50).dimmed());
107    println!("  {} {}", "Address:".bold(), address.to_string().green());
108    println!("  {} {:?}", "Network:".bold(), btc_network);
109    if !is_change {
110        println!("  {} {}", "hBTC Recipient:".bold(), recipient);
111    }
112    println!("{}", "━".repeat(50).dimmed());
113
114    Ok(())
115}

117async fn request(
118    config: &CliConfig,
119    _tx_opts: &TxOptions,
120    txid: &str,
121    vout: u32,
122    amount: u64,
123    recipient: Option<&str>,
124) -> Result<()> {
125    config.validate()?;
126
127    let hashi_ids = crate::config::HashiIds {
128        package_id: config.package_id(),
129        hashi_object_id: config.hashi_object_id(),
130    };
131
132    let signer = config
133        .load_keypair()?
134        .context("Keypair required for deposit request. Set keypair_path in config.")?;
135
136    let derivation_path = match recipient {
137        Some(r) => Some(
138            r.parse::<sui_sdk_types::Address>()
139                .context("Invalid recipient Sui address")?,
140        ),
141        None => {
142            let addr = signer.public_key().derive_address();
143            print_info(&format!(
144                "No --recipient specified, defaulting to signer address {}",
145                addr
146            ));
147            Some(addr)
148        }
149    };
150
151    let client = sui_rpc::Client::new(&config.sui_rpc_url)?;
152    let mut executor = crate::sui_tx_executor::SuiTxExecutor::new(client, signer, hashi_ids);
153
154    let parsed_txid: bitcoin::Txid = txid.parse().context("Invalid txid")?;
155    let txid_address = sui_sdk_types::Address::new(parsed_txid.to_byte_array());
156
157    print_info("Submitting deposit request on Sui...");
158
159    let request_id = executor
160        .execute_create_deposit_request(txid_address, vout, amount, derivation_path)
161        .await?;
162
163    print_success(&format!("Deposit request created: {}", request_id));
164
165    Ok(())
166}

168async fn status(config: &CliConfig, request_id: &str) -> Result<()> {
169    let client = HashiClient::new(config).await?;
170
171    let req_addr = request_id
172        .parse::<sui_sdk_types::Address>()
173        .context("Invalid request ID")?;
174
175    let deposits = client.fetch_deposit_requests();
176    let deposit = deposits.iter().find(|d| d.id == req_addr);
177
178    println!("\n{}", "Deposit Request Status".bold());
179    println!("{}", "━".repeat(60).dimmed());
180
181    let Some(dep) = deposit else {
182        print_info("Deposit request not found in pending queue (may be confirmed or expired).");
183        println!("{}", "━".repeat(60).dimmed());
184        return Ok(());
185    };
186
187    let txid_bytes: [u8; 32] = dep.utxo.id.txid.into();
188    let txid = bitcoin::Txid::from_byte_array(txid_bytes);
189    println!(
190        "  {} {}",
191        "Request ID:".bold(),
192        display::format_address_full(&dep.id)
193    );
194    println!("  {} {}:{}", "UTXO:".bold(), txid, dep.utxo.id.vout);
195    println!("  {} {} sats", "Amount:".bold(), dep.utxo.amount);
196    println!(
197        "  {} {}",
198        "Requested:".bold(),
199        display::format_timestamp(dep.timestamp_ms)
200    );
201    println!("  {} {}", "Status:".bold(), "Pending".yellow());
202
203    // BTC context if available
204    if let Ok(Some(btc_rpc)) = config.btc_rpc_client() {
205        println!();
206        println!("  {}", "BTC Context:".bold());
207        use bitcoincore_rpc::RpcApi;
208        match btc_rpc.get_raw_transaction_info(&txid, None) {
209            Ok(info) => {
210                let confirmations = info.confirmations.unwrap_or(0);
211                println!("    {} {}", "Confirmations:".bold(), confirmations);
212                if let Some(ref blockhash) = info.blockhash {
213                    println!("    {} {}", "Block:".bold(), blockhash);
214                }
215            }
216            Err(_) => {
217                println!("    {}", "(transaction not found on BTC node)".dimmed());
218            }
219        }
220    }
221
222    println!("{}", "━".repeat(60).dimmed());
223    Ok(())
224}

226async fn list(config: &CliConfig) -> Result<()> {
227    let client = HashiClient::new(config).await?;
228
229    let deposits = client.fetch_deposit_requests();
230
231    println!("\n{}", "Deposit Requests".bold());
232    println!("{}", "━".repeat(100).dimmed());
233
234    if deposits.is_empty() {
235        print_info("No pending deposit requests.");
236    } else {
237        println!(
238            "  {:<20} {:<14} {:<12} {:<10} {:<20} {}",
239            "Request ID".bold(),
240            "Amount (sats)".bold(),
241            "UTXO".bold(),
242            "Status".bold(),
243            "Caller".bold(),
244            "Requested".bold()
245        );
246        for dep in &deposits {
247            let txid_bytes: [u8; 32] = dep.utxo.id.txid.into();
248            let txid = bitcoin::Txid::from_byte_array(txid_bytes);
249            let txid_str = txid.to_string();
250            println!(
251                "  {:<20} {:<14} {}:{:<3} {:<10} {:<20} {}",
252                display::format_address_full(&dep.id),
253                dep.utxo.amount,
254                txid_str,
255                dep.utxo.id.vout,
256                "Pending".yellow(),
257                display::format_address_full(&dep.requester_address),
258                display::format_timestamp(dep.timestamp_ms)
259            );
260        }
261        println!("\n  {} deposit(s)", deposits.len());
262    }
263
264    println!("{}", "━".repeat(100).dimmed());
265    Ok(())
