76    pub async fn wait_for_key_ready(&self) -> G {
77        let mut rx = self.key_ready_rx.clone();
78        loop {
79            {
80                let value = rx.borrow();
81                if let Some(pk) = value.as_ref() {
82                    return *pk;
83                }
84            }
85            if rx.changed().await.is_err() {
86                std::future::pending().await
87            }
88        }
89    }

91    pub fn public_key(&self) -> Option<G> {
92        *self.key_ready_rx.borrow()
93    }

130    #[tracing::instrument(name = "mpc_service", skip_all)]
131    async fn run(mut self) {
132        let pending = self.get_pending_epoch_change();
133        let is_in_committee = self.inner.is_in_current_committee();
134        info!(
135            "MPC service starting: pending_epoch_change={pending:?}, \
136             is_in_current_committee={is_in_committee}",
137        );
138        if let Some(epoch) = pending {
139            info!("Entering handle_reconfig for epoch {epoch}");
140            self.handle_reconfig(epoch).await;
141        } else if self.is_awaiting_genesis() {
142            // No committee has been formed yet (epoch 0, no committee for epoch 0).
143            // Wait for enough validators to register then trigger genesis reconfig.
144            info!("No initial committee yet; waiting for enough validators to register...");
145            self.try_submit_genesis_reconfig().await;
146        } else if self.inner.is_in_current_committee() {
147            loop {
148                if let Some(epoch) = self.get_pending_epoch_change() {
149                    self.handle_reconfig(epoch).await;
150                    continue;
151                }
152                self.sync_if_stale().await;
153                let epoch = self.inner.onchain_state().epoch();
154                if self.inner.signing_manager_for(epoch).is_some() {
155                    break;
156                }
157                tokio::time::sleep(RETRY_INTERVAL).await;
158            }
159        } else {
160            info!("Node is not in the current committee, waiting for reconfig notification...");
161        }
162        let mut notifications = self.inner.onchain_state().subscribe();
163        let mut checkpoint_rx = self.inner.onchain_state().subscribe_checkpoint();
164        let mut reconcile_tick = tokio::time::interval(RECONCILE_TICK);
165        reconcile_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
166        loop {
167            // Check for pending reconfig before blocking on `recv()`.
168            if let Some(epoch) = self.get_pending_epoch_change() {
169                self.handle_reconfig(epoch).await;
170                continue;
171            }
172            tokio::select! {
173                notification = notifications.recv() => {
174                    match notification {
175                        Ok(notification) => match notification {
176                            Notification::StartReconfig(epoch) => {
177                                self.handle_reconfig(epoch).await;
178                            }
179                            Notification::SuiEpochChanged(sui_epoch) => {
180                                self.try_submit_start_reconfig(sui_epoch).await;
181                            }
182                            _ => {}
183                        },
184                        Err(e) => {
185                            error!("MPC notification recv error: {e:?}, resubscribing");
186                            notifications = self.inner.onchain_state().subscribe();
187                        }
188                    }
189                }
190                Ok(()) = checkpoint_rx.changed() => {
191                    self.sync_if_stale().await;
192                }
193                _ = reconcile_tick.tick() => {
194                    self.sync_if_stale().await;
195                }
196                Ok(()) = self.refill_rx.changed() => {
197                    let next_batch = *self.refill_rx.borrow();
198                    self.refill_with_retries(next_batch).await;
199                }
200            }
201        }
202    }

204    async fn refill_with_retries(&self, next_batch: u32) {
205        for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
206            if let Err(e) = self.bail_if_reconfig_pending() {
207                info!("presignature refill stopped: {e}");
208                return;
209            }
210            match self.refill_presignatures(next_batch).await {
211                Ok(()) => break,
212                Err(e) => {
213                    error!(
214                        "Presignature refill attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} failed: {e}"
215                    );
216                    if attempt < MAX_PROTOCOL_ATTEMPTS {
217                        tokio::time::sleep(RETRY_INTERVAL).await;
218                    }
219                }
220            }
221        }
222    }

226            tokio::time::sleep(RETRY_INTERVAL).await;
227        }

246    /// Wait for enough validators to register, then submit `start_reconfig`
247    /// to form the initial committee. Blocks until a pending epoch change
248    /// appears (either from our own submission or another node's).
249    async fn try_submit_genesis_reconfig(&self) {
250        loop {
251            if self.get_pending_epoch_change().is_some() {
252                return;
253            }
254            match self.inner.next_reconfig_epoch().await {
255                Ok(target) => {
256                    if let Err(e) = self.inner.prepare_and_register_keys(target).await {
257                        debug!(
258                            "Encryption/signing key registration for epoch {target} failed: {e}; \
259                             will retry on next genesis_reconfig iteration"
260                        );
261                    }
262                }
263                Err(e) => debug!("Failed to compute next_reconfig_epoch: {e}"),
264            }
265            // Attempt to submit start_reconfig. This will fail on-chain until
266            // the publisher sends finish_publish (the launch switch).
267            let result = async {
268                let mut executor =
269                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
270                executor.execute_start_reconfig().await
271            };
272            match result.await {
273                Ok(()) => {
274                    info!("Genesis start_reconfig submitted successfully");
275                    return;
276                }
277                Err(e) => {
278                    debug!("Genesis start_reconfig not yet possible: {e}");
279                    // Poll for pending epoch change while waiting, in case
280                    // another node submitted start_reconfig.
281                    let polls = (RETRY_INTERVAL.as_millis()
282                        / START_RECONFIG_POLL_INTERVAL.as_millis())
283                        as u32;
284                    for _ in 0..polls {
285                        if self.get_pending_epoch_change().is_some() {
286                            return;
287                        }
288                        tokio::time::sleep(START_RECONFIG_POLL_INTERVAL).await;
289                    }
290                }
291            }
292        }
293    }

295    async fn recover_mpc_state(&self) -> anyhow::Result<MpcOutput> {
296        let onchain_state = self.inner.onchain_state().clone();
297        let epoch = onchain_state.epoch();
298        let certs = fetch_key_generation_certificates(&onchain_state, epoch)
299            .await
300            .map_err(|e| {
301                anyhow::anyhow!("failed to fetch reconfig certs for epoch {epoch}: {e}")
302            })?;
303        let is_key_rotation = matches!(certs.first(), Some((_, CertificateV1::Rotation(_))));
304        let onchain_mpc_key = onchain_state.mpc_public_key();
305        info!(
306            "recover_mpc_state: epoch={epoch}, is_key_rotation={is_key_rotation}, \
307             onchain_mpc_key_len={}",
308            onchain_mpc_key.len(),
309        );
310        let output = if is_key_rotation {
311            self.recover_current_rotation(epoch, &onchain_mpc_key).await
312        } else {
313            self.recover_current_dkg(epoch, &onchain_mpc_key).await
314        }?;
315        info!(
316            "recover_mpc_state: recovered vk={}",
317            hex::encode(output.public_key.to_byte_array())
318        );
319        Ok(output)
320    }

322    async fn recover_current_dkg(
323        &self,
324        epoch: u64,
325        onchain_mpc_key: &[u8],
326    ) -> anyhow::Result<MpcOutput> {
327        self.setup_initial_dkg(epoch)?;
328        let onchain_state = self.inner.onchain_state().clone();
329        let certs: Vec<CertificateV1> =
330            fetch_certificates(&onchain_state, epoch, None, move_types::ProtocolType::Dkg)
331                .await
332                .map_err(|e| anyhow::anyhow!("failed to fetch DKG certs for epoch {epoch}: {e}"))?
333                .into_iter()
334                .map(|(_, cert)| cert)
335                .collect();
336        let mpc_manager = self
337            .inner
338            .mpc_manager()
339            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized for DKG recovery"))?;
340        match MpcManager::reconstruct_current_dkg_output(&mpc_manager, &certs, onchain_mpc_key) {
341            MpcOutputRecoveryOutcome::Recovered(output) => {
342                info!(
343                    "recover_current_dkg: recovered current epoch {epoch} from local DKG \
344                     messages (no peers)"
345                );
346                Ok(output)
347            }
348            MpcOutputRecoveryOutcome::NotApplicable => self.run_dkg(epoch).await,
349            MpcOutputRecoveryOutcome::Suspicious(reason) => {
350                error!(
351                    "recover_current_dkg: local DKG state for epoch {epoch} contradicts on-chain \
352                     truth ({reason}); observing this epoch, will recover at the next rotation"
353                );
354                self.inner.metrics.mpc_recovery_suspicious_total.inc();
355                Err(anyhow::anyhow!(
356                    "suspicious local DKG state for epoch {epoch}: {reason}"
357                ))
358            }
359        }
360    }

381        .map_err(|e| anyhow::anyhow!("failed to fetch rotation certs for epoch {epoch}: {e}"))?
382        .into_iter()
383        .map(|(_, cert)| cert)
384        .collect();
385        let previous_certs: Vec<CertificateV1> =
386            fetch_key_generation_certificates(&onchain_state, previous_epoch)
387                .await
388                .map_err(|e| {
389                    anyhow::anyhow!(
390                        "failed to fetch certs for previous epoch {previous_epoch}: {e}"
391                    )
392                })?
393                .into_iter()
394                .map(|(_, cert)| cert)
395                .collect();
396        match MpcManager::reconstruct_current_rotation_output(
397            &mpc_manager,
398            &current_certs,
399            &previous_certs,
400            onchain_mpc_key,
401        ) {
402            MpcOutputRecoveryOutcome::Recovered(output) => {
403                info!(
404                    "recover_current_rotation: recovered current epoch {epoch} and previous \
405                     {previous_epoch} from local rotation messages (no peers)"
406                );
407                Ok(output)
408            }
409            MpcOutputRecoveryOutcome::NotApplicable => self.run_key_rotation(epoch).await,
410            MpcOutputRecoveryOutcome::Suspicious(reason) => {
411                error!(
412                    "recover_current_rotation: local rotation state for epoch {epoch} contradicts \
413                     on-chain truth ({reason}); observing this epoch, will recover at the next \
414                     rotation"
415                );
416                self.inner.metrics.mpc_recovery_suspicious_total.inc();
417                Err(anyhow::anyhow!(
418                    "suspicious local rotation state for epoch {epoch}: {reason}"
419                ))
420            }
421        }
422    }

453    async fn generate_presignatures(
454        &self,
455        epoch: u64,
456        batch_index: u32,
457    ) -> anyhow::Result<(Committee, Presignatures)> {
458        let onchain_state = self.inner.onchain_state().clone();
459        let committee = onchain_state
460            .state()
461            .hashi()
462            .committees
463            .committees()
464            .get(&epoch)
465            .ok_or_else(|| anyhow::anyhow!("No committee found for epoch {}", epoch))?
466            .clone();
467        let mpc_manager = self
468            .inner
469            .mpc_manager()
470            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized"))?;
471        let signer = self.inner.config.operator_private_key()?;
472        let p2p_channel =
473            RpcP2PChannel::new(onchain_state.clone(), epoch, MPC_LABEL_NONCE_GENERATION);
474        let mut tob_channel = SuiTobChannel::new(
475            self.inner.config.hashi_ids(),
476            onchain_state,
477            epoch,
478            Some(batch_index),
479            move_types::ProtocolType::NonceGeneration,
480            signer,
481        )
482        .with_idle_timeout(NONCE_RECEIVE_IDLE_TIMEOUT);
483        let metrics = &self.inner.metrics;
484        let _timer = metrics
485            .mpc_total_duration_seconds
486            .with_label_values(&[MPC_LABEL_NONCE_GENERATION])
487            .start_timer();
488        let nonce_result = MpcManager::run_nonce_generation(
489            &mpc_manager,
490            batch_index,
491            &p2p_channel,
492            &mut tob_channel,
493            metrics,
494        )
495        .await;
496        drop(_timer);
497        let nonce_outputs =
498            nonce_result.map_err(|e| anyhow::anyhow!("Nonce generation failed: {e}"))?;
499        let (batch_size_per_weight, params, use_legacy) = {
500            let mgr = mpc_manager.read().unwrap();
501            (
502                mgr.batch_size_per_weight,
503                Parameters {
504                    t: mgr.mpc_config.threshold,
505                    f: mgr.mpc_config.max_faulty,
506                },
507                mgr.mpc_config.presignature_derivation_version.use_legacy(),
508            )
509        };
510        let _timer = metrics
511            .mpc_presig_conversion_duration_seconds
512            .with_label_values(&[MPC_LABEL_NONCE_GENERATION])
513            .start_timer();
514        let presignatures =
515            Presignatures::new(nonce_outputs, batch_size_per_weight, params, use_legacy)
516                .map_err(|e| anyhow::anyhow!("Failed to create presignatures: {e}"))?;
517        drop(_timer);
518        Ok((committee, presignatures))
519    }

524        let signing_manager = SigningManager::new(
525            address,
526            committee,
527            output.threshold,
528            output.key_shares.clone(),

533            PRESIG_REFILL_DIVISOR,

536        self.inner.store_signing_manager(signing_manager);
537        Ok(())
538    }

557    async fn recover_presigning_state(&self, output: &MpcOutput) -> anyhow::Result<()> {
558        let (num_consumed, epoch, committee, pending) = {
559            let state = self.inner.onchain_state().state();
560            let hashi = state.hashi();
561            let num_consumed = hashi.num_consumed_presigs;
562            let epoch = hashi.committees.epoch();
563            let committee = hashi
564                .committees
565                .committees()
566                .get(&epoch)
567                .ok_or_else(|| anyhow::anyhow!("No committee found for epoch {epoch}"))?
568                .clone();
569            let mut pending: HashSet<u64> = HashSet::new();
570            for txn in hashi.withdrawal_queue.withdrawal_txns().values() {
571                if txn.signing_epoch() != epoch {
572                    continue;
573                }
574                for signature in &txn.signing.signatures {
575                    if let move_types::MpcSig::Pending(index) = signature {
576                        anyhow::ensure!(
577                            *index < num_consumed,
578                            "pending presig index {index} is not below allocation cursor {num_consumed}",
579                        );
580                        anyhow::ensure!(
581                            pending.insert(*index),
582                            "pending presig index {index} is assigned more than once",
583                        );
584                    }
585                }
586            }
587            (num_consumed, epoch, committee, pending)
588        };
589        let mpc_manager = self
590            .inner
591            .mpc_manager()
592            .ok_or_else(|| anyhow::anyhow!("MpcManager not initialized"))?;
593        let (batch_size_per_weight, params, use_legacy, protocol, floor) = {
594            let mgr = mpc_manager.read().unwrap();
595            (
596                mgr.batch_size_per_weight,
597                Parameters {
598                    t: mgr.mpc_config.threshold,
599                    f: mgr.mpc_config.max_faulty,
600                },
601                mgr.mpc_config.presignature_derivation_version.use_legacy(),
602                mgr.mpc_config.nonce_generation_protocol,
603                mgr.required_nonce_weight(),
604            )
605        };
606        let mut boundaries: Vec<(u32, u64, usize)> = Vec::new();
607        let mut batch_start = 0u64;
608        let mut batch_index = 0u32;
609        loop {
610            self.bail_if_reconfig_pending()?;
611            let size = self
612                .nonce_batch_size_from_certs(
613                    &mpc_manager,
614                    epoch,
615                    batch_index,
616                    protocol,
617                    params,
618                    use_legacy,
619                    batch_size_per_weight,
620                    floor,
621                )
622                .await?;
623            let Some(size) = size else {
624                anyhow::ensure!(
625                    batch_start >= num_consumed,
626                    "nonce batch {batch_index} at start {batch_start} read sub-floor below cursor \
627                     {num_consumed} — partial cert fetch",
628                );
629                break;
630            };
631            boundaries.push((batch_index, batch_start, size));
632            if num_consumed < batch_start + size as u64 {
633                break;
634            }
635            batch_start += size as u64;
636            batch_index += 1;
637        }
638        anyhow::ensure!(
639            !boundaries.is_empty(),
640            "no certified nonce batches to recover for epoch {epoch} at cursor {num_consumed}",
641        );
642        let recovered_end = boundaries
643            .last()
644            .map_or(0, |&(_, start, size)| start + size as u64);
645        let first_pending = boundaries
646            .iter()
647            .position(|&(_, start, size)| {
648                let end = start + size as u64;
649                pending.iter().any(|&p| p >= start && p < end)
650            })
651            .unwrap_or_else(|| boundaries.len().saturating_sub(1));
652        let mut retained: Vec<(Presignatures, u32, u64)> = Vec::new();
653        // TODO(IOP-529): Avoid the double cert-fetch in presig recovery.
654        for &(bidx, start, size) in boundaries.iter().skip(first_pending) {
655            self.bail_if_reconfig_pending()?;
656            let presigs = self
657                .recover_presignatures_from_certs(
658                    &mpc_manager,
659                    epoch,
660                    bidx,
661                    batch_size_per_weight,
662                    params,
663                )
664                .await?;
665            anyhow::ensure!(
666                presigs.len() == size,
667                "batch {bidx} boundary size {size} (Phase 1) != reconstructed len {} (Phase 2)",
668                presigs.len(),
669            );
670            retained.push((presigs, bidx, start));
671        }
672        anyhow::ensure!(
673            self.inner.onchain_state().epoch() == epoch,
674            "epoch changed during presigning recovery",
675        );
676        let latest_cursor = self
677            .inner
678            .onchain_state()
679            .state()
680            .hashi()
681            .num_consumed_presigs;
682        anyhow::ensure!(
683            latest_cursor <= recovered_end,
684            "cursor {latest_cursor} advanced past recovered end {recovered_end} during recovery",
685        );
686        let address = self.inner.config.validator_address()?;
687        let retained_count = retained.len();
688        let signing_manager = SigningManager::new_recovered(
689            address,
690            committee,
691            output.threshold,
692            output.key_shares.clone(),
693            output.public_key,
694            retained,
695            num_consumed,
696            &pending,
697            PRESIG_REFILL_DIVISOR,
698            self.refill_tx.clone(),
699        )?;
700        self.inner.store_signing_manager(signing_manager);
701        info!(
702            "Recovered presigning state: {retained_count} of {} batch(es) retained from \
703             first_pending_batch={first_pending}, recovered_end={recovered_end}, \
704             num_consumed_presigs={num_consumed}, pending={}",
705            boundaries.len(),
706            pending.len(),
707        );
708        Ok(())
709    }

711    #[allow(clippy::too_many_arguments)]
712    async fn nonce_batch_size_from_certs(
713        &self,
714        mpc_manager: &Arc<std::sync::RwLock<MpcManager>>,
715        epoch: u64,
716        batch_index: u32,
717        protocol: NonceGenerationProtocol,
718        params: Parameters,
719        use_legacy: bool,
720        batch_size_per_weight: u16,
721        floor: u32,
722    ) -> anyhow::Result<Option<usize>> {
723        let onchain_state = self.inner.onchain_state().clone();
724        let weight = match protocol {
725            NonceGenerationProtocol::Vanilla => {
726                let Some(certs) = onchain_state
727                    .fetch_certs(
728                        epoch,
729                        Some(batch_index),
730                        move_types::ProtocolType::NonceGeneration,
731                    )
732                    .await?
733                else {
734                    return Ok(None);
735                };
736                let certs = MpcManager::verified_nonce_certs(mpc_manager, epoch, certs).await;
737                certified_nonce_weight(mpc_manager, &certs)
738            }
739            NonceGenerationProtocol::Avid => {
740                let certs = fetch_certificates(
741                    &onchain_state,
742                    epoch,
743                    Some(batch_index),
744                    move_types::ProtocolType::NonceGeneration,
745                )
746                .await?;
747                if certs.is_empty() {
748                    return Ok(None);
749                }
750                let certs = MpcManager::verified_nonce_certs(mpc_manager, epoch, certs).await;
751                certified_nonce_weight(mpc_manager, &certs)
752            }
753        };
754        if weight < floor {
755            return Ok(None);
756        }
757        Ok(Some(presig_count(
758            weight as usize,
759            params,
760            use_legacy,
761            batch_size_per_weight,
762        )))
763    }

765    async fn sync_if_stale(&self) {
766        if self.is_awaiting_genesis() || !self.inner.is_in_current_committee() {
767            return;
768        }
769        let epoch = self.inner.onchain_state().epoch();
770        if self.inner.signing_manager_for(epoch).is_some() {
771            return;
772        }
773        let _guard = match self.reconciling.try_lock() {
774            Ok(g) => g,
775            Err(_) => return,
776        };
777        let epoch = self.inner.onchain_state().epoch();
778        if self.inner.signing_manager_for(epoch).is_some() {
779            return;
780        }
781        info!("sync_if_stale: rebuilding SigningManager for epoch {epoch}");
782        let output = match self.recover_mpc_state().await {
783            Ok(output) => output,
784            Err(e) => {
785                if let Some(p) = self.get_pending_epoch_change() {
786                    info!(
787                        "sync_if_stale: recover_mpc_state for epoch {epoch} superseded by \
788                         pending reconfig to {p} ({e})"
789                    );
790                    return;
791                }
792                error!("sync_if_stale: recover_mpc_state failed for epoch {epoch}: {e}");
793                self.re_register_keys_if_lost().await;
794                return;
795            }
796        };
797        match self.recover_presigning_state(&output).await {
798            Ok(()) => {
799                info!("sync_if_stale: recovered epoch {epoch} via certs path");
800            }
801            Err(certs_err) => {
802                if let Some(p) = self.get_pending_epoch_change() {
803                    info!(
804                        "sync_if_stale: recovery for epoch {epoch} superseded by pending \
805                         reconfig to {p} ({certs_err})"
806                    );
807                    return;
808                }
809                let num_consumed = self
810                    .inner
811                    .onchain_state()
812                    .state()
813                    .hashi()
814                    .num_consumed_presigs;
815                if num_consumed > 0 {
816                    error!(
817                        "sync_if_stale: recovery failed for epoch {epoch} at cursor \
818                         {num_consumed} ({certs_err}); retrying on the next tick (no genesis fallback)"
819                    );
820                    return;
821                }
822                info!(
823                    "sync_if_stale: certs path failed for epoch {epoch} ({certs_err}), \
824                     falling back to protocol"
825                );
826                if let Err(e) = self.prepare_signing(epoch, &output).await {
827                    error!("sync_if_stale: protocol fallback failed for epoch {epoch}: {e}");
828                    return;
829                }
830                info!("sync_if_stale: recovered epoch {epoch} via protocol path");
831            }
832        }
833        let _ = self.key_ready_tx.send(Some(output.public_key));
834    }

836    async fn re_register_keys_if_lost(&self) {
837        let committee = self
838            .inner
839            .onchain_state()
840            .state()
841            .hashi()
842            .committees
843            .current_committee()
844            .cloned();
845        let Some(committee) = committee else { return };
846        let Ok(me) = self.inner.config.validator_address() else {
847            return;
848        };
849        if !self.inner.committee_key_lost(&committee, me) {
850            return;
851        }
852        self.inner.metrics.mpc_committee_key_lost_total.inc();
853        let mut target = match self.inner.next_reconfig_epoch().await {
854            Ok(target) => target,
855            Err(e) => {
856                warn!("cannot determine next reconfig epoch for key re-registration: {e}");
857                return;
858            }
859        };
860        for _ in 0..MAX_KEY_REREGISTRATION_BUMPS {
861            if self
862                .replacement_keys_target_epoch
863                .lock()
864                .unwrap()
865                .is_some_and(|recorded| recorded >= target)
866            {
867                return;
868            }
869            warn!(
870                "no DB encryption or signing key matches the current committee record; \
871                 registering fresh keys for epoch {target} so the node rejoins at that reconfig"
872            );
873            if let Err(e) = self.inner.prepare_and_register_keys(target).await {
874                warn!("failed to register replacement keys for epoch {target}: {e}; will retry");
875                return;
876            }
877            match self
878                .inner
879                .onchain_state()
880                .scrape_committee_for_epoch(target)
881                .await
882            {
883                Ok(Some(frozen)) if self.inner.committee_key_lost(&frozen, me) => {
884                    self.inner.metrics.mpc_key_reregistration_bumps_total.inc();
885                    info!(
886                        "epoch {target} committee already snapshotted without replacement keys; \
887                         re-targeting {}",
888                        target + 1
889                    );
890                    target += 1;
891                }
892                Ok(Some(frozen)) => {
893                    if frozen.members().iter().any(|m| m.validator_address() == me) {
894                        info!("replacement keys frozen into the epoch {target} committee");
895                    } else {
896                        warn!(
897                            "not a member of the epoch {target} committee; replacement keys \
898                             registered for a future committee"
899                        );
900                    }
901                    *self.replacement_keys_target_epoch.lock().unwrap() = Some(target);
902                    return;
903                }
904                Ok(None) => {
905                    info!(
906                        "replacement keys registered; the epoch {target} committee is not yet \
907                         snapshotted and will include them"
908                    );
909                    *self.replacement_keys_target_epoch.lock().unwrap() = Some(target);
910                    return;
911                }
912                Err(e) => {
913                    warn!(
914                        "cannot verify the epoch {target} committee after key registration: {e}; \
915                         re-verifying next tick"
916                    );
917                    return;
918                }
919            }
920        }
921        warn!(
922            "replacement keys still excluded after {MAX_KEY_REREGISTRATION_BUMPS} registration \
923             attempts; will retry next tick"
924        );
925    }

941    async fn recover_presignatures_from_certs(
942        &self,
943        mpc_manager: &Arc<std::sync::RwLock<MpcManager>>,
944        epoch: u64,
945        batch_index: u32,
946        batch_size_per_weight: u16,
947        params: Parameters,
948    ) -> anyhow::Result<Presignatures> {
949        let onchain_state = self.inner.onchain_state().clone();
950        let p2p_channel = RpcP2PChannel::new(
951            self.inner.onchain_state().clone(),
952            epoch,
953            MPC_LABEL_NONCE_GENERATION,
954        );
955        let (protocol, use_legacy, floor) = {
956            let mgr = mpc_manager.read().unwrap();
957            (
958                mgr.mpc_config.nonce_generation_protocol,
959                mgr.mpc_config.presignature_derivation_version.use_legacy(),
960                mgr.required_nonce_weight(),
961            )
962        };
963        let expected_from = |weight: u32| -> anyhow::Result<usize> {
964            anyhow::ensure!(
965                weight >= floor,
966                "nonce batch {batch_index} for epoch {epoch} refetched below floor \
967                 ({weight} < {floor}); certificate set shrank during recovery",
968            );
969            Ok(presig_count(
970                weight as usize,
971                params,
972                use_legacy,
973                batch_size_per_weight,
974            ))
975        };
976        let (outputs, expected_size) = match protocol {
977            NonceGenerationProtocol::Vanilla => {
978                let certs = onchain_state
979                    .fetch_certs(
980                        epoch,
981                        Some(batch_index),
982                        move_types::ProtocolType::NonceGeneration,
983                    )
984                    .await?
985                    .ok_or_else(|| {
986                        anyhow::anyhow!(
987                            "No nonce gen certificates on TOB for epoch {epoch} batch {batch_index}"
988                        )
989                    })?;
990                let certs = MpcManager::verified_nonce_certs(mpc_manager, epoch, certs).await;
991                let expected_size = expected_from(certified_nonce_weight(mpc_manager, &certs))?;
992                let outputs = MpcManager::reconstruct_presignatures_with_complaint_recovery(
993                    mpc_manager,
994                    epoch,
995                    batch_index,
996                    &certs,
997                    &p2p_channel,
998                )
999                .await?;
1000                (outputs, Some(expected_size))
1001            }
1002            NonceGenerationProtocol::Avid => {
1003                let certs = fetch_certificates(
1004                    &onchain_state,
1005                    epoch,
1006                    Some(batch_index),
1007                    move_types::ProtocolType::NonceGeneration,
1008                )
1009                .await?;
1010                if certs.is_empty() {
1011                    return Err(anyhow::anyhow!(
1012                        "No nonce gen certificates on TOB for epoch {epoch} batch {batch_index}"
1013                    ));
1014                }
1015                let certs = MpcManager::verified_nonce_certs(mpc_manager, epoch, certs).await;
1016                let expected_size = expected_from(certified_nonce_weight(mpc_manager, &certs))?;
1017                let mut prefetched = PrefetchedTobChannel::new(certs);
1018                let outputs = MpcManager::run_nonce_generation(
1019                    mpc_manager,
1020                    batch_index,
1021                    &p2p_channel,
1022                    &mut prefetched,
1023                    &self.inner.metrics,
1024                )
1025                .await
1026                .map_err(|e| anyhow::anyhow!("AVID nonce recovery from certs failed: {e}"))?;
1027                (outputs, Some(expected_size))
1028            }
1029        };
1030        if outputs.is_empty() {
1031            return Err(anyhow::anyhow!(
1032                "No valid nonce outputs after reconstruction for epoch {epoch} batch {batch_index}"
1033            ));
1034        }
1035        let presignatures = Presignatures::new(outputs, batch_size_per_weight, params, use_legacy)
1036            .map_err(|e| anyhow::anyhow!("Failed to create presignatures: {e}"))?;
1037        if let Some(expected) = expected_size {
1038            anyhow::ensure!(
1039                presignatures.len() == expected,
1040                "Reconstructed nonce batch {batch_index} for epoch {epoch} has {} presigs but \
1041                 certificates imply {expected}; message-incomplete reconstruction",
1042                presignatures.len(),
1043            );
1044        }
1045        Ok(presignatures)
1046    }

1048    async fn try_submit_start_reconfig(&self, sui_epoch: u64) {
1049        if self.get_pending_epoch_change().is_some() {
1050            return;
1051        }
1052        let hashi_epoch = self
1053            .inner
1054            .onchain_state()
1055            .state()
1056            .hashi()
1057            .committees
1058            .epoch();
1059        if hashi_epoch >= sui_epoch {
1060            return;
1061        }
1062        if let Err(e) = self.inner.prepare_and_register_keys(sui_epoch).await {
1063            warn!(
1064                "Failed to prepare/register encryption+signing keys for epoch {sui_epoch}: {e}; \
1065                 will retry on next trigger"
1066            );
1067        }
1068        for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
1069            let result = async {
1070                let mut executor =
1071                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
1072                executor.execute_start_reconfig().await
1073            };
1074            match result.await {
1075                Ok(()) => {
1076                    return;
1077                }
1078                Err(e) => {
1079                    warn!("start_reconfig attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} failed: {e}");
1080                    if attempt < MAX_PROTOCOL_ATTEMPTS {
1081                        // Poll for pending epoch change while waiting, so we can
1082                        // return early if another node submitted start_reconfig.
1083                        let polls = (RETRY_INTERVAL.as_millis()
1084                            / START_RECONFIG_POLL_INTERVAL.as_millis())
1085                            as u32;
1086                        for _ in 0..polls {
1087                            if self.get_pending_epoch_change().is_some() {
1088                                return;
1089                            }
1090                            tokio::time::sleep(START_RECONFIG_POLL_INTERVAL).await;
1091                        }
1092                    }
1093                }
1094            }
1095        }
1096    }

1098    #[tracing::instrument(level = "info", skip_all, fields(target_epoch))]
1099    async fn handle_reconfig(&self, target_epoch: u64) {
1100        let run_dkg = self
1101            .inner
1102            .onchain_state()
1103            .state()
1104            .hashi()
1105            .committees
1106            .mpc_public_key()
1107            .is_empty();
1108        let protocol_label = if run_dkg {
1109            MPC_LABEL_DKG
1110        } else {
1111            MPC_LABEL_KEY_ROTATION
1112        };
1113        if self.get_pending_epoch_change() != Some(target_epoch) {
1114            info!(
1115                "handle_reconfig: epoch {target_epoch} no longer pending at entry, aborting before start",
1116            );
1117            return;
1118        }
1119        // A pending epoch change implies the launch (finish_publish) has
1120        // happened, so the on-chain bitcoin_chain_id exists now even if this
1121        // node booted pre-launch and the startup check was skipped. Refuse
1122        // to participate on mismatch — signing for the wrong Bitcoin network
1123        // must not happen; the caller re-enters while the epoch change is
1124        // pending, surfacing the error every RETRY_INTERVAL until an
1125        // operator fixes the config.
1126        if let Err(e) = self.inner.verify_bitcoin_chain_id() {
1127            error!("refusing to participate in reconfig for epoch {target_epoch}: {e}");
1128            self.sleep_if_still_pending(target_epoch).await;
1129            return;
1130        }
1131        let metrics = &self.inner.metrics;
1132        let _reconfig_timer = metrics
1133            .mpc_reconfig_total_duration_seconds
1134            .with_label_values(&[protocol_label])
1135            .start_timer();
1136        info!("handle_reconfig: epoch={target_epoch}, run_dkg={run_dkg}, entering retry loop",);
1137        let output = loop {
1138            if self.get_pending_epoch_change() != Some(target_epoch) {
1139                info!("handle_reconfig: epoch {target_epoch} no longer pending, aborting",);
1140                return;
1141            }
1142            let needs_fresh_manager = match self.inner.mpc_manager() {
1143                None => true,
1144                Some(mgr) => mgr.read().unwrap().mpc_config.epoch != target_epoch,
1145            };
1146            if needs_fresh_manager {
1147                let setup_result = if run_dkg {
1148                    self.setup_initial_dkg(target_epoch)
1149                } else {
1150                    self.setup_key_rotation(target_epoch)
1151                };
1152                if let Err(e) = setup_result {
1153                    error!(
1154                        "Failed to set up MPC manager for epoch {}: {e}, retrying...",
1155                        target_epoch
1156                    );
1157                    self.sleep_if_still_pending(target_epoch).await;
1158                    continue;
1159                }
1160            }
1161            let _timer = metrics
1162                .mpc_total_duration_seconds
1163                .with_label_values(&[protocol_label])
1164                .start_timer();
1165            let result = if run_dkg {
1166                self.run_dkg(target_epoch).await
1167            } else {
1168                self.run_key_rotation(target_epoch).await
1169            };
1170            drop(_timer);
1171            match result {
1172                Ok(output) => break output,
1173                Err(e) => {
1174                    error!(
1175                        "MPC protocol for epoch {} failed: {e}, retrying...",
1176                        target_epoch
1177                    );
1178                    self.sleep_if_still_pending(target_epoch).await;
1179                }
1180            }
1181        };
1182        if !self.reconfig_target_live(target_epoch) {
1183            info!("handle_reconfig: epoch {target_epoch} aborted after protocol completion");
1184            return;
1185        }
1186        let _ = self.key_ready_tx.send(Some(output.public_key));
1187        info!("MPC key ready for epoch {target_epoch}, submitting end_reconfig");
1188        let _end_reconfig_timer = metrics
1189            .mpc_end_reconfig_duration_seconds
1190            .with_label_values(&[protocol_label])
1191            .start_timer();
1192        loop {
1193            if self.get_pending_epoch_change() != Some(target_epoch) {
1194                break;
1195            }
1196            match self.submit_end_reconfig(target_epoch, &output).await {
1197                Ok(()) => break,
1198                Err(e) => match classify_reconfig_submission_error(&e) {
1199                    ReconfigSubmissionErrorKind::NonMoveAbort => {
1200                        warn!(
1201                            "submit_end_reconfig for epoch {} failed: {e}, retrying...",
1202                            target_epoch
1203                        );
1204                        self.sleep_if_still_pending(target_epoch).await;
1205                    }
1206                    ReconfigSubmissionErrorKind::NonRetryableMoveAbort
1207                    | ReconfigSubmissionErrorKind::CommitteeHandoffAlreadySubmitted
1208                    | ReconfigSubmissionErrorKind::EndReconfigAlreadyCompleted => {
1209                        let msg = format!(
1210                            "submit_end_reconfig for epoch {target_epoch} failed with non-retryable error: {e}"
1211                        );
1212                        error!("{msg}");
1213                        panic!("{msg}");
1214                    }
1215                },
1216            }
1217        }
1218        drop(_end_reconfig_timer);
1219        let next_epoch = target_epoch + 1;
1220        if let Err(e) = self.inner.prepare_and_register_keys(next_epoch).await {
1221            warn!(
1222                "Failed to prepare/register encryption+signing keys for epoch {next_epoch}: {e}; \
1223                 will retry at next trigger"
1224            );
1225        }
1226        self.backup_handle.backup_after_epoch_change(target_epoch);
1227        info!("end_reconfig complete for epoch {target_epoch}, running prepare_signing");
1228        let pruning_references = {
1229            let state = self.inner.onchain_state().state();
1230            let committee_set = &state.hashi().committees;
1231            let mut referenced = crate::db::PruningReferences::default();
1232            for committee in committee_set.committees().values() {
1233                for member in committee.members() {
1234                    referenced
1235                        .add_member_pubkeys(member.encryption_public_key(), member.public_key());
1236                }
1237            }
1238            for member_info in committee_set.members().values() {
1239                referenced.add_pending_registration(
1240                    member_info.next_epoch_encryption_public_key(),
1241                    member_info.next_epoch_public_key(),
1242                );
1243            }
1244            if let Some((prev_epoch, _)) = committee_set.previous_committee_for_target(target_epoch)
1245            {
1246                referenced.add_committee_epoch(prev_epoch);
1247            }
1248            referenced
1249        };
1250        if let Err(e) = self
1251            .inner
1252            .db
1253            .prune_messages_below(target_epoch, &pruning_references)
1254        {
1255            error!("Failed to prune old MPC messages below epoch {target_epoch}: {e}");
1256        }
1257        if !self.reconfig_target_live(target_epoch) {
1258            info!(
1259                "handle_reconfig: epoch {target_epoch} no longer pending nor current; \
1260                 skipping signing setup"
1261            );
1262            return;
1263        }
1264        let _prepare_signing_timer = metrics
1265            .mpc_prepare_signing_duration_seconds
1266            .with_label_values(&[protocol_label])
1267            .start_timer();
1268        for attempt in 1..=MAX_PROTOCOL_ATTEMPTS {
1269            match self.prepare_signing(target_epoch, &output).await {
1270                Ok(()) => break,
1271                Err(e) => {
1272                    error!(
1273                        "prepare_signing attempt {attempt}/{MAX_PROTOCOL_ATTEMPTS} \
1274                         for epoch {target_epoch}: {e}"
1275                    );
1276                    if attempt < MAX_PROTOCOL_ATTEMPTS {
1277                        tokio::time::sleep(RETRY_INTERVAL).await;
1278                    } else {
1279                        error!(
1280                            "All prepare_signing attempts exhausted for epoch {target_epoch}. \
1281                             Node cannot sign until next recovery trigger."
1282                        );
1283                    }
1284                }
1285            }
1286        }
1287        drop(_prepare_signing_timer);
1288        drop(_reconfig_timer);
1289    }

1354    async fn submit_end_reconfig(&self, epoch: u64, output: &MpcOutput) -> anyhow::Result<()> {
1355        let mpc_public_key =
1356            bcs::to_bytes(&output.public_key).expect("public key serialization should succeed");
1357        let target_committee = self
1358            .inner
1359            .onchain_state()
1360            .state()
1361            .hashi()
1362            .committees
1363            .committees()
1364            .get(&epoch)
1365            .cloned()
1366            .ok_or_else(|| anyhow::anyhow!("no committee found for epoch {}", epoch))?;
1367        let message = ReconfigCompletionMessage {
1368            epoch,
1369            mpc_public_key: mpc_public_key.clone(),
1370        };
1371        let my_address = self.inner.config.validator_address()?;
1372        let signing_key =
1373            self.inner
1374                .find_signing_key_for_committee(&target_committee, my_address, epoch)?;
1375        let my_sig = signing_key.sign(epoch, my_address, &message);
1376        self.inner
1377            .store_reconfig_signature(epoch, my_sig.signature().as_bytes().to_vec());
1378        let cert = loop {
1379            if self.get_pending_epoch_change() != Some(epoch) {
1380                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
1381            }
1382            match self
1383                .collect_reconfig_signatures(epoch, &mpc_public_key, &target_committee)
1384                .await
1385            {
1386                Ok(cert) => break cert,
1387                Err(e) => {
1388                    warn!(
1389                        "Signature collection for epoch {} failed: {e}, retrying...",
1390                        epoch
1391                    );
1392                    self.sleep_if_still_pending(epoch).await;
1393                }
1394            }
1395        };
1396        self.submit_committee_handoff_if_needed(epoch).await?;
1397        loop {
1398            if self.get_pending_epoch_change() != Some(epoch) {
1399                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
1400            }
1401            let result = async {
1402                let mut executor =
1403                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
1404                executor
1405                    .execute_end_reconfig(&mpc_public_key, cert.committee_signature())
1406                    .await
1407            };
1408            match result.await {
1409                Ok(()) => return Ok(()),
1410                Err(e) => match classify_reconfig_submission_error(&e) {
1411                    ReconfigSubmissionErrorKind::EndReconfigAlreadyCompleted => {
1412                        warn!(
1413                            "end_reconfig submission for epoch {epoch} found reconfig already completed; rescraping on-chain state: {e}"
1414                        );
1415                        self.inner.onchain_state().rescrape().await?;
1416                        if self.get_pending_epoch_change() != Some(epoch) {
1417                            return Ok(());
1418                        }
1419                        Err(e).with_context(|| {
1420                            format!(
1421                                "end_reconfig submission for epoch {epoch} failed with ENotReconfiguring, but epoch is still pending after rescrape"
1422                            )
1423                        })?;
1424                    }
1425                    ReconfigSubmissionErrorKind::NonRetryableMoveAbort
1426                    | ReconfigSubmissionErrorKind::CommitteeHandoffAlreadySubmitted => {
1427                        Err(e).with_context(|| {
1428                            format!(
1429                                "end_reconfig submission for epoch {epoch} failed with non-retryable error"
1430                            )
1431                        })?;
1432                    }
1433                    ReconfigSubmissionErrorKind::NonMoveAbort => {
1434                        warn!(
1435                            "end_reconfig submission for epoch {} failed: {e}, retrying...",
1436                            epoch
1437                        );
1438                        self.sleep_if_still_pending(epoch).await;
1439                    }
1440                },
1441            }
1442        }
1443    }

1445    async fn submit_committee_handoff_if_needed(&self, epoch: u64) -> anyhow::Result<()> {
1446        let from_epoch = self.inner.onchain_state().epoch();
1447        let requires_committee_handoff = !self
1448            .inner
1449            .onchain_state()
1450            .state()
1451            .hashi()
1452            .committees
1453            .mpc_public_key()
1454            .is_empty();
1455        if !requires_committee_handoff {
1456            return Ok(());
1457        }
1458
1459        let committee_handoff = loop {
1460            if self.get_pending_epoch_change() != Some(epoch) {
1461                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
1462            }
1463            match crate::leader::LeaderService::collect_committee_transition_signatures(
1464                &self.inner,
1465                from_epoch,
1466            )
1467            .await
1468            {
1469                Ok(handoff) => break handoff,
1470                Err(e) => {
1471                    warn!(
1472                        from_epoch,
1473                        "Committee handoff signature collection failed: {e}, retrying..."
1474                    );
1475                    self.sleep_if_still_pending(epoch).await;
1476                }
1477            }
1478        };
1479        loop {
1480            if self.get_pending_epoch_change() != Some(epoch) {
1481                return Err(anyhow::anyhow!("epoch {} no longer pending", epoch));
1482            }
1483            let result = async {
1484                let mut executor =
1485                    crate::sui_tx_executor::SuiTxExecutor::from_hashi(self.inner.clone())?;
1486                executor
1487                    .execute_submit_committee_handoff(committee_handoff.committee_signature())
1488                    .await
1489            };
1490            match result.await {
1491                Ok(()) => return Ok(()),
1492                Err(e) => match classify_reconfig_submission_error(&e) {
1493                    ReconfigSubmissionErrorKind::CommitteeHandoffAlreadySubmitted => {
1494                        warn!(
1495                            "submit_committee_handoff submission for epoch {epoch} found handoff already submitted: {e}"
1496                        );
1497                        return Ok(());
1498                    }
1499                    ReconfigSubmissionErrorKind::NonRetryableMoveAbort
1500                    | ReconfigSubmissionErrorKind::EndReconfigAlreadyCompleted => {
1501                        Err(e).with_context(|| {
1502                            format!(
1503                                "submit_committee_handoff submission for epoch {epoch} failed with non-retryable error"
1504                            )
1505                        })?;
1506                    }
1507                    ReconfigSubmissionErrorKind::NonMoveAbort => {
1508                        warn!(
1509                            "submit_committee_handoff submission for epoch {} failed: {e}, retrying...",
1510                            epoch
1511                        );
1512                        self.sleep_if_still_pending(epoch).await;
1513                    }
1514                },
1515            }
1516        }
1517    }

1519    async fn collect_reconfig_signatures(
1520        &self,
1521        epoch: u64,
1522        mpc_public_key: &[u8],
1523        committee: &Committee,
1524    ) -> anyhow::Result<hashi_types::committee::SignedMessage<ReconfigCompletionMessage>> {
1525        let message = ReconfigCompletionMessage {
1526            epoch,
1527            mpc_public_key: mpc_public_key.to_vec(),
1528        };
1529        let my_address = self.inner.config.validator_address()?;
1530        let my_sig_bytes = self
1531            .inner
1532            .get_reconfig_signature(epoch)
1533            .expect("own signature must be stored before collecting");
1534        let my_sig =
1535            BLS12381Signature::from_bytes(&my_sig_bytes).expect("stored signature must be valid");
1536        let mut aggregator = BlsSignatureAggregator::new(committee, message.clone());
1537        aggregator
1538            .add_signature_from(my_address, my_sig)
1539            .map_err(|e| anyhow::anyhow!("failed to add own signature: {e}"))?;
1540        let required_weight = certificate_threshold(committee.total_weight());
1541        while aggregator.weight() < required_weight {
1542            if self.get_pending_epoch_change() != Some(epoch) {
1543                return Err(anyhow::anyhow!(
1544                    "epoch {epoch} no longer pending during signature collection"
1545                ));
1546            }
1547            let other_members: Vec<_> = committee
1548                .members()
1549                .iter()
1550                .filter(|m| m.validator_address() != my_address)
1551                .collect();
1552            let futures = other_members.iter().map(|member| {
1553                let address = member.validator_address();
1554                async move {
1555                    let result = tokio::time::timeout(RPC_TIMEOUT, async {
1556                        let client = self
1557                            .inner
1558                            .onchain_state()
1559                            .state()
1560                            .hashi()
1561                            .committees
1562                            .client(&address)
1563                            .ok_or_else(|| anyhow::anyhow!("client not found for {}", address))?;
1564                        client
1565                            .get_reconfig_completion_signature(epoch)
1566                            .await
1567                            .map_err(|e| anyhow::anyhow!("RPC failed: {e}"))
1568                    })
1569                    .await
1570                    .unwrap_or_else(|_| Err(anyhow::anyhow!("RPC timed out")));
1571                    (
1572                        address,
1573                        result.and_then(|opt| {
1574                            opt.ok_or_else(|| anyhow::anyhow!("signature not ready"))
1575                        }),
1576                    )
1577                }
1578            });
1579            let results = join_all(futures).await;
1580            for (address, result) in results {
1581                if let Ok(sig_bytes) = result {
1582                    match BLS12381Signature::from_bytes(&sig_bytes) {
1583                        Ok(sig) => {
1584                            if let Err(e) = aggregator.add_signature_from(address, sig) {
1585                                info!("Signature from {} rejected: {e}", address);
1586                            }
1587                        }
1588                        Err(e) => {
1589                            info!("Invalid signature bytes from {}: {e}", address);
1590                        }
1591                    }
1592                }
1593            }
1594            if aggregator.weight() < required_weight {
1595                tokio::time::sleep(RETRY_INTERVAL).await;
1596            }
1597        }
1598        aggregator
1599            .finish()
1600            .map_err(|e| anyhow::anyhow!("failed to finalize certificate: {e}"))
1601    }
1602}

1630enum ReconfigSubmissionErrorKind {
1631    NonMoveAbort,
1632    NonRetryableMoveAbort,
1633    CommitteeHandoffAlreadySubmitted,
1634    EndReconfigAlreadyCompleted,
1635}
1636
1637fn classify_reconfig_submission_error(err: &anyhow::Error) -> ReconfigSubmissionErrorKind {
1638    let Some(tx_err) = err.downcast_ref::<crate::sui_tx_executor::TransactionExecutionError>()
1639    else {
1640        return ReconfigSubmissionErrorKind::NonMoveAbort;
1641    };
1642
1643    let Some(error) = tx_err.status().error_opt() else {
1644        return ReconfigSubmissionErrorKind::NonMoveAbort;
1645    };
1646    if error
1647        .kind
1648        .and_then(|kind| ExecutionErrorKind::try_from(kind).ok())
1649        != Some(ExecutionErrorKind::MoveAbort)
1650    {
1651        return ReconfigSubmissionErrorKind::NonMoveAbort;
1652    }
1653
1654    let Some(abort) = error.abort_opt() else {
1655        return ReconfigSubmissionErrorKind::NonRetryableMoveAbort;
1656    };
1657    let location = abort.location();
1658
1659    let abort_constant_name = abort
1660        .clever_error
1661        .as_ref()
1662        .and_then(|clever| clever.constant_name.as_deref());
1663    match (
1664        location.module_opt(),
1665        location.function_name_opt(),
1666        abort_constant_name,
1667    ) {
1668        (Some("committee_set"), Some("set_pending_committee_handoff_cert"), _) => {
1669            ReconfigSubmissionErrorKind::CommitteeHandoffAlreadySubmitted
1670        }
1671        (Some("reconfig"), Some("end_reconfig"), Some(RECONFIG_E_NOT_RECONFIGURING)) => {
1672            ReconfigSubmissionErrorKind::EndReconfigAlreadyCompleted
1673        }
1674        _ => ReconfigSubmissionErrorKind::NonRetryableMoveAbort,
1675    }
1676}
