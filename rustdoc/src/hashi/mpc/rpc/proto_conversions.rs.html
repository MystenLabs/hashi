20/// Get a required field from a proto message.
21#[allow(clippy::result_large_err)]
22fn required<T>(field: Option<T>, name: &str) -> Result<T, TryFromProtoError> {
23    field.ok_or_else(|| TryFromProtoError::missing(name))
24}
25
26/// Parse an address from a hex string.
27#[allow(clippy::result_large_err)]
28fn parse_address(s: &str, field: &str) -> Result<Address, TryFromProtoError> {
29    s.parse::<Address>()
30        .map_err(|e| TryFromProtoError::invalid(field, e))
31}
32
33/// Deserialize a BCS-encoded proto field.
34#[allow(clippy::result_large_err)]
35fn deserialize_bcs<'de, T: Deserialize<'de>>(
36    bcs: &'de Bcs,
37    field: &str,
38) -> Result<T, TryFromProtoError> {
39    bcs.deserialize()
40        .map_err(|e| TryFromProtoError::invalid(field, e))
41}
42
43/// Serialize a value to BCS with a type name.
44fn serialize_bcs<T: Serialize>(value: &T) -> Bcs {
45    Bcs::serialize(value)
46        .expect("serialization should succeed")
47        .with_name(std::any::type_name::<T>())
48}

50/// Parse a share index map from proto.
51#[allow(clippy::result_large_err)]
52fn parse_rotation_messages_map(
53    map: &std::collections::HashMap<u32, Bcs>,
54) -> Result<BTreeMap<ShareIndex, avss::Message>, TryFromProtoError> {
55    let mut messages = BTreeMap::new();
56    for (&index, bcs) in map {
57        let share_index = ShareIndex::new(index as u16).ok_or_else(|| {
58            TryFromProtoError::invalid("rotation_messages.key", "index must be non-zero")
59        })?;
60        let message: avss::Message = deserialize_bcs(bcs, "rotation_messages.value")?;
61        messages.insert(share_index, message);
62    }
63    Ok(messages)
64}
65
66/// Convert rotation messages map to proto format.
67fn rotation_messages_to_proto(
68    messages: &BTreeMap<ShareIndex, avss::Message>,
69) -> std::collections::HashMap<u32, Bcs> {
70    messages
71        .iter()
72        .map(|(idx, msg)| (idx.get() as u32, serialize_bcs(msg)))
73        .collect()
74}

76/// Convert a domain AVID nonce message to its grouped proto form.
77fn avid_nonce_message_to_proto(avid: &types::AvidNonceMessage) -> proto::AvidNonceMessage {
78    use proto::avid_nonce_message::Kind;
79    let kind = match &avid.kind {
80        types::AvidNonceMessageKind::Optimistic(message) => {
81            Kind::Optimistic(proto::AvidNonceOptimistic {
82                message: Some(serialize_bcs(message)),
83            })
84        }
85        types::AvidNonceMessageKind::Dispersal {
86            dispersal,
87            confirm_cert,
88        } => Kind::Dispersal(proto::AvidNonceDispersal {
89            dispersal: Some(serialize_bcs(dispersal)),
90            confirm_cert: Some(serialize_bcs(confirm_cert)),
91        }),
92        types::AvidNonceMessageKind::Echo { dealer, echo } => Kind::Echo(proto::AvidNonceEcho {
93            dealer: Some(dealer.to_string()),
94            echo: Some(serialize_bcs(echo)),
95        }),
96    };
97    proto::AvidNonceMessage {
98        batch_index: Some(avid.batch_index),
99        kind: Some(kind),
100    }
101}

103/// Parse a grouped proto AVID nonce message into its domain form.
104#[allow(clippy::result_large_err)]
105fn avid_nonce_message_from_proto(
106    avid: &proto::AvidNonceMessage,
107) -> Result<types::AvidNonceMessage, TryFromProtoError> {
108    use proto::avid_nonce_message::Kind;
109    let batch_index = required(avid.batch_index, "avid_nonce_message.batch_index")?;
110    let kind = match avid.kind.as_ref() {
111        Some(Kind::Optimistic(optimistic)) => {
112            let message: batch_avss_avid::AvssMessage = deserialize_bcs(
113                required(
114                    optimistic.message.as_ref(),
115                    "avid_nonce_message.optimistic.message",
116                )?,
117                "avid_nonce_message.optimistic.message",
118            )?;
119            types::AvidNonceMessageKind::Optimistic(message)
120        }
121        Some(Kind::Dispersal(dispersal_msg)) => {
122            let dispersal: batch_avss_avid::Dispersal = deserialize_bcs(
123                required(
124                    dispersal_msg.dispersal.as_ref(),
125                    "avid_nonce_message.dispersal.dispersal",
126                )?,
127                "avid_nonce_message.dispersal.dispersal",
128            )?;
129            let confirm_cert: types::AvidConfirmCertificate = deserialize_bcs(
130                required(
131                    dispersal_msg.confirm_cert.as_ref(),
132                    "avid_nonce_message.dispersal.confirm_cert",
133                )?,
134                "avid_nonce_message.dispersal.confirm_cert",
135            )?;
136            types::AvidNonceMessageKind::Dispersal {
137                dispersal,
138                confirm_cert,
139            }
140        }
141        Some(Kind::Echo(echo_msg)) => {
142            let dealer = parse_address(
143                required(echo_msg.dealer.as_ref(), "avid_nonce_message.echo.dealer")?,
144                "avid_nonce_message.echo.dealer",
145            )?;
146            let echo: batch_avss_avid::Echo = deserialize_bcs(
147                required(echo_msg.echo.as_ref(), "avid_nonce_message.echo.echo")?,
148                "avid_nonce_message.echo.echo",
149            )?;
150            types::AvidNonceMessageKind::Echo { dealer, echo }
151        }
152        None => return Err(TryFromProtoError::missing("avid_nonce_message.kind")),
153    };
154    Ok(types::AvidNonceMessage { batch_index, kind })
155}

161impl types::SendMessagesRequest {
162    pub fn to_proto(&self, epoch: u64) -> proto::SendMessagesRequest {
163        use proto::send_messages_request::Messages;
164        let messages = match &self.messages {
165            types::Messages::Dkg(message) => Messages::DkgMessage(serialize_bcs(message)),
166            types::Messages::Rotation(messages) => {
167                Messages::RotationMessages(proto::RotationMessages {
168                    messages: rotation_messages_to_proto(messages),
169                })
170            }
171            types::Messages::NonceGeneration(nonce) => {
172                Messages::NonceMessage(proto::NonceMessage {
173                    batch_index: Some(nonce.batch_index),
174                    message: Some(serialize_bcs(&nonce.message)),
175                })
176            }
177            types::Messages::NonceGenerationAvid(avid) => {
178                Messages::AvidNonceMessage(avid_nonce_message_to_proto(avid))
179            }
180        };
181        proto::SendMessagesRequest {
182            epoch: Some(epoch),
183            messages: Some(messages),
184        }
185    }
186}

188impl TryFrom<&proto::SendMessagesRequest> for types::SendMessagesRequest {
189    type Error = TryFromProtoError;
190
191    fn try_from(value: &proto::SendMessagesRequest) -> Result<Self, Self::Error> {
192        use proto::send_messages_request::Messages;
193        let messages = match &value.messages {
194            Some(Messages::DkgMessage(dkg_message)) => {
195                let message: avss::Message = deserialize_bcs(dkg_message, "dkg_message")?;
196                types::Messages::Dkg(message)
197            }
198            Some(Messages::RotationMessages(rotation)) => {
199                types::Messages::Rotation(parse_rotation_messages_map(&rotation.messages)?)
200            }
201            Some(Messages::NonceMessage(nonce)) => {
202                let batch_index = required(nonce.batch_index, "nonce_message.batch_index")?;
203                let message: batch_avss::Message = deserialize_bcs(
204                    required(nonce.message.as_ref(), "nonce_message.message")?,
205                    "nonce_message.message",
206                )?;
207                types::Messages::NonceGeneration(types::NonceMessage {
208                    batch_index,
209                    message,
210                })
211            }
212            Some(Messages::AvidNonceMessage(avid)) => {
213                types::Messages::NonceGenerationAvid(avid_nonce_message_from_proto(avid)?)
214            }
215            None => {
216                return Err(TryFromProtoError::missing("messages"));
217            }
218        };
219        Ok(Self { messages })
220    }
221}

280fn mpc_protocol_type_to_proto(pt: types::ProtocolTypeIndicator) -> proto::MpcProtocolType {
281    match pt {
282        types::ProtocolTypeIndicator::Dkg => proto::MpcProtocolType::Dkg,
283        types::ProtocolTypeIndicator::KeyRotation => proto::MpcProtocolType::KeyRotation,
284        types::ProtocolTypeIndicator::NonceGeneration => proto::MpcProtocolType::NonceGeneration,
285    }

288#[allow(clippy::result_large_err)]
289fn mpc_protocol_type_from_proto(
290    value: i32,
291) -> Result<types::ProtocolTypeIndicator, TryFromProtoError> {
292    match proto::MpcProtocolType::try_from(value) {
293        Ok(proto::MpcProtocolType::Dkg) => Ok(types::ProtocolTypeIndicator::Dkg),
294        Ok(proto::MpcProtocolType::KeyRotation) => Ok(types::ProtocolTypeIndicator::KeyRotation),
295        Ok(proto::MpcProtocolType::NonceGeneration) => {
296            Ok(types::ProtocolTypeIndicator::NonceGeneration)
297        }
298        _ => Err(TryFromProtoError::missing("valid protocol_type")),
299    }

374impl types::ComplainRequest {
375    pub fn to_proto(&self) -> proto::ComplainRequest {
376        proto::ComplainRequest {
377            epoch: Some(self.epoch),
378            dealer: Some(self.dealer.to_string()),
379            share_index: self.share_index.map(|idx| idx.get() as u32),
380            complaint: Some(match &self.complaint {
381                types::ProtocolComplaint::Avss(c) => serialize_bcs(c),
382                types::ProtocolComplaint::BatchedAvss(c) => serialize_bcs(c),
383            }),
384            protocol_type: Some(mpc_protocol_type_to_proto(self.protocol_type) as i32),
385            batch_index: self.batch_index,
386        }
387    }
388}
389
390impl TryFrom<&proto::ComplainRequest> for types::ComplainRequest {
391    type Error = TryFromProtoError;
392
393    fn try_from(value: &proto::ComplainRequest) -> Result<Self, Self::Error> {
394        let epoch = *required(value.epoch.as_ref(), "epoch")?;
395        let dealer = parse_address(required(value.dealer.as_ref(), "dealer")?, "dealer")?;
396        let share_index = if let Some(idx) = value.share_index {
397            Some(
398                std::num::NonZeroU16::new(idx as u16)
399                    .ok_or_else(|| TryFromProtoError::invalid("share_index", "must be non-zero"))?,
400            )
401        } else {
402            None
403        };
404        let protocol_type =
405            mpc_protocol_type_from_proto(required(value.protocol_type, "protocol_type")?)?;
406        let complaint_bytes = required(value.complaint.as_ref(), "complaint")?;
407        let complaint = match protocol_type {
408            types::ProtocolTypeIndicator::Dkg | types::ProtocolTypeIndicator::KeyRotation => {
409                types::ProtocolComplaint::Avss(deserialize_bcs(complaint_bytes, "complaint")?)
410            }
411            types::ProtocolTypeIndicator::NonceGeneration => types::ProtocolComplaint::BatchedAvss(
412                deserialize_bcs(complaint_bytes, "complaint")?,
413            ),
414        };
415        Ok(Self {
416            dealer,
417            share_index,
418            batch_index: value.batch_index,
419            complaint,
420            protocol_type,
421            epoch,
422        })
423    }
424}

515impl TryFrom<&proto::GetPublicMpcOutputResponse> for types::GetPublicMpcOutputResponse {
516    type Error = TryFromProtoError;
517
518    fn try_from(value: &proto::GetPublicMpcOutputResponse) -> Result<Self, Self::Error> {
519        use fastcrypto_tbls::threshold_schnorr::G;
520
521        let public_key = deserialize_bcs(
522            required(value.public_key.as_ref(), "public_key")?,
523            "public_key",
524        )?;
525        let mut commitments = BTreeMap::new();
526        for (&index, bcs) in &value.commitments {
527            let share_index = ShareIndex::new(index as u16).ok_or_else(|| {
528                TryFromProtoError::invalid("commitments.key", "index must be non-zero")
529            })?;
530            let commitment_value: G = deserialize_bcs(bcs, "commitments.value")?;
531            commitments.insert(share_index, commitment_value);
532        }
533        Ok(Self {
534            output: types::PublicMpcOutput {
535                public_key,
536                commitments,
537            },
538        })
539    }

643        batch_avss_avid::Echo,
644    ) {
645        let (t, f, n, batch) = (3u16, 3u16, 10u16, 3u16);
646        let mut rng = rand::thread_rng();
647        let sks: Vec<_> = (0..n)
648            .map(|_| ecies_v1::PrivateKey::<types::EncryptionGroupElement>::new(&mut rng))
649            .collect();
650        let nodes = Nodes::new(
651            sks.iter()
652                .enumerate()
653                .map(|(id, sk)| Node {
654                    id: id as u16,
655                    pk: ecies_v1::PublicKey::from_private_key(sk),
656                    weight: 1,
657                })
658                .collect(),
659        )
660        .unwrap();
661        let sid = b"avid conversion test".to_vec();
662        let params = Parameters { t, f };
663        let dealer =
664            batch_avss_avid::Dealer::new(nodes.clone(), 0, params, sid.clone(), batch).unwrap();
665        let (state, optimistic) = dealer.create_avss_messages(&mut rng).unwrap();
666        let cert = StubAvssCert {
667            voters: (0u16..=7).collect(),
668            vote: batch_avss_avid::AvssVote {
669                common_message_hash: state.common.hash(),
670            },
671        };
672        let messages = dealer.create_avid_messages(&state, cert).unwrap();
673        // Receiver 0 processes the message addressed to it (its shards verify against its
674        // own Merkle leaf), then echoes for a pending recipient.
675        let pending: PartyId = 8;
676        let avid_message = messages.message_for(0).unwrap();
677        let dispersal = avid_message.dispersal.clone();
678        let receiver =
679            batch_avss_avid::Receiver::new(nodes, 0, 0, params, sid, sks[0].clone(), batch)
680                .unwrap();
681        let verified_common = receiver
682            .verify_common_message(state.common.clone())
683            .unwrap();
684        let (echo_builder, _) = receiver
685            .process_avid_message(Some(&verified_common), avid_message)
686            .unwrap()
687            .unwrap();
688        let echo = echo_builder.create_echo(pending).unwrap();
689        (optimistic.get(&0).unwrap().clone(), dispersal, echo)
690    }

692    fn confirm_cert() -> types::AvidConfirmCertificate {
693        // Cert contents are arbitrary here — the round-trip test only checks BCS serialization.
694        let message = types::DealerMessagesHash {
695            dealer_address: Address::new([0u8; 32]),
696            messages_hash: Blake2b256::digest(b"confirm cert").digest.into(),
697        };
698        let signature = BLS12381AggregateSignature::default();
699        SignedMessage::new(7, message, signature.as_bytes(), &[1u8]).unwrap()
700    }

703        let request = types::SendMessagesRequest { messages };

706        assert_eq!(
