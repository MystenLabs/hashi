52/// Build a one-certificate-per-KP roster while retaining the matching secret
53/// keys so tests can prove the returned armored shares are decryptable.
54#[cfg(test)]
55pub fn mock_kp_certs_roster_with_secrets(num_kps: usize) -> (KpCertsRoster, MockKpSecretKeys) {
56    let mut secret_keys = MockKpSecretKeys::new();
57    let cert_sets = (0..num_kps)
58        .map(|_| {
59            let (public, secret) = hashi_types::pgp::test_utils::mock_pgp_keypair();
60            let cert = PgpPublicCert::new(public).expect("mock public cert should parse");
61            let fingerprint = cert.fingerprint().to_hex();
62            assert!(
63                secret_keys.insert(fingerprint, secret).is_none(),
64                "mock PGP fingerprints should be unique"
65            );
66            KpCerts::new(vec![cert]).expect("one mock cert forms a valid KP cert set")
67        })
68        .collect();
69    (
70        KpCertsRoster::new(cert_sets).expect("mock cert sets form a valid roster"),
71        secret_keys,
72    )
73}

75/// Decrypt every ciphertext in a KP-share roster and return one share per ID.
76/// If a share has multiple recipient certs, all ciphertexts must decrypt to the
77/// same scalar.
78#[cfg(test)]
79pub fn decrypt_kp_shares(
80    encrypted_shares: &KPEncryptedSharesRoster,
81    secret_keys: &MockKpSecretKeys,
82) -> Vec<Share> {
83    encrypted_shares
84        .iter()
85        .map(|encrypted_share| {
86            let mut share_value = None;
87            for (fingerprint, ciphertext) in &encrypted_share.ciphertexts_by_fingerprint {
88                let secret_key = secret_keys
89                    .get(fingerprint)
90                    .expect("every ciphertext should have a matching mock secret key");
91                let mut decryptor = decrypt_with_secret_key(
92                    std::io::Cursor::new(ciphertext.clone().into_bytes()),
93                    secret_key.as_bytes(),
94                )
95                .expect("mock KP should decrypt its armored share");
96                let mut plaintext = Vec::new();
97                decryptor
98                    .read_to_end(&mut plaintext)
99                    .expect("decrypted share should be readable");
100                let value = ScalarPrimitive::<K256Secp256k1>::from_slice(&plaintext)
101                    .map(k256::Scalar::from)
102                    .expect("decrypted share should be a valid scalar");
103                if let Some(expected) = share_value {
104                    assert_eq!(
105                        value, expected,
106                        "all recipients must receive the same share"
107                    );
108                } else {
109                    share_value = Some(value);
110                }
111            }
112            Share {
113                id: encrypted_share.id,
114                value: share_value.expect("each encrypted share should have a recipient"),
115            }
116        })
117        .collect()
118}

120/// Mock S3 logger that captures every PutObject's (key, body) into the returned
121/// Vec. Lets tests assert on what was written. Body is captured via `match_requests`
122/// (same Mutex side-channel trick as `mock_logger_with_layout`).
123pub fn mock_logger_capturing() -> (GuardianS3Client, CapturedPuts) {
124    use aws_sdk_s3::operation::put_object::PutObjectOutput;
125    use aws_sdk_s3::Client;
126    use aws_smithy_mocks::mock;
127    use aws_smithy_mocks::mock_client;
128    use aws_smithy_mocks::RuleMode;
129    use hashi_types::guardian::S3Config;
130
131    let captures: CapturedPuts = Arc::new(std::sync::Mutex::new(Vec::new()));
132    let captures_w = captures.clone();
133
134    let put_ok = mock!(Client::put_object)
135        .match_requests(move |req| {
136            let key = req.key().expect("put_object missing key").to_string();
137            let body = req
138                .body()
139                .bytes()
140                .expect("body should be in-memory in tests")
141                .to_vec();
142            captures_w.lock().unwrap().push((key, body));
143            true
144        })
145        .then_output(|| PutObjectOutput::builder().build());
146    let client = mock_client!(aws_sdk_s3, RuleMode::MatchAny, &[&put_ok]);
147    let logger = GuardianS3Client::from_client_for_tests(S3Config::mock_for_testing(), client);
148    (logger, captures)
149}

151/// Mock S3 logger whose `list_objects_v2(delimiter='/')` and
152/// `list_object_versions` responses are computed from an in-memory key set —
153/// useful for testing layered prefix tree-walks. PutObject also succeeds.
154///
155/// The dynamic responses depend on inspecting the request `prefix`; we capture
156/// it in a Mutex from `match_requests` and read it in `then_output` (the
157/// smithy-mocks API doesn't surface the request inside `then_output`). This
158/// is sound under a single-threaded async runtime — each S3 call's predicate
159/// runs immediately before its output factory.
160pub fn mock_logger_with_layout(keys: impl IntoIterator<Item = String>) -> GuardianS3Client {
161    use aws_sdk_s3::operation::list_object_versions::ListObjectVersionsOutput;
162    use aws_sdk_s3::operation::list_objects_v2::ListObjectsV2Output;
163    use aws_sdk_s3::operation::put_object::PutObjectOutput;
164    use aws_sdk_s3::types::CommonPrefix;
165    use aws_sdk_s3::types::ObjectVersion;
166    use aws_sdk_s3::Client;
167    use aws_smithy_mocks::mock;
168    use aws_smithy_mocks::mock_client;
169    use aws_smithy_mocks::RuleMode;
170    use hashi_types::guardian::S3Config;
171    use std::collections::BTreeSet;
172    use std::sync::Arc;
173    use std::sync::Mutex;
174
175    let keys: Arc<BTreeSet<String>> = Arc::new(keys.into_iter().collect());
176
177    let v2_prefix: Arc<Mutex<Option<String>>> = Arc::new(Mutex::new(None));
178    let v2_prefix_w = v2_prefix.clone();
179    let v2_prefix_r = v2_prefix.clone();
180    let v2_keys = keys.clone();
181    let list_v2 = mock!(Client::list_objects_v2)
182        .match_requests(move |req| {
183            if req.delimiter() != Some("/") {
184                return false;
185            }
186            *v2_prefix_w.lock().unwrap() = req.prefix().map(|s| s.to_string());
187            true
188        })
189        .then_output(move || {
190            let prefix = v2_prefix_r.lock().unwrap().clone().unwrap_or_default();
191            let mut children: BTreeSet<String> = BTreeSet::new();
192            for key in v2_keys.iter() {
193                let Some(rest) = key.strip_prefix(&prefix) else {
194                    continue;
195                };
196                if let Some(slash) = rest.find('/') {
197                    let mut child = prefix.clone();
198                    child.push_str(&rest[..=slash]);
199                    children.insert(child);
200                }
201            }
202            let common_prefixes: Vec<CommonPrefix> = children
203                .into_iter()
204                .map(|c| CommonPrefix::builder().prefix(c).build())
205                .collect();
206            ListObjectsV2Output::builder()
207                .set_common_prefixes(Some(common_prefixes))
208                .build()
209        });
210
211    let lv_prefix: Arc<Mutex<Option<String>>> = Arc::new(Mutex::new(None));
212    let lv_prefix_w = lv_prefix.clone();
213    let lv_prefix_r = lv_prefix.clone();
214    let lv_keys = keys.clone();
215    let list_versions = mock!(Client::list_object_versions)
216        .match_requests(move |req| {
217            *lv_prefix_w.lock().unwrap() = req.prefix().map(|s| s.to_string());
218            true
219        })
220        .then_output(move || {
221            let prefix = lv_prefix_r.lock().unwrap().clone().unwrap_or_default();
222            let versions: Vec<ObjectVersion> = lv_keys
223                .iter()
224                .filter(|k| k.starts_with(&prefix))
225                .map(|k| ObjectVersion::builder().key(k).is_latest(true).build())
226                .collect();
227            ListObjectVersionsOutput::builder()
228                .set_versions(Some(versions))
229                .build()
230        });
231
232    let put_ok = mock!(Client::put_object).then_output(|| PutObjectOutput::builder().build());
233
234    let client = mock_client!(
235        aws_sdk_s3,
236        RuleMode::MatchAny,
237        &[&list_v2, &list_versions, &put_ok]
238    );
239    GuardianS3Client::from_client_for_tests(S3Config::mock_for_testing(), client)
240}

255fn dummy_secret_sharing_instance() -> SecretSharingInstance {
256    let params = SecretSharingParams::new(TEST_N, TEST_T).unwrap();

356    /// Apply operator_init's installs to an existing enclave (mirrors `operator_init`'s
357    /// withdraw-mode commit). Lets a harness defer operator-init until DKG output exists.
358    pub fn install_operator_init_for_testing(&self, args: OperatorInitTestArgs) {
359        self.config.set_s3_logger(args.s3_logger).unwrap();
360        crate::operator_init::OIWithdrawModeInstall::from_parts(
361            args.config,
362            args.ceremony_state,
363            args.genesis_state,
364        )
365        .install_into(self);
366        self.advance_lifecycle_into(WithdrawStage::OperatorInitialized.into())
367            .expect("operator init test setup should advance lifecycle");
368    }

371        let enclave = Self::create_with_random_keys_for_mode(EnclaveMode::Ceremony);
372        enclave.config.set_s3_logger(s3_logger).unwrap();
373        enclave
374            .advance_lifecycle_into(CeremonyStage::OperatorInitialized.into())
375            .expect("ceremony operator init test setup should advance lifecycle");
376        enclave
377    }

392/// Generate and set a fresh BTC keypair on an operator-init'd enclave.
393/// Returns the x-only pubkey so callers can publish it on-chain before
394/// the rest of provisioner-init has run (i.e. before DKG completes and
395/// `finalize_enclave` can be called). Idempotent: returns the existing
396/// pubkey if the keypair has already been set.
397pub fn set_or_get_enclave_btc_pubkey(enclave: &Arc<Enclave>) -> GuardianResult<BitcoinPubkey> {
398    if let Ok(pk) = enclave.config.enclave_btc_pubkey() {
399        return Ok(pk);
400    }
401    let secp = Secp256k1::new();
402    let mut sk_bytes = [0u8; 32];
403    rand::thread_rng().fill_bytes(&mut sk_bytes);
404    let enclave_btc_keypair = Keypair::from_secret_key(
405        &secp,
406        &SecretKey::from_slice(&sk_bytes).expect("random bytes form a valid secp256k1 key"),
407    );
408    let pk = enclave_btc_keypair.x_only_public_key().0;
409    enclave.config.set_btc_keypair(enclave_btc_keypair)?;
410    Ok(pk)
411}
412
413/// Drive an operator-initialized enclave through provisioner-init by setting the
414/// BTC keypair. The live serving state is installed separately by OA helpers. The
415/// keypair may already exist from an earlier [`set_or_get_enclave_btc_pubkey`]
416/// (idempotent).
417pub fn finalize_enclave(enclave: &Arc<Enclave>) -> GuardianResult<()> {
418    let _ = set_or_get_enclave_btc_pubkey(enclave)?;
419    enclave.advance_lifecycle_into(WithdrawStage::ProvisionerInitialized.into())?;
420    Ok(())
421}

423/// Install activation-derived live state for tests that need normal operation.
424pub fn activate_enclave_for_testing(
425    enclave: &Arc<Enclave>,
426    committee: HashiCommittee,
427    limiter_config: LimiterConfig,
428    limiter_state: LimiterState,
429) -> GuardianResult<()> {
430    let rate_limiter = RateLimiter::new(limiter_config, limiter_state)?;
431
432    enclave.state.init(committee, rate_limiter)?;
433    enclave.clear_temporary_init_state();
434    enclave.advance_lifecycle_into(WithdrawStage::Activated.into())?;
435    Ok(())
436}

438/// Operator-init + finalize in one shot.
439pub async fn create_fully_initialized_enclave(args: FullyInitializedArgs) -> Arc<Enclave> {
440    let FullyInitializedArgs {
441        network,
442        committee,
443        master_pubkey,
444        limiter_config,
445        limiter_state,
446    } = args;
447
448    let config = InitConfig::from_parts_for_testing(limiter_config, master_pubkey, network);
449    let enclave =
450        create_operator_initialized_enclave(OperatorInitTestArgs::default().with_config(config))
451            .await;
452
453    finalize_enclave(&enclave).expect("finalize_enclave should succeed on a fresh enclave");
454    activate_enclave_for_testing(&enclave, committee, limiter_config, limiter_state)
455        .expect("activate_enclave_for_testing should succeed on a fresh enclave");
456
457    assert!(enclave.is_fully_initialized());
458    assert!(enclave.temporary_init_state().is_err());
459    enclave
