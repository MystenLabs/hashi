4//! Core enclave types: `Enclave` holds all guardian state (durable config,
5//! mutable runtime state, and temporary initialization state). Lives in the
6//! library so external crates
7//! (integration test harnesses, ops tooling) can construct and drive an
8//! enclave without going through `main`.

36/// Enclave's config & state
37pub struct Enclave {
38    /// Immutable config (set once during init)
39    pub config: EnclaveConfig,
40    /// Mutable state
41    pub state: EnclaveState,
42    /// Temporary state retained until operator activation commits.
43    temporary_init_state: RwLock<Option<TemporaryInitState>>,
44    /// Serializes lifecycle and control-plane transitions so concurrent
45    /// operations cannot race a check-then-set.
46    control_lock: tokio::sync::Mutex<()>,
47}
48
49/// Configuration set during initialization (immutable after set)
50pub struct EnclaveConfig {
51    /// Ephemeral signing keypair generated at boot.
52    signing_keys: GuardianSignKeyPair,
53    /// Ephemeral encryption keypair generated at boot.
54    encryption_keys: GuardianEncKeyPair,
55    /// S3 client & config (set in operator_init)
56    s3_logger: OnceLock<GuardianS3Client>,
57    /// Enclave BTC private key (set in provisioner_init)
58    enclave_btc_keypair: OnceLock<Keypair>,
59    /// BTC network: mainnet, testnet, regtest (set in operator_init)
60    btc_network: OnceLock<Network>,
61    /// Raw MPC verifying key as a curve point. Stored with y-parity so the
62    /// 2-of-2 child-key derivation matches the MPC's signing protocol.
63    /// Set in operator_init.
64    hashi_btc_master_pubkey: OnceLock<HashiMasterG>,
65    /// Guardian build PCR pins used to verify attested guardian sessions.
66    pcr_allowlist: OnceLock<PcrAllowlist>,
67    /// Operator-supplied limiter configuration.
68    /// Note: This struct is duplicated in two places: `RateLimiter` stores a copy after activation.
69    limiter_config: OnceLock<LimiterConfig>,
70}

72/// Mutable state that changes during operation.
73/// Committee + rate limiter are installed during operator_activate.
74pub struct EnclaveState {
75    /// Authoritative mode-specific lifecycle, initialized at boot.
76    lifecycle: RwLock<EnclaveLifecycle>,
77    /// Current Hashi committee.
78    committee: RwLock<Option<Arc<HashiCommittee>>>,
79    /// Rate limiter. Set once during operator_activate.
80    /// Uses `Arc<tokio::Mutex>` so the guard can be held across `.await`.
81    rate_limiter: OnceLock<Arc<tokio::sync::Mutex<RateLimiter>>>,
82}

137    /// Sign a BTC tx. Returns an Err if enclave btc keypair or hashi btc pk is not set.
138    pub fn btc_sign(&self, tx_utxos: &TxUTXOs) -> GuardianResult<(Txid, Vec<BitcoinSignature>)> {
139        let enclave_keypair = self
140            .enclave_btc_keypair
141            .get()
142            .ok_or(InvalidInputs("Bitcoin key is not initialized".into()))?;
143        let hashi_btc_pk = self
144            .hashi_btc_master_pubkey
145            .get()
146            .ok_or(InvalidInputs("Hashi BTC public key not set".into()))?;
147
148        let enclave_btc_pk = enclave_keypair.x_only_public_key().0;
149        let (messages, txid) = tx_utxos.signing_messages_and_txid(&enclave_btc_pk, hashi_btc_pk);
150        Ok((txid, sign_btc_tx(&messages, enclave_keypair)))
151    }

154        self.enclave_btc_keypair.get().is_some()

179    /// Install the activation-derived committee + rate limiter. Called from operator_activate.
180    pub fn init(&self, committee: HashiCommittee, rate_limiter: RateLimiter) -> GuardianResult<()> {
181        self.set_committee(committee)?;
182        self.set_rate_limiter(rate_limiter)?;
183        Ok(())

186    // ========================================================================
187    // Committee Management
188    // ========================================================================
189
190    /// Get the current committee.
191    pub fn get_committee(&self) -> GuardianResult<Arc<HashiCommittee>> {
192        let guard = self
193            .committee
194            .read()
195            .expect("rwlock should never throw an error");
196        guard
197            .as_ref()
198            .cloned()
199            .ok_or_else(|| InvalidInputs("committee not initialized".into()))
200    }
201
202    /// Whether the committee is installed, without cloning the `Arc`.
203    fn has_committee(&self) -> bool {
204        self.committee
205            .read()
206            .expect("rwlock should never throw an error")
207            .is_some()
208    }
209
210    /// Set committee. Called only from `init` during operator activation.
211    fn set_committee(&self, committee: HashiCommittee) -> GuardianResult<()> {
212        info!("Setting committee for epoch {}.", committee.epoch());
213
214        let mut guard = self
215            .committee
216            .write()
217            .expect("rwlock should never throw an error");
218        if guard.is_some() {
219            return Err(InvalidInputs("committee already initialized".into()));
220        }
221        *guard = Some(Arc::new(committee));
222        Ok(())
223    }
224
225    /// Replace an already-initialized committee. Rejects the swap unless
226    /// the in-memory epoch matches `expected_current_epoch`.
227    pub fn replace_committee(
228        &self,
229        committee: HashiCommittee,
230        expected_current_epoch: u64,
231    ) -> GuardianResult<()> {
232        info!("Replacing committee for epoch {}.", committee.epoch());
233
234        let mut guard = self
235            .committee
236            .write()
237            .expect("rwlock should never throw an error");
238        let current_epoch = guard
239            .as_ref()
240            .ok_or_else(|| InvalidInputs("committee not initialized".into()))?
241            .epoch();
242        if current_epoch != expected_current_epoch {
243            return Err(InvalidInputs(format!(
244                "committee epoch mismatch: expected {expected_current_epoch}, actual {current_epoch}"
245            )));
246        }
247        *guard = Some(Arc::new(committee));
248        Ok(())
249    }
250
251    // ========================================================================
252    // Rate Limiter Management
253    // ========================================================================
254
255    fn set_rate_limiter(&self, limiter: RateLimiter) -> GuardianResult<()> {
256        info!("Setting rate limiter.");
257
258        self.rate_limiter
259            .set(Arc::new(tokio::sync::Mutex::new(limiter)))
260            .map_err(|_| InvalidInputs("rate_limiter already initialized".into()))
261    }
262
263    /// Timeout for acquiring the limiter lock. If a withdrawal is in progress and
264    /// takes longer than this, callers bail rather than wait indefinitely.
265    const LIMITER_LOCK_TIMEOUT: Duration = Duration::from_secs(10);
266
267    /// Acquire exclusive access to the limiter, bounded by
268    /// `LIMITER_LOCK_TIMEOUT`.
269    async fn lock_limiter(&self) -> GuardianResult<OwnedMutexGuard<RateLimiter>> {
270        let rate_limiter = self
271            .rate_limiter
272            .get()
273            .ok_or_else(|| InternalError("rate limiter not initialized".into()))?;
274        tokio::time::timeout(
275            Self::LIMITER_LOCK_TIMEOUT,
276            rate_limiter.clone().lock_owned(),
277        )
278        .await
279        .map_err(|_| Unavailable("timed out waiting for rate limiter lock".into()))
280    }
281
282    /// Acquire exclusive access to the limiter, consume tokens, and return a guard.
283    /// The guard is held through signing and durable logging so no other withdrawal
284    /// can start until this one is durably logged or the enclave aborts.
285    pub async fn consume_from_limiter(
286        &self,
287        seq: u64,
288        timestamp: u64,
289        amount_sats: u64,
290    ) -> GuardianResult<OwnedMutexGuard<RateLimiter>> {
291        let mut guard = self.lock_limiter().await?;
292        guard.consume(seq, timestamp, amount_sats)?;
293        Ok(guard)
294    }
295
296    /// Return the limiter state for status reporting, or `None` if the limiter
297    /// is uninitialized or remains locked through the acquisition deadline.
298    pub async fn limiter_state(&self) -> Option<LimiterState> {
299        Some(*self.lock_limiter().await.ok()?.state())
300    }
301}

332    /// Spawn a root-owned Tokio task so caller cancellation only detaches the
333    /// waiter and does not cancel accepted work. Use this when the task needs
334    /// no serialization or owns a narrower lock for the state it mutates.
335    /// E.g., used in `standard_withdrawal` that owns a separate serialization lock inside.
336    pub async fn spawn_task<Input, Output, Task, Fut>(
337        self: Arc<Self>,
338        input: Input,
339        task: Task,
340    ) -> GuardianResult<Output>
341    where
342        Input: Send + 'static,
343        Output: Send + 'static,
344        Task: FnOnce(Arc<Self>, Input) -> Fut + Send + 'static,
345        Fut: Future<Output = GuardianResult<Output>> + Send + 'static,
346    {
347        tokio::spawn(async move { task(self, input).await })
348            .await
349            .expect("guardian task failed")
350    }
351
352    /// Spawn a root-owned Tokio task that holds the control lock while running.
353    /// Use this for lifecycle and control-plane transitions whose shared-state
354    /// checks and mutations must not interleave with another control task.
355    pub async fn spawn_control_task<Input, Output, Task, Fut>(
356        self: Arc<Self>,
357        input: Input,
358        task: Task,
359    ) -> GuardianResult<Output>
360    where
361        Input: Send + 'static,
362        Output: Send + 'static,
363        Task: FnOnce(Arc<Self>, Input) -> Fut + Send + 'static,
364        Fut: Future<Output = GuardianResult<Output>> + Send + 'static,
365    {
366        self.spawn_task(input, move |enclave, input| async move {
367            let task_enclave = enclave.clone();
368            let _guard = enclave.control_lock.lock().await;
369            task(task_enclave, input).await
370        })
371        .await
372    }

391    /// Require an exact mode and lifecycle stage.
392    pub fn require_lifecycle(&self, expected: EnclaveLifecycle) -> GuardianResult<()> {
393        let actual = self.lifecycle();
394        if actual != expected {
395            return Err(InvalidInputs(format!(
396                "expected enclave lifecycle {expected:?}, got {actual:?}"
397            )));
398        }
399        Ok(())
400    }
401
402    /// Transition after the operation's durable log succeeds. The lifecycle is
403    /// the single source of completion state.
404    pub fn advance_lifecycle_into(&self, next: EnclaveLifecycle) -> GuardianResult<()> {
405        let expected = next
406            .predecessor()
407            .ok_or_else(|| InvalidInputs(format!("cannot advance lifecycle into {next:?}")))?;
408        let mut lifecycle = self
409            .state
410            .lifecycle
411            .write()
412            .expect("lifecycle lock poisoned");
413        if *lifecycle != expected {
414            return Err(InvalidInputs(format!(
415                "expected enclave lifecycle {expected:?}, got {:?}",
416                *lifecycle
417            )));
418        }
419        self.assert_state_installed_for(next);
420        *lifecycle = next;
421        Ok(())
422    }
423
424    fn assert_state_installed_for(&self, next: EnclaveLifecycle) {
425        let installed = match next {
426            EnclaveLifecycle::Ceremony(CeremonyStage::Uninitialized)
427            | EnclaveLifecycle::Withdraw(WithdrawStage::Uninitialized) => {
428                unreachable!("cannot advance lifecycle into {next:?}")
429            }
430            EnclaveLifecycle::Ceremony(CeremonyStage::OperatorInitialized) => {
431                self.operator_init_state_installed(EnclaveMode::Ceremony)
432            }
433            EnclaveLifecycle::Withdraw(WithdrawStage::OperatorInitialized) => {
434                self.operator_init_state_installed(EnclaveMode::Withdraw)
435            }
436            // Ceremony handlers advance only after writing their output to S3.
437            // The lifecycle itself is the completion state.
438            EnclaveLifecycle::Ceremony(CeremonyStage::Completed) => return,
439            EnclaveLifecycle::Withdraw(WithdrawStage::ProvisionerInitialized) => {
440                self.config.is_enclave_btc_keypair_set() && self.temporary_init_state_is_available()
441            }
442            EnclaveLifecycle::Withdraw(WithdrawStage::Activated) => {
443                self.state.has_committee()
444                    && self.state.rate_limiter.get().is_some()
445                    && !self.temporary_init_state_is_available()
446            }
447        };
448        assert!(
449            installed,
450            "cannot advance lifecycle to {next:?}: state is incomplete"
451        );
452    }
453
454    /// Whether every field operator_init installs is present (mode-aware).
455    fn operator_init_state_installed(&self, mode: EnclaveMode) -> bool {
456        // Both modes install the S3 logger; a ceremony enclave installs nothing else.
457        if self.config.s3_logger.get().is_none() {
458            return false;
459        }
460        match mode {
461            EnclaveMode::Ceremony => true,
462            // Withdraw enclaves additionally install their stable configuration.
463            EnclaveMode::Withdraw => {
464                self.config.btc_network.get().is_some()
465                    && self.config.pcr_allowlist.get().is_some()
466                    && self.config.limiter_config.get().is_some()
467                    && self.temporary_init_state_is_available()
468                    && self.config.hashi_btc_master_pubkey.get().is_some()
469            }
470        }
471    }
472
473    pub fn is_fully_initialized(&self) -> bool {
474        self.lifecycle() == WithdrawStage::Activated.into()
475    }

506    /// Build identity for `GuardianInfo.untrusted_git_revision` / the
507    /// `PcrAllowlist` key. A real ceremony enclave is a distinct measured build
508    /// (its own PCR0) from the same-commit withdraw enclave, so it reports a
509    /// distinct identity — the allowlist forbids two entries per revision, so
510    /// otherwise the withdraw enclave and KPs couldn't pin both PCR0s.
511    /// `test`/`non-enclave-dev` skip attestation and share one entry, so the
512    /// suffix is compiled out (existing mock flow unchanged).
513    fn reported_git_revision(&self) -> String {
514        // Injected at build time (docker/CI); defaults outside a real build.
515        let base = option_env!("GIT_REVISION").unwrap_or("unknown");
516        if cfg!(not(any(test, feature = "non-enclave-dev"))) && self.mode() == EnclaveMode::Ceremony
517        {
518            format!("{base}-ceremony")
519        } else {
520            base.to_string()
521        }
522    }

524    pub async fn info(&self) -> GuardianInfo {
525        let temporary_init_state = self.temporary_init_state().ok();
526        GuardianInfo {
527            lifecycle: self.lifecycle(),
528            secret_sharing_instance: temporary_init_state
529                .as_ref()
530                .map(|state| state.ceremony_state.secret_sharing_instance.clone()),
531            bucket_info: self
532                .config
533                .s3_logger()
534                .ok()
535                .map(|l| l.bucket_info().clone()),
536            encryption_pubkey: self.encryption_public_key().to_bytes().to_vec(),
537            config_hash: temporary_init_state.as_ref().map(|state| state.config_hash),
538            genesis_state_hash: temporary_init_state
539                .as_ref()
540                .and_then(|state| state.genesis_state.as_ref().map(GenesisState::digest)),
541            untrusted_git_revision: self.reported_git_revision(),
542            enclave_btc_pubkey: self.config.enclave_btc_pubkey().ok(),
543            limiter_state: self.state.limiter_state().await,
544            limiter_config: self.limiter_config().ok(),
545            current_committee_epoch: self.state.get_committee().ok().map(|c| c.epoch()),
546            mpc_master_g: self.config.hashi_btc_master_pubkey.get().copied(),
547        }
548    }

624    /// Return an owned initialization snapshot. Activation takes the stored
625    /// value, so this capability is unavailable once the enclave is active.
626    /// Keep this as the only accessor for `TemporaryInitState`. Its readers are
627    /// restricted to provisioner initialization, operator activation, and
628    /// status reporting; active request handlers must not depend on it.
629    pub fn temporary_init_state(&self) -> GuardianResult<TemporaryInitState> {
630        self.temporary_init_state
631            .read()
632            .expect("temporary initialization lock poisoned")
633            .clone()
634            .ok_or_else(|| InvalidInputs("Temporary initialization state not set".into()))
635    }
636
637    /// Operator initialization is the sole installer of temporary state.
638    pub fn set_temporary_init_state(&self, state: TemporaryInitState) -> GuardianResult<()> {
639        let mut slot = self
640            .temporary_init_state
641            .write()
642            .expect("temporary initialization lock poisoned");
643        if slot.is_some() {
644            return Err(InvalidInputs(
645                "Temporary initialization state already set".into(),
646            ));
647        }
648        *slot = Some(state);
649        Ok(())
650    }
651
652    fn temporary_init_state_is_available(&self) -> bool {
653        self.temporary_init_state
654            .read()
655            .expect("temporary initialization lock poisoned")
656            .is_some()
657    }
658
659    pub fn clear_temporary_init_state(&self) {
660        self.temporary_init_state
661            .write()
662            .expect("temporary initialization lock poisoned")
663            .take()
664            .expect("temporary initialization state must exist before activation");
665    }

671    /// Construct a verified reader with the enclave's fixed S3 client and PCR
672    /// allowlist. Each operation gets a fresh, operation-scoped session cache.
673    pub fn new_guardian_reader(&self) -> GuardianResult<GuardianReader> {
674        let s3 = self.config.s3_logger()?.clone();
675        let pcr_allowlist = self
676            .config
677            .pcr_allowlist
678            .get()
679            .cloned()
680            .ok_or_else(|| InvalidInputs("PCR allowlist is uninitialized".into()))?;
681        Ok(GuardianReader::from_s3_client(s3, pcr_allowlist))
682    }
683
684    pub fn limiter_config(&self) -> GuardianResult<LimiterConfig> {
685        self.config
686            .limiter_config
687            .get()
688            .copied()
689            .ok_or_else(|| InvalidInputs("Limiter config is uninitialized".into()))
690    }
691
692    pub fn install_config(
693        &self,
694        network: Network,
695        hashi_btc_master_pubkey: HashiMasterG,
696        pcr_allowlist: PcrAllowlist,
697        limiter_config: LimiterConfig,
698    ) -> GuardianResult<()> {
699        self.config
700            .btc_network
701            .set(network)
702            .map_err(|_| InvalidInputs("Network is already initialized".into()))?;
703        self.config
704            .hashi_btc_master_pubkey
705            .set(hashi_btc_master_pubkey)
706            .map_err(|_| InvalidInputs("Hashi BTC key is already initialized".into()))?;
707        self.config
708            .pcr_allowlist
709            .set(pcr_allowlist)
710            .map_err(|_| InvalidInputs("PCR allowlist is already initialized".into()))?;
711        self.config
712            .limiter_config
713            .set(limiter_config)
714            .map_err(|_| InvalidInputs("Limiter config is already initialized".into()))
715    }
