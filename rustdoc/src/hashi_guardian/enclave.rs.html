4//! Core enclave types: `Enclave` holds all guardian state (immutable config
5//! set during operator/provisioner-init, mutable runtime state, and the
6//! one-time init scratchpad). Lives in the library so external crates
7//! (integration test harnesses, ops tooling) can construct and drive an
8//! enclave without going through `main`.

31/// Enclave's config & state
32pub struct Enclave {
33    /// Immutable config (set once during init)
34    pub config: EnclaveConfig,
35    /// Mutable state
36    pub state: EnclaveState,
37    /// Initialization scratchpad
38    pub scratchpad: Scratchpad,
39    /// Serializes control-plane RPCs (operator_init, update_committee) so
40    /// concurrent callers can't race a check-then-set. Held across the handler.
41    /// TODO: fold provisioner_init / ceremony serialization in here too, once
42    /// their dual-purpose locks (share accumulator / one-shot bool) are untangled.
43    pub control_lock: tokio::sync::Mutex<()>,
44}
45
46/// Configuration set during initialization (immutable after set)
47pub struct EnclaveConfig {
48    /// Enclave mode (set on boot).
49    mode: EnclaveMode,
50    /// Ephemeral keypair (set on boot)
51    eph_keys: EphemeralKeyPairs,
52    /// S3 client & config (set in operator_init)
53    s3_logger: OnceLock<GuardianS3Client>,
54    /// Enclave BTC private key (set in provisioner_init)
55    enclave_btc_keypair: OnceLock<Keypair>,
56    /// BTC network: mainnet, testnet, regtest (set in operator_init)
57    btc_network: OnceLock<Network>,
58    /// Raw MPC verifying key as a curve point. Stored with y-parity so the
59    /// 2-of-2 child-key derivation matches the MPC's signing protocol.
60    /// Set in operator_init.
61    hashi_btc_master_pubkey: OnceLock<HashiMasterG>,
62}
63
64/// Mutable state that changes during operation.
65/// Note: committee + rate limiter are installed during operator_init.
66pub struct EnclaveState {
67    /// Current Hashi committee.
68    committee: RwLock<Option<Arc<HashiCommittee>>>,
69    /// Rate limiter. Set once during operator_init.
70    /// Uses `Arc<tokio::Mutex>` so the guard can be held across `.await`.
71    rate_limiter: OnceLock<Arc<tokio::sync::Mutex<RateLimiter>>>,
72}
73
74/// Scratchpad used only during initialization. The OnceLock flags retain their
75/// state for the life of the enclave.
76#[derive(Default)]
77pub struct Scratchpad {
78    /// Secret-sharing instance (commitments + N + T) set by operator_init.
79    pub secret_sharing_instance: OnceLock<SecretSharingInstance>,
80    /// Digest of the operator-supplied WithdrawModeState (set in operator_init);
81    /// the AAD KPs bind on their share submissions.
82    pub state_hash: OnceLock<[u8; 32]>,
83    /// Set once operator_init has successfully written all logs to S3.
84    /// This prevents heartbeats from being emitted before operator_init logs.
85    pub operator_init_logging_complete: OnceLock<()>,
86    /// Set once the provisioner init flow has successfully logged EnclaveFullyInitialized.
87    /// This prevents withdrawals from starting before provisioner_init logs.
88    pub provisioner_init_logging_complete: OnceLock<()>,
89    /// Guards the single ceremony per enclave: `setup_new_key` (genesis) or
90    /// `rotate_kps` (rotation), never both. Each holds this guard across its
91    /// flow so the two can't interleave; the `bool` flips true once a ceremony
92    /// finalizes, making it one-shot per enclave instance (the operator
93    /// restarts to run another).
94    pub ceremony_complete: tokio::sync::Mutex<bool>,
95    /// Encrypted shares produced by the ceremony (`setup_new_key` or
96    /// `rotate_kps`), served to KPs from `get_guardian_info`. Set once.
97    pub latest_encrypted_shares: OnceLock<KPEncryptedShares>,
98}

162    /// Sign a BTC tx. Returns an Err if enclave btc keypair or hashi btc pk is not set.
163    pub fn btc_sign(&self, tx_utxos: &TxUTXOs) -> GuardianResult<(Txid, Vec<BitcoinSignature>)> {
164        let enclave_keypair = self
165            .enclave_btc_keypair
166            .get()
167            .ok_or(InvalidInputs("Bitcoin key is not initialized".into()))?;
168        let hashi_btc_pk = self
169            .hashi_btc_master_pubkey
170            .get()
171            .ok_or(InvalidInputs("Hashi BTC public key not set".into()))?;
172
173        let enclave_btc_pk = enclave_keypair.x_only_public_key().0;
174        let (messages, txid) = tx_utxos.signing_messages_and_txid(&enclave_btc_pk, hashi_btc_pk);
175        Ok((txid, sign_btc_tx(&messages, enclave_keypair)))
176    }

199impl EnclaveState {
200    /// Install the operator-supplied committee + rate limiter. Called from operator_init.
201    pub fn init(&self, committee: HashiCommittee, rate_limiter: RateLimiter) -> GuardianResult<()> {
202        self.set_committee(committee)?;
203        self.set_rate_limiter(rate_limiter)?;
204        Ok(())
205    }
206
207    // ========================================================================
208    // Committee Management
209    // ========================================================================
210
211    /// Get the current committee.
212    pub fn get_committee(&self) -> GuardianResult<Arc<HashiCommittee>> {
213        let guard = self
214            .committee
215            .read()
216            .expect("rwlock should never throw an error");
217        guard
218            .as_ref()
219            .cloned()
220            .ok_or_else(|| InvalidInputs("committee not initialized".into()))
221    }
222
223    /// Whether the committee is installed, without cloning the `Arc` — used by the
224    /// operator_init completeness check, which runs on the heartbeat/withdrawal path.
225    fn has_committee(&self) -> bool {
226        self.committee
227            .read()
228            .expect("rwlock should never throw an error")
229            .is_some()
230    }
231
232    /// Set committee. Called only from `init` (operator_init).
233    fn set_committee(&self, committee: HashiCommittee) -> GuardianResult<()> {
234        info!("Setting committee for epoch {}.", committee.epoch());
235
236        let mut guard = self
237            .committee
238            .write()
239            .expect("rwlock should never throw an error");
240        if guard.is_some() {
241            return Err(InvalidInputs("committee already initialized".into()));
242        }
243        *guard = Some(Arc::new(committee));
244        Ok(())
245    }
246
247    /// Replace an already-initialized committee. Rejects the swap unless
248    /// the in-memory epoch matches `expected_current_epoch`.
249    pub fn replace_committee(
250        &self,
251        committee: HashiCommittee,
252        expected_current_epoch: u64,
253    ) -> GuardianResult<()> {
254        info!("Replacing committee for epoch {}.", committee.epoch());
255
256        let mut guard = self
257            .committee
258            .write()
259            .expect("rwlock should never throw an error");
260        let current_epoch = guard
261            .as_ref()
262            .ok_or_else(|| InvalidInputs("committee not initialized".into()))?
263            .epoch();
264        if current_epoch != expected_current_epoch {
265            return Err(InvalidInputs(format!(
266                "committee epoch mismatch: expected {expected_current_epoch}, actual {current_epoch}"
267            )));
268        }
269        *guard = Some(Arc::new(committee));
270        Ok(())
271    }
272
273    // ========================================================================
274    // Rate Limiter Management
275    // ========================================================================
276
277    fn set_rate_limiter(&self, limiter: RateLimiter) -> GuardianResult<()> {
278        info!("Setting rate limiter.");
279
280        self.rate_limiter
281            .set(Arc::new(tokio::sync::Mutex::new(limiter)))
282            .map_err(|_| InvalidInputs("rate_limiter already initialized".into()))
283    }
284
285    /// Acquire exclusive access to the limiter, consume tokens, and return a guard.
286    /// The guard holds the mutex lock — no other withdrawal can start until it is
287    /// committed or dropped (which reverts).
288    /// Timeout for acquiring the limiter lock. If a withdrawal is in progress and
289    /// takes longer than this, we bail rather than queue up requests indefinitely.
290    const LIMITER_LOCK_TIMEOUT: Duration = Duration::from_secs(10);
291
292    pub async fn consume_from_limiter(
293        &self,
294        seq: u64,
295        timestamp: u64,
296        amount_sats: u64,
297    ) -> GuardianResult<LimiterGuard> {
298        let rate_limiter = self
299            .rate_limiter
300            .get()
301            .ok_or_else(|| InvalidInputs("rate_limiter not initialized".into()))?;
302        let mut guard = tokio::time::timeout(
303            Self::LIMITER_LOCK_TIMEOUT,
304            rate_limiter.clone().lock_owned(),
305        )
306        .await
307        .map_err(|_| InvalidInputs("timed out waiting for rate limiter lock".into()))?;
308        guard.consume(seq, timestamp, amount_sats)?;
309        Ok(LimiterGuard::new(guard))
310    }
311
312    pub async fn limiter_state(&self) -> Option<LimiterState> {
313        let limiter = self.rate_limiter.get()?;
314        Some(*limiter.lock().await.state())
315    }
316
317    pub async fn limiter_config(&self) -> Option<LimiterConfig> {
318        let limiter = self.rate_limiter.get()?;
319        Some(*limiter.lock().await.config())
320    }
321}
322
323/// RAII guard that holds the limiter mutex via an owned guard, returned by
324/// [`EnclaveState::consume_from_limiter`]. Reverts on drop unless committed.
325pub struct LimiterGuard {
326    guard: tokio::sync::OwnedMutexGuard<RateLimiter>,
327    committed: bool,
328}
329
330impl LimiterGuard {
331    pub(crate) fn new(guard: tokio::sync::OwnedMutexGuard<RateLimiter>) -> Self {
332        Self {
333            guard,
334            committed: false,
335        }
336    }
337
338    /// Mark this withdrawal as successful. Prevents revert on drop.
339    pub fn commit(mut self) {
340        self.committed = true;
341    }
342
343    pub fn state(&self) -> &LimiterState {
344        self.guard.state()
345    }
346}
347
348impl Drop for LimiterGuard {
349    fn drop(&mut self) {
350        if !self.committed {
351            self.guard.revert();
352        }
353    }
354}

377    /// Which flows this enclave serves (fixed at boot).
378    pub fn mode(&self) -> EnclaveMode {
379        self.config.mode
380    }
381
382    /// Provisioner_init is complete: the reconstructed BTC keypair is set and
383    /// its installation has been logged.
384    pub fn is_provisioner_init_complete(&self) -> bool {
385        self.config.is_enclave_btc_keypair_set()
386            && self
387                .scratchpad
388                .provisioner_init_logging_complete
389                .get()
390                .is_some()
391    }
392
393    pub fn is_operator_init_complete(&self) -> bool {
394        let logged = self
395            .scratchpad
396            .operator_init_logging_complete
397            .get()
398            .is_some();
399        // commit_operator_init sets this flag last in an all-or-nothing commit, so
400        // a set flag implies every installed field is present. This backstop holds
401        // in prod too (a violation aborts via `abort_on_panic`) and never fires
402        // transiently, since the flag is the last thing set. The converse is NOT
403        // checked: a normal commit installs the state before its (slow) S3 logging
404        // completes, and a lock-free reader (e.g. the heartbeat) can observe that
405        // window with state present but the flag unset.
406        assert!(
407            !logged || self.operator_init_state_installed(),
408            "operator_init_logging_complete set but operator_init state is incomplete"
409        );
410        logged
411    }
412
413    /// Whether every field operator_init installs is present (mode-aware). Only
414    /// used to assert the `operator_init_logging_complete` invariant above.
415    fn operator_init_state_installed(&self) -> bool {
416        // Both modes install the S3 logger; a ceremony enclave installs nothing else.
417        if self.config.s3_logger.get().is_none() {
418            return false;
419        }
420        match self.config.mode {
421            EnclaveMode::Ceremony => true,
422            // Withdraw enclaves additionally install the WithdrawModeConfig.
423            EnclaveMode::Withdraw => {
424                self.config.btc_network.get().is_some()
425                    && self.scratchpad.secret_sharing_instance.get().is_some()
426                    && self.scratchpad.state_hash.get().is_some()
427                    && self.config.hashi_btc_master_pubkey.get().is_some()
428                    && self.state.has_committee()
429                    && self.state.rate_limiter.get().is_some()
430            }
431        }
432    }
433
434    pub fn is_fully_initialized(&self) -> bool {
435        self.is_provisioner_init_complete() && self.is_operator_init_complete()
436    }
