4use std::time::Duration;
5
6/// Interval between successful heartbeat writes.
7pub const HEARTBEAT_INTERVAL: Duration = Duration::from_mins(1);
8/// Maximum write failure interval before the enclave aborts.
9pub const MAX_S3_WRITE_FAILURE_INTERVAL: Duration = Duration::from_mins(5);
10/// The live session's latest heartbeat must be at most 3 minutes old.
11pub const LIVE_SESSION_LATEST_HEARTBEAT_MAX_AGE: Duration = Duration::from_mins(3);
12/// Silence required before another session is considered inactive.
13pub const OTHER_SESSION_QUIET_PERIOD: Duration = Duration::from_mins(10);
14
15// An enclave that cannot write its next heartbeat must abort before another
16// session is allowed to treat it as quiet.
17const _: () = assert!(
18    HEARTBEAT_INTERVAL.as_secs() + MAX_S3_WRITE_FAILURE_INTERVAL.as_secs()
19        < OTHER_SESSION_QUIET_PERIOD.as_secs()
20);
