4//! Verified read layer over the guardian's S3 logs.
5//!
6//! The guardian writes its logs via [`GuardianS3Client`]; off-enclave readers
7//! (the monitor auditor, KP/operator init tooling) replay them here through a
8//! [`GuardianReader`], which owns the S3 client and the trusted-key cache so a
9//! session's attestation is verified once across every read. Streams are
10//! hour-partitioned (`withdraw/`/`heartbeat/`); [`withdraw_cursor`]/[`heartbeat_cursor`]
11//! open a cursor that the caller advances/retreats and feeds to [`GuardianReader::read_dir`].
12

35/// Open an hour-scoped cursor at `start` over the `withdraw/` stream. Advance/
36/// retreat with [`S3HourScopedDirectory::next_dir`]/`prev_dir`, gate on
37/// `write_completion_time`, and read via [`GuardianReader::read_dir`].
38pub fn withdraw_cursor(start: UnixSeconds) -> S3HourScopedDirectory {
39    S3HourScopedDirectory::new(S3_DIR_WITHDRAW, start)
40}
41
42/// Like [`withdraw_cursor`], but over the `heartbeat/` stream.
43pub fn heartbeat_cursor(start: UnixSeconds) -> S3HourScopedDirectory {
44    S3HourScopedDirectory::new(S3_DIR_HEARTBEAT, start)
45}

47/// Verified reader over the guardian's S3 logs. Owns the S3 client and a private
48/// session-key cache, so each session's (eventually expensive) attestation check
49/// happens at most once across every read. Thread a single `&mut GuardianReader`
50/// through a run.
51pub struct GuardianReader {
52    s3: GuardianS3Client,
53    pubkey_cache: GuardianSessionKeyCache,
54}
55
56impl GuardianReader {
57    pub async fn new(config: &S3Config) -> anyhow::Result<Self> {
58        let s3 = GuardianS3Client::new_checked(config)
59            .await
60            .map_err(|e| anyhow::anyhow!(e))
61            .context("failed to verify guardian S3 connectivity")?;
62        Ok(Self::from_s3_client(s3))
63    }
64
65    pub fn from_s3_client(s3: GuardianS3Client) -> Self {
66        Self {
67            s3,
68            pubkey_cache: GuardianSessionKeyCache::default(),
69        }
70    }
71
72    /// Raw S3 client, for unverified listing/reads (e.g. the limiter's bucket walk).
73    pub fn s3(&self) -> &GuardianS3Client {
74        &self.s3
75    }
76
77    /// Read and verify every record in `dir`, resolving each writing session's
78    /// signing pubkey via the cache (loaded once per session).
79    pub async fn read_dir(
80        &mut self,
81        dir: &S3HourScopedDirectory,
82    ) -> anyhow::Result<Vec<VerifiedLogRecord>> {
83        let all_logs = self
84            .s3
85            .list_all_objects_in_dir::<LogRecord>(dir)
86            .await
87            .with_context(|| format!("failed to list guardian logs in {dir}"))?;
88
89        let mut out = Vec::with_capacity(all_logs.len());
90        for log in all_logs {
91            out.push(self.pubkey_cache.verify_record(&self.s3, log).await?);
92        }
93        Ok(out)
94    }
95
96    /// Read and verify a session's signed `OIGuardianInfo` log (written at
97    /// operator-init). Implements check B of IOP-225.
98    pub async fn get_info(&mut self, session_id: &str) -> anyhow::Result<GuardianInfo> {
99        let key = InitLogMessage::guardian_info_object_key(session_id);
100        let record = self.s3.get_log_record(&key).await?;
101        self.pubkey_cache
102            .verify_record(&self.s3, record)
103            .await?
104            .message
105            .into_init_log()
106            .and_then(|x| match x {
107                InitLogMessage::OIGuardianInfo(info) => Some(*info),
108                _ => None,
109            })
110            .with_context(|| format!("expected OIGuardianInfo at {key}"))
111    }
112
113    /// The latest secret-sharing instance from `ceremony/` — the max-`sharing_seq`
114    /// (lex-last) entry, attestation- and signature-verified. `None` if no ceremony
115    /// has been logged yet. Written from initial setup onward, so present whenever a
116    /// key exists.
117    pub async fn read_latest_ceremony_instance(
118        &mut self,
119    ) -> anyhow::Result<Option<SecretSharingInstance>> {
120        let keys = self
121            .s3
122            .list_all_keys_in_dir(&format!("{}/", S3_DIR_CEREMONY))
123            .await?;
124        let Some(key) = pick_latest_key(keys, S3_DIR_CEREMONY) else {
125            return Ok(None);
126        };
127        let record = self.s3.get_log_record(&key).await?;
128        let record = self.pubkey_cache.verify_record(&self.s3, record).await?;
129        let LogMessage::Ceremony(msg) = record.message else {
130            anyhow::bail!("expected a ceremony log at {key}");
131        };
132        Ok(Some(match *msg {
133            CeremonyLogMessage::NewKey { instance } => instance,
134            CeremonyLogMessage::Rotate { new_instance, .. } => new_instance,
135        }))
136    }
137
138    /// The latest applied committee from `committee-update/` — the lex-last
139    /// non-`failure-` (i.e. highest-epoch Success) entry, attestation- and
140    /// signature-verified. `None` if no committee update has been logged (e.g. a
141    /// fresh deployment whose committee still only exists in the operator's boot
142    /// config).
143    pub async fn read_latest_committee(&mut self) -> anyhow::Result<Option<Committee>> {
144        let keys = self
145            .s3
146            .list_all_keys_in_dir(&format!("{}/", S3_DIR_COMMITTEE_UPDATE))
147            .await?;
148        let Some(key) = pick_latest_key(keys, S3_DIR_COMMITTEE_UPDATE) else {
149            return Ok(None);
150        };
151        let record = self.s3.get_log_record(&key).await?;
152        let record = self.pubkey_cache.verify_record(&self.s3, record).await?;
153        let LogMessage::CommitteeUpdate(msg) = record.message else {
154            anyhow::bail!("expected a committee-update log at {key}");
155        };
156        match *msg {
157            CommitteeUpdateLogMessage::Success { new_committee, .. } => Ok(Some(new_committee)),
158            CommitteeUpdateLogMessage::Failure { .. } => {
159                anyhow::bail!("lex-last non-failure key resolved to a Failure log at {key}")
160            }
161        }
162    }
163}

165/// Enclave signing pubkeys trusted after their attestation was verified once,
166/// keyed by session. Internal to [`GuardianReader`].
167///
168/// TODO(check C): make this the trust engine — construct it with a trusted
169/// `commit -> ExpectedPcrs` map and pin each session's attested PCRs against the
170/// entry for its `/info`-reported commit.
171#[derive(Default)]
172struct GuardianSessionKeyCache {
173    keys: HashMap<SessionID, GuardianPubKey>,
174}
175
176impl GuardianSessionKeyCache {
177    /// The session's signing pubkey, verifying and caching its attestation on first use.
178    async fn get_or_load_pubkey(
179        &mut self,
180        s3: &GuardianS3Client,
181        session_id: &str,
182    ) -> anyhow::Result<&GuardianPubKey> {
183        if !self.keys.contains_key(session_id) {
184            let pubkey = s3.get_verified_enclave_pubkey(session_id).await?;
185            self.keys.insert(session_id.to_string(), pubkey);
186        }
187        Ok(&self.keys[session_id])
188    }
189
190    /// Verify `record`'s signature under its session's trusted pubkey.
191    async fn verify_record(
192        &mut self,
193        s3: &GuardianS3Client,
194        record: LogRecord,
195    ) -> anyhow::Result<VerifiedLogRecord> {
196        let signing_pubkey = self.get_or_load_pubkey(s3, &record.session_id).await?;
197        record
198            .verify(signing_pubkey)
199            .with_context(|| "failed to verify guardian enclave signature")
200    }
201}

203/// Pick the lex-greatest key, skipping any whose name starts with `<dir>/failure-`.
204/// Keys are zero-padded (ceremony `sharing_seq`, committee `epoch`), so the lex-max
205/// non-failure key is the latest successful entry.
206fn pick_latest_key(keys: Vec<String>, dir: &str) -> Option<String> {
207    let failure_prefix = format!("{dir}/failure-");
208    keys.into_iter()
209        .filter(|k| !k.starts_with(&failure_prefix))
210        .max()
211}

213#[cfg(test)]
214mod tests {
215    use super::*;
216
217    fn success_key(epoch: u64) -> String {
218        format!("{S3_DIR_COMMITTEE_UPDATE}/{epoch:020}-sess.json")
219    }
220    fn failure_key(epoch: u64) -> String {
221        format!("{S3_DIR_COMMITTEE_UPDATE}/failure-{epoch:020}-sess-abcd1234.json")
222    }
223
224    #[test]
225    fn pick_latest_key_none_when_empty() {
226        assert_eq!(pick_latest_key(vec![], S3_DIR_COMMITTEE_UPDATE), None);
227    }
228
229    #[test]
230    fn pick_latest_key_takes_lex_max() {
231        let keys = vec![success_key(3), success_key(7), success_key(5)];
232        assert_eq!(
233            pick_latest_key(keys, S3_DIR_COMMITTEE_UPDATE),
234            Some(success_key(7))
235        );
236    }
237
238    #[test]
239    fn pick_latest_key_skips_higher_failure() {
240        // A later-epoch failure (lex-greater than any success) must not win.
241        let keys = vec![success_key(5), failure_key(9)];
242        assert_eq!(
243            pick_latest_key(keys, S3_DIR_COMMITTEE_UPDATE),
244            Some(success_key(5))
245        );
246    }
247
248    #[test]
249    fn pick_latest_key_none_when_all_failures() {
250        assert_eq!(
251            pick_latest_key(vec![failure_key(9)], S3_DIR_COMMITTEE_UPDATE),
252            None
253        );
254    }
