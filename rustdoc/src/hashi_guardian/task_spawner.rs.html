4//! Cancellation-safe task spawning for state-changing guardian RPCs.
5//!
6//! Transport conversion stays in `rpc`; endpoint modules contain domain logic.
7//! This module routes each mutation through the enclave's appropriate
8//! cancellation-safe execution policy.
9//!
10//! Control-plane mutations use `spawn_control_task` because they share enclave
11//! lifecycle and configuration state. Standard withdrawal uses `spawn_task`
12//! because its limiter guard is the narrower serialization boundary: requests
13//! may validate concurrently, but limiter mutation through durable logging is
14//! still exclusive.
15

127#[cfg(test)]
128mod tests {
129    use super::*;
130    use hashi_types::guardian::EnclaveMode;
131    use hashi_types::guardian::GuardianEncKeyPair;
132    use hashi_types::guardian::GuardianSignKeyPair;
133    use std::time::Duration;
134    use tokio::sync::oneshot;
135
136    /// A task body that reports when it starts, waits for the test to let it
137    /// continue, then reports completion.
138    struct PausedTask {
139        started: oneshot::Sender<()>,
140        resume: oneshot::Receiver<()>,
141        finished: oneshot::Sender<()>,
142    }
143
144    fn test_enclave() -> Arc<Enclave> {
145        Arc::new(Enclave::new(
146            GuardianSignKeyPair::new(rand::thread_rng()),
147            GuardianEncKeyPair::random(&mut rand::thread_rng()),
148            EnclaveMode::Withdraw,
149        ))
150    }
151
152    async fn pause_after_start(_enclave: Arc<Enclave>, task: PausedTask) -> GuardianResult<()> {
153        task.started.send(()).unwrap();
154        task.resume.await.unwrap();
155        task.finished.send(()).unwrap();
156        Ok(())
157    }
158
159    async fn signal_started(
160        _enclave: Arc<Enclave>,
161        started: oneshot::Sender<()>,
162    ) -> GuardianResult<()> {
163        started.send(()).unwrap();
164        Ok(())
165    }
166
167    #[tokio::test]
168    async fn root_owned_task_survives_caller_cancellation() {
169        let (started_tx, started_rx) = oneshot::channel();
170        let (resume_tx, resume_rx) = oneshot::channel();
171        let (finished_tx, finished_rx) = oneshot::channel();
172
173        // This outer task represents the Tonic RPC handler awaiting the
174        // independently spawned guardian task.
175        let caller = tokio::spawn(test_enclave().spawn_task(
176            PausedTask {
177                started: started_tx,
178                resume: resume_rx,
179                finished: finished_tx,
180            },
181            pause_after_start,
182        ));
183        // Ensure the guardian accepted and started the task before simulating
184        // the client disconnect.
185        started_rx.await.unwrap();
186
187        // Cancelling the RPC handler drops only its waiter. The guardian task
188        // spawned by `spawn_task` must continue independently.
189        caller.abort();
190        assert!(caller.await.unwrap_err().is_cancelled());
191
192        // Allow the guardian task to finish and prove that caller cancellation
193        // did not cancel it.
194        resume_tx.send(()).unwrap();
195        tokio::time::timeout(Duration::from_secs(1), finished_rx)
196            .await
197            .expect("root-owned task should finish")
198            .unwrap();
199    }
200
201    #[tokio::test]
202    async fn control_tasks_are_serialized() {
203        let enclave = test_enclave();
204        let (first_started_tx, first_started_rx) = oneshot::channel();
205        let (first_resume_tx, first_resume_rx) = oneshot::channel();
206        let (first_finished_tx, first_finished_rx) = oneshot::channel();
207
208        // The first task acquires the control lock, then pauses while holding it.
209        let first = tokio::spawn(enclave.clone().spawn_control_task(
210            PausedTask {
211                started: first_started_tx,
212                resume: first_resume_rx,
213                finished: first_finished_tx,
214            },
215            pause_after_start,
216        ));
217        first_started_rx.await.unwrap();
218
219        // A second control task is accepted and spawned, but must wait for the
220        // first task to release the control lock.
221        let (second_started_tx, mut second_started_rx) = oneshot::channel();
222        let second = tokio::spawn(enclave.spawn_control_task(second_started_tx, signal_started));
223        // Yield this test task to give Tokio an opportunity to poll the second
224        // task and let it reach the control lock. This is a scheduling hint,
225        // not proof that the second task reached the lock-waiting point.
226        tokio::task::yield_now().await;
227        assert!(matches!(
228            second_started_rx.try_recv(),
229            Err(oneshot::error::TryRecvError::Empty)
230        ));
231
232        // Finishing the first task releases the lock, after which the second
233        // task may enter and signal that it started.
234        first_resume_tx.send(()).unwrap();
235        first_finished_rx.await.unwrap();
236        second_started_rx.await.unwrap();
237        first.await.unwrap().unwrap();
238        second.await.unwrap().unwrap();
239    }
