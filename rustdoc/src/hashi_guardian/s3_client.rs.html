32/// Maximum attempts the AWS SDK makes before returning one write failure.
33const MAX_RETRY_ATTEMPTS: u32 = 5;
34/// Delay between application-level retries of an immutable S3 log write.
35const S3_WRITE_RETRY_INTERVAL: Duration = Duration::from_secs(10);

52    pub async fn new(config: &S3Config) -> Self {
53        info!("S3 Configuration:");
54        info!("   Bucket: {}", config.bucket_name());
55        info!("   Region: {}", config.region());
56
57        let mut creds = CredentialsBuilder::default()
58            .access_key_id(config.access_key.clone())
59            .secret_access_key(config.secret_key.clone())
60            .provider_name("hashi-guardian");
61        creds.set_session_token(config.session_token.clone());
62        let creds = creds.build();
63
64        let retry_config = RetryConfig::standard().with_max_attempts(MAX_RETRY_ATTEMPTS); // default is 3
65
66        let aws_config = aws_config::defaults(aws_config::BehaviorVersion::latest())
67            .region(aws_config::Region::new(config.region().to_string()))
68            .credentials_provider(SharedCredentialsProvider::new(creds))
69            .retry_config(retry_config)
70            .load()
71            .await;
72
73        // A custom endpoint implies an S3-compatible service (MinIO, LocalStack), which
74        // need path-style addressing.
75        let mut s3_builder = aws_sdk_s3::config::Builder::from(&aws_config);
76        if std::env::var_os("AWS_ENDPOINT_URL_S3").is_some() {
77            s3_builder = s3_builder.force_path_style(true);
78        }
79        let client = S3Client::from_conf(s3_builder.build());
80
81        Self {
82            client,
83            config: config.clone(),
84            object_lock_policy: S3ObjectLockPolicy::for_environment(config.retention_environment),
85        }
86    }

118    /// Attempt one immutable log write, relying on the AWS SDK retry policy.
119    pub async fn write_log_record(&self, log: LogRecord) -> GuardianResult<()> {
120        let key = log.object_key().to_string();
121        let object_lock_duration = self.object_lock_duration(&log);
122        self.write_at_key(&key, &log, object_lock_duration).await
123    }

125    /// Retry an immutable log write through the grace period, then abort.
126    /// State-changing RPCs call this from their root-owned task.
127    pub async fn write_log_record_or_abort(&self, log: LogRecord) -> GuardianResult<()> {
128        self.write_log_record_or_abort_inner(
129            log,
130            MAX_S3_WRITE_FAILURE_INTERVAL,
131            S3_WRITE_RETRY_INTERVAL,
132        )
133        .await
134    }
135
136    async fn write_log_record_or_abort_inner(
137        &self,
138        log: LogRecord,
139        max_failure_interval: Duration,
140        retry_interval: Duration,
141    ) -> GuardianResult<()> {
142        let object_lock_duration = self.object_lock_duration(&log);
143        let key = log.object_key();
144        let write_until_success = async {
145            loop {
146                match self.write_at_key(key, &log, object_lock_duration).await {
147                    Ok(()) => return Ok(()),
148                    Err(error) => {
149                        warn!(%key, ?error, "S3 log write failed; retrying");
150                        tokio::time::sleep(retry_interval).await;
151                    }
152                }
153            }
154        };
155
156        match tokio::time::timeout(max_failure_interval, write_until_success).await {
157            Ok(result) => result,
158            Err(_) => panic!(
159                "S3 log {} was not written within {:?}",
160                key, max_failure_interval
161            ),
162        }
163    }
164
165    fn object_lock_duration(&self, log: &LogRecord) -> Duration {
166        log.object_lock_duration(self.object_lock_policy)
167    }

169    /// Write a value to S3 at an explicit key.
170    ///
171    /// This is intended for ordered log streams where the caller determines the key.
172    async fn write_at_key<T: Serialize>(
173        &self,
174        key: &str,
175        value: &T,
176        object_lock_duration: Duration,
177    ) -> GuardianResult<()> {
178        let s3_client = &self.client;
179        let s3_config = &self.config;
180
181        info!("Logging to {}", key);
182
183        let expiry_time = SystemTime::now()
184            .checked_add(object_lock_duration)
185            .expect("Cant overflow");
186
187        let body = serde_json::to_vec(value).expect("Cant serialize to JSON");
188
189        // `If-None-Match: *` makes retries safe: a lost-ack write that already
190        // landed returns 412 instead of creating another version. A 412 is only
191        // success if the existing immutable object is exactly this record.
192        let result = s3_client
193            .put_object()
194            .bucket(s3_config.bucket_name())
195            .key(key)
196            .content_type("application/json")
197            .object_lock_mode(ObjectLockMode::Compliance)
198            .object_lock_retain_until_date(DateTime::from(expiry_time))
199            .if_none_match("*")
200            .body(ByteStream::from(body.clone()))
201            .send()
202            .await;
203        if let Err(e) = result {
204            let already_written = e
205                .raw_response()
206                .is_some_and(|resp| resp.status().as_u16() == 412);
207            if !already_written {
208                // DisplayErrorContext displays the full error returned by the SDK
209                return Err(S3Error(format!(
210                    "Failed to write to s3: {}",
211                    DisplayErrorContext(&e)
212                )));
213            }
214            self.verify_existing_write(key, &body).await?;
215            info!("Object {} already contains the intended record", key);
216        }
217
218        info!("Logged entry to immutable storage");
219        info!("Object locked until: {:?}", expiry_time);
220        info!(
221            "Public URL: https://{}.s3.amazonaws.com/{}",
222            s3_config.bucket_name(),
223            key
224        );
225
226        Ok(())
227    }

229    /// Similar to `get_object_unsafe`, but compares the raw bytes and treats
230    /// invalid lock metadata as a fatal conflict at this write-once key.
231    async fn verify_existing_write(&self, key: &str, expected_body: &[u8]) -> GuardianResult<()> {
232        let response = self
233            .client
234            .get_object()
235            .bucket(self.config.bucket_name())
236            .key(key)
237            .send()
238            .await
239            .map_err(|e| {
240                S3Error(format!(
241                    "Failed to get object {}: {}",
242                    key,
243                    DisplayErrorContext(&e)
244                ))
245            })?;
246        let has_compliance_lock = response.object_lock_mode() == Some(&ObjectLockMode::Compliance)
247            && response.object_lock_retain_until_date().is_some();
248        let actual_body = response.body.collect().await.map_err(|e| {
249            S3Error(format!(
250                "Failed to read object body for key {}: {}",
251                key,
252                DisplayErrorContext(&e)
253            ))
254        })?;
255
256        if actual_body.into_bytes().as_ref() != expected_body {
257            // A 412 revealed different content at this write-once key. Retrying
258            // cannot replace it, so continuing would violate log durability.
259            panic!("existing object {key} differs from the intended record");
260        }
261        if !has_compliance_lock {
262            // The intended record exists but is not immutable. Retrying cannot
263            // replace it, so it cannot satisfy the durable-write requirement.
264            panic!("existing object {key} is missing a valid compliance lock");
265        }
266
267        Ok(())
268    }

278    /// Verify that the S3 bucket has object lock enabled and returns an Err if not.
279    /// Can be used as a test for S3 connectivity.
280    pub async fn assert_object_lock_enabled(&self) -> GuardianResult<()> {
281        let s3_client = &self.client;
282        let s3_config = &self.config;
283
284        // Verify bucket exists and has Object Lock enabled
285        let bucket_config = s3_client
286            .get_object_lock_configuration()
287            .bucket(s3_config.bucket_name())
288            .send()
289            .await;
290
291        match bucket_config {
292            Ok(config) => {
293                let object_lock_config = config.object_lock_configuration().ok_or_else(|| {
294                    S3Error("Object lock configuration missing in S3 response".into())
295                })?;
296
297                let object_lock_enabled_config =
298                    object_lock_config.object_lock_enabled().ok_or_else(|| {
299                        S3Error("Object lock enabled field missing in S3 response".into())
300                    })?;
301
302                match object_lock_enabled_config {
303                    ObjectLockEnabled::Enabled => {
304                        info!("Bucket {} has Object Lock enabled", s3_config.bucket_name());
305                    }
306                    other => {
307                        return Err(S3Error(format!(
308                            "Unexpected object lock enabled config: {:?}",
309                            other
310                        )))
311                    }
312                }
313            }
314            Err(e) => {
315                return Err(S3Error(format!(
316                    "Failed to verify Object Lock configuration: {}",
317                    DisplayErrorContext(&e)
318                )));
319            }
320        }
321
322        Ok(())
323    }

325    /// List up to 10 objects in the bucket.
326    /// This is intended as a lightweight connectivity/debug helper (primarily for testing).
327    pub async fn list_objects_sample(&self) -> GuardianResult<()> {
328        let s3_client = &self.client;
329        let s3_config = &self.config;
330
331        let bucket_objects = s3_client
332            .list_objects_v2()
333            .bucket(s3_config.bucket_name())
334            .max_keys(10)
335            .send()
336            .await
337            .map_err(|e| {
338                S3Error(format!(
339                    "Failed to list objects: {}",
340                    DisplayErrorContext(&e)
341                ))
342            })?;
343
344        let objects = bucket_objects.contents();
345
346        if objects.is_empty() {
347            info!(
348                "Bucket {} has no objects (or no access to list)",
349                s3_config.bucket_name()
350            );
351            return Ok(());
352        }
353
354        info!(
355            "Bucket {}: listing {} object(s) (max 10)",
356            s3_config.bucket_name(),
357            objects.len()
358        );
359
360        for (i, obj) in objects.iter().enumerate() {
361            let key = obj.key().unwrap_or("<missing key>");
362            info!(
363                "  {}. key={} size={:?} last_modified={:?} etag={:?}",
364                i + 1,
365                key,
366                obj.size(),
367                obj.last_modified(),
368                obj.e_tag()
369            );
370        }
371
372        Ok(())
373    }

376/// Controls whether an S3 read requires Compliance-mode object-lock metadata.
377pub(crate) enum LockCheck {
378    /// Reject the object unless its Compliance lock is still unexpired.
379    Required,
380    /// Do not inspect lock metadata. Used for signed records whose short lock
381    /// is expected to expire, such as KP-share state.
382    Skipped,
383}
384
385/// Controls whether an S3 read validates that the key has no overwrite or
386/// deletion history.
387pub(crate) enum HistoryCheck {
388    /// Validate the exact key's history before fetching it.
389    Required,
390    /// The caller already validated the history of the enclosing prefix.
391    AlreadyChecked,
392}

399    /// Lists immediate subdirectories under `prefix` (S3 `CommonPrefixes`,
400    /// returned by `list_objects_v2` with `delimiter='/'`). Used to tree-walk
401    /// the hour-partitioned withdraw layout (`withdraw/YYYY/MM/DD/HH/`)
402    /// without paginating every object key. Returned prefixes are unique and
403    /// sorted lexicographically.
404    pub async fn list_common_prefixes(&self, prefix: &str) -> GuardianResult<Vec<String>> {
405        let mut continuation_token: Option<String> = None;
406        let mut out: BTreeSet<String> = BTreeSet::new();
407        loop {
408            let mut req = self
409                .client
410                .list_objects_v2()
411                .bucket(self.config.bucket_name())
412                .prefix(prefix)
413                .delimiter("/");
414            if let Some(ref token) = continuation_token {
415                req = req.continuation_token(token);
416            }
417            let response = req.send().await.map_err(|e| {
418                S3Error(format!(
419                    "Failed to list common prefixes under {}: {}",
420                    prefix,
421                    DisplayErrorContext(&e)
422                ))
423            })?;
424            for cp in response.common_prefixes() {
425                if let Some(p) = cp.prefix() {
426                    out.insert(p.to_string());
427                }
428            }
429            if response.is_truncated() != Some(true) {
430                break;
431            }
432            let Some(token) = response.next_continuation_token() else {
433                return Err(S3Error(format!(
434                    "Truncated response but no next_continuation_token for prefix {}",
435                    prefix
436                )));
437            };
438            continuation_token = Some(token.to_string());
439        }
440        Ok(out.into_iter().collect())
441    }

443    /// Lists every key under `prefix`, refusing to proceed if any matching
444    /// object has a delete marker or non-latest version. The prefix may be a
445    /// directory or a complete object key. Returned keys are unique and sorted.
446    pub async fn validate_prefix_history_and_list_keys(
447        &self,
448        prefix: &str,
449    ) -> GuardianResult<Vec<String>> {
450        let s3_client = &self.client;
451        let s3_config = &self.config;
452
453        let mut key_marker: Option<String> = None;
454        let mut version_id_marker: Option<String> = None;
455        let mut seen_keys: BTreeSet<String> = BTreeSet::new();
456
457        loop {
458            let mut req = s3_client
459                .list_object_versions()
460                .bucket(s3_config.bucket_name())
461                .prefix(prefix);
462            if let Some(ref marker) = key_marker {
463                req = req.key_marker(marker);
464            }
465            if let Some(ref marker) = version_id_marker {
466                req = req.version_id_marker(marker);
467            }
468
469            let response = req.send().await.map_err(|e| {
470                S3Error(format!(
471                    "Failed to list object versions for prefix {}: {}",
472                    prefix,
473                    DisplayErrorContext(&e)
474                ))
475            })?;
476
477            if !response.delete_markers().is_empty() {
478                return Err(S3Error(format!(
479                    "Delete marker found under prefix {}",
480                    prefix
481                )));
482            }
483
484            // https://docs.aws.amazon.com/AmazonS3/latest/API/API_ObjectVersion.html
485            for version in response.versions() {
486                let key = version.key().ok_or_else(|| {
487                    S3Error("Missing key in list_object_versions response".into())
488                })?;
489
490                // NOTE: If an object's lock expires, then all bets are off.
491                // For example, is_latest could be true even though an older version of it was deleted (post lock expiry).
492                if version.is_latest() != Some(true) {
493                    return Err(S3Error(format!(
494                        "Non-latest version found for key {} under prefix {}",
495                        key, prefix
496                    )));
497                }
498
499                if !seen_keys.insert(key.to_string()) {
500                    // this check is redundant as we ensure is_latest = true above
501                    return Err(S3Error(format!(
502                        "Duplicate version found for key {} under prefix {}",
503                        key, prefix
504                    )));
505                }
506            }
507
508            if response.is_truncated() != Some(true) {
509                break;
510            }
511
512            key_marker = response.next_key_marker().map(ToString::to_string);
513            version_id_marker = response.next_version_id_marker().map(ToString::to_string);
514
515            if key_marker.is_none() {
516                return Err(S3Error(format!(
517                    "Truncated response but no next_key_marker for prefix {}",
518                    prefix
519                )));
520            }
521        }
522
523        Ok(seen_keys.into_iter().collect())
524    }

526    /// Batch read. Callers must ensure that all objects with prefix `dir.to_string()` have
527    /// unexpired compliance-mode object locks.
528    ///
529    /// Each returned record's signed object key is checked against the actual
530    /// S3 key from which it was read.
531    pub async fn list_all_log_records_in_dir(
532        &self,
533        dir: &S3HourScopedDirectory,
534    ) -> GuardianResult<Vec<LogRecord>> {
535        let prefix = dir.to_string();
536        let keys = self.validate_prefix_history_and_list_keys(&prefix).await?;
537        let mut out = Vec::with_capacity(keys.len());
538        for key in keys {
539            // The prefix history was checked above. Immutable batch logs are
540            // also expected to remain under Compliance lock.
541            out.push(
542                self.get_log_record_inner(&key, LockCheck::Required, HistoryCheck::AlreadyChecked)
543                    .await?,
544            );
545        }
546        Ok(out)
547    }

549    /// Fetches and deserializes a record with explicit S3 integrity checks,
550    /// always rejecting a mismatch between its signed intended key and the
551    /// actual S3 key. `HistoryCheck::AlreadyChecked` requires the caller to have
552    /// validated the key's enclosing prefix.
553    pub(crate) async fn get_log_record_inner(
554        &self,
555        key: &str,
556        lock_check: LockCheck,
557        history_check: HistoryCheck,
558    ) -> GuardianResult<LogRecord> {
559        if matches!(history_check, HistoryCheck::Required) {
560            let keys = self.validate_prefix_history_and_list_keys(key).await?;
561            if keys.len() != 1 || keys[0] != key {
562                return Err(S3Error(format!(
563                    "expected exactly one object for key {}, found {:?}",
564                    key, keys
565                )));
566            }
567        }
568
569        let response = self
570            .client
571            .get_object()
572            .bucket(self.config.bucket_name())
573            .key(key)
574            .send()
575            .await
576            .map_err(|e| {
577                S3Error(format!(
578                    "Failed to get object {}: {}",
579                    key,
580                    DisplayErrorContext(&e)
581                ))
582            })?;
583
584        if matches!(lock_check, LockCheck::Required)
585            && !has_unexpired_compliance_lock(
586                response.object_lock_mode(),
587                response.object_lock_retain_until_date(),
588                SystemTime::now(),
589            )
590        {
591            return Err(S3Error(format!(
592                "Missing, invalid, or expired object lock metadata for key {}",
593                key
594            )));
595        }
596
597        let bytes = response.body.collect().await.map_err(|e| {
598            S3Error(format!(
599                "Failed to read object body for key {}: {}",
600                key,
601                DisplayErrorContext(&e)
602            ))
603        })?;
604
605        let record = serde_json::from_slice::<LogRecord>(&bytes.into_bytes()).map_err(|e| {
606            S3Error(format!(
607                "Failed to deserialize object {} into target type: {}",
608                key, e
609            ))
610        })?;
611        record.validate_actual_object_key(key)?;
612        Ok(record)
613    }

615    /// Reads an immutable-log object, asserting its Compliance lock is unexpired.

617        self.get_log_record_inner(key, LockCheck::Required, HistoryCheck::Required)
618            .await

621    /// Resolve a session's [`VerifiedSessionInfo`]: read the AWS-self-signed
622    /// attestation (anchoring the signing pubkey), then the signed `GuardianInfo`,
623    /// and pin the attestation's PCR0 against the `allowlist` entry named by the
624    /// info's reported build. No caller needs the raw attestation bytes.
625    pub(crate) async fn get_verified_session_info(
626        &self,
627        session_id: &str,
628        allowlist: &PcrAllowlist,
629    ) -> GuardianResult<VerifiedSessionInfo> {
630        // 1. Attestation (unsigned: authenticated by AWS, not the enclave key) →
631        //    the signing pubkey it commits to.
632        let att_key = InitLogMessage::attestation_object_key(session_id);
633        let attestation_record = self.get_log_record(&att_key).await?;
634        let (_, _, attestation_message) = attestation_record.validate_unsigned()?;
635        let (attestation, signing_pubkey) = attestation_message
636            .into_init_log()
637            .and_then(|x| match x {
638                InitLogMessage::OIAttestationUnsigned {
639                    attestation,
640                    signing_public_key,
641                } => Some((attestation, signing_public_key)),
642                _ => None,
643            })
644            .ok_or_else(|| S3Error(format!("expected OIAttestationUnsigned at key {att_key}")))?;
645
646        // 2. GuardianInfo, signature-verified under that pubkey → the reported build.
647        let info_key = InitLogMessage::guardian_info_object_key(session_id);
648        let info_record = self.get_log_record(&info_key).await?;
649        let (_, _, info_message) = info_record.verify(&signing_pubkey)?;
650        let info = info_message
651            .into_init_log()
652            .and_then(|x| match x {
653                InitLogMessage::OIGuardianInfo(info) => Some(*info),
654                _ => None,
655            })
656            .ok_or_else(|| S3Error(format!("expected OIGuardianInfo at key {info_key}")))?;
657
658        // 3. Anchor the pubkey and pin PCR0 to the allowlist entry for the
659        //    reported build. This replays a logged attestation whose short-lived
660        //    leaf cert has typically expired, so the chain is checked at the
661        //    document's own signed timestamp, not now.
662        let build_pcrs = allowlist.resolve(&info.untrusted_git_revision)?.clone();
663        attestation.verify_replay(&signing_pubkey, &build_pcrs)?;
664
665        Ok(VerifiedSessionInfo {
666            signing_pubkey,
667            info,
668            build_pcrs,
669        })
670    }

674// retain-until date covers the record timestamp plus this log type's duration
675// from `object_lock_policy`.
676fn has_unexpired_compliance_lock(
677    mode: Option<&ObjectLockMode>,
678    retain_until: Option<&DateTime>,
679    now: SystemTime,
680) -> bool {
681    mode == Some(&ObjectLockMode::Compliance)
682        && retain_until.is_some_and(|retain_until| *retain_until > DateTime::from(now))
683}

925    #[test]
926    fn compliance_lock_expiry_is_strict() {
927        let now = SystemTime::UNIX_EPOCH + Duration::from_secs(1_000);
928        let expired = DateTime::from(now - Duration::from_secs(1));
929        let expires_now = DateTime::from(now);
930        let unexpired = DateTime::from(now + Duration::from_secs(1));
931
932        assert!(!has_unexpired_compliance_lock(
933            Some(&ObjectLockMode::Compliance),
934            Some(&expired),
935            now,
936        ));
937        assert!(!has_unexpired_compliance_lock(
938            Some(&ObjectLockMode::Compliance),
939            Some(&expires_now),
940            now,
941        ));
942        assert!(has_unexpired_compliance_lock(
943            Some(&ObjectLockMode::Compliance),
944            Some(&unexpired),
945            now,
946        ));
947    }
