4//! `operator_init`: receives S3 config and (in withdraw mode) the stable
5//! `InitConfig`, installs arming state, and writes the init logs. Enabled in
6//! both modes. The companion `provisioner_init` (withdraw-only) lives in
7//! `withdraw::provisioner_init`.

22/// Complete operator-init state ready for its fail-stop commit.

28/// Withdraw-mode arming state built from `InitConfig` and the ceremony logs.
29pub struct OIWithdrawModeInstall {

32    genesis_state: Option<GenesisState>,
33}

35impl OIInstall {

39            withdraw_mode,

44impl OIWithdrawModeInstall {
45    pub fn from_parts(
46        init_config: InitConfig,
47        ceremony_state: CeremonyState,
48        genesis_state: Option<GenesisState>,
49    ) -> Self {
50        Self {
51            init_config,
52            ceremony_state,
53            genesis_state,
54        }
55    }

57    /// Build the arming bundle from the stable config and S3-derived ceremony +
58    /// KP share state.
59    pub async fn from_config(
60        logger: &GuardianS3Client,
61        config: InitConfig,
62        genesis_state: Option<GenesisState>,
63    ) -> GuardianResult<Self> {
64        let mut reader =
65            GuardianReader::from_s3_client(logger.clone(), config.pcr_allowlist().clone());
66        let ceremony_state = reader
67            .read_latest_ceremony_state(BuildPolicy::AnyAllowlisted)
68            .await
69            .map_err(|e| InvalidInputs(format!("read latest ceremony and KP share state: {e}")))?
70            .ok_or_else(|| InvalidInputs("no ceremony log found for withdraw init".into()))?;
71
72        Ok(Self::from_parts(config, ceremony_state, genesis_state))
73    }
74
75    /// Install the bundle onto a fresh enclave. Infallible by design (see the
76    /// `operator_init` invariant): every set runs once on a fresh enclave.
77    pub fn install_into(self, enclave: &Enclave) {
78        let config_hash = self.init_config.digest();
79        let (limiter_config, hashi_btc_master_pubkey, pcr_allowlist, network) =
80            self.init_config.into_parts();
81
82        info!(
83            "Setting secret-sharing instance: n={}, t={}, {} commitments.",
84            self.ceremony_state.secret_sharing_instance.num_shares(),
85            self.ceremony_state.secret_sharing_instance.threshold(),
86            self.ceremony_state
87                .secret_sharing_instance
88                .commitments()
89                .len()
90        );
91        if let Some(genesis_state) = &self.genesis_state {
92            info!(
93                genesis_state_hash = hex::encode(genesis_state.digest()),
94                "Storing genesis state."
95            );
96        }
97        enclave
98            .set_temporary_init_state(TemporaryInitState {
99                ceremony_state: self.ceremony_state,
100                genesis_state: self.genesis_state,
101                config_hash,
102            })
103            .expect("Unable to set temporary initialization state");
104
105        info!(?network, "Setting enclave configuration.");
106        enclave
107            .install_config(
108                network,
109                hashi_btc_master_pubkey,
110                pcr_allowlist,
111                limiter_config,
112            )
113            .expect("Unable to set enclave configuration");
114    }
115}

117/// Receives S3 API keys and — for a withdraw-mode enclave — the stable `InitConfig`;
118/// installs arming state and fixes the `config_hash`. A ceremony enclave carries only S3.
119///
120/// Invariant: operator_init never returns an `Err` from a partially-initialized
121/// enclave. Every fallible step (request validation, S3 connectivity, S3 reads)
122/// runs before any state is mutated, so an early `Err` leaves the enclave
123/// untouched and retryable. The mutation then happens entirely in
124/// `commit_operator_init`, which returns `()` — it cannot report an error, so a
125/// half-mutated enclave is never observed via an `Err`.
126/// Validate and commit operator initialization under the cancellation-safe
127/// control lock so concurrent callers cannot race the check-then-commit.
128pub async fn operator_init(
129    enclave: Arc<Enclave>,
130    request: OperatorInitRequest,
131) -> GuardianResult<()> {
132    info!("/operator_init - Received request.");
133
134    let uninitialized = match enclave.mode() {
135        EnclaveMode::Ceremony => CeremonyStage::Uninitialized.into(),
136        EnclaveMode::Withdraw => WithdrawStage::Uninitialized.into(),
137    };
138    enclave.require_lifecycle(uninitialized)?;
139    info!("Lifecycle stage validated.");
140
141    // ---- Validate & build: Nothing in this phase mutates enclave state, so any
142    // error here leaves the enclave untouched. ----
143
144    // A withdraw-mode enclave must carry the config and a ceremony enclave must not.
145    request.validate(enclave.mode())?;
146
147    let (s3_config, init_config, genesis_state) = request.into_parts();
148    let logger = GuardianS3Client::new_checked(&s3_config).await?;
149    info!("S3 connectivity check complete.");
150
151    // Build the withdraw-mode install bundle up front; `None` for a ceremony enclave.
152    let withdraw_mode = match init_config {
153        Some(config) => {
154            Some(OIWithdrawModeInstall::from_config(&logger, config, genesis_state).await?)
155        }
156        None => None,
157    };
158    let install = OIInstall::new(logger, withdraw_mode);
159
160    // ---- All-or-nothing Commit: Nothing in this phase errors out. ----
161    info!("Committing S3 logger and mode-specific initialization state.");
162    commit_operator_init(&enclave, install).await;
163
164    info!("Operator initialization complete.");
165    Ok(())
166}

168/// Install the validated config on the enclave and write the operator_init logs.
169/// Infallible by design (returns `()`, see the `operator_init` invariant): every
170/// `set` here runs on a fresh enclave under the control lock, and the I/O steps
171/// (attestation, S3 logging) panic on failure rather than return.
172async fn commit_operator_init(enclave: &Enclave, install: OIInstall) {
173    let OIInstall {
174        logger,
175        withdraw_mode,
176    } = install;
177
178    enclave
179        .config
180        .set_s3_logger(logger)
181        .expect("Unable to set logger");
182
183    // A ceremony enclave has no withdraw-mode arming state.
184    if let Some(withdraw_mode) = withdraw_mode {
185        withdraw_mode.install_into(enclave);
186    }
187
188    // Log to S3!
189    // 1) Attestation and pub key help authenticate all subsequent enclave-signed messages.
190    let signing_pk = enclave.signing_pubkey();
191    enclave
192        .log_init(OIAttestationUnsigned {
193            attestation: get_attestation(&signing_pk).expect("Unable to get attestation"),
194            signing_public_key: signing_pk,
195        })
196        .await
197        .expect("Unable to log OperatorInitAttestationUnsigned");
198
199    // 2) Share commitments help KPs confirm that the right private key will be constructed.
200    // This pre-transition snapshot reports `Uninitialized`; successfully
201    // writing it completes operator initialization.
202    // TODO(testnet-wipe): Replace the full GuardianInfo snapshot with a
203    // purpose-built OI payload containing only the data readers and KPs need;
204    // the evolving status response should not define the durable log schema.
205    // Include the retention environment in that payload for audit provenance.
206    enclave
207        .log_init(OIGuardianInfo(Box::new(enclave.info().await)))
208        .await
209        .expect("Unable to log GuardianInfo");
210
211    let initialized = match enclave.mode() {
212        EnclaveMode::Ceremony => CeremonyStage::OperatorInitialized.into(),
213        EnclaveMode::Withdraw => WithdrawStage::OperatorInitialized.into(),
214    };
215    enclave
216        .advance_lifecycle_into(initialized)
217        .expect("operator_init should advance an uninitialized enclave");
218}

225    /// Run commit_operator_init on a fresh enclave for the given mode (withdraw =>
226    /// carries the InitConfig install bundle; ceremony => none).
227    async fn commit_for_mode(mode: EnclaveMode) -> (Arc<Enclave>, CapturedPuts) {
228        let enclave = Arc::new(Enclave::new(
229            GuardianSignKeyPair::new(rand::thread_rng()),
230            GuardianEncKeyPair::random(&mut rand::thread_rng()),
231            mode,
232        ));
233
234        let (logger, captures) = crate::test_utils::mock_logger_capturing();
235        let install = match mode {
236            EnclaveMode::Withdraw => {
237                let config = InitConfig::mock_for_testing(None);
238                let args = crate::test_utils::OperatorInitTestArgs::default();
239                OIInstall::new(
240                    logger,
241                    Some(OIWithdrawModeInstall::from_parts(
242                        config,
243                        args.ceremony_state,
244                        None,
245                    )),
246                )
247            }
248            EnclaveMode::Ceremony => OIInstall::new(logger, None),
249        };
250
251        commit_operator_init(&enclave, install).await;
252        (enclave, captures)
253    }
254
255    fn assert_operator_init_logs(
256        enclave: &Enclave,
257        captures: &CapturedPuts,
258        expected_lifecycle: EnclaveLifecycle,
259    ) {
260        let captured = captures.lock().unwrap();
261        assert_eq!(captured.len(), 2, "operator init should write two records");
262        let session_id = enclave.s3_session_id();
263        assert_eq!(
264            captured[0].0,
265            InitLogMessage::attestation_object_key(&session_id)
266        );
267        assert_eq!(
268            captured[1].0,
269            InitLogMessage::guardian_info_object_key(&session_id)
270        );
271
272        let attestation: LogRecord = serde_json::from_slice(&captured[0].1).unwrap();
273        assert!(matches!(
274            attestation.message,
275            VersionedLogMessage::V2(LogMessage::Init(message))
276                if matches!(*message, OIAttestationUnsigned { .. })
277        ));
278
279        let guardian_info: LogRecord = serde_json::from_slice(&captured[1].1).unwrap();
280        let VersionedLogMessage::V2(LogMessage::Init(message)) = guardian_info.message else {
281            panic!("expected V2 init record");
282        };
283        let OIGuardianInfo(info) = *message else {
284            panic!("expected operator-init GuardianInfo record");
285        };
286        assert_eq!(info.lifecycle, expected_lifecycle);
287    }
288
289    #[tokio::test]
290    async fn commit_marks_operator_init_complete_withdraw_mode() {
291        let (enclave, captures) = commit_for_mode(EnclaveMode::Withdraw).await;
292        assert_eq!(
293            enclave.lifecycle(),
294            WithdrawStage::OperatorInitialized.into()
295        );
296        assert_operator_init_logs(&enclave, &captures, WithdrawStage::Uninitialized.into());
297    }
