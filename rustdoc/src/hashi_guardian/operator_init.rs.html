4//! `operator_init`: receives S3 config and (in withdraw mode) the
5//! `WithdrawModeConfig`, installs them, and writes the init logs. Enabled in
6//! both modes. The companion `provisioner_init` (withdraw-only) lives in
7//! `withdraw::provisioner_init`.
8
9use crate::attestation::get_attestation;
10use crate::Enclave;
11use crate::GuardianS3Client;
12use hashi_types::bitcoin::HashiMasterG;
13use hashi_types::guardian::InitLogMessage::OIAttestationUnsigned;
14use hashi_types::guardian::InitLogMessage::OIGuardianInfo;
15use hashi_types::guardian::*;
16use std::sync::Arc;
17use tracing::info;
18use GuardianError::*;

20/// The withdraw-mode state to install, built from `WithdrawModeConfig` (incl. the
21/// computed `state_hash` and the constructed rate limiter).
22pub(crate) struct WithdrawInstall {
23    state_hash: [u8; 32],
24    network: bitcoin::Network,
25    secret_sharing_instance: SecretSharingInstance,
26    hashi_btc_master_pubkey: HashiMasterG,
27    committee: HashiCommittee,
28    rate_limiter: RateLimiter,
29}
30
31impl WithdrawInstall {
32    /// Decompose a `WithdrawModeConfig` into the install bundle, constructing the
33    /// rate limiter (the only fallible step). Shared by `operator_init` and tests.
34    pub(crate) fn from_config(config: WithdrawModeConfig) -> GuardianResult<Self> {
35        let (withdraw_state, secret_sharing_instance, network) = config.into_parts();
36        let state_hash = withdraw_state.digest();
37        let (committee, limiter_config, limiter_state, hashi_btc_master_pubkey) =
38            withdraw_state.into_parts();
39        let rate_limiter = RateLimiter::new(limiter_config, limiter_state)?;
40        Ok(Self {
41            state_hash,
42            network,
43            secret_sharing_instance,
44            hashi_btc_master_pubkey,
45            committee,
46            rate_limiter,
47        })
48    }
49
50    /// Install the bundle onto a fresh enclave. Infallible by design (see the
51    /// `operator_init` invariant): every set runs once on a fresh enclave.
52    pub(crate) fn install_into(self, enclave: &Enclave) {
53        info!("Setting bitcoin network to {:?}.", self.network);
54        enclave
55            .config
56            .set_bitcoin_network(self.network)
57            .expect("Unable to set network");
58
59        info!(
60            "Storing secret-sharing instance: n={}, t={}, {} commitments.",
61            self.secret_sharing_instance.num_shares(),
62            self.secret_sharing_instance.threshold(),
63            self.secret_sharing_instance.commitments().len()
64        );
65        enclave
66            .set_secret_sharing_instance(self.secret_sharing_instance)
67            .expect("Unable to set secret-sharing instance");
68
69        info!("Setting state hash.");
70        enclave
71            .set_state_hash(self.state_hash)
72            .expect("Unable to set state hash");
73
74        info!("Setting hashi BTC master pubkey.");
75        enclave
76            .config
77            .set_hashi_btc_pk(self.hashi_btc_master_pubkey)
78            .expect("Unable to set hashi BTC master pubkey");
79
80        info!("Installing committee and rate limiter.");
81        enclave
82            .state
83            .init(self.committee, self.rate_limiter)
84            .expect("Unable to init enclave state");
85    }
86}

88/// Receives S3 API keys and — for a withdraw-mode enclave — the `WithdrawModeConfig`
89/// (committee, limiter, BTC master pubkey, secret-sharing instance, network);
90/// installs them and fixes the `state_hash`. A ceremony enclave carries only S3.
91///
92/// Invariant: operator_init never returns an `Err` from a partially-initialized
93/// enclave. Every fallible step (request validation, S3 connectivity, rate-limiter
94/// construction) runs before any state is mutated, so an early `Err` leaves the
95/// enclave untouched and retryable. The mutation then happens entirely in
96/// `commit_operator_init`, which returns `()` — it cannot report an error, so a
97/// half-mutated enclave is never observed via an `Err`.
98pub async fn operator_init(
99    enclave: Arc<Enclave>,
100    request: OperatorInitRequest,
101) -> GuardianResult<()> {
102    info!("/operator_init - Received request.");
103
104    // Serialize so concurrent callers can't race the check-then-commit below.
105    let _guard = enclave.control_lock.lock().await;
106
107    if enclave.is_operator_init_complete() {
108        return Err(InvalidInputs("operator_init already complete".into()));
109    }
110
111    // ---- Validate & build: Nothing in this phase mutates enclave state, so any
112    // error here leaves the enclave untouched. ----
113
114    // A withdraw-mode enclave must carry the config and a ceremony enclave must not.
115    request.validate(enclave.mode())?;
116
117    let (s3_config, state) = request.into_parts();
118    let logger = GuardianS3Client::new_checked(&s3_config).await?;
119    info!("S3 connectivity check complete.");
120
121    // Build the withdraw-mode install bundle (incl. the rate limiter, the last
122    // fallible step) up front; `None` for a ceremony enclave.
123    let withdraw = match state {
124        Some(config) => Some(WithdrawInstall::from_config(config)?),
125        None => None,
126    };
127
128    // ---- All-or-nothing Commit: Nothing in this phase errors out. ----
129    commit_operator_init(&enclave, logger, withdraw).await;
130    Ok(())
131}

133/// Install the validated config on the enclave and write the operator_init logs.
134/// Infallible by design (returns `()`, see the `operator_init` invariant): every
135/// `set` here runs on a fresh enclave under the control lock, and the I/O steps
136/// (attestation, S3 logging) panic on failure rather than return.
137async fn commit_operator_init(
138    enclave: &Enclave,
139    logger: GuardianS3Client,
140    withdraw: Option<WithdrawInstall>,
141) {
142    info!("Storing S3 configuration.");
143    enclave
144        .config
145        .set_s3_logger(logger)
146        .expect("Unable to set logger");
147
148    // Withdraw-mode state (committee, limiter, BTC master pubkey, instance, network,
149    // state_hash); a ceremony enclave installs none of it.
150    if let Some(install) = withdraw {
151        install.install_into(enclave);
152    }
153
154    // Log to S3!
155    // 1) Attestation and pub key help authenticate all subsequent enclave-signed messages.
156    let signing_pk = enclave.signing_pubkey();
157    enclave
158        .log_init(OIAttestationUnsigned {
159            attestation: get_attestation(&signing_pk).expect("Unable to get attestation"),
160            signing_public_key: signing_pk,
161        })
162        .await
163        .expect("Unable to log OperatorInitAttestationUnsigned");
164
165    // 2) Share commitments help KPs confirm that the right private key will be constructed.
166    enclave
167        .log_init(OIGuardianInfo(Box::new(enclave.info().await)))
168        .await
169        .expect("Unable to log GuardianInfo");
170
171    enclave
172        .scratchpad
173        .operator_init_logging_complete
174        .set(())
175        .expect("operator_init_logging_complete should only be set once");
176
177    info!("Operator initialization complete.");
178}

180#[cfg(test)]
181mod tests {
182    use super::*;
183
184    /// Run commit_operator_init on a fresh enclave for the given mode (withdraw =>
185    /// carries the WithdrawModeConfig install bundle; ceremony => none).
186    async fn commit_for_mode(mode: EnclaveMode) -> Arc<Enclave> {
187        let enclave = Arc::new(Enclave::new(
188            GuardianSignKeyPair::new(rand::thread_rng()),
189            GuardianEncKeyPair::random(&mut rand::thread_rng()),
190            mode,
191        ));
192
193        let withdraw = match mode {
194            EnclaveMode::Withdraw => Some(
195                WithdrawInstall::from_config(WithdrawModeConfig::mock_for_testing(None)).unwrap(),
196            ),
197            EnclaveMode::Ceremony => None,
198        };
199
200        commit_operator_init(&enclave, crate::test_utils::mock_logger(), withdraw).await;
201        enclave
202    }
203
204    #[tokio::test]
205    async fn commit_marks_operator_init_complete_withdraw_mode() {
206        let enclave = commit_for_mode(EnclaveMode::Withdraw).await;
207        assert!(enclave.is_operator_init_complete());
208    }
209
210    #[tokio::test]
211    async fn commit_marks_operator_init_complete_ceremony_mode() {
212        let enclave = commit_for_mode(EnclaveMode::Ceremony).await;
213        assert!(enclave.is_operator_init_complete());
214    }
