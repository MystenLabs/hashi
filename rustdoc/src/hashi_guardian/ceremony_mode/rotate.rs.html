15/// Operator-relayed setup-mode rotation: the operator submits the current KPs'
16/// encrypted old shares in one call. The enclave verifies them, reconstructs and
17/// re-splits the BTC key to the new KP set, logs the new instance, returns the new shares.
18pub async fn rotate_kps(
19    enclave: Arc<Enclave>,
20    request: RotateKpsRequest,
21) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
22    info!("/rotate_kps - Received request.");
23
24    enclave.require_lifecycle(CeremonyStage::OperatorInitialized.into())?;
25
26    let (encrypted_old_shares, old_instance, state) = request.into_parts();
27    let old_t = old_instance.threshold();
28
29    let sk = enclave.encryption_secret_key();
30    let state_hash = state.digest();
31
32    // Decrypt and verify every submission. A share only decrypts if its KP bound
33    // this same `state` as AAD, so the decrypted shares all agree on the target.
34    // TODO: Move RotateKps to the PI model: collect typed, signed per-KP
35    // submissions and verify them in the enclave. That first requires a
36    // per-KP rotation submission/aggregation path; today the operator supplies
37    // this batch directly, so AAD remains its authorization binding.
38    let old_shares = decrypt_verify_shares(
39        &encrypted_old_shares,
40        sk,
41        Some(&state_hash),
42        old_instance.commitments(),
43        old_t,
44    )?;
45    info!(
46        "Verified {} old shares (threshold {old_t}).",
47        old_shares.len()
48    );
49
50    let response = finalize_rotation(&enclave, &old_shares, &old_instance, state).await?;
51    enclave
52        .advance_lifecycle_into(CeremonyStage::Completed.into())
53        .expect("rotate_kps should complete a ceremony lifecycle");
54    Ok(response)
55}

57async fn finalize_rotation(
58    enclave: &Arc<Enclave>,
59    old_shares: &[Share],
60    old_instance: &SecretSharingInstance,
61    state: RotateKpsState,
62) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
63    info!("Threshold reached, reconstructing BTC key.");
64
65    let k256_sk =
66        combine_shares(old_shares, old_instance.threshold()).expect("threshold shares reach");
67
68    // Rotation re-shares the same key, so its x-only pubkey is unchanged; record it.
69    let btc_master_pubkey = k256_sk_to_btc_xonly_pubkey(&k256_sk);
70
71    let (new_certs_roster, new_params) = state.into_parts();
72    let n = new_params.num_shares();
73    let t = new_params.threshold();
74    let certificate_count: usize = new_certs_roster
75        .iter()
76        .map(|set| set.pgp_certs().len())
77        .sum();
78    info!(
79        share_count = n,
80        threshold = t,
81        certificate_count,
82        "Received new key provisioner OpenPGP certificate roster."
83    );
84    for (index, cert_set) in new_certs_roster.iter().enumerate() {
85        info!(
86            share_id = index + 1,
87            certificate_count = cert_set.pgp_certs().len(),
88            recipient_fingerprints = ?cert_set.fingerprints(),
89            "Received new KP certificate set."
90        );
91    }
92
93    // Confine the !Send `ThreadRng` to a sync scope so the surrounding async
94    // future stays Send.
95    let (encrypted_shares, share_commitments) = {
96        let mut rng = rand::thread_rng();
97        split_and_encrypt_for_kps(&k256_sk, &new_certs_roster, &new_params, &mut rng)
98    };
99    info!(
100        share_count = encrypted_shares.share_count(),
101        ciphertext_count = encrypted_shares.ciphertext_count(),
102        "Re-encrypted each share once per new KP certificate."
103    );
104
105    let new_sharing_seq = old_instance.sharing_seq() + 1;
106    let new_instance = SecretSharingInstance::new(share_commitments, n, t, new_sharing_seq)?;
107    info!(
108        "Persisting rotation sharing_seq={new_sharing_seq} cert_seq=0 to kp-shares/ + ceremony/."
109    );
110    enclave
111        .log_kp_share_state(new_sharing_seq, 0, encrypted_shares.clone())
112        .await?;
113
114    enclave
115        .log_ceremony(CeremonyLogMessage::Rotate {
116            old_instance: old_instance.clone(),
117            new_instance,
118            btc_master_pubkey,
119        })
120        .await?;
121
122    info!("Rotation complete.");
123    Ok(enclave.sign(RotateKpsResponse { encrypted_shares }))
124}

182    /// Bundle one submission per share, all bound to `state.digest()` as AAD —
183    /// i.e. what the operator assembles from the current KPs.
184    fn build_request(
185        shares: &[Share],
186        enclave: &Enclave,
187        old_instance: &SecretSharingInstance,
188        state: RotateKpsState,
189    ) -> RotateKpsRequest {
190        let submissions = shares
191            .iter()
192            .map(|s| {
193                RotateKpsRequest::build_from_share_and_state(
194                    s,
195                    enclave.encryption_public_key(),
196                    &state,
197                    &mut rand::thread_rng(),
198                )
199            })
200            .collect();
201        RotateKpsRequest::new(submissions, old_instance.clone(), state)
202    }

204    /// Run one rotation and return its verified response shares.
205    async fn rotate_and_verify(
206        enclave: &Arc<Enclave>,
207        req: RotateKpsRequest,
208    ) -> KPEncryptedSharesRoster {
209        let signed = rotate_kps(enclave.clone(), req).await.expect("ok");
210        signed
211            .verify(&enclave.signing_pubkey())
212            .expect("response signed by enclave")
213            .encrypted_shares
214    }

216    /// Assert the rotation returned `new_n` PGP-armored shares and produced
217    /// exactly one `ceremony/` log at `sharing_seq = 1` carrying the instance
218    /// only (no ciphertexts).
219    fn assert_rotation_output(
220        captures: &CapturedPuts,
221        response_shares: &KPEncryptedSharesRoster,
222        secret_keys: &MockKpSecretKeys,
223        new_n: usize,
224        new_t: usize,
225    ) {
226        assert_eq!(response_shares.share_count(), new_n);
227        for enc in response_shares.iter() {
228            for ciphertext in enc.ciphertexts_by_fingerprint.values() {
229                assert!(
230                    ciphertext.starts_with("-----BEGIN PGP MESSAGE-----"),
231                    "expected a PGP-armored share in the response"
232                );
233            }
234        }
235
236        let captured = captures.lock().unwrap();
237        let ceremony_logs: Vec<_> = captured
238            .iter()
239            .filter(|(k, _)| k.starts_with("ceremony/"))
240            .collect();
241        assert_eq!(ceremony_logs.len(), 1, "expected one ceremony/ log");
242        let (key, body) = ceremony_logs[0];
243        assert!(
244            key.starts_with("ceremony/00000000000000000001-"),
245            "expected sharing_seq=1, got key {key}"
246        );
247        assert!(
248            !std::str::from_utf8(body)
249                .unwrap()
250                .contains("BEGIN PGP MESSAGE"),
251            "ceremony log must not contain ciphertexts"
252        );
253
254        let record: LogRecord = serde_json::from_slice(body).unwrap();
255        let VersionedLogMessage::V2(LogMessage::Ceremony(ceremony)) = record.message else {
256            panic!("expected V2 Ceremony variant");
257        };
258        let CeremonyLogMessage::Rotate {
259            old_instance,
260            new_instance,
261            btc_master_pubkey,
262        } = *ceremony
263        else {
264            panic!("expected Rotate variant");
265        };
266        // The consumed old instance is recorded for chain auditability.
267        assert_eq!(old_instance.sharing_seq(), 0);
268        assert_eq!(old_instance.num_shares(), TEST_N);
269        assert_eq!(old_instance.threshold(), TEST_T);
270        assert_eq!(new_instance.sharing_seq(), 1);
271        assert_eq!(new_instance.num_shares(), new_n);
272        assert_eq!(new_instance.threshold(), new_t);
273
274        let decrypted_shares = decrypt_kp_shares(response_shares, secret_keys);
275        for share in &decrypted_shares {
276            new_instance
277                .commitments()
278                .verify_share(share)
279                .expect("decrypted rotation share should match its commitment");
280        }
281        let reconstructed = combine_shares(&decrypted_shares[..new_t], new_t).unwrap();
282        assert_eq!(
283            k256_sk_to_btc_xonly_pubkey(&reconstructed),
284            btc_master_pubkey,
285            "threshold decrypted rotation shares should reconstruct the original key"
286        );
287
288        // The new shares are persisted to kp-shares/ keyed by the new sharing_seq
289        // and initial cert_seq=0.
290        let shares_logs: Vec<_> = captured
291            .iter()
292            .filter(|(k, _)| k.starts_with("kp-shares/"))
293            .collect();
294        assert_eq!(shares_logs.len(), 1, "expected one kp-shares/ log");
295        let (shares_key, shares_body) = shares_logs[0];
296        assert!(
297            shares_key.starts_with("kp-shares/00000000000000000001/00000000000000000000-"),
298            "expected sharing_seq=1 cert_seq=0, got key {shares_key}"
299        );
300        let shares_record: LogRecord = serde_json::from_slice(shares_body).unwrap();
301        let VersionedLogMessage::V2(LogMessage::KpShareState(shares)) = shares_record.message
302        else {
303            panic!("expected V2 KpShareState variant");
304        };
305        assert_eq!(shares.sharing_seq, 1);
306        assert_eq!(shares.cert_seq, 0);
307        assert_eq!(shares.encrypted_shares, *response_shares);
308        assert_eq!(shares.encrypted_shares.share_count(), new_n);
309    }

330    #[tokio::test]
331    async fn rejects_second_call_after_complete() {
332        let (shares, old_instance, captures, enclave) = setup_rotation_enclave().await;
333
334        // First call reaches threshold and finalizes.
335        let req = build_request(&shares[..TEST_T], &enclave, &old_instance, build_state());
336        rotate_kps(enclave.clone(), req).await.expect("ok");
337
338        // A second call is rejected outright — no re-split.
339        let req2 = build_request(&shares[..TEST_T], &enclave, &old_instance, build_state());
340        let err = rotate_kps(enclave, req2).await.expect_err("should reject");
341        assert!(matches!(err, InvalidInputs(_)));
342
343        let captured = captures.lock().unwrap();
344        let count = captured
345            .iter()
346            .filter(|(k, _)| k.starts_with("ceremony/"))
347            .count();
348        assert_eq!(count, 1, "rotation must finalize exactly once");
349    }

399        // A submission bound to a different `state` won't decrypt under the
400        // request's `state` AAD, so the whole request is rejected gracefully

406
407        let mut rng = rand::thread_rng();
