15/// Operator-relayed setup-mode rotation: the operator submits the current KPs'
16/// encrypted old shares in one call. The enclave verifies them, reconstructs and
17/// re-splits the BTC key to the new KP set, logs the new instance, returns the new shares.
18pub async fn rotate_kps(
19    enclave: Arc<Enclave>,
20    request: RotateKpsRequest,
21) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
22    info!("/rotate_kps - Received request.");
23
24    // Hold the ceremony guard for the whole call: blocks a concurrent
25    // setup_new_key, and the flag rejects re-entry after one ceremony finalizes.
26    let mut ceremony_complete = enclave.scratchpad.ceremony_complete.lock().await;
27    if !enclave.is_operator_init_complete() {
28        return Err(InvalidInputs("call operator_init first".into()));
29    }
30    if *ceremony_complete {
31        return Err(InvalidInputs("setup or rotation already complete".into()));
32    }
33
34    let (encrypted_old_shares, old_instance, state) = request.into_parts();
35    let old_t = old_instance.threshold();
36
37    let sk = enclave.encryption_secret_key();
38    let state_hash = state.digest();
39
40    // Decrypt and verify every submission. A share only decrypts if its KP bound
41    // this same `state` as AAD, so the decrypted shares all agree on the target.
42    let old_shares = decrypt_verify_shares(
43        &encrypted_old_shares,
44        sk,
45        &state_hash,
46        old_instance.commitments(),
47        old_t,
48    )?;
49    info!(
50        "Verified {} old shares (threshold {old_t}).",
51        old_shares.len()
52    );
53
54    let response = finalize_rotation(&enclave, &old_shares, &old_instance, state).await?;
55    *ceremony_complete = true;
56    Ok(response)
57}
58
59async fn finalize_rotation(
60    enclave: &Arc<Enclave>,
61    old_shares: &[Share],
62    old_instance: &SecretSharingInstance,
63    state: RotateKpsState,
64) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
65    info!("Threshold reached, reconstructing BTC key.");
66
67    let k256_sk =
68        combine_shares(old_shares, old_instance.threshold()).expect("threshold shares reach");
69
70    let (new_certs, new_params) = state.into_parts();
71    let n = new_params.num_shares();
72    let t = new_params.threshold();
73    info!("Re-splitting for {n} new KPs (threshold: {t}).");
74
75    // Confine the !Send `ThreadRng` to a sync scope so the surrounding async
76    // future stays Send.
77    let (encrypted_shares, share_commitments) = {
78        let mut rng = rand::thread_rng();
79        split_and_encrypt_for_kps(&k256_sk, &new_certs, &new_params, &mut rng)
80    };
81
82    let new_sharing_seq = old_instance.sharing_seq() + 1;
83    let new_instance = SecretSharingInstance::new(share_commitments, n, t, new_sharing_seq)?;
84    info!("Writing CeremonyLogMessage sharing_seq={new_sharing_seq} to ceremony/.");
85    enclave
86        .log_ceremony(CeremonyLogMessage::Rotate {
87            old_instance: old_instance.clone(),
88            new_instance,
89        })
90        .await?;
91
92    enclave.set_latest_encrypted_shares(encrypted_shares.clone())?;
93
94    info!("Rotation complete.");
95    Ok(enclave.sign(RotateKpsResponse { encrypted_shares }))
96}

113    /// Build (original sk, old shares, captures, enclave).
114    async fn setup_rotation_enclave() -> (SecretKey, Vec<Share>, CapturedPuts, Arc<Enclave>) {
115        let sk = SecretKey::random(&mut rand::thread_rng());
116        let params = SecretSharingParams::new(TEST_N, TEST_T).unwrap();
117        let shares = split_secret(&sk, &params, &mut rand::thread_rng());
118        let commitments = ShareCommitments::from_shares(&shares).unwrap();
119        let (logger, captures) = mock_logger_capturing();
120        let enclave = Enclave::create_operator_initialized_with(
121            OperatorInitTestArgs::default()
122                .with_commitments(commitments)
123                .with_s3_logger(logger),
124        )
125        .await;
126        (sk, shares, captures, enclave)
127    }

133    /// Bundle one submission per share, all bound to `state.digest()` as AAD —
134    /// i.e. what the operator assembles from the current KPs.
135    fn build_request(
136        shares: &[Share],
137        enclave: &Enclave,
138        state: RotateKpsState,
139    ) -> RotateKpsRequest {
140        let submissions = shares
141            .iter()
142            .map(|s| {
143                RotateKpsRequest::build_from_share_and_state(
144                    s,
145                    enclave.encryption_public_key(),
146                    &state,
147                    &mut rand::thread_rng(),
148                )
149            })
150            .collect();
151        // The old instance the operator would read from `ceremony/`; here it's the
152        // one the enclave was set up with (matches the shares being submitted).
153        let old_instance = enclave.secret_sharing_instance().unwrap().clone();
154        RotateKpsRequest::new(submissions, old_instance, state)
155    }

157    /// Run one rotation and return its verified response shares.
158    async fn rotate_and_verify(
159        enclave: &Arc<Enclave>,
160        req: RotateKpsRequest,
161    ) -> Vec<KPEncryptedShare> {
162        let signed = rotate_kps(enclave.clone(), req).await.expect("ok");
163        signed
164            .verify(&enclave.signing_pubkey())
165            .expect("response signed by enclave")
166            .encrypted_shares
167    }

169    /// Assert the rotation returned `new_n` PGP-armored shares and produced
170    /// exactly one `ceremony/` log at `sharing_seq = 1` carrying the instance
171    /// only (no ciphertexts).
172    ///
173    /// TODO: strengthen this (and setup_new_key's test) to decrypt the armored
174    /// shares with the new KPs' PGP secret keys and verify they reconstruct the
175    /// original BTC key, once a PGP-decrypt test helper exists.
176    fn assert_rotation_output(
177        captures: &CapturedPuts,
178        response_shares: &[KPEncryptedShare],
179        new_n: usize,
180        new_t: usize,
181    ) {
182        assert_eq!(response_shares.len(), new_n);
183        for enc in response_shares {
184            assert!(
185                enc.armored_ciphertext
186                    .starts_with("-----BEGIN PGP MESSAGE-----"),
187                "expected a PGP-armored share in the response"
188            );
189        }
190
191        let captured = captures.lock().unwrap();
192        let ceremony_logs: Vec<_> = captured
193            .iter()
194            .filter(|(k, _)| k.starts_with("ceremony/"))
195            .collect();
196        assert_eq!(ceremony_logs.len(), 1, "expected one ceremony/ log");
197        let (key, body) = ceremony_logs[0];
198        assert!(
199            key.starts_with("ceremony/00000000000000000001-"),
200            "expected sharing_seq=1, got key {key}"
201        );
202        assert!(
203            !std::str::from_utf8(body)
204                .unwrap()
205                .contains("BEGIN PGP MESSAGE"),
206            "ceremony log must not contain ciphertexts"
207        );
208
209        let record: LogRecord = serde_json::from_slice(body).unwrap();
210        let LogMessage::Ceremony(ceremony) = record.message else {
211            panic!("expected Ceremony variant");
212        };
213        let CeremonyLogMessage::Rotate {
214            old_instance,
215            new_instance,
216        } = *ceremony
217        else {
218            panic!("expected Rotate variant");
219        };
220        // The consumed old instance is recorded for chain auditability.
221        assert_eq!(old_instance.sharing_seq(), 0);
222        assert_eq!(old_instance.num_shares(), TEST_N);
223        assert_eq!(old_instance.threshold(), TEST_T);
224        assert_eq!(new_instance.sharing_seq(), 1);
225        assert_eq!(new_instance.num_shares(), new_n);
226        assert_eq!(new_instance.threshold(), new_t);
227    }

247    #[tokio::test]
248    async fn rejects_second_call_after_complete() {
249        let (_sk, shares, captures, enclave) = setup_rotation_enclave().await;
250
251        // First call reaches threshold and finalizes.
252        let req = build_request(&shares[..TEST_T], &enclave, build_state());
253        rotate_kps(enclave.clone(), req).await.expect("ok");
254
255        // A second call is rejected outright — no re-split.
256        let req2 = build_request(&shares[..TEST_T], &enclave, build_state());
257        let err = rotate_kps(enclave, req2).await.expect_err("should reject");
258        assert!(matches!(err, InvalidInputs(_)));
259
260        let captured = captures.lock().unwrap();
261        let count = captured
262            .iter()
263            .filter(|(k, _)| k.starts_with("ceremony/"))
264            .count();
265        assert_eq!(count, 1, "rotation must finalize exactly once");
266    }

319
320        let mut rng = rand::thread_rng();
