15/// Operator-relayed setup-mode rotation: the operator submits the current KPs'
16/// encrypted old shares in one call. The enclave verifies them, reconstructs and
17/// re-splits the BTC key to the new KP set, logs the new instance, returns the new shares.
18pub async fn rotate_kps(
19    enclave: Arc<Enclave>,
20    request: RotateKpsRequest,
21) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
22    info!("/rotate_kps - Received request.");
23
24    // Hold the ceremony guard for the whole call: blocks a concurrent
25    // setup_new_key, and the flag rejects re-entry after one ceremony finalizes.
26    let mut ceremony_complete = enclave.scratchpad.ceremony_complete.lock().await;
27    if !enclave.is_operator_init_complete() {
28        return Err(InvalidInputs("call operator_init first".into()));
29    }
30    if *ceremony_complete {
31        return Err(InvalidInputs("setup or rotation already complete".into()));
32    }
33
34    let (encrypted_old_shares, old_instance, state) = request.into_parts();
35    let old_t = old_instance.threshold();
36
37    let sk = enclave.encryption_secret_key();
38    let state_hash = state.digest();
39
40    // Decrypt and verify every submission. A share only decrypts if its KP bound
41    // this same `state` as AAD, so the decrypted shares all agree on the target.
42    let old_shares = decrypt_verify_shares(
43        &encrypted_old_shares,
44        sk,
45        &state_hash,
46        old_instance.commitments(),
47        old_t,
48    )?;
49    info!(
50        "Verified {} old shares (threshold {old_t}).",
51        old_shares.len()
52    );
53
54    let response = finalize_rotation(&enclave, &old_shares, &old_instance, state).await?;
55    *ceremony_complete = true;
56    Ok(response)
57}

59async fn finalize_rotation(
60    enclave: &Arc<Enclave>,
61    old_shares: &[Share],
62    old_instance: &SecretSharingInstance,
63    state: RotateKpsState,
64) -> GuardianResult<GuardianSigned<RotateKpsResponse>> {
65    info!("Threshold reached, reconstructing BTC key.");
66
67    let k256_sk =
68        combine_shares(old_shares, old_instance.threshold()).expect("threshold shares reach");
69
70    let (new_certs, new_params) = state.into_parts();
71    let n = new_params.num_shares();
72    let t = new_params.threshold();
73    info!("Re-splitting for {n} new KPs (threshold: {t}).");
74
75    // Confine the !Send `ThreadRng` to a sync scope so the surrounding async
76    // future stays Send.
77    let (encrypted_shares, share_commitments) = {
78        let mut rng = rand::thread_rng();
79        split_and_encrypt_for_kps(&k256_sk, &new_certs, &new_params, &mut rng)
80    };
81
82    let new_sharing_seq = old_instance.sharing_seq() + 1;
83    let new_instance = SecretSharingInstance::new(share_commitments, n, t, new_sharing_seq)?;
84    info!("Persisting rotation sharing_seq={new_sharing_seq} to shares/ + ceremony/.");
85    enclave
86        .log_shares(new_sharing_seq, encrypted_shares.clone())
87        .await?;
88
89    enclave
90        .log_ceremony(CeremonyLogMessage::Rotate {
91            old_instance: old_instance.clone(),
92            new_instance,
93            roster: encrypted_shares.recipient_roster(),
94        })
95        .await?;
96
97    enclave
98        .set_latest_encrypted_shares(encrypted_shares.clone())
99        .expect("set_latest_encrypted_shares should work if ceremony_complete=false");
100
101    info!("Rotation complete.");
102    Ok(enclave.sign(RotateKpsResponse { encrypted_shares }))
103}

120    /// Build (original sk, old shares, captures, enclave).
121    async fn setup_rotation_enclave() -> (SecretKey, Vec<Share>, CapturedPuts, Arc<Enclave>) {
122        let sk = SecretKey::random(&mut rand::thread_rng());
123        let params = SecretSharingParams::new(TEST_N, TEST_T).unwrap();
124        let shares = split_secret(&sk, &params, &mut rand::thread_rng());
125        let commitments = ShareCommitments::from_shares(&shares).unwrap();
126        let (logger, captures) = mock_logger_capturing();

128            OperatorInitTestArgs::default()
129                .with_commitments(commitments)
130                .with_s3_logger(logger),
131        )
132        .await;
133        (sk, shares, captures, enclave)
134    }

140    /// Bundle one submission per share, all bound to `state.digest()` as AAD —
141    /// i.e. what the operator assembles from the current KPs.
142    fn build_request(
143        shares: &[Share],
144        enclave: &Enclave,
145        state: RotateKpsState,
146    ) -> RotateKpsRequest {
147        let submissions = shares
148            .iter()
149            .map(|s| {
150                RotateKpsRequest::build_from_share_and_state(
151                    s,
152                    enclave.encryption_public_key(),
153                    &state,
154                    &mut rand::thread_rng(),
155                )
156            })
157            .collect();
158        // The old instance the operator would read from `ceremony/`; here it's the
159        // one the enclave was set up with (matches the shares being submitted).
160        let old_instance = enclave.secret_sharing_instance().unwrap().clone();
161        RotateKpsRequest::new(submissions, old_instance, state)
162    }

164    /// Run one rotation and return its verified response shares.
165    async fn rotate_and_verify(enclave: &Arc<Enclave>, req: RotateKpsRequest) -> KPEncryptedShares {
166        let signed = rotate_kps(enclave.clone(), req).await.expect("ok");
167        signed
168            .verify(&enclave.signing_pubkey())
169            .expect("response signed by enclave")
170            .encrypted_shares
171    }

173    /// Assert the rotation returned `new_n` PGP-armored shares and produced
174    /// exactly one `ceremony/` log at `sharing_seq = 1` carrying the instance
175    /// only (no ciphertexts).
176    ///
177    /// TODO: strengthen this (and setup_new_key's test) to decrypt the armored
178    /// shares with the new KPs' PGP secret keys and verify they reconstruct the
179    /// original BTC key, once a PGP-decrypt test helper exists.
180    fn assert_rotation_output(
181        captures: &CapturedPuts,
182        response_shares: &[KPEncryptedShare],
183        new_n: usize,
184        new_t: usize,
185    ) {
186        assert_eq!(response_shares.len(), new_n);
187        for enc in response_shares {
188            assert!(
189                enc.armored_ciphertext
190                    .starts_with("-----BEGIN PGP MESSAGE-----"),
191                "expected a PGP-armored share in the response"
192            );
193        }
194
195        let captured = captures.lock().unwrap();
196        let ceremony_logs: Vec<_> = captured
197            .iter()
198            .filter(|(k, _)| k.starts_with("ceremony/"))
199            .collect();
200        assert_eq!(ceremony_logs.len(), 1, "expected one ceremony/ log");
201        let (key, body) = ceremony_logs[0];
202        assert!(
203            key.starts_with("ceremony/00000000000000000001-"),
204            "expected sharing_seq=1, got key {key}"
205        );
206        assert!(
207            !std::str::from_utf8(body)
208                .unwrap()
209                .contains("BEGIN PGP MESSAGE"),
210            "ceremony log must not contain ciphertexts"
211        );
212
213        let record: LogRecord = serde_json::from_slice(body).unwrap();
214        let LogMessage::Ceremony(ceremony) = record.message else {
215            panic!("expected Ceremony variant");
216        };
217        let CeremonyLogMessage::Rotate {
218            old_instance,
219            new_instance,
220            roster,
221        } = *ceremony
222        else {
223            panic!("expected Rotate variant");
224        };
225        // The consumed old instance is recorded for chain auditability.
226        assert_eq!(old_instance.sharing_seq(), 0);
227        assert_eq!(old_instance.num_shares(), TEST_N);
228        assert_eq!(old_instance.threshold(), TEST_T);
229        assert_eq!(new_instance.sharing_seq(), 1);
230        assert_eq!(new_instance.num_shares(), new_n);
231        assert_eq!(new_instance.threshold(), new_t);
232        // Roster commits one recipient fingerprint per new share.
233        assert_eq!(roster.len(), new_n);
234
235        // The new shares are persisted to shares/ keyed by the new sharing_seq.
236        let shares_logs: Vec<_> = captured
237            .iter()
238            .filter(|(k, _)| k.starts_with("shares/"))
239            .collect();
240        assert_eq!(shares_logs.len(), 1, "expected one shares/ log");
241        let (shares_key, shares_body) = shares_logs[0];
242        assert!(
243            shares_key.starts_with("shares/00000000000000000001-"),
244            "expected sharing_seq=1, got key {shares_key}"
245        );
246        let shares_record: LogRecord = serde_json::from_slice(shares_body).unwrap();
247        let LogMessage::Shares(shares) = shares_record.message else {
248            panic!("expected Shares variant");
249        };
250        assert_eq!(shares.sharing_seq, 1);
251        assert_eq!(shares.encrypted_shares.len(), new_n);
252    }

272    #[tokio::test]
273    async fn rejects_second_call_after_complete() {
274        let (_sk, shares, captures, enclave) = setup_rotation_enclave().await;
275
276        // First call reaches threshold and finalizes.
277        let req = build_request(&shares[..TEST_T], &enclave, build_state());
278        rotate_kps(enclave.clone(), req).await.expect("ok");
279
280        // A second call is rejected outright — no re-split.
281        let req2 = build_request(&shares[..TEST_T], &enclave, build_state());
282        let err = rotate_kps(enclave, req2).await.expect_err("should reject");
283        assert!(matches!(err, InvalidInputs(_)));
284
285        let captured = captures.lock().unwrap();
286        let count = captured
287            .iter()
288            .filter(|(k, _)| k.starts_with("ceremony/"))
289            .count();
290        assert_eq!(count, 1, "rotation must finalize exactly once");
291    }
