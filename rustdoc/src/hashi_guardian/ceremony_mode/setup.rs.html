7use hashi_types::guardian::*;
8use k256::SecretKey;
9use std::sync::Arc;
10use tracing::info;

12/// Set up a new BTC key. Flow:
13///     1. KPs send their OpenPGP certificates to the operator
14///     2. Operator calls setup_new_key
15///     3. KPs fetch commitments/roster from `ceremony/` and ciphertexts from `shares/`
16pub async fn setup_new_key(
17    enclave: Arc<Enclave>,
18    request: SetupNewKeyRequest,
19) -> GuardianResult<GuardianSigned<SetupNewKeyResponse>> {
20    info!("/setup_new_key - Received request.");
21    // Hold the ceremony guard across the whole flow so concurrent setup_new_key
22    // /rotate_kps callers can't both pass the completion check below and run.
23    let mut ceremony_complete = enclave.scratchpad.ceremony_complete.lock().await;
24    if !enclave.is_operator_init_complete() {
25        return Err(InvalidInputs("call operator_init first".into()));
26    }
27    if *ceremony_complete {
28        return Err(InvalidInputs("setup or rotation already complete".into()));
29    }
30
31    let params = request.params();
32    let n = params.num_shares();
33    let t = params.threshold();
34    let key_provisioner_certs = request.pgp_certs();
35    info!(
36        "Received {} OpenPGP certificates.",
37        key_provisioner_certs.len()
38    );
39
40    info!("Generating new Bitcoin private key.");
41    // Confine the !Send `ThreadRng` to a sync scope so the surrounding async
42    // future stays Send.
43    let (encrypted_shares, share_commitments, fingerprint_hex) = {
44        let mut rng = rand::thread_rng();
45        let sk = SecretKey::random(&mut rng);
46        let fp = format!("{:x}", fingerprint(&sk));
47        info!("Splitting secret into {n} shares (threshold: {t}).");
48        let (encrypted, commitments) =
49            split_and_encrypt_for_kps(&sk, key_provisioner_certs, params, &mut rng);
50        (encrypted, commitments, fp)
51    };
52    info!(
53        "Bitcoin key generated with fingerprint {}; all {} shares encrypted.",
54        fingerprint_hex, n
55    );
56
57    let ss_instance = SecretSharingInstance::new(share_commitments.clone(), n, t, 0)
58        .expect("(n, t) validated by SetupNewKeyRequest; commitments produced with matching count");
59
60    info!("Persisting setup sharing_seq=0 to shares/ + ceremony/.");
61    enclave.log_shares(0, encrypted_shares.clone()).await?;
62
63    enclave
64        .log_ceremony(CeremonyLogMessage::NewKey {
65            instance: ss_instance.clone(),
66            roster: encrypted_shares.recipient_roster(),
67        })
68        .await?;
69
70    enclave
71        .set_latest_encrypted_shares(encrypted_shares.clone())
72        .expect("set_latest_encrypted_shares should work if ceremony_complete=false");
73
74    let response = enclave.sign(SetupNewKeyResponse {
75        encrypted_shares,
76        secret_sharing_instance: ss_instance,
77    });
78
79    *ceremony_complete = true;
80    info!("Setup complete.");
81    Ok(response)
82}

84#[cfg(test)]
85mod tests {
86    use super::*;
87    use crate::mock_logger_capturing;
88    use crate::OperatorInitTestArgs;
89    use hashi_types::guardian::LogMessage;
90    use hashi_types::guardian::LogRecord;
91    use hashi_types::pgp::test_utils::mock_pgp_certs;
92
93    const TEST_N: usize = 5;
94    const TEST_T: usize = 3;
95
96    fn mock_setup_new_key_request() -> SetupNewKeyRequest {
97        SetupNewKeyRequest::new(mock_pgp_certs(TEST_N), TEST_N, TEST_T).unwrap()
98    }
99
100    #[tokio::test]
101    async fn test_setup_new_key() {
102        let (logger, captures) = mock_logger_capturing();
103        let enclave = Enclave::create_operator_initialized_with(
104            OperatorInitTestArgs::default().with_s3_logger(logger),
105        )
106        .await;
107        let verification_key = &enclave.signing_pubkey();
108        let request = mock_setup_new_key_request();
109        let resp = setup_new_key(enclave.clone(), request).await.unwrap();
110        let validated_resp = resp.verify(verification_key).unwrap();
111
112        // Response still carries the armored ciphertexts.
113        assert_eq!(validated_resp.encrypted_shares.len(), TEST_N);
114        assert_eq!(validated_resp.secret_sharing_instance.num_shares(), TEST_N);
115        assert_eq!(validated_resp.secret_sharing_instance.threshold(), TEST_T);
116        assert_eq!(validated_resp.secret_sharing_instance.sharing_seq(), 0);
117        assert_eq!(
118            validated_resp.secret_sharing_instance.commitments().len(),
119            TEST_N
120        );
121        for enc_share in validated_resp.encrypted_shares.iter() {
122            assert!(enc_share
123                .armored_ciphertext
124                .starts_with("-----BEGIN PGP MESSAGE-----"));
125        }
126
127        // The ceremony log records the instance only — no ciphertexts.
128        let captured = captures.lock().unwrap();
129        let ceremony_logs: Vec<_> = captured
130            .iter()
131            .filter(|(k, _)| k.starts_with("ceremony/"))
132            .collect();
133        assert_eq!(ceremony_logs.len(), 1, "expected one ceremony/ log");
134        let (_key, body) = ceremony_logs[0];
135        assert!(
136            !std::str::from_utf8(body)
137                .unwrap()
138                .contains("BEGIN PGP MESSAGE"),
139            "ceremony log must not contain ciphertexts"
140        );
141
142        let record: LogRecord = serde_json::from_slice(body).unwrap();
143        let LogMessage::Ceremony(ceremony) = record.message else {
144            panic!("expected Ceremony variant");
145        };
146        let CeremonyLogMessage::NewKey { instance, roster } = *ceremony else {
147            panic!("expected NewKey variant");
148        };
149        assert_eq!(instance.sharing_seq(), 0);
150        assert_eq!(instance.num_shares(), TEST_N);
151        assert_eq!(instance.threshold(), TEST_T);
152        // The roster commits one recipient fingerprint per share.
153        assert_eq!(roster.len(), TEST_N);
154
155        // The encrypted shares are persisted to shares/ keyed by sharing_seq,
156        // and carry the ciphertexts the ceremony log omits.
157        let shares_logs: Vec<_> = captured
158            .iter()
159            .filter(|(k, _)| k.starts_with("shares/"))
160            .collect();
161        assert_eq!(shares_logs.len(), 1, "expected one shares/ log");
162        let (shares_key, shares_body) = shares_logs[0];
163        assert_eq!(
164            *shares_key,
165            format!("shares/{:020}-{}.json", 0, enclave.s3_session_id())
166        );
167        let shares_record: LogRecord = serde_json::from_slice(shares_body).unwrap();
168        let LogMessage::Shares(shares) = shares_record.message else {
169            panic!("expected Shares variant");
170        };
171        assert_eq!(shares.sharing_seq, 0);
172        assert_eq!(shares.encrypted_shares.len(), TEST_N);
173        assert!(std::str::from_utf8(shares_body)
174            .unwrap()
175            .contains("BEGIN PGP MESSAGE"));
176    }
177}
