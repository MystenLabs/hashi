4//! `provisioner_init` (withdraw mode): collects the current KPs' encrypted
5//! shares, decrypts/verifies them under the enclave's `state_hash`, and
6//! reconstructs the BTC key once threshold shares are present. Runs after the
7//! shared `crate::operator_init`.
8
9use crate::Enclave;
10use hashi_types::guardian::crypto::combine_shares;
11use hashi_types::guardian::crypto::decrypt_verify_shares;
12use hashi_types::guardian::crypto::k256_sk_to_btc_keypair;
13use hashi_types::guardian::crypto::Share;
14use hashi_types::guardian::InitLogMessage::PIEnclaveFullyInitialized;
15use hashi_types::guardian::*;
16use std::sync::Arc;
17use tracing::info;
18use GuardianError::*;

20/// Receives the current KPs' encrypted shares in one submission. Decrypts each
21/// under the enclave's state_hash (set at operator_init) as AAD — so only shares
22/// from KPs that agreed on the operator-supplied state decrypt — verifies them
23/// against the commitments, and reconstructs the BTC key once threshold shares
24/// are present.
25pub async fn provisioner_init(
26    enclave: Arc<Enclave>,
27    request: ProvisionerInitRequest,
28) -> GuardianResult<()> {
29    info!("/provisioner_init - Received request.");
30
31    // Serialize so concurrent callers can't race the check-then-finalize below.
32    let _guard = enclave.control_lock.lock().await;
33
34    if !enclave.is_operator_init_complete() {
35        return Err(InvalidInputs("Do operator init first".into()));
36    }
37    if enclave.is_provisioner_init_complete() {
38        return Err(InvalidInputs("Provisioner init already complete".into()));
39    }
40    info!("Enclave state validated.");
41
42    let sk = enclave.encryption_secret_key();
43    let instance = enclave
44        .secret_sharing_instance()
45        .expect("secret-sharing instance should be set after operator_init");
46    let threshold = instance.threshold();
47    // Always set here: provisioner_init is withdraw-mode only, and the
48    // operator_init check above guarantees a withdraw-mode enclave installed it.
49    let state_hash = *enclave
50        .state_hash()
51        .expect("withdraw-mode operator_init installs the state_hash");
52
53    // Decrypt and verify every submission. A share only decrypts if its KP bound
54    // the enclave's state_hash as AAD, so the decrypted shares all agree on the
55    // operator-supplied state.
56    let shares = decrypt_verify_shares(
57        request.encrypted_shares(),
58        sk,
59        &state_hash,
60        instance.commitments(),
61        threshold,
62    )?;
63    info!("Verified {} shares (threshold {threshold}).", shares.len());
64
65    let share_ids = shares.iter().map(|s| s.id).collect();
66    finalize_init(&shares, threshold, &enclave).await;
67    // Log to S3 indicating that withdrawals can be expected henceforth. This log
68    // does not serve any security purpose.
69    enclave
70        .log_init(PIEnclaveFullyInitialized { share_ids })
71        .await
72        .expect("Unable to log EnclaveFullyInitialized");
73
74    enclave
75        .scratchpad
76        .provisioner_init_logging_complete
77        .set(())
78        .expect("provisioner_init_logging_complete should only be set once");
79
80    Ok(())
81}
82
83/// Reconstruct the BTC key from the threshold shares and install it. The rest of
84/// the enclave state was set at operator_init.
85/// Panics upon an error as the enclaves state is irrecoverable at this point.
86async fn finalize_init(shares: &[Share], threshold: usize, enclave: &Arc<Enclave>) {
87    info!("Threshold reached, combining shares.");
88    let enclave_k256_sk = combine_shares(shares, threshold).expect("Unable to combine shares");
89    let enclave_btc_keypair = k256_sk_to_btc_keypair(&enclave_k256_sk);
90
91    info!("Setting enclave keypair.");
92    enclave
93        .config
94        .set_btc_keypair(enclave_btc_keypair)
95        .expect("Unable to set enclave keypair");
96
97    info!("Enclave initialization complete.");
98}

100#[cfg(test)]
101mod tests {
102    use super::*;
103    use crate::OperatorInitTestArgs;
104    use k256::SecretKey;
105
106    const TEST_N: usize = 5;
107    const TEST_T: usize = 3;
108
109    /// Helper: Generate test shares and initialized enclave
110    /// Returns (shares, enclave)
111    async fn setup_test_shares_and_enclave() -> (Vec<Share>, Arc<Enclave>) {
112        let sk = SecretKey::random(&mut rand::thread_rng());
113        let params = SecretSharingParams::new(TEST_N, TEST_T).unwrap();
114        let shares = split_secret(&sk, &params, &mut rand::thread_rng());
115        let share_commitments = ShareCommitments::from_shares(&shares).unwrap();
116        let enclave = Enclave::create_operator_initialized_with(
117            OperatorInitTestArgs::default().with_commitments(share_commitments),
118        )
119        .await;
120        (shares, enclave)
121    }
122
123    /// Bundle one submission per share, all bound to the enclave's state_hash as
124    /// AAD — i.e. what the relay assembles from the current KPs.
125    fn build_request(shares: &[Share], enclave: &Enclave) -> ProvisionerInitRequest {
126        let state_hash = *enclave.state_hash().unwrap();
127        let submissions = shares
128            .iter()
129            .map(|s| {
130                ProvisionerInitRequest::build_from_share(
131                    s,
132                    enclave.encryption_public_key(),
133                    state_hash,
134                    &mut rand::thread_rng(),
135                )
136            })
137            .collect();
138        ProvisionerInitRequest::new(submissions)
139    }

141    #[tokio::test]
142    async fn happy_path_threshold_reached() {
143        let (shares, enclave) = setup_test_shares_and_enclave().await;
144        let req = build_request(&shares[..TEST_T], &enclave);
145
146        provisioner_init(enclave.clone(), req).await.expect("ok");
147        assert!(
148            enclave.config.is_enclave_btc_keypair_set(),
149            "Bitcoin key should be set after threshold"
150        );
151        assert!(enclave.is_fully_initialized(), "fully initialized");
152    }
153
154    #[tokio::test]
155    async fn rejects_second_call_after_complete() {
156        let (shares, enclave) = setup_test_shares_and_enclave().await;
157
158        let req = build_request(&shares[..TEST_T], &enclave);
159        provisioner_init(enclave.clone(), req).await.expect("ok");
160
161        // A second call is rejected outright (already complete).
162        let req2 = build_request(&shares[..TEST_T], &enclave);
163        let err = provisioner_init(enclave, req2)
164            .await
165            .expect_err("should reject");
166        assert!(matches!(err, InvalidInputs(_)));
167    }

169    #[tokio::test]
170    async fn rejects_below_threshold() {
171        let (shares, enclave) = setup_test_shares_and_enclave().await;
172        let req = build_request(&shares[..TEST_T - 1], &enclave);
173        let err = provisioner_init(enclave.clone(), req)
174            .await
175            .expect_err("should fail");
176        assert!(matches!(err, InvalidInputs(_)));
177        assert!(
178            !enclave.config.is_enclave_btc_keypair_set(),
179            "Bitcoin key should not be set below threshold"
180        );
181    }

183    #[tokio::test]
184    async fn rejects_before_operator_init() {
185        // Enclave without operator_init: rejected before any AAD is used.
186        let enclave = Enclave::create_with_random_keys();
187        let share = Share {
188            id: std::num::NonZeroU16::new(1).unwrap(),
189            value: k256::Scalar::ONE,
190        };
191        let enc = ProvisionerInitRequest::build_from_share(
192            &share,
193            enclave.encryption_public_key(),
194            [0u8; 32],
195            &mut rand::thread_rng(),
196        );
197        let req = ProvisionerInitRequest::new(vec![enc]);
198
199        let err = provisioner_init(enclave, req)
200            .await
201            .expect_err("should fail");
202        assert!(matches!(err, InvalidInputs(_)));
203    }

205    #[tokio::test]
206    async fn rejects_share_with_mismatched_state_hash() {
207        let (shares, enclave) = setup_test_shares_and_enclave().await;
208
209        // A KP that binds a state_hash differing from the enclave's (i.e. it
210        // disagreed on the operator-supplied state) produces a share that fails
211        // to decrypt — rejected gracefully, not via a panic.
212        let wrong_state_hash = [0xABu8; 32];
213        assert_ne!(&wrong_state_hash, enclave.state_hash().unwrap());
214        let enc = ProvisionerInitRequest::build_from_share(
215            &shares[0],
216            enclave.encryption_public_key(),
217            wrong_state_hash,
218            &mut rand::thread_rng(),
219        );
220        let req = ProvisionerInitRequest::new(vec![enc]);
221
222        let err = provisioner_init(enclave, req)
223            .await
224            .expect_err("should fail");
225        assert!(matches!(err, InvalidInputs(_)));
226    }
