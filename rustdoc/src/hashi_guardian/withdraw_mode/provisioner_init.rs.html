4//! `provisioner_init` (withdraw mode): verifies the current KPs' signed share
5//! submissions and reconstructs the BTC key once threshold shares are present.
6//! Runs after the shared `crate::operator_init`.
7
8use crate::withdraw_mode::genesis::ensure_no_serving_committee;
9use crate::Enclave;
10use hashi_types::guardian::crypto::combine_shares;
11use hashi_types::guardian::crypto::decrypt_verify_shares;
12use hashi_types::guardian::crypto::k256_sk_to_btc_keypair;
13use hashi_types::guardian::InitLogMessage::PIEnclaveFullyInitialized;
14use hashi_types::guardian::*;
15use std::sync::Arc;
16use tracing::info;

18/// Validated provisioner-init state ready for its fail-stop commit.
19///
20/// Construction performs every request-dependent fallible operation without
21/// mutating the enclave. Once built, installation must either complete or abort
22/// the enclave process.
23struct PIInstall {
24    enclave_btc_keypair: bitcoin::secp256k1::Keypair,
25    genesis_log: Option<GenesisLogMessage>,
26    completion_log: InitLogMessage,
27}
28
29impl PIInstall {
30    async fn from_request(
31        enclave: &Enclave,
32        request: ProvisionerInitRequest,
33    ) -> GuardianResult<Self> {
34        let initialization = enclave
35            .temporary_init_state()
36            .expect("temporary initialization state should be set after operator_init");
37        let ceremony_state = &initialization.ceremony_state;
38        let instance = &ceremony_state.secret_sharing_instance;
39        let threshold = instance.threshold();
40        let sharing_seq = instance.sharing_seq();
41        let config_hash = initialization.config_hash;
42        let session_id = enclave.s3_session_id();
43        let genesis_state = initialization.genesis_state.clone();
44        let genesis_state_hash = genesis_state.as_ref().map(GenesisState::digest);
45
46        let encrypted_shares = verify_signed_submissions(
47            &request,
48            &session_id,
49            &config_hash,
50            genesis_state_hash,
51            &ceremony_state.encrypted_shares,
52        )?;
53        let shares = decrypt_verify_shares(
54            &encrypted_shares,
55            enclave.encryption_secret_key(),
56            None,
57            instance.commitments(),
58            threshold,
59        )?;
60        info!("Verified {} shares (threshold {threshold}).", shares.len());
61
62        info!("Threshold reached, combining shares.");
63        let enclave_k256_sk = combine_shares(&shares, threshold)?;
64        let enclave_btc_keypair = k256_sk_to_btc_keypair(&enclave_k256_sk);
65        let enclave_btc_pubkey = enclave_btc_keypair.x_only_public_key().0;
66        let share_ids = shares.iter().map(|share| share.id).collect();
67
68        if genesis_state.is_some() {
69            ensure_no_serving_committee(enclave).await?;
70        }
71
72        Ok(Self {
73            enclave_btc_keypair,
74            genesis_log: genesis_state.map(|state| GenesisLogMessage {
75                committee: state.into_committee(),
76            }),
77            completion_log: PIEnclaveFullyInitialized {
78                sharing_seq,
79                share_ids,
80                enclave_btc_pubkey,
81            },
82        })
83    }
84}

86/// Receives the current KPs' signed share submissions in one batch. The relay
87/// may pre-verify them as a DoS guard, but the enclave authoritatively verifies
88/// each signature and session/config binding before decrypting and
89/// commitment-checking any share.
90pub async fn provisioner_init(
91    enclave: Arc<Enclave>,
92    request: ProvisionerInitRequest,
93) -> GuardianResult<()> {
94    info!("/provisioner_init - Received request.");
95
96    enclave.require_lifecycle(WithdrawStage::OperatorInitialized.into())?;
97    info!("Lifecycle stage validated.");
98
99    // ---- Validate & build: Nothing in this phase mutates enclave state, so any
100    // error here leaves the enclave untouched. ----
101    let install = PIInstall::from_request(&enclave, request).await?;
102
103    // ---- All-or-nothing Commit: Nothing in this phase errors out. ----
104    info!("Committing enclave BTC keypair.");
105    commit_provisioner_init(&enclave, install).await;
106
107    info!("Provisioner initialization complete.");
108    Ok(())
109}
110
111/// Install the prepared key, durably mark PI complete, and then expose the new
112/// lifecycle. This fail-stop phase never returns an error after mutation begins.
113async fn commit_provisioner_init(enclave: &Enclave, install: PIInstall) {
114    enclave
115        .config
116        .set_btc_keypair(install.enclave_btc_keypair)
117        .expect("Unable to set enclave keypair");
118
119    if let Some(genesis_log) = install.genesis_log {
120        let epoch = genesis_log.committee.epoch;
121        enclave
122            .log_genesis(genesis_log)
123            .await
124            .expect("Unable to log KP-authorized genesis state");
125        info!(epoch, "KP-authorized genesis committee written");
126    }
127
128    // OA waits for this durable marker before activating the enclave.
129    enclave
130        .log_init(install.completion_log)
131        .await
132        .expect("Unable to log EnclaveFullyInitialized");
133
134    enclave
135        .advance_lifecycle_into(WithdrawStage::ProvisionerInitialized.into())
136        .expect("provisioner_init should advance an operator-initialized enclave");
137}

139fn verify_signed_submissions(
140    request: &ProvisionerInitRequest,
141    live_session_id: &SessionID,
142    live_config_hash: &[u8; 32],
143    live_genesis_state_hash: Option<[u8; 32]>,
144    expected_kp_encrypted_shares: &KPEncryptedSharesRoster,
145) -> GuardianResult<Vec<GuardianEncryptedShare>> {
146    request
147        .0
148        .iter()
149        .map(|signed| {
150            let signer_fingerprint = signed.signer_fingerprint().to_hex();
151            signed.verify_signature()?;
152            let submission = &signed.data;
153
154            if submission.expected_session_id() != live_session_id.as_str() {
155                return Err(GuardianError::InvalidInputs(format!(
156                    "PI submission expected guardian session {}, live session is {}",
157                    submission.expected_session_id(),
158                    live_session_id
159                )));
160            }
161            if submission.expected_config_hash() != live_config_hash {
162                return Err(GuardianError::InvalidInputs(format!(
163                    "PI submission expected config hash {}, live config hash is {}",
164                    hex::encode(submission.expected_config_hash()),
165                    hex::encode(live_config_hash)
166                )));
167            }
168            if submission.expected_genesis_state_hash() != live_genesis_state_hash {
169                return Err(GuardianError::InvalidInputs(format!(
170                    "PI submission expected genesis state hash {:?}, live genesis state hash is {:?}",
171                    submission.expected_genesis_state_hash().map(hex::encode),
172                    live_genesis_state_hash.map(hex::encode)
173                )));
174            }
175
176            let share_id = submission.encrypted_share().id;
177            let (assigned_share, _) = expected_kp_encrypted_shares
178                .find_by_fingerprint(&signer_fingerprint)
179                .ok_or_else(|| {
180                    GuardianError::InvalidInputs(format!(
181                        "PI submission signer {signer_fingerprint} is not in the current KP \
182                         encrypted-share roster"
183                    ))
184                })?;
185            if assigned_share.id != share_id {
186                return Err(GuardianError::InvalidInputs(format!(
187                    "PI submission signer {signer_fingerprint} is assigned to KP share id {}, \
188                     not submitted share id {}",
189                    assigned_share.id.get(),
190                    share_id.get()
191                )));
192            }
193
194            info!(
195                share_id = share_id.get(),
196                signer_fingerprint, "verified signed PI submission"
197            );
198            Ok(submission.encrypted_share().clone())
199        })
200        .collect()
201}

337            expected_config_hash: [u8; 32],

345                expected_genesis_state_hash,

349        fn signed_submission_with_key_and_genesis_hash(
350            &self,
351            share: &Share,
352            signer: &(PgpPublicCert, String),
353            expected_session_id: SessionID,
354            expected_config_hash: [u8; 32],
355            expected_genesis_state_hash: Option<[u8; 32]>,
356        ) -> KpSigned<SingleProvisionerInitRequest> {
357            let request = SingleProvisionerInitRequest::build_from_share(
358                expected_session_id,
359                expected_config_hash,
360                expected_genesis_state_hash,
361                share,
362                self.enclave.encryption_public_key(),
363                &mut rand::thread_rng(),
364            );
365            let (cert, secret) = signer;
366            KpSigned {
367                signature: sign_detached_in_process(secret, &KpSigned::signed_bytes(&request)),
368                data: request,
369                signer_cert: cert.clone(),
370            }
371        }
372
373        fn request(&self, shares: &[Share]) -> ProvisionerInitRequest {
374            let session_id = self.enclave.s3_session_id();
375            let config_hash = self.config_hash();
376            let submissions = shares
377                .iter()
378                .map(|share| {
379                    self.signed_submission(
380                        share,
381                        usize::from(share.id.get() - 1),
382                        session_id.clone(),
383                        config_hash,
384                    )
385                })
386                .collect();
387            ProvisionerInitRequest(submissions)
388        }
389
390        async fn provision(&self, request: ProvisionerInitRequest) -> GuardianResult<()> {
391            provisioner_init(self.enclave.clone(), request).await
392        }
393    }

395    #[tokio::test]
396    async fn happy_path_threshold_reached() {
397        let ctx = setup().await;
398        ctx.provision(ctx.request(&ctx.shares[..TEST_T]))
399            .await
400            .expect("ok");
401        assert!(
402            ctx.enclave.config.is_enclave_btc_keypair_set(),
403            "Bitcoin key should be set after threshold"
404        );
405        assert_eq!(
406            ctx.enclave.lifecycle(),
407            WithdrawStage::ProvisionerInitialized.into(),
408            "provisioner init complete"
409        );
410        assert!(!ctx.enclave.is_fully_initialized(), "not active before OA");
411
412        let captured = ctx.captures.lock().unwrap();
413        assert_eq!(
414            captured.len(),
415            1,
416            "provisioner init should write one record"
417        );
418        let record: LogRecord = serde_json::from_slice(&captured[0].1).unwrap();
419        let VersionedLogMessage::V2(LogMessage::Init(message)) = record.message else {
420            panic!("expected V2 init record");
421        };
422        assert_eq!(
423            captured[0].0,
424            message.object_key(&ctx.enclave.s3_session_id())
425        );
426        let PIEnclaveFullyInitialized {
427            sharing_seq,
428            share_ids,
429            enclave_btc_pubkey,
430        } = *message
431        else {
432            panic!("expected provisioner-init completion record");
433        };
434        assert_eq!(sharing_seq, 0);
435        assert_eq!(
436            share_ids,
437            ctx.shares[..TEST_T]
438                .iter()
439                .map(|share| share.id)
440                .collect::<Vec<_>>()
441        );
442        assert_eq!(
443            enclave_btc_pubkey,
444            ctx.enclave.config.enclave_btc_pubkey().unwrap()
445        );
446    }
