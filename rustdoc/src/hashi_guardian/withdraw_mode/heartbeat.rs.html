4use crate::Enclave;
5use hashi_types::guardian::EnclaveMode;
6use hashi_types::guardian::GuardianResult;
7use hashi_types::guardian::HeartbeatLogMessage;
8use hashi_types::guardian::WithdrawStage;
9use std::sync::Arc;
10use std::time::Duration;
11
12/// Stateful heartbeat writer.
13pub struct HeartbeatWriter {
14    pub enclave: Arc<Enclave>,
15    /// the next sequence number used in s3 logs
16    pub next_seq: u64,
17}
18
19impl HeartbeatWriter {
20    pub fn new(enclave: Arc<Enclave>) -> Self {
21        assert_eq!(
22            enclave.mode(),
23            EnclaveMode::Withdraw,
24            "heartbeats are only supported in withdraw mode"
25        );
26        Self {
27            enclave,
28            next_seq: 0,
29        }
30    }
31
32    /// Attempt to send one heartbeat.
33    ///
34    /// - If operator init is not complete, this is a no-op.
35    ///
36    /// The shared S3 writer retries failures and aborts the process if its grace
37    /// period expires.
38    pub async fn tick(&mut self) -> GuardianResult<()> {
39        if self.enclave.lifecycle() == WithdrawStage::Uninitialized.into() {
40            return Ok(());
41        }
42
43        self.enclave
44            .log_heartbeat(HeartbeatLogMessage::new(self.next_seq))
45            .await?;
46        self.next_seq += 1;
47        Ok(())
48    }
49
50    /// Run the periodic heartbeat loop.
51    pub async fn run(mut self, interval: Duration) {
52        let mut delay = Duration::ZERO;
53        loop {
54            tokio::time::sleep(delay).await;
55            self.tick()
56                .await
57                .expect("heartbeat write failed unexpectedly");
58            delay = interval;
59        }
60    }
61}

63#[cfg(test)]
64mod tests {
65    use super::*;
66    use crate::OperatorInitTestArgs;
67    use hashi_types::guardian::LogMessage;
68    use hashi_types::guardian::LogRecord;
69    use hashi_types::guardian::VersionedLogMessage;
70
71    #[tokio::test]
72    async fn heartbeat_is_a_noop_before_operator_init() {
73        let mut writer = HeartbeatWriter::new(Enclave::create_with_random_keys());
74        writer.tick().await.unwrap();
75        assert_eq!(writer.next_seq, 0);
76    }
77
78    #[tokio::test]
79    async fn heartbeat_advances_after_durable_write() {
80        let (logger, captures) = crate::test_utils::mock_logger_capturing();
81        let enclave = Enclave::create_operator_initialized_with(
82            OperatorInitTestArgs::default().with_s3_logger(logger),
83        )
84        .await;
85        let mut writer = HeartbeatWriter::new(enclave);
86        writer.tick().await.unwrap();
87        assert_eq!(writer.next_seq, 1);
88
89        let captured = captures.lock().unwrap();
90        assert_eq!(captured.len(), 1, "heartbeat tick should write one record");
91        let record: LogRecord = serde_json::from_slice(&captured[0].1).unwrap();
92        assert_eq!(captured[0].0, record.object_key());
93        let VersionedLogMessage::V2(LogMessage::Heartbeat(message)) = record.message else {
94            panic!("expected V2 heartbeat record");
95        };
96        assert_eq!(message.seq, 0);
97    }
98}
