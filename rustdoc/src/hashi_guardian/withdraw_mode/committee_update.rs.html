20/// Advance the committee to a future epoch with a cert from the outgoing
21/// committee. Hashi epochs can skip values (reconfig is sparse), so the
22/// proposed epoch is only required to be strictly greater than the
23/// current one; sequentiality is not enforced.
24/// Idempotent on already-applied or older transitions.
25pub async fn update_committee(
26    enclave: Arc<Enclave>,
27    signed: HashiSigned<CommitteeTransitionRequest>,
28) -> GuardianResult<u64> {
29    // Serialize so a stalled call can't roll the committee backwards.
30    let _guard = enclave.control_lock.lock().await;
31
32    if !enclave.is_fully_initialized() {
33        return Err(EnclaveUninitialized);
34    }
35
36    let current = enclave.state.get_committee()?;
37    let current_epoch = current.epoch();
38    let proposed_epoch = signed.message().new_committee.epoch;
39
40    if proposed_epoch <= current_epoch {
41        info!(current_epoch, proposed_epoch, "update_committee: no-op");
42        return Ok(current_epoch);
43    }
44
45    // Match hashi's leader, which bails its sig collection at
46    // `certificate_threshold(from_committee.total_weight())`. Using a
47    // higher (configured) threshold here would reject otherwise-valid
48    // certs.
49    let threshold = certificate_threshold(current.total_weight());
50    if let Err(e) = verify_hashi_cert(current.clone(), threshold, &signed) {
51        log_failure(&enclave, current_epoch, &signed, &e).await?;
52        return Err(e);
53    }
54
55    let new_committee: HashiCommittee = signed
56        .message()
57        .new_committee
58        .clone()
59        .try_into()
60        .map_err(|e| InvalidInputs(format!("invalid new committee in transition: {e}")))?;
61
62    if new_committee.epoch() != proposed_epoch {
63        let err = InvalidInputs(format!(
64            "new committee epoch ({}) does not match transition epoch ({proposed_epoch})",
65            new_committee.epoch()
66        ));
67        log_failure(&enclave, current_epoch, &signed, &err).await?;
68        return Err(err);
69    }
70
71    // Log before the in-memory swap so failed S3 writes don't advance the committee.
72    log_success(&enclave, current_epoch, &signed).await?;
73    enclave
74        .state
75        .replace_committee(new_committee, current_epoch)
76        .expect("committee initialized at current_epoch under the update lock");
77
78    info!(
79        from_epoch = current_epoch,
80        to_epoch = proposed_epoch,
81        "Committee updated"
82    );
83    Ok(proposed_epoch)
84}

86pub async fn update_committee_chain(
87    enclave: Arc<Enclave>,
88    transitions: Vec<HashiSigned<CommitteeTransitionRequest>>,
89) -> GuardianResult<u64> {
90    let mut current_epoch = enclave.state.get_committee()?.epoch();
91    for signed in transitions {
92        current_epoch = update_committee(enclave.clone(), signed).await?;
93    }
94    Ok(current_epoch)
95}
96
97async fn log_success(
98    enclave: &Enclave,
99    from_epoch: u64,
100    signed: &HashiSigned<CommitteeTransitionRequest>,
101) -> GuardianResult<()> {
102    let msg = CommitteeUpdateLogMessage::Success {
103        from_epoch,
104        new_committee: signed.message().new_committee.clone(),
105        request_sign: signed.committee_signature().clone(),
106    };
107    enclave.log_committee_update(msg).await
108}
109
110async fn log_failure(
111    enclave: &Enclave,
112    from_epoch: u64,
113    signed: &HashiSigned<CommitteeTransitionRequest>,
114    err: &GuardianError,
115) -> GuardianResult<()> {
116    let msg = CommitteeUpdateLogMessage::Failure {
117        from_epoch,
118        new_committee: signed.message().new_committee.clone(),
119        request_sign: signed.committee_signature().clone(),
120        error: err.clone(),
121    };
122    if let Err(log_err) = enclave.log_committee_update(msg).await {
123        error!(
124            from_epoch,
125            "failed to log committee update failure to S3: {log_err:?}"
126        );
127        return Err(InternalError(format!(
128            "Failed to log committee update error {err} due to S3 logging error {log_err}"
129        )));
130    }
131    Ok(())
132}

154    fn mock_signer_address() -> SuiAddress {
155        SuiAddress::new([1u8; 32])
156    }
157
158    fn mock_bls_sk() -> Bls12381PrivateKey {
159        use rand::SeedableRng;
160        let mut rng = rand::rngs::StdRng::seed_from_u64(0x000C_0FFE_EBAD_F00D);
161        Bls12381PrivateKey::generate(&mut rng)
162    }
163
164    fn mock_encryption_pk() -> EncryptionPublicKey {
165        use rand::SeedableRng;
166        let mut rng = rand::rngs::StdRng::seed_from_u64(0xDEAD_BEEF);
167        let sk = hashi_types::committee::EncryptionPrivateKey::new(&mut rng);
168        EncryptionPublicKey::from_private_key(&sk)
169    }
170
171    fn committee_at(epoch: u64) -> HashiCommittee {
172        let pk = mock_bls_sk().public_key();
173        let member = HashiCommitteeMember::new(mock_signer_address(), pk, mock_encryption_pk(), 10);
174        HashiCommittee::new(
175            vec![member],
176            epoch,
177            DEFAULT_MPC_THRESHOLD_IN_BASIS_POINTS,
178            DEFAULT_MPC_WEIGHT_REDUCTION_ALLOWED_DELTA,
179            DEFAULT_MPC_MAX_FAULTY_IN_BASIS_POINTS,
180            VANILLA_MPC_NONCE_GENERATION_PROTOCOL,
181        )
182    }
183
184    fn sign_transition_at(
185        signing_epoch: u64,
186        new_committee: HashiCommittee,
187    ) -> HashiSigned<CommitteeTransitionRequest> {
188        let outgoing = committee_at(signing_epoch);
189        let transition = CommitteeTransitionRequest {
190            new_committee: hashi_types::move_types::Committee::from(&new_committee),
191        };
192        let sk = mock_bls_sk();
193        let sig = sk.sign(signing_epoch, mock_signer_address(), &transition);
194        let mut agg = BlsSignatureAggregator::new(&outgoing, transition);
195        agg.add_signature(sig).expect("member sig should verify");
196        agg.finish().expect("threshold should be met")
197    }

200        let kp = create_btc_keypair_for_test(&[1u8; 32]);
201        let master_pubkey = hashi_master_g_from_btc_xonly_for_test(&kp.x_only_public_key().0);
202        create_fully_initialized_enclave(FullyInitializedArgs {
203            network: Network::Regtest,
204            committee: committee_at(epoch),
205            master_pubkey,
206            limiter_config: LimiterConfig {

215        })

217    }
